52auto flow_controller::available_send_window() const noexcept -> uint64_t
53{
54 if (bytes_sent_ >= send_limit_) {
55 return 0;
56 }
57 return send_limit_ - bytes_sent_;
58}
59
60auto flow_controller::consume_send_window(uint64_t bytes) -> VoidResult
61{
62 if (bytes == 0) {
63 return ok();
64 }
65
66 const uint64_t available = available_send_window();
67 if (bytes > available) {
68 return error_void(flow_control_error::send_blocked,
69 "Send window exhausted",
70 "quic::flow_controller");
71 }
72
73 bytes_sent_ += bytes;
74 data_blocked_sent_ = false; // Reset blocked flag on successful send
75 return ok();
76}
77
78void flow_controller::update_send_limit(uint64_t max_data)
79{
80 // Only update if new limit is higher (MAX_DATA can only increase)
81 if (max_data > send_limit_) {
82 send_limit_ = max_data;
83 data_blocked_sent_ = false; // Can send again, reset blocked flag
84 }
85}
86
87auto flow_controller::is_send_blocked() const noexcept -> bool
88{
89 return bytes_sent_ >= send_limit_;
90}

96auto flow_controller::record_received(uint64_t bytes) -> VoidResult
97{
98 if (bytes == 0) {
99 return ok();
100 }
101
102 const uint64_t new_total = bytes_received_ + bytes;
103 if (new_total > receive_limit_) {
104 return error_void(flow_control_error::receive_overflow,
105 "Received data exceeds flow control limit",
106 "quic::flow_controller");
107 }
108
109 bytes_received_ = new_total;
110 return ok();
111}
112
113void flow_controller::record_consumed(uint64_t bytes)
114{
115 bytes_consumed_ += bytes;
116 // Consumed cannot exceed received
117 if (bytes_consumed_ > bytes_received_) {
118 bytes_consumed_ = bytes_received_;
119 }
120}

126auto flow_controller::should_send_max_data() const noexcept -> bool
127{
128 // Send MAX_DATA when we've consumed enough of the window
129 const uint64_t threshold = static_cast<uint64_t>(window_size_ * update_threshold_);
130
131 // Calculate how much of the advertised window has been consumed
132 const uint64_t consumed_from_last = bytes_consumed_ - (last_sent_max_data_ - window_size_);
133
134 return consumed_from_last >= threshold;
135}
136
137auto flow_controller::generate_max_data() -> std::optional<uint64_t>
138{
139 if (!should_send_max_data()) {
140 return std::nullopt;
141 }
142
143 // New limit is current consumption point plus window size
144 const uint64_t new_limit = bytes_consumed_ + window_size_;
145
146 // Only send if it's an increase
147 if (new_limit <= last_sent_max_data_) {
148 return std::nullopt;
149 }
150
151 receive_limit_ = new_limit;
152 last_sent_max_data_ = new_limit;
153 return new_limit;
154}

156auto flow_controller::should_send_data_blocked() const noexcept -> bool

159}

202auto get_flow_control_stats(const flow_controller& fc) -> flow_control_stats
203{
204 flow_control_stats stats;
205
206 // Send side
207 stats.send_limit = fc.send_limit();
208 stats.bytes_sent = fc.bytes_sent();
209 stats.send_window_available = fc.available_send_window();
210 stats.send_blocked = fc.is_send_blocked();
211
212 // Receive side
213 stats.receive_limit = fc.receive_limit();
214 stats.bytes_received = fc.bytes_received();
215 stats.bytes_consumed = fc.bytes_consumed();
216 stats.receive_window_available = fc.receive_limit() - fc.bytes_received();

218 return stats;
