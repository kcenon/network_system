70dtls_socket::dtls_socket(asio::ip::udp::socket socket, SSL_CTX* ssl_ctx)
71 : socket_(std::move(socket))
72 , ssl_ctx_(ssl_ctx)
73 , ssl_(nullptr)
74 , rbio_(nullptr)
75 , wbio_(nullptr)
76{
77 // Create SSL object
78 ssl_ = SSL_new(ssl_ctx_);
79 if (!ssl_)
80 {
81 throw std::runtime_error("Failed to create SSL object");
82 }
83
84 // Create memory BIOs for non-blocking I/O
85 rbio_ = BIO_new(BIO_s_mem());
86 wbio_ = BIO_new(BIO_s_mem());
87 if (!rbio_ || !wbio_)
88 {
89 if (rbio_) BIO_free(rbio_);
90 if (wbio_) BIO_free(wbio_);
91 SSL_free(ssl_);
92 throw std::runtime_error("Failed to create BIO objects");
93 }
94
95 // Set BIOs to non-blocking mode
96 BIO_set_nbio(rbio_, 1);
97 BIO_set_nbio(wbio_, 1);
98
99 // Connect BIOs to SSL object (SSL takes ownership)
100 SSL_set_bio(ssl_, rbio_, wbio_);
101
102 // Enable DTLS cookie exchange for servers (DoS protection)
103 // This is optional and can be configured later
104}

117auto dtls_socket::async_handshake(
118 handshake_type type,
119 std::function<void(std::error_code)> handler) -> void
120{
121 handshake_type_ = type;
122 handshake_callback_ = std::move(handler);
123 handshake_in_progress_.store(true);
124
125 {
126 std::lock_guard<std::mutex> lock(ssl_mutex_);
127
128 if (type == handshake_type::client)
129 {
130 SSL_set_connect_state(ssl_);
131 }
132 else
133 {
134 SSL_set_accept_state(ssl_);
135 }
136 }
137
138 // Start receiving for handshake packets
139 start_receive();
140
141 // Initiate handshake
142 continue_handshake();
143}

145auto dtls_socket::continue_handshake() -> void
146{
147 if (!handshake_in_progress_.load())
148 {
149 return;
150 }
151
152 int result;
153 {
154 std::lock_guard<std::mutex> lock(ssl_mutex_);
155 result = SSL_do_handshake(ssl_);
156 }
157
158 if (result == 1)
159 {
160 // Handshake complete
161 handshake_in_progress_.store(false);
162 handshake_complete_.store(true);
163
164 // Flush any remaining output
165 flush_bio_output();
166
167 std::function<void(std::error_code)> callback;
168 {
169 std::lock_guard<std::mutex> lock(callback_mutex_);
170 callback = std::move(handshake_callback_);
171 }
172
173 if (callback)
174 {
175 callback(std::error_code{});
176 }
177 return;
178 }
179
180 int ssl_error;
181 {
182 std::lock_guard<std::mutex> lock(ssl_mutex_);
183 ssl_error = SSL_get_error(ssl_, result);
184 }
185
186 // Flush any output generated by the handshake
187 flush_bio_output();
188
189 if (ssl_error == SSL_ERROR_WANT_READ || ssl_error == SSL_ERROR_WANT_WRITE)
190 {
191 // Need more data - continue receiving
192 return;
193 }
194
195 // Handshake failed
196 handshake_in_progress_.store(false);
197
198 std::function<void(std::error_code)> callback;
199 {
200 std::lock_guard<std::mutex> lock(callback_mutex_);
201 callback = std::move(handshake_callback_);
202 }
203
204 if (callback)
205 {
206 callback(make_ssl_error());
207 }
208}

239auto dtls_socket::do_receive() -> void
240{
241 if (!is_receiving_.load())
242 {
243 return;
244 }
245
246 auto self = shared_from_this();
247 socket_.async_receive_from(
248 asio::buffer(read_buffer_),
249 sender_endpoint_,
250 [this, self](std::error_code ec, std::size_t length)
251 {
252 if (!is_receiving_.load())
253 {
254 return;
255 }
256
257 if (ec)
258 {
259 std::function<void(std::error_code)> callback;
260 {
261 std::lock_guard<std::mutex> lock(callback_mutex_);
262 callback = error_callback_;
263 }
264 if (callback)
265 {
266 callback(ec);
267 }
268 return;
269 }
270
271 if (length > 0)
272 {
273 std::vector<uint8_t> data(read_buffer_.begin(),
274 read_buffer_.begin() + length);
275 process_received_data(data, sender_endpoint_);
276 }
277
278 // Continue receiving
279 if (is_receiving_.load())
280 {
281 do_receive();
282 }
283 });
284}

286auto dtls_socket::process_received_data(const std::vector<uint8_t>& data,
287 const asio::ip::udp::endpoint& sender) -> void
288{
289 // Write received data to the read BIO
290 {
291 std::lock_guard<std::mutex> lock(ssl_mutex_);
292 int written = BIO_write(rbio_, data.data(), static_cast<int>(data.size()));
293 if (written <= 0)
294 {
295 // BIO write failed
296 return;
297 }
298 }
299
300 // If handshake is in progress, continue it
301 if (handshake_in_progress_.load())
302 {
303 continue_handshake();
304 return;
305 }
306
307 // Handshake complete, try to read decrypted data
308 if (handshake_complete_.load())
309 {
310 std::vector<uint8_t> decrypted;
311 decrypted.resize(65536);
312
313 int read_len;
314 {
315 std::lock_guard<std::mutex> lock(ssl_mutex_);
316 read_len = SSL_read(ssl_, decrypted.data(), static_cast<int>(decrypted.size()));
317 }
318
319 if (read_len > 0)
320 {
321 decrypted.resize(static_cast<std::size_t>(read_len));
322
323 std::function<void(const std::vector<uint8_t>&, const asio::ip::udp::endpoint&)> callback;
324 {
325 std::lock_guard<std::mutex> lock(callback_mutex_);
326 callback = receive_callback_;
327 }
328 if (callback)
329 {
330 callback(decrypted, sender);
331 }
332 }
333 else
334 {
335 int ssl_error;
336 {
337 std::lock_guard<std::mutex> lock(ssl_mutex_);
338 ssl_error = SSL_get_error(ssl_, read_len);
339 }
340
341 if (ssl_error != SSL_ERROR_WANT_READ && ssl_error != SSL_ERROR_WANT_WRITE)
342 {
343 // Real error
344 std::function<void(std::error_code)> callback;
345 {
346 std::lock_guard<std::mutex> lock(callback_mutex_);
347 callback = error_callback_;
348 }
349 if (callback)
350 {
351 callback(make_ssl_error());
352 }
353 }
354 }
355 }

358auto dtls_socket::flush_bio_output() -> void
359{
360 // Check if there's data to send in the write BIO
361 std::vector<uint8_t> output;
362 {
363 std::lock_guard<std::mutex> lock(ssl_mutex_);
364 int pending = BIO_ctrl_pending(wbio_);
365 if (pending <= 0)
366 {
367 return;
368 }
369
370 output.resize(static_cast<std::size_t>(pending));
371 int read_len = BIO_read(wbio_, output.data(), pending);
372 if (read_len <= 0)
373 {
374 return;
375 }
376 output.resize(static_cast<std::size_t>(read_len));
377 }
378
379 // Send the encrypted data
380 asio::ip::udp::endpoint target;
381 {
382 std::lock_guard<std::mutex> lock(endpoint_mutex_);
383 target = peer_endpoint_;
384 }
385
386 if (target.port() != 0)
387 {
388 auto buffer = std::make_shared<std::vector<uint8_t>>(std::move(output));
389 socket_.async_send_to(
390 asio::buffer(*buffer),
391 target,
392 [buffer](std::error_code /*ec*/, std::size_t /*bytes*/)
393 {
394 // Fire and forget for handshake messages
395 });
396 }
397}

412auto dtls_socket::async_send_to(
413 std::vector<uint8_t>&& data,
414 const asio::ip::udp::endpoint& endpoint,
415 std::function<void(std::error_code, std::size_t)> handler) -> void
416{
417 if (!handshake_complete_.load())
418 {
419 if (handler)
420 {
421 handler(std::make_error_code(std::errc::not_connected), 0);
422 }
423 return;
424 }
425
426 // Encrypt the data
427 int written;
428 {
429 std::lock_guard<std::mutex> lock(ssl_mutex_);
430 written = SSL_write(ssl_, data.data(), static_cast<int>(data.size()));
431 }
432
433 if (written <= 0)
434 {
435 if (handler)
436 {
437 handler(make_ssl_error(), 0);
438 }
439 return;
440 }
441
442 // Get encrypted data from write BIO
443 std::vector<uint8_t> encrypted;
444 {
445 std::lock_guard<std::mutex> lock(ssl_mutex_);
446 int pending = BIO_ctrl_pending(wbio_);
447 if (pending <= 0)
448 {
449 if (handler)
450 {
451 handler(std::make_error_code(std::errc::io_error), 0);
452 }
453 return;
454 }
455
456 encrypted.resize(static_cast<std::size_t>(pending));
457 int read_len = BIO_read(wbio_, encrypted.data(), pending);
458 if (read_len <= 0)
459 {
460 if (handler)
461 {
462 handler(std::make_error_code(std::errc::io_error), 0);
463 }
464 return;
465 }
466 encrypted.resize(static_cast<std::size_t>(read_len));
467 }
468
469 // Send encrypted data
470 auto self = shared_from_this();
471 auto buffer = std::make_shared<std::vector<uint8_t>>(std::move(encrypted));
472 auto original_size = data.size();
473
474 socket_.async_send_to(
475 asio::buffer(*buffer),
476 endpoint,
477 [handler = std::move(handler), buffer, original_size](
478 std::error_code ec, std::size_t /*bytes_sent*/)
479 {
480 if (handler)
481 {
482 // Report original (plaintext) size to the handler
483 handler(ec, ec ? 0 : original_size);
484 }
485 });
486}
