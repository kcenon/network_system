24auto ws_listener_adapter::start(const endpoint_info& bind_address) -> VoidResult
25{
26 if (!server_) {
27 return error_void(
28 static_cast<int>(std::errc::invalid_argument),
29 "Server is not initialized"
30 );
31 }
32
33 if (server_->is_running()) {
34 return error_void(
35 static_cast<int>(std::errc::already_connected),
36 "Already listening"
37 );
38 }
39
40 {
41 std::lock_guard lock(endpoint_mutex_);
42 local_endpoint_ = bind_address;
43 }
44
45 {
46 std::lock_guard lock(connections_mutex_);
47 connections_.clear();
48 }
49
50 return server_->start_server(bind_address.port, ws_path_);
51}

58auto ws_listener_adapter::stop() noexcept -> void
59{
60 if (server_ && server_->is_running()) {
61 // Notify disconnection for all connections
62 {
63 std::lock_guard cb_lock(callbacks_mutex_);
64 std::lock_guard conn_lock(connections_mutex_);
65
66 if (callbacks_.on_disconnect) {
67 for (const auto& [conn_id, _] : connections_) {
68 callbacks_.on_disconnect(conn_id);
69 }
70 }
71 connections_.clear();
72 }
73
74 (void)server_->stop_server();
75 }
76}

111auto ws_listener_adapter::send_to(std::string_view connection_id,
112 std::span<const std::byte> data) -> VoidResult
113{
114 std::shared_ptr<core::ws_connection> conn;
115 {
116 std::lock_guard lock(connections_mutex_);
117 auto it = connections_.find(std::string(connection_id));
118 if (it == connections_.end()) {
119 return error_void(
120 static_cast<int>(std::errc::no_such_device_or_address),
121 "Connection not found: " + std::string(connection_id)
122 );
123 }
124 conn = it->second;
125 }
126
127 if (!conn || !conn->is_connected()) {
128 return error_void(
129 static_cast<int>(std::errc::not_connected),
130 "Connection is no longer valid"
131 );
132 }
133
134 std::vector<uint8_t> buffer(data.size());
135 std::memcpy(buffer.data(), data.data(), data.size());
136
137 return conn->send_binary(std::move(buffer));
138}

156auto ws_listener_adapter::close_connection(std::string_view connection_id) noexcept -> void
157{
158 std::shared_ptr<core::ws_connection> conn;
159 {
160 std::lock_guard lock(connections_mutex_);
161 auto it = connections_.find(std::string(connection_id));
162 if (it != connections_.end()) {
163 conn = it->second;
164 connections_.erase(it);
165 }
166 }
167
168 if (conn) {
169 conn->close();
170 }
171}

185auto ws_listener_adapter::setup_internal_callbacks() -> void
186{
187 if (!server_) {
188 return;
189 }
190
191 // Bridge connection callback
192 server_->set_connection_callback(
193 [this](std::shared_ptr<interfaces::i_websocket_session> session) {
194 if (!session) {
195 return;
196 }
197
198 // Use the session's ID as connection ID
199 std::string conn_id(session->id());
200
201 // Try to get the actual ws_connection from server
202 auto ws_conn = server_->get_connection(conn_id);
203 if (ws_conn) {
204 std::lock_guard lock(connections_mutex_);
205 connections_[conn_id] = ws_conn;
206 }
207
208 // Call user callbacks
209 {
210 std::lock_guard lock(callbacks_mutex_);
211 if (callbacks_.on_accept) {
212 callbacks_.on_accept(conn_id);
213 }
214 }
215 });
216
217 // Bridge disconnection callback
218 server_->set_disconnection_callback(
219 [this](std::string_view session_id, uint16_t /*code*/, std::string_view /*reason*/) {
220 std::string conn_id(session_id);
221
222 {
223 std::lock_guard lock(connections_mutex_);
224 connections_.erase(conn_id);
225 }
226
227 {
228 std::lock_guard lock(callbacks_mutex_);
229 if (callbacks_.on_disconnect) {
230 callbacks_.on_disconnect(conn_id);
231 }
232 }
233 });
234
235 // Bridge binary message callback
236 server_->set_binary_callback(
237 [this](std::string_view session_id, const std::vector<uint8_t>& data) {
238 std::string conn_id(session_id);
239
240 std::lock_guard lock(callbacks_mutex_);
241 if (callbacks_.on_data) {
242 std::span<const std::byte> byte_span(
243 reinterpret_cast<const std::byte*>(data.data()),
244 data.size()
245 );
246 callbacks_.on_data(conn_id, byte_span);
247 }
248 });
249
250 // Also bridge text messages as binary data for unified interface
251 server_->set_text_callback(
252 [this](std::string_view session_id, const std::string& text) {
253 std::string conn_id(session_id);
254
255 std::lock_guard lock(callbacks_mutex_);
256 if (callbacks_.on_data) {
257 std::span<const std::byte> byte_span(
258 reinterpret_cast<const std::byte*>(text.data()),
259 text.size()
260 );
261 callbacks_.on_data(conn_id, byte_span);
262 }
263 });
264
265 // Bridge error callback
266 server_->set_error_callback(
267 [this](std::string_view session_id, std::error_code ec) {
268 std::string conn_id(session_id);
269
270 std::lock_guard lock(callbacks_mutex_);
271 if (callbacks_.on_error) {
272 callbacks_.on_error(conn_id, ec);
273 }
274 });
275}
