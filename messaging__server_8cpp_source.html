54 messaging_server::~messaging_server() noexcept
55 {
56 if (is_running())
57 {
58 (void)stop_server();
59 }
60 // Additional cleanup for resources not managed by lifecycle manager.
61 try
62 {
63 // Ensure all resources are cleaned up even if stop_server() returned early
64 // This handles the case where start_server() failed and is_running() is false
65 // but resources were partially created
66 // Lock to prevent race with do_accept() accessing the acceptor
67 {
68 std::lock_guard<std::mutex> lock(acceptor_mutex_);
69 if (acceptor_)
70 {
71 asio::error_code ec;
72 acceptor_->cancel(ec);
73 if (acceptor_->is_open())
74 {
75 acceptor_->close(ec);
76 }
77 acceptor_.reset();
78 }
79 }
80
81 if (cleanup_timer_)
82 {
83 cleanup_timer_->cancel();
84 cleanup_timer_.reset();
85 }
86
87 if (work_guard_)
88 {
89 work_guard_.reset();
90 }
91
92 if (io_context_)
93 {
94 io_context_->stop();
95 io_context_.reset();
96 }
97 }
98 catch (...)
99 {
100 // Destructor must not throw - swallow all exceptions
101 }
102 }

108 auto messaging_server::start_server(unsigned short port) -> VoidResult
109 {
110 // Create tracing span for server start operation
111 auto span = tracing::is_tracing_enabled()
112 ? std::make_optional(tracing::trace_context::create_span("tcp.server.start"))
113 : std::nullopt;
114 if (span)
115 {
116 span->set_attribute("net.host.port", static_cast<int64_t>(port))
117 .set_attribute("net.transport", "tcp")
118 .set_attribute("server.id", server_id_);
119 }
120
121 // Use base class do_start which handles lifecycle management
122 auto result = do_start(port);
123 if (result.is_err())
124 {
125 if (span)
126 {
127 span->set_error(result.error().message);
128 }
129 }
130 else
131 {
132 if (span)
133 {
134 span->set_status(tracing::span_status::ok);
135 }
136 }
137
138 return result;
139 }

217 auto messaging_server::do_start_impl(unsigned short port) -> VoidResult
218 {
219 // This method handles TCP-specific initialization.
220 try
221 {
222 // Create io_context and acceptor
223 io_context_ = std::make_shared<asio::io_context>();
224 // Create work guard to keep io_context running
225 work_guard_ = std::make_unique<asio::executor_work_guard<asio::io_context::executor_type>>(
226 asio::make_work_guard(*io_context_)
227 );
228 acceptor_ = std::make_unique<tcp::acceptor>(
229 *io_context_, tcp::endpoint(tcp::v4(), port));
230
231 // Create cleanup timer
232 cleanup_timer_ = std::make_unique<asio::steady_timer>(*io_context_);
233
234 // Begin accepting connections
235 do_accept();
236
237 // Start periodic cleanup timer
238 start_cleanup_timer();
239
240 // Use io_context_thread_manager for unified thread management
241 io_context_future_ = integration::io_context_thread_manager::instance()
242 .run_io_context(io_context_, "messaging_server:" + server_id());
243
244 NETWORK_LOG_INFO("[messaging_server] Started listening on port " + std::to_string(port));
245 return ok();
246 }
247 catch (const std::system_error& e)
248 {
249 // Cleanup partially created resources to prevent memory corruption
250 // Resources may have been created before the exception was thrown
251 acceptor_.reset();
252 cleanup_timer_.reset();
253 work_guard_.reset();
254 io_context_.reset();
255
256 // Check for specific error codes (ASIO uses asio::error category)
257 if (e.code() == asio::error::address_in_use ||
258 e.code() == std::errc::address_in_use)
259 {
260 return error_void(
261 error_codes::network_system::bind_failed,
262 "Failed to bind to port: address already in use",
263 "messaging_server::do_start",
264 "Port: " + std::to_string(port)
265 );
266 }
267 else if (e.code() == asio::error::access_denied ||
268 e.code() == std::errc::permission_denied)
269 {
270 return error_void(
271 error_codes::network_system::bind_failed,
272 "Failed to bind to port: permission denied",
273 "messaging_server::do_start",
274 "Port: " + std::to_string(port)
275 );
276 }
277
278 return error_void(
279 error_codes::common_errors::internal_error,
280 "Failed to start server: " + std::string(e.what()),
281 "messaging_server::do_start",
282 "Port: " + std::to_string(port)
283 );
284 }
285 catch (const std::exception& e)
286 {
287 // Cleanup partially created resources to prevent memory corruption
288 acceptor_.reset();
289 cleanup_timer_.reset();
290 work_guard_.reset();
291 io_context_.reset();
292
293 return error_void(
294 error_codes::common_errors::internal_error,
295 "Failed to start server: " + std::string(e.what()),
296 "messaging_server::do_start",
297 "Port: " + std::to_string(port)
298 );
299 }
300 }

302 auto messaging_server::do_stop_impl() -> VoidResult
303 {
304 // This method handles TCP-specific cleanup.
305 try
306 {
307 // Step 1: Cancel and close the acceptor to stop accepting new connections
308 // Lock to prevent race with do_accept() accessing the acceptor
309 {
310 std::lock_guard<std::mutex> lock(acceptor_mutex_);
311 if (acceptor_)
312 {
313 asio::error_code ec;
314 // Cancel pending async_accept operations to prevent memory leaks
315 acceptor_->cancel(ec);
316 if (acceptor_->is_open())
317 {
318 acceptor_->close(ec);
319 }
320 }
321 }
322
323 // Cancel cleanup timer
324 if (cleanup_timer_)
325 {
326 cleanup_timer_->cancel();
327 }
328
329 // Step 2: Stop all active sessions (close sockets, stop reading)
330 // NOTE: Do NOT clear sessions here - they may still be referenced by
331 // pending async callbacks. We must wait for io_context to finish first.
332 {
333 std::lock_guard<std::mutex> lock(sessions_mutex_);
334 for (auto& sess : sessions_)
335 {
336 if (sess)
337 {
338 sess->stop_session();
339 }
340 }
341 }
342
343 // Step 3: Release work guard to allow io_context to finish
344 if (work_guard_)
345 {
346 work_guard_.reset();
347 }
348
349 // Step 4: Stop io_context via unified manager
350 if (io_context_)
351 {
352 integration::io_context_thread_manager::instance().stop_io_context(io_context_);
353 }
354
355 // Step 5: Wait for io_context task to complete
356 // This ensures all pending async callbacks have finished executing
357 if (io_context_future_.valid())
358 {
359 io_context_future_.wait();
360 }
361
362 // Step 5.5: Run remaining handlers to clean up pending async operations.
363 // When io_context::stop() is called, pending handlers may not have
364 // executed yet. Running poll() ensures all handlers are invoked and
365 // their captured resources (resolver/socket) are properly destroyed
366 // before io_context is destroyed, preventing heap corruption.
367 if (io_context_)
368 {
369 try
370 {
371 io_context_->restart();
372 io_context_->poll();
373 }
374 catch (...)
375 {
376 // Ignore exceptions during cleanup
377 }
378 }
379
380 // Step 6: NOW it's safe to clear sessions - all async operations are done
381 {
382 std::lock_guard<std::mutex> lock(sessions_mutex_);
383 sessions_.clear();
384 }
385
386 // Step 7: Release resources explicitly to ensure cleanup
387 acceptor_.reset();
388 cleanup_timer_.reset();
389 io_context_.reset();
390
391 NETWORK_LOG_INFO("[messaging_server] Stopped.");
392 return ok();
393 }
394 catch (const std::exception& e)
395 {
396 return error_void(
397 error_codes::common_errors::internal_error,
398 "Failed to stop server: " + std::string(e.what()),
399 "messaging_server::do_stop",
400 "Server ID: " + server_id()
401 );
402 }

405 auto messaging_server::do_accept() -> void
406 {
407 // Lock to prevent race with do_stop() closing the acceptor
408 std::lock_guard<std::mutex> lock(acceptor_mutex_);
409
410 // Early return if server is stopping or acceptor is invalid
411 if (!is_running() || !acceptor_ || !acceptor_->is_open())
412 {
413 return;
414 }
415
416 auto self = shared_from_this();
417 acceptor_->async_accept(
418 [this, self](std::error_code ec, tcp::socket sock)
419 { on_accept(ec, std::move(sock)); });
420 }

422 auto messaging_server::on_accept(std::error_code ec, tcp::socket socket)
423 -> void
424 {
425 if (!is_running())
426 {
427 return;
428 }
429
430 // Create tracing span for accept operation
431 auto span = tracing::is_tracing_enabled()
432 ? std::make_optional(tracing::trace_context::create_span("tcp.server.accept"))
433 : std::nullopt;
434 if (span)
435 {
436 span->set_attribute("net.transport", "tcp")
437 .set_attribute("server.id", server_id_);
438 }
439
440 if (ec)
441 {
442 if (span)
443 {
444 span->set_error(ec.message());
445 }
446 NETWORK_LOG_ERROR("[messaging_server] Accept error: " + ec.message());
447#if KCENON_WITH_COMMON_SYSTEM
448 // Record connection error
449 connection_errors_.fetch_add(1, std::memory_order_relaxed);
450 if (monitor_) {
451 monitor_->record_metric("connection_errors", static_cast<double>(connection_errors_.load()));
452 }
453#endif
454 // Report metrics
455 metrics::metric_reporter::report_connection_failed(ec.message());
456 return;
457 }
458
459 // Add peer information to span
460 if (span)
461 {
462 try
463 {
464 auto endpoint = socket.remote_endpoint();
465 span->set_attribute("net.peer.ip", endpoint.address().to_string())
466 .set_attribute("net.peer.port", static_cast<int64_t>(endpoint.port()));
467 }
468 catch (...)
469 {
470 // Ignore errors getting endpoint info
471 }
472 }
473
474 // Report successful connection
475 metrics::metric_reporter::report_connection_accepted();
476
477 // Cleanup dead sessions before adding new one
478 cleanup_dead_sessions();
479
480 // Create a new messaging_session
481 auto new_session = std::make_shared<kcenon::network::session::messaging_session>(
482 std::move(socket), server_id());
483
484 // Set up session callbacks to forward to server callbacks
485 // Use base class getters to avoid direct member access
486 auto self = shared_from_this();
487
488 // Get copies of callbacks via thread-safe getters from base class
489 auto local_receive_callback = get_receive_callback();
490 auto local_disconnection_callback = get_disconnection_callback();
491 auto local_error_callback = get_error_callback();
492
493 // Set up callbacks without holding any locks to avoid lock-order-inversion
494 // Lambdas use captured local callbacks directly to avoid locking callback_mutex_
495 if (local_receive_callback)
496 {
497 new_session->set_receive_callback(
498 [self, new_session, local_receive_callback](const std::vector<uint8_t>& data)
499 {
500 // Call the captured callback directly without locking to prevent deadlock
501 local_receive_callback(new_session, data);
502 });
503 }
504
505 if (local_disconnection_callback)
506 {
507 new_session->set_disconnection_callback(
508 [self, local_disconnection_callback](const std::string& session_id)
509 {
510 // Call the captured callback directly without locking to prevent deadlock
511 local_disconnection_callback(session_id);
512 });
513 }
514
515 if (local_error_callback)
516 {
517 new_session->set_error_callback(
518 [self, new_session, local_error_callback](std::error_code ec)
519 {
520 // Call the captured callback directly without locking to prevent deadlock
521 local_error_callback(new_session, ec);
522 });
523 }
524
525 // Track it in our sessions_ vector (protected by mutex)
526 {
527 std::lock_guard<std::mutex> lock(sessions_mutex_);
528 sessions_.push_back(new_session);
529 }
530
531 // Start the session
532 new_session->start_session();
533
534 // Invoke connection callback via base class helper
535 invoke_connection_callback(new_session);
536
537#if KCENON_WITH_COMMON_SYSTEM
538 // Record active connections metric
539 {
540 std::lock_guard<std::mutex> lock(sessions_mutex_);
541 if (monitor_) {
542 monitor_->record_metric("active_connections", static_cast<double>(sessions_.size()));
543 }
544 }
545#endif
546
547 // Report active connections to metrics system
548 {
549 std::lock_guard<std::mutex> lock(sessions_mutex_);
550 metrics::metric_reporter::report_active_connections(sessions_.size());
551 }
552
553 // Mark span as successful before accepting next connection
554 if (span)
555 {
556 span->set_status(tracing::span_status::ok);
557 }
558
559 // Accept next connection
560 do_accept();
561 }

563 auto messaging_server::cleanup_dead_sessions() -> void
564 {
565 std::lock_guard<std::mutex> lock(sessions_mutex_);
566
567 // Remove sessions that have been stopped
568 sessions_.erase(
569 std::remove_if(sessions_.begin(), sessions_.end(),
570 [](const auto& session) {
571 return session && session->is_stopped();
572 }),
573 sessions_.end()
574 );
575
576 NETWORK_LOG_DEBUG("[messaging_server] Cleaned up dead sessions. Active: " +
577 std::to_string(sessions_.size()));
578
579#if KCENON_WITH_COMMON_SYSTEM
580 // Update active connections metric after cleanup
581 if (monitor_) {
582 monitor_->record_metric("active_connections", static_cast<double>(sessions_.size()));
583 }
584#endif
585 }

587 auto messaging_server::start_cleanup_timer() -> void
588 {
589 if (!cleanup_timer_ || !is_running())
590 {
591 return;
592 }
593
594 // Schedule cleanup every 30 seconds
595 cleanup_timer_->expires_after(std::chrono::seconds(30));
596
597 auto self = shared_from_this();
598 cleanup_timer_->async_wait(
599 [this, self](const std::error_code& ec)
600 {
601 if (!ec && is_running())
602 {
603 cleanup_dead_sessions();
604 start_cleanup_timer(); // Reschedule
605 }
606 }
607 );
608 }
