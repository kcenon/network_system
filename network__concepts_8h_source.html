83template <typename T>
84concept ByteBuffer = requires(const T t) {
85 { t.data() } -> std::convertible_to<const void*>;
86 { t.size() } -> std::convertible_to<std::size_t>;
87};
88
102template <typename T>
103concept MutableByteBuffer = ByteBuffer<T> && requires(T t, std::size_t n) {
104 { t.resize(n) };
105 { t.data() } -> std::convertible_to<void*>;
106};

260template <typename T>
261concept NetworkClient = requires(T t, std::vector<uint8_t> data) {
262 { t.is_connected() } -> std::convertible_to<bool>;
263 { t.send_packet(std::move(data)) };
264 { t.stop_client() };
265};
266
284template <typename T>
285concept NetworkServer = requires(T t, unsigned short port) {
286 { t.start_server(port) };
287 { t.stop_server() };
288};
289
303template <typename T>
304concept NetworkSession = requires(T t) {
305 { t.get_session_id() } -> std::convertible_to<std::string>;
306 { t.start_session() };
307 { t.stop_session() };
308};

329template <typename T>
330concept DataTransformer = requires(T t, std::vector<uint8_t>& data) {
331 { t.transform(data) } -> std::convertible_to<bool>;
332};
333
347template <typename T>
348concept ReversibleDataTransformer =
349 DataTransformer<T> && requires(T t, std::vector<uint8_t>& data) {
350 { t.reverse_transform(data) } -> std::convertible_to<bool>;
351 };
