64 auto websocket_protocol::process_data(std::span<const uint8_t> data) -> void
65 {
66 // Append incoming data to buffer (single copy from span view)
67 buffer_.insert(buffer_.end(), data.begin(), data.end());
68
69 // Process frames from buffer
70 process_frames();
71 }

73 auto websocket_protocol::process_frames() -> void
74 {
75 while (true)
76 {
77 // Try to decode frame header
78 auto header_opt = websocket_frame::decode_header(buffer_);
79 if (!header_opt.has_value())
80 {
81 // Not enough data for header, wait for more
82 return;
83 }
84
85 const auto& header = header_opt.value();
86
87 // Calculate total frame size
88 const size_t header_size =
89 websocket_frame::calculate_header_size(header.payload_len, header.mask);
90 const size_t frame_size = header_size + header.payload_len;
91
92 // Check if we have the complete frame
93 if (buffer_.size() < frame_size)
94 {
95 // Incomplete frame, wait for more data
96 return;
97 }
98
99 // Decode payload
100 auto payload = websocket_frame::decode_payload(header, buffer_);
101
102 // Remove processed frame from buffer
103 buffer_.erase(buffer_.begin(), buffer_.begin() + frame_size);
104
105 // Dispatch frame based on opcode
106 if (header.opcode == ws_opcode::text || header.opcode == ws_opcode::binary ||
107 header.opcode == ws_opcode::continuation)
108 {
109 handle_data_frame(header, payload);
110 }
111 else
112 {
113 handle_control_frame(header, payload);
114 }
115 }
116 }

118 auto websocket_protocol::handle_data_frame(const ws_frame_header& header,
119 const std::vector<uint8_t>& payload)
120 -> void
121 {
122 // Handle fragmentation
123 if (header.opcode == ws_opcode::continuation)
124 {
125 // Continuation frame
126 if (fragmented_message_.empty())
127 {
128 // Protocol error: continuation without initial frame
129 return;
130 }
131
132 // Append to fragmented message
133 fragmented_message_.insert(fragmented_message_.end(), payload.begin(),
134 payload.end());
135
136 if (header.fin)
137 {
138 // Final fragment - complete message
139 ws_message msg;
140 msg.type = (fragmented_type_ == ws_opcode::text) ? ws_message_type::text
141 : ws_message_type::binary;
142 msg.data = std::move(fragmented_message_);
143
144 // Validate UTF-8 for text messages
145 if (msg.type == ws_message_type::text && !is_valid_utf8(msg.data))
146 {
147 // Invalid UTF-8, protocol error
148 fragmented_message_.clear();
149 return;
150 }
151
152 // Clear fragmentation state
153 fragmented_message_.clear();
154 fragmented_type_ = ws_opcode::continuation;
155
156 // Invoke callback
157 if (message_callback_)
158 {
159 message_callback_(msg);
160 }
161 }
162 }
163 else
164 {
165 // Initial frame (text or binary)
166 if (!header.fin)
167 {
168 // Start of fragmented message
169 fragmented_type_ = header.opcode;
170 fragmented_message_ = payload;
171 }
172 else
173 {
174 // Complete message in single frame
175 ws_message msg;
176 msg.type = (header.opcode == ws_opcode::text) ? ws_message_type::text
177 : ws_message_type::binary;
178 msg.data = payload;
179
180 // Validate UTF-8 for text messages
181 if (msg.type == ws_message_type::text && !is_valid_utf8(msg.data))
182 {
183 // Invalid UTF-8, protocol error
184 return;
185 }
186
187 // Invoke callback
188 if (message_callback_)
189 {
190 message_callback_(msg);
191 }
192 }
193 }
194 }

196 auto websocket_protocol::handle_control_frame(const ws_frame_header& header,
197 const std::vector<uint8_t>& payload)
198 -> void
199 {
200 // Control frames must not be fragmented
201 if (!header.fin)
202 {
203 // Protocol error
204 return;
205 }
206
207 // Control frames must have payload <= 125 bytes
208 if (header.payload_len > 125)
209 {
210 // Protocol error
211 return;
212 }
213
214 switch (header.opcode)
215 {
216 case ws_opcode::ping:
217 handle_ping(payload);
218 break;
219 case ws_opcode::pong:
220 handle_pong(payload);
221 break;
222 case ws_opcode::close:
223 handle_close(payload);
224 break;
225 default:
226 // Unknown control frame
227 break;
228 }
229 }

247 auto websocket_protocol::handle_close(const std::vector<uint8_t>& payload) -> void
248 {
249 ws_close_code code = ws_close_code::normal;
250 std::string reason;
251
252 if (payload.size() >= 2)
253 {
254 // Extract close code (network byte order)
255 code = static_cast<ws_close_code>((payload[0] << 8) | payload[1]);
256
257 // Extract reason (if present)
258 if (payload.size() > 2)
259 {
260 reason = std::string(payload.begin() + 2, payload.end());
261
262 // Validate UTF-8 in reason
263 std::vector<uint8_t> reason_bytes(reason.begin(), reason.end());
264 if (!is_valid_utf8(reason_bytes))
265 {
266 // Invalid UTF-8 in close reason
267 code = ws_close_code::protocol_error;
268 reason.clear();
269 }
270 }
271 }
272
273 if (close_callback_)
274 {
275 close_callback_(code, reason);
276 }
277 }

330 auto websocket_protocol::create_close(ws_close_code code, std::string&& reason)
331 -> std::vector<uint8_t>
332 {
333 std::vector<uint8_t> payload;
334
335 // Encode close code (network byte order)
336 payload.push_back(static_cast<uint8_t>((static_cast<uint16_t>(code) >> 8) & 0xFF));
337 payload.push_back(static_cast<uint8_t>(static_cast<uint16_t>(code) & 0xFF));
338
339 // Append reason (if provided)
340 if (!reason.empty())
341 {
342 // Validate UTF-8 in reason
343 std::vector<uint8_t> reason_bytes(reason.begin(), reason.end());
344 if (is_valid_utf8(reason_bytes))
345 {
346 payload.insert(payload.end(), reason.begin(), reason.end());
347 }
348 }
349
350 // Ensure payload <= 125 bytes
351 if (payload.size() > 125)
352 {
353 payload.resize(125);
354 }
355
356 return websocket_frame::encode_frame(ws_opcode::close, std::move(payload), true,
357 is_client_);
358 }

384 auto websocket_protocol::is_valid_utf8(const std::vector<uint8_t>& data) -> bool
385 {
386 size_t i = 0;
387 while (i < data.size())
388 {
389 uint8_t byte = data[i];
390
391 // Determine the number of bytes in this UTF-8 character
392 int bytes_to_follow = 0;
393
394 if ((byte & 0x80) == 0x00)
395 {
396 // 1-byte character (0xxxxxxx)
397 bytes_to_follow = 0;
398 }
399 else if ((byte & 0xE0) == 0xC0)
400 {
401 // 2-byte character (110xxxxx)
402 bytes_to_follow = 1;
403 }
404 else if ((byte & 0xF0) == 0xE0)
405 {
406 // 3-byte character (1110xxxx)
407 bytes_to_follow = 2;
408 }
409 else if ((byte & 0xF8) == 0xF0)
410 {
411 // 4-byte character (11110xxx)
412 bytes_to_follow = 3;
413 }
414 else
415 {
416 // Invalid UTF-8 start byte
417 return false;
418 }
419
420 // Check if we have enough bytes
421 if (i + bytes_to_follow >= data.size())
422 {
423 return false;
424 }
425
426 // Validate continuation bytes
427 for (int j = 1; j <= bytes_to_follow; ++j)
428 {
429 if ((data[i + j] & 0xC0) != 0x80)
430 {
431 // Invalid continuation byte (should be 10xxxxxx)
432 return false;
433 }
434 }
435
436 i += bytes_to_follow + 1;
437 }
438
439 return true;
440 }
