56VoidResult ObservabilityBridge::initialize(const BridgeConfig& config) {
57 if (initialized_.load()) {
58 return error_void(
59 error_codes::common_errors::already_exists,
60 "ObservabilityBridge already initialized",
61 "ObservabilityBridge::initialize");
62 }
63
64 if (!logger_) {
65 return error_void(
66 error_codes::common_errors::invalid_argument,
67 "Logger is null",
68 "ObservabilityBridge::initialize");
69 }
70
71 if (!monitor_) {
72 return error_void(
73 error_codes::common_errors::invalid_argument,
74 "Monitor is null",
75 "ObservabilityBridge::initialize");
76 }
77
78 // Check if bridge is enabled (default: true)
79 auto enabled_it = config.properties.find("enabled");
80 if (enabled_it != config.properties.end() && enabled_it->second == "false") {
81 return error_void(
82 error_codes::common_errors::invalid_argument,
83 "Bridge is disabled in configuration",
84 "ObservabilityBridge::initialize");
85 }
86
87 // Check if monitoring is enabled (default: true)
88 auto monitoring_enabled_it = config.properties.find("enable_monitoring");
89 monitoring_enabled_ = (monitoring_enabled_it == config.properties.end() ||
90 monitoring_enabled_it->second != "false");
91
92 // Initialize metrics
93 std::lock_guard<std::mutex> lock(metrics_mutex_);
94 cached_metrics_.is_healthy = true;
95 cached_metrics_.last_activity = std::chrono::steady_clock::now();
96 cached_metrics_.custom_metrics["backend_type"] = static_cast<double>(backend_type_);
97 cached_metrics_.custom_metrics["monitoring_enabled"] = monitoring_enabled_ ? 1.0 : 0.0;
98 cached_metrics_.custom_metrics["logger_available"] = logger_ ? 1.0 : 0.0;
99 cached_metrics_.custom_metrics["monitor_available"] = monitor_ ? 1.0 : 0.0;
100
101 initialized_.store(true);
102 return ok();
103}

105VoidResult ObservabilityBridge::shutdown() {
106 if (!initialized_.load()) {
107 return ok(); // Idempotent: already shut down
108 }
109
110 // Flush logger before shutdown
111 if (logger_) {
112 try {
113 logger_->flush();
114 } catch (...) {
115 // Ignore flush errors during shutdown
116 }
117 }
118
119 // Update metrics to reflect shutdown state
120 {
121 std::lock_guard<std::mutex> lock(metrics_mutex_);
122 cached_metrics_.is_healthy = false;
123 cached_metrics_.last_activity = std::chrono::steady_clock::now();
124 }
125
126 initialized_.store(false);
127 return ok();
128}

134BridgeMetrics ObservabilityBridge::get_metrics() const {
135 std::lock_guard<std::mutex> lock(metrics_mutex_);
136
137 if (!initialized_.load() || !logger_ || !monitor_) {
138 BridgeMetrics metrics;
139 metrics.is_healthy = false;
140 metrics.last_activity = cached_metrics_.last_activity;
141 return metrics;
142 }
143
144 // Update metrics with current state
145 BridgeMetrics metrics = cached_metrics_;
146 metrics.is_healthy = (logger_ && monitor_);
147 metrics.last_activity = std::chrono::steady_clock::now();
148 metrics.custom_metrics["backend_type"] = static_cast<double>(backend_type_);
149 metrics.custom_metrics["monitoring_enabled"] = monitoring_enabled_ ? 1.0 : 0.0;
150 metrics.custom_metrics["logger_available"] = logger_ ? 1.0 : 0.0;
151 metrics.custom_metrics["monitor_available"] = monitor_ ? 1.0 : 0.0;
152
153 // Update cached metrics for next call
154 cached_metrics_ = metrics;
155
156 return metrics;
157}

172std::shared_ptr<ObservabilityBridge> ObservabilityBridge::from_common_system(
173 std::shared_ptr<::kcenon::common::interfaces::ILogger> logger,
174 std::shared_ptr<::kcenon::common::interfaces::IMonitor> monitor) {
175 if (!logger) {
176 throw std::invalid_argument("ObservabilityBridge::from_common_system requires non-null logger");
177 }
178 if (!monitor) {
179 throw std::invalid_argument("ObservabilityBridge::from_common_system requires non-null monitor");
180 }
181
182 // Adapt common_system logger and monitor to network_system interfaces
183 auto adapted_logger = std::make_shared<common_system_logger_adapter>();
184
185 // For monitor, we use basic_monitoring since there's no direct adapter yet
186 // This is a known limitation - future work could create a proper adapter
187 auto adapted_monitor = std::make_shared<basic_monitoring>();
188
189 return std::make_shared<ObservabilityBridge>(
190 adapted_logger,
191 adapted_monitor,
192 BackendType::CommonSystem);
193}
