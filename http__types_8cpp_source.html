41 // Convert string to lowercase for case-insensitive comparison
42 auto to_lower(std::string str) -> std::string
43 {
44 std::transform(str.begin(), str.end(), str.begin(),
45 [](unsigned char c) { return std::tolower(c); });
46 return str;
47 }
48 }

52 auto http_request::get_header(const std::string& name) const -> std::optional<std::string>
53 {
54 auto lower_name = to_lower(name);
55 for (const auto& [key, value] : headers)
56 {
57 if (to_lower(key) == lower_name)
58 {
59 return value;
60 }
61 }
62 return std::nullopt;
63 }

65 auto http_request::set_header(const std::string& name, const std::string& value) -> void
66 {
67 // Remove existing header with same name (case-insensitive)
68 auto lower_name = to_lower(name);
69 for (auto it = headers.begin(); it != headers.end();)
70 {
71 if (to_lower(it->first) == lower_name)
72 {
73 it = headers.erase(it);
74 }
75 else
76 {
77 ++it;
78 }
79 }
80
81 // Add new header
82 headers[name] = value;
83 }

159 auto string_to_http_method(const std::string& method_str) -> ::kcenon::network::internal::Result<http_method>
160 {
161 auto upper_method = method_str;
162 std::transform(upper_method.begin(), upper_method.end(), upper_method.begin(),
163 [](unsigned char c) { return std::toupper(c); });
164
165 if (upper_method == "GET") return ::kcenon::network::ok(http_method::HTTP_GET);
166 if (upper_method == "POST") return ::kcenon::network::ok(http_method::HTTP_POST);
167 if (upper_method == "PUT") return ::kcenon::network::ok(http_method::HTTP_PUT);
168 if (upper_method == "DELETE") return ::kcenon::network::ok(http_method::HTTP_DELETE);
169 if (upper_method == "HEAD") return ::kcenon::network::ok(http_method::HTTP_HEAD);
170 if (upper_method == "OPTIONS") return ::kcenon::network::ok(http_method::HTTP_OPTIONS);
171 if (upper_method == "PATCH") return ::kcenon::network::ok(http_method::HTTP_PATCH);
172 if (upper_method == "CONNECT") return ::kcenon::network::ok(http_method::HTTP_CONNECT);
173 if (upper_method == "TRACE") return ::kcenon::network::ok(http_method::HTTP_TRACE);
174
175 return ::kcenon::network::error<http_method>(
176 ::kcenon::network::error_codes::common_errors::invalid_argument,
177 "Unknown HTTP method: " + method_str,
178 "http_types");
179 }

264 // cookie methods
265 auto cookie::to_header_value() const -> std::string
266 {
267 std::string result = name + "=" + value;
268
269 if (!path.empty()) {
270 result += "; Path=" + path;
271 }
272 if (!domain.empty()) {
273 result += "; Domain=" + domain;
274 }
275 if (!expires.empty()) {
276 result += "; Expires=" + expires;
277 }
278 if (max_age >= 0) {
279 result += "; Max-Age=" + std::to_string(max_age);
280 }
281 if (http_only) {
282 result += "; HttpOnly";
283 }
284 if (secure) {
285 result += "; Secure";
286 }
287 if (!same_site.empty()) {
288 result += "; SameSite=" + same_site;
289 }
290
291 return result;
292 }
