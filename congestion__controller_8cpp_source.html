94auto congestion_controller::on_packet_acked(const sent_packet& packet,
95 std::chrono::steady_clock::time_point ack_time)
96 -> void
97{
98 // Update bytes in flight
99 if (packet.in_flight && bytes_in_flight_ >= packet.sent_bytes)
100 {
101 bytes_in_flight_ -= packet.sent_bytes;
102 }
103
104 // Don't increase cwnd in recovery
105 if (is_in_recovery(packet.sent_time))
106 {
107 return;
108 }
109
110 // Exit recovery if we're receiving ACKs for packets sent after recovery started
111 if (state_ == congestion_state::recovery)
112 {
113 state_ = congestion_state::congestion_avoidance;
114 }
115
116 // Congestion window increase (RFC 9002 Section 7.3.1)
117 if (state_ == congestion_state::slow_start)
118 {
119 // Slow start: increase cwnd by the number of bytes acknowledged
120 cwnd_ += packet.sent_bytes;
121
122 // Check if we've exceeded slow start threshold
123 if (cwnd_ >= ssthresh_)
124 {
125 state_ = congestion_state::congestion_avoidance;
126 }
127 }
128 else
129 {
130 // Congestion avoidance: AIMD
131 // cwnd += max_datagram_size * acked_bytes / cwnd
132 auto increment = (max_datagram_size_ * packet.sent_bytes) / cwnd_;
133 if (increment == 0)
134 {
135 increment = 1; // Always increase by at least 1 byte
136 }
137 cwnd_ += increment;
138 }
139}

141auto congestion_controller::on_packet_lost(const sent_packet& packet) -> void
142{
143 // Update bytes in flight
144 if (packet.in_flight && bytes_in_flight_ >= packet.sent_bytes)
145 {
146 bytes_in_flight_ -= packet.sent_bytes;
147 }
148
149 // Trigger congestion event
150 on_congestion_event(packet.sent_time);
151}

153auto congestion_controller::on_congestion_event(
154 std::chrono::steady_clock::time_point sent_time) -> void
155{
156 // Only respond to congestion once per RTT (RFC 9002 Section 7.3.2)
157 if (is_in_recovery(sent_time))
158 {
159 return;
160 }
161
162 // Enter recovery
163 congestion_recovery_start_ = std::chrono::steady_clock::now();
164 state_ = congestion_state::recovery;
165
166 // Reduce cwnd and set ssthresh
167 // ssthresh = cwnd * kLossReductionFactor
168 // cwnd = max(ssthresh, minimum_window)
169 ssthresh_ = static_cast<size_t>(
170 static_cast<double>(cwnd_) * kLossReductionFactor);
171 cwnd_ = std::max(ssthresh_, minimum_window_);
172}

174auto congestion_controller::on_ecn_congestion(
175 std::chrono::steady_clock::time_point sent_time) -> void
176{
177 // RFC 9002 Section 7.1: Processing ECN Information
178 // ECN-CE marks indicate congestion without packet loss
179 // Respond same as packet loss, but only once per RTT
180
181 // Only respond to congestion once per RTT
182 if (is_in_recovery(sent_time))
183 {
184 return;
185 }
186
187 // Enter recovery
188 congestion_recovery_start_ = std::chrono::steady_clock::now();
189 state_ = congestion_state::recovery;
190
191 // Reduce cwnd and set ssthresh
192 // RFC 9002 Section 7.1: "a sender that receives an ACK frame with
193 // ECN feedback indicating the CE codepoint MUST enter congestion
194 // avoidance"
195 ssthresh_ = static_cast<size_t>(
196 static_cast<double>(cwnd_) * kLossReductionFactor);
197 cwnd_ = std::max(ssthresh_, minimum_window_);
198}

200auto congestion_controller::on_persistent_congestion(const rtt_estimator& rtt) -> void
201{
202 // RFC 9002 Section 7.6: Reset cwnd to minimum
203 cwnd_ = minimum_window_;
204 ssthresh_ = cwnd_;
205 state_ = congestion_state::slow_start;
206 congestion_recovery_start_ = std::chrono::steady_clock::time_point{};
207}

209auto congestion_controller::is_in_recovery(
210 std::chrono::steady_clock::time_point sent_time) const noexcept -> bool
211{
212 if (state_ != congestion_state::recovery)
213 {
214 return false;
215 }
216
217 // We're in recovery if the packet was sent before recovery started
218 return sent_time <= congestion_recovery_start_;
219}
