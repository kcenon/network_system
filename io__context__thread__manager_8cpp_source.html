48class io_context_thread_manager::impl {
49public:
50 struct context_entry {
51 std::weak_ptr<asio::io_context> context;
52 std::string component_name;
53 };

55 impl() : total_started_(0), total_completed_(0) {}
56
57 ~impl() {
58 stop_all();
59 wait_all();

62 std::shared_ptr<thread_pool_interface> get_thread_pool() {
63 std::lock_guard<std::mutex> lock(mutex_);
64 if (!thread_pool_) {
65 thread_pool_ = kcenon::network::core::network_context::instance().get_thread_pool();
66 if (!thread_pool_) {
67 // Fallback to basic thread pool with size for concurrent io_contexts
68 // Tests like ConnectionScaling need 20+ clients + server
69 // Use larger size to avoid thread exhaustion
70 auto pool_size = std::max(32u, std::thread::hardware_concurrency() * 4);
71 thread_pool_ = std::make_shared<basic_thread_pool>(pool_size);
72 }
73 }
74 return thread_pool_;
75 }

77 std::future<void> run_io_context(
78 std::shared_ptr<asio::io_context> io_context,
79 const std::string& component_name
80 ) {
81 auto pool = get_thread_pool();
82 if (!pool) {
83 auto promise = std::make_shared<std::promise<void>>();
84 promise->set_exception(
85 std::make_exception_ptr(
86 std::runtime_error("Thread pool not available")
87 )
88 );
89 return promise->get_future();
90 }
91
92 // Submit io_context::run to thread pool
93 auto ctx_weak = std::weak_ptr<asio::io_context>(io_context);
94 auto name = component_name.empty() ? "unnamed" : component_name;
95
96 // Create a shared promise for the caller's future
97 auto caller_promise = std::make_shared<std::promise<void>>();
98 auto caller_future = caller_promise->get_future();
99
100 // Track the entry for metrics and stop_all functionality
101 {
102 std::lock_guard<std::mutex> lock(mutex_);
103 context_entry entry;
104 entry.context = io_context;
105 entry.component_name = name;
106 entries_.push_back(std::move(entry));
107 total_started_.fetch_add(1, std::memory_order_relaxed);
108 }
109
110 // Submit the io_context::run task - this is the ONLY task we submit
111 // No separate monitoring task to avoid thread pool exhaustion
112 // Note: Capture total_completed_ by pointer to avoid capturing 'this',
113 // which can cause heap corruption during static destruction.
114 auto* completed_ptr = &total_completed_;
115 // Submit io_context::run task to thread pool
116 // IMPORTANT: Avoid logging inside this task to prevent heap corruption during
117 // static destruction. When common_system's GlobalLoggerRegistry is destroyed
118 // before this thread pool task completes, logging causes heap corruption.
119 // Use detail::static_destruction_guard::is_logging_safe() for any necessary logging.
120 pool->submit([ctx_weak, caller_promise, completed_ptr]() {
121 auto ctx = ctx_weak.lock();
122 if (!ctx) {
123 // io_context expired - this can happen during rapid shutdown
124 caller_promise->set_value();
125 return;
126 }
127
128 try {
129 ctx->run();
130 caller_promise->set_value();
131 } catch (...) {
132 caller_promise->set_exception(std::current_exception());
133 }
134
135 completed_ptr->fetch_add(1, std::memory_order_relaxed);
136 });
137
138 // Clean up expired entries periodically
139 cleanup_expired();
140
141 return caller_future;
142 }

144 void stop_io_context(std::shared_ptr<asio::io_context> io_context) {
145 // NOTE: No logging here to prevent heap corruption during static
146 // destruction. This may be called when GlobalLoggerRegistry is destroyed.
147 if (io_context) {
148 io_context->stop();
149 }
150 }
151
152 void stop_all() {
153 // NOTE: No logging here to prevent heap corruption during static
154 // destruction. This may be called when GlobalLoggerRegistry is destroyed.
155 std::lock_guard<std::mutex> lock(mutex_);
156 for (auto& entry : entries_) {
157 auto ctx = entry.context.lock();
158 if (ctx) {
159 ctx->stop();
160 }
161 }
162 }
163
164 void wait_all() {
165 // Since futures are returned directly to callers, wait_all just
166 // waits for all tracked io_contexts to be stopped.
167 // The callers should wait on their own futures.
168 auto deadline = std::chrono::steady_clock::now() + std::chrono::seconds(10);
169
170 while (std::chrono::steady_clock::now() < deadline) {
171 {
172 std::lock_guard<std::mutex> lock(mutex_);
173 bool all_stopped = true;
174 for (const auto& entry : entries_) {
175 auto ctx = entry.context.lock();
176 if (ctx && !ctx->stopped()) {
177 all_stopped = false;
178 break;
179 }
180 }
181 if (all_stopped) {
182 entries_.clear();
183 break;
184 }
185 }
186 std::this_thread::sleep_for(std::chrono::milliseconds(10));
187 }
188
189 // Clear any remaining expired entries
190 std::lock_guard<std::mutex> lock(mutex_);
191 entries_.clear();
192 }
193
194 size_t active_count() const {
195 std::lock_guard<std::mutex> lock(mutex_);
196 size_t count = 0;
197 for (const auto& entry : entries_) {
198 auto ctx = entry.context.lock();
199 if (ctx && !ctx->stopped()) {
200 ++count;
201 }
202 }
203 return count;
204 }
205
206 bool is_active(std::shared_ptr<asio::io_context> io_context) const {
207 if (!io_context) {
208 return false;
209 }
210
211 std::lock_guard<std::mutex> lock(mutex_);
212 for (const auto& entry : entries_) {
213 auto ctx = entry.context.lock();
214 if (ctx && ctx.get() == io_context.get() && !ctx->stopped()) {
215 return true;
216 }
217 }
218 return false;
219 }

221 void set_thread_pool(std::shared_ptr<thread_pool_interface> pool) {
222 std::lock_guard<std::mutex> lock(mutex_);
223 thread_pool_ = pool;
224 }
225
226 io_context_thread_manager::metrics get_metrics() const {
227 io_context_thread_manager::metrics m;
228 m.active_contexts = active_count();
229 m.total_started = total_started_.load(std::memory_order_relaxed);
230 m.total_completed = total_completed_.load(std::memory_order_relaxed);
231 return m;
232 }
233
234private:
235 void cleanup_expired() {
236 std::lock_guard<std::mutex> lock(mutex_);
237 entries_.erase(
238 std::remove_if(entries_.begin(), entries_.end(),
239 [](const context_entry& entry) {
240 return entry.context.expired();
241 }),
242 entries_.end()
243 );
244 }
245
246 mutable std::mutex mutex_;
247 std::shared_ptr<thread_pool_interface> thread_pool_;
248 std::vector<context_entry> entries_;
249 std::atomic<size_t> total_started_;
250 std::atomic<size_t> total_completed_;
251};

258io_context_thread_manager::io_context_thread_manager()
259 // Intentional Leak pattern: Use no-op deleter to prevent destruction
260 // during static destruction phase. This avoids heap corruption when
261 // thread pool tasks still reference impl's members (e.g., completed counter).
262 // Memory impact: ~few KB (reclaimed by OS on process termination)
263 : pimpl_(new impl(), [](impl*) { /* no-op deleter - intentional leak */ }) {
264}
