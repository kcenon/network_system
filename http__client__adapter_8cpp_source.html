93auto http_client_adapter::start(std::string_view host, uint16_t port) -> VoidResult
94{
95 if (!client_)
96 {
97 return error_void(
98 error_codes::common_errors::internal_error,
99 "Client not initialized",
100 "http_client_adapter::start"
101 );
102 }
103
104 // Store the connection details for building URLs
105 host_ = std::string(host);
106 port_ = port;
107 is_running_.store(true, std::memory_order_release);
108
109 // Notify connected callback (HTTP is "connected" when we have a target)
110 {
111 std::shared_ptr<interfaces::connection_observer> observer_copy;
112 connected_callback_t callback_copy;
113 {
114 std::lock_guard<std::mutex> lock(callbacks_mutex_);
115 observer_copy = observer_;
116 callback_copy = connected_callback_;
117 }
118
119 if (observer_copy)
120 {
121 observer_copy->on_connected();
122 }
123 if (callback_copy)
124 {
125 callback_copy();
126 }
127 }
128
129 return ok();
130}

168auto http_client_adapter::send(std::vector<uint8_t>&& data) -> VoidResult
169{
170 if (!client_)
171 {
172 return error_void(
173 error_codes::common_errors::internal_error,
174 "Client not initialized",
175 "http_client_adapter::send"
176 );
177 }
178
179 if (!is_running_.load(std::memory_order_acquire))
180 {
181 return error_void(
182 error_codes::common_errors::invalid_argument,
183 "Client is not running - call start() first",
184 "http_client_adapter::send"
185 );
186 }
187
188 // Build the full URL
189 const std::string url = build_url();
190
191 // Perform HTTP POST request with the binary data
192 auto result = client_->post(url, data);
193 if (result.is_err())
194 {
195 notify_error(std::make_error_code(std::errc::io_error));
196 return error_void(
197 error_codes::common_errors::internal_error,
198 "HTTP POST request failed: " + result.error().message,
199 "http_client_adapter::send"
200 );
201 }
202
203 // Notify receive callback with response body
204 const auto& response = result.value();
205 notify_receive(response.body);
206
207 return ok();
208}

249auto http_client_adapter::build_url() const -> std::string
250{
251 std::ostringstream oss;
252 oss << (use_ssl_ ? "https://" : "http://")
253 << host_ << ":" << port_ << path_;
254 return oss.str();
255}
