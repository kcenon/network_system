266// =============================================================================
267// Static destruction guard
268//
269// When KCENON_WITH_COMMON_SYSTEM is enabled, LOG_* macros access common_system's
270// GlobalLoggerRegistry singleton. During static destruction, this singleton may
271// be destroyed before network_system's singletons, causing heap corruption when
272// thread pool threads try to log.
273//
274// This guard prevents logging during static destruction by tracking if the
275// logging subsystem is still valid.
276// =============================================================================
277
278namespace detail {
279
287class static_destruction_guard {
288public:
289 static_destruction_guard() { counter_.fetch_add(1, std::memory_order_relaxed); }
290 ~static_destruction_guard() { counter_.fetch_sub(1, std::memory_order_relaxed); }
291
296 static bool is_logging_safe()
297 {
298 return counter_.load(std::memory_order_relaxed) > 0;
299 }
300
301private:
302 static inline std::atomic<int> counter_{0};
303};
304
305// This static instance ensures the counter is incremented at program start
306// and decremented during static destruction. When counter_ reaches 0,
307// we know static destruction has begun.
308inline static_destruction_guard global_guard_;
309
310} // namespace detail
311
312} // namespace kcenon::network::integration
