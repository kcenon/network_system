100struct method_descriptor

136struct service_descriptor
137{
139 std::string name;
140
142 std::string package;
143
145 std::vector<method_descriptor> methods;
146
152 auto find_method(std::string_view method_name) const
153 -> const method_descriptor*
154 {
155 for (const auto& method : methods)
156 {
157 if (method.name == method_name)
158 {
159 return &method;
160 }
161 }
162 return nullptr;
163 }
164
169 auto full_name() const -> std::string
170 {
171 if (package.empty())
172 {
173 return name;
174 }
175 return package + "." + name;
176 }
177};
