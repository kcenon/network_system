50 auto http2_server_stream::send_headers(int status_code,
51 const std::vector<http_header>& headers,
52 bool end_stream) -> VoidResult
53 {
54 std::lock_guard<std::mutex> lock(mutex_);
55
56 if (state_ == stream_state::closed ||
57 state_ == stream_state::half_closed_local) {
58 return error_void(
59 error_codes::common_errors::invalid_argument,
60 "Stream is closed for sending",
61 "http2_server_stream");
62 }
63
64 if (headers_sent_) {
65 return error_void(
66 error_codes::common_errors::invalid_argument,
67 "Headers already sent",
68 "http2_server_stream");
69 }
70
71 // Build response headers
72 auto response_headers = build_response_headers(status_code, headers);
73
74 // Encode headers using HPACK
75 auto encoded = encoder_->encode(response_headers);
76
77 // Create HEADERS frame
78 headers_frame frame(stream_id_, std::move(encoded), end_stream, true);
79 auto result = frame_sender_(frame);
80
81 if (result.is_err()) {
82 return result;
83 }
84
85 headers_sent_ = true;
86
87 if (end_stream) {
88 state_ = stream_state::half_closed_local;
89 }
90
91 return ok();
92 }

94 auto http2_server_stream::send_data(const std::vector<uint8_t>& data,
95 bool end_stream) -> VoidResult
96 {
97 std::lock_guard<std::mutex> lock(mutex_);
98
99 if (state_ == stream_state::closed ||
100 state_ == stream_state::half_closed_local) {
101 return error_void(
102 error_codes::common_errors::invalid_argument,
103 "Stream is closed for sending",
104 "http2_server_stream");
105 }
106
107 if (!headers_sent_) {
108 return error_void(
109 error_codes::common_errors::invalid_argument,
110 "Headers must be sent before data",
111 "http2_server_stream");
112 }
113
114 // Split data into frames respecting max_frame_size
115 size_t offset = 0;
116 while (offset < data.size()) {
117 size_t chunk_size = std::min(
118 static_cast<size_t>(max_frame_size_),
119 data.size() - offset);
120
121 std::vector<uint8_t> chunk(data.begin() + static_cast<std::ptrdiff_t>(offset),
122 data.begin() + static_cast<std::ptrdiff_t>(offset + chunk_size));
123
124 bool is_last = (offset + chunk_size >= data.size());
125 bool frame_end_stream = end_stream && is_last;
126
127 data_frame frame(stream_id_, std::move(chunk), frame_end_stream);
128 auto result = frame_sender_(frame);
129
130 if (result.is_err()) {
131 return result;
132 }
133
134 offset += chunk_size;
135 }
136
137 // Handle empty data case with end_stream
138 if (data.empty() && end_stream) {
139 data_frame frame(stream_id_, std::vector<uint8_t>{}, true);
140 auto result = frame_sender_(frame);
141 if (result.is_err()) {
142 return result;
143 }
144 }
145
146 if (end_stream) {
147 state_ = stream_state::half_closed_local;
148 }
149
150 return ok();
151 }

175 auto http2_server_stream::reset(uint32_t err_code) -> VoidResult
176 {
177 std::lock_guard<std::mutex> lock(mutex_);
178
179 if (state_ == stream_state::closed) {
180 return ok();
181 }
182
183 rst_stream_frame frame(stream_id_, err_code);
184 auto result = frame_sender_(frame);
185
186 state_ = stream_state::closed;
187
188 return result;

247 auto http2_server_stream::build_response_headers(int status_code,
248 const std::vector<http_header>& additional)
249 -> std::vector<http_header>
250 {
251 std::vector<http_header> headers;
252 headers.reserve(additional.size() + 1);
253
254 // :status pseudo-header must be first
255 headers.emplace_back(":status", std::to_string(status_code));
256
257 // Add additional headers
258 for (const auto& header : additional) {
259 headers.push_back(header);
260 }
261
262 return headers;
263 }
