53 VoidResult initialize(const NetworkSystemBridgeConfig& config) {
54 std::lock_guard<std::mutex> lock(mutex_);
55
56 if (initialized_.load()) {
57 return error_void(
58 error_codes::common_errors::already_exists,
59 "NetworkSystemBridge already initialized",
60 "NetworkSystemBridge::initialize");
61 }
62
63 // Store configuration
64 config_ = config;
65
66 // Initialize thread pool bridge if enabled
67 if (config.enable_thread_pool && thread_pool_bridge_) {
68 BridgeConfig thread_config;
69 thread_config.integration_name = config.integration_name;
70 thread_config.properties = config.thread_pool_properties;
71
72 auto result = thread_pool_bridge_->initialize(thread_config);
73 if (result.is_err()) {
74 return error_void(
75 error_codes::common_errors::internal_error,
76 "Failed to initialize thread pool bridge: " + result.error().message,
77 "NetworkSystemBridge::initialize");
78 }
79 initialized_bridges_.push_back("thread_pool");
80 }
81
82 // Future: Initialize logger bridge if enabled
83 if (config.enable_logger && logger_) {
84 // Logger initialization logic here
85 initialized_bridges_.push_back("logger");
86 }
87
88 // Future: Initialize monitoring bridge if enabled
89 if (config.enable_monitoring && monitoring_) {
90 // Monitoring initialization logic here
91 initialized_bridges_.push_back("monitoring");
92 }
93
94 initialized_.store(true);
95 return ok();
96 }
97
98 VoidResult shutdown() {
99 std::lock_guard<std::mutex> lock(mutex_);
100
101 if (!initialized_.load()) {
102 return ok(); // Idempotent
103 }
104
105 // Shutdown in reverse order
106 std::reverse(initialized_bridges_.begin(), initialized_bridges_.end());
107
108 for (const auto& bridge_name : initialized_bridges_) {
109 if (bridge_name == "thread_pool" && thread_pool_bridge_) {
110 thread_pool_bridge_->shutdown();
111 }
112 // Future: Handle other bridges
113 }
114
115 initialized_bridges_.clear();
116 initialized_.store(false);
117 return ok();
118 }
119
120 bool is_initialized() const {
121 return initialized_.load();
122 }
123
124 BridgeMetrics get_metrics() const {
125 std::lock_guard<std::mutex> lock(mutex_);
126
127 BridgeMetrics aggregated;
128 aggregated.is_healthy = true;
129 aggregated.last_activity = std::chrono::steady_clock::now();
130
131 // Aggregate thread pool metrics
132 if (thread_pool_bridge_ && thread_pool_bridge_->is_initialized()) {
133 auto thread_metrics = thread_pool_bridge_->get_metrics();
134 aggregated.is_healthy &= thread_metrics.is_healthy;
135
136 for (const auto& [key, value] : thread_metrics.custom_metrics) {
137 aggregated.custom_metrics["thread_pool." + key] = value;
138 }
139 }
140
141 // Future: Aggregate logger and monitoring metrics
142
143 return aggregated;
144 }

169 VoidResult set_thread_pool_bridge(std::shared_ptr<ThreadPoolBridge> bridge) {
170 std::lock_guard<std::mutex> lock(mutex_);
171
172 if (initialized_.load()) {
173 return error_void(
174 error_codes::common_errors::already_exists,
175 "Cannot set thread pool bridge after initialization",
176 "NetworkSystemBridge::set_thread_pool_bridge");
177 }
178
179 if (!bridge) {
180 return error_void(
181 error_codes::common_errors::invalid_argument,
182 "Thread pool bridge cannot be null",
183 "NetworkSystemBridge::set_thread_pool_bridge");
184 }
185
186 thread_pool_bridge_ = std::move(bridge);
187 return ok();
188 }
189

191 std::lock_guard<std::mutex> lock(mutex_);
192
193 if (initialized_.load()) {
194 return error_void(

196 "Cannot set logger after initialization",
197 "NetworkSystemBridge::set_logger");
198 }
199
200 logger_ = std::move(logger);
201 return ok();
202 }
203
204 VoidResult set_monitoring(std::shared_ptr<monitoring_interface> monitoring) {
205 std::lock_guard<std::mutex> lock(mutex_);
206
207 if (initialized_.load()) {
208 return error_void(
209 error_codes::common_errors::already_exists,
210 "Cannot set monitoring after initialization",
211 "NetworkSystemBridge::set_monitoring");
212 }
213
214 monitoring_ = std::move(monitoring);
215 return ok();
216 }

361std::shared_ptr<NetworkSystemBridge> NetworkSystemBridge::with_common_system(
362 std::shared_ptr<::kcenon::common::interfaces::IExecutor> executor,
363 std::shared_ptr<::kcenon::common::interfaces::ILogger> logger,
364 std::shared_ptr<::kcenon::common::interfaces::IMonitor> monitor) {
365
366 auto bridge = std::make_shared<NetworkSystemBridge>();
367
368 // Set up thread pool bridge
369 if (executor) {
370 auto thread_pool_bridge = ThreadPoolBridge::from_common_system(executor);
371 bridge->set_thread_pool_bridge(thread_pool_bridge);
372 }
373
374 // Set up logger

376 auto logger_adapter = std::make_shared<common_logger_adapter>(logger);
377 bridge->set_logger(logger_adapter);
378 }
379
380 // Set up monitoring
381 if (monitor) {
382 auto monitoring_adapter = std::make_shared<common_monitoring_adapter>(monitor);
383 bridge->set_monitoring(monitoring_adapter);
384 }
385
386 return bridge;

390std::shared_ptr<NetworkSystemBridge> NetworkSystemBridge::with_custom(
391 std::shared_ptr<thread_pool_interface> thread_pool,
392 std::shared_ptr<logger_interface> logger,
393 std::shared_ptr<monitoring_interface> monitoring) {
394
395 auto bridge = std::make_shared<NetworkSystemBridge>();
396
397 if (thread_pool) {
398 auto thread_pool_bridge = std::make_shared<ThreadPoolBridge>(
399 thread_pool, ThreadPoolBridge::BackendType::Custom);
400 bridge->set_thread_pool_bridge(thread_pool_bridge);
401 }
402
403 if (logger) {
404 bridge->set_logger(logger);
405 }
406
407 if (monitoring) {
408 bridge->set_monitoring(monitoring);
409 }
410
411 return bridge;
412}
