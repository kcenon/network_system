133// Helper to escape JSON strings
134static auto escape_json_string(const std::string& input) -> std::string
135{
136 std::ostringstream oss;
137 for (char c : input)
138 {
139 switch (c)
140 {
141 case '"':
142 oss << "\\\"";
143 break;
144 case '\\':
145 oss << "\\\\";
146 break;
147 case '\b':
148 oss << "\\b";
149 break;
150 case '\f':
151 oss << "\\f";
152 break;
153 case '\n':
154 oss << "\\n";
155 break;
156 case '\r':
157 oss << "\\r";
158 break;
159 case '\t':
160 oss << "\\t";
161 break;
162 default:
163 if (static_cast<unsigned char>(c) < 0x20)
164 {
165 oss << "\\u" << std::hex << std::setw(4) << std::setfill('0')
166 << static_cast<int>(c);
167 }
168 else
169 {
170 oss << c;
171 }
172 break;
173 }
174 }
175 return oss.str();
176}

178// Helper to escape HTML strings
179static auto escape_html_string(const std::string& input) -> std::string
180{
181 std::ostringstream oss;
182 for (char c : input)
183 {
184 switch (c)
185 {
186 case '&':
187 oss << "&amp;";
188 break;
189 case '<':
190 oss << "&lt;";
191 break;
192 case '>':
193 oss << "&gt;";
194 break;
195 case '"':
196 oss << "&quot;";
197 break;
198 case '\'':
199 oss << "&#39;";
200 break;
201 default:
202 oss << c;
203 break;
204 }
205 }
206 return oss.str();
207}

209auto http_error_response::build_json_error(const http_error& error) -> http_response
210{
211 http_response response;
212 response.status_code = error.status_code();
213 response.status_message = std::string(get_error_status_text(error.code));
214
215 // Build RFC 7807 compliant JSON response
216 std::ostringstream json;
217 json << "{\n";
218 json << " \"type\": \"about:blank\",\n";
219 json << " \"title\": \"" << escape_json_string(response.status_message) << "\",\n";
220 json << " \"status\": " << response.status_code << ",\n";
221 json << " \"detail\": \"" << escape_json_string(error.detail.empty() ? error.message : error.detail)
222 << "\"";
223
224 if (!error.request_id.empty())
225 {
226 json << ",\n \"instance\": \"" << escape_json_string(error.request_id) << "\"";
227 }
228
229 // Add timestamp
230 auto time_t_val = std::chrono::system_clock::to_time_t(error.timestamp);
231 std::tm tm_val{};
232#ifdef _WIN32
233 gmtime_s(&tm_val, &time_t_val);
234#else
235 gmtime_r(&time_t_val, &tm_val);
236#endif
237 char time_buf[32];
238 std::strftime(time_buf, sizeof(time_buf), "%Y-%m-%dT%H:%M:%SZ", &tm_val);
239 json << ",\n \"timestamp\": \"" << time_buf << "\"";
240
241 json << "\n}\n";
242
243 response.set_body_string(json.str());
244 response.set_header("Content-Type", "application/problem+json; charset=utf-8");
245
246 return response;
247}

249auto http_error_response::build_html_error(const http_error& error) -> http_response
250{
251 http_response response;
252 response.status_code = error.status_code();
253 response.status_message = std::string(get_error_status_text(error.code));
254
255 std::ostringstream html;
256 html << "<!DOCTYPE html>\n";
257 html << "<html lang=\"en\">\n";
258 html << "<head>\n";
259 html << " <meta charset=\"utf-8\">\n";
260 html << " <meta name=\"viewport\" content=\"width=device-width, initial-scale=1\">\n";
261 html << " <title>" << response.status_code << " " << escape_html_string(response.status_message)
262 << "</title>\n";
263 html << " <style>\n";
264 html << " body { font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif; "
265 << "margin: 40px auto; max-width: 650px; line-height: 1.6; color: #333; padding: 0 10px; }\n";
266 html << " h1 { color: #c00; border-bottom: 2px solid #eee; padding-bottom: 10px; }\n";
267 html << " .detail { background: #f9f9f9; padding: 15px; border-radius: 5px; }\n";
268 html << " .meta { color: #666; font-size: 0.9em; margin-top: 20px; }\n";
269 html << " </style>\n";
270 html << "</head>\n";
271 html << "<body>\n";
272 html << " <h1>" << response.status_code << " " << escape_html_string(response.status_message)
273 << "</h1>\n";
274
275 if (!error.message.empty())
276 {
277 html << " <p>" << escape_html_string(error.message) << "</p>\n";
278 }
279
280 if (!error.detail.empty())
281 {
282 html << " <div class=\"detail\">\n";
283 html << " <strong>Details:</strong> " << escape_html_string(error.detail) << "\n";
284 html << " </div>\n";
285 }
286
287 html << " <div class=\"meta\">\n";
288 if (!error.request_id.empty())
289 {
290 html << " <p>Request ID: " << escape_html_string(error.request_id) << "</p>\n";
291 }
292 html << " <p><em>NetworkSystem HTTP Server</em></p>\n";
293 html << " </div>\n";
294 html << "</body>\n";
295 html << "</html>\n";
296
297 response.set_body_string(html.str());
298 response.set_header("Content-Type", "text/html; charset=utf-8");
299
300 return response;
301}
