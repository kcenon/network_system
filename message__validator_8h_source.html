147public:
155 [[nodiscard]] static bool validate_size(
156 size_t size,
157 size_t max_size = message_limits::MAX_MESSAGE_SIZE) noexcept {
158 return size <= max_size;
159 }

168 static void validate_size_or_throw(
169 size_t size,
170 size_t max_size = message_limits::MAX_MESSAGE_SIZE) {
171 if (size > max_size) {
172 throw std::length_error(
173 "Message size " + std::to_string(size) +
174 " exceeds limit " + std::to_string(max_size));
175 }
176 }
177
190 [[nodiscard]] static size_t safe_copy(
191 void* dest,
192 size_t dest_size,
193 const void* src,
194 size_t src_size) noexcept {
195 if (!dest || !src || dest_size == 0 || src_size == 0) {
196 return 0;
197 }
198
199 size_t copy_size = std::min(dest_size, src_size);
200 std::memcpy(dest, src, copy_size);
201 return copy_size;
202 }
203
212 [[nodiscard]] static size_t safe_strcpy(
213 char* dest,
214 size_t dest_size,
215 const char* src) noexcept {
216 if (!dest || dest_size == 0 || !src) {
217 return 0;
218 }
219
220 size_t src_len = std::strlen(src);
221 size_t copy_len = std::min(dest_size - 1, src_len);
222
223 std::memcpy(dest, src, copy_len);
224 dest[copy_len] = '\0';
225
226 return copy_len;
227 }
228
240 [[nodiscard]] static validation_result validate_http_header(
241 std::string_view header) noexcept {
242 // Check size
243 if (header.size() > message_limits::MAX_HEADER_SIZE) {
244 return validation_result::size_exceeded;
245 }
246
247 // Check for NULL bytes (NULL byte injection attack)
248 if (header.find('\0') != std::string_view::npos) {
249 return validation_result::null_byte_detected;
250 }
251
252 // Check for invalid control characters (except \r\n\t)
253 for (char c : header) {
254 if (c < 0x20 && c != '\r' && c != '\n' && c != '\t') {
255 return validation_result::invalid_character;
256 }
257 }
258
259 return validation_result::ok;
260 }
261
268 [[nodiscard]] static bool validate_header_count(size_t count) noexcept {
269 return count <= message_limits::MAX_HEADER_COUNT;
270 }
271
279 [[nodiscard]] static bool validate_websocket_frame(
280 size_t payload_length,
281 size_t max_size = message_limits::MAX_WEBSOCKET_FRAME) noexcept {
282 return payload_length <= max_size;
283 }
284
291 [[nodiscard]] static validation_result validate_url(
292 std::string_view url) noexcept {
293 if (url.size() > message_limits::MAX_URL_LENGTH) {
294 return validation_result::size_exceeded;
295 }
296
297 if (url.find('\0') != std::string_view::npos) {
298 return validation_result::null_byte_detected;
299 }
300
301 return validation_result::ok;
302 }
303
313 [[nodiscard]] static bool contains_suspicious_pattern(
314 std::string_view data) noexcept {
315 // Check for NULL byte injection
316 if (data.find('\0') != std::string_view::npos) {
317 return true;
318 }
319
320 // Check for HTTP response splitting
321 if (data.find("\r\n\r\n") != std::string_view::npos) {
322 return true;
323 }
324
325 return false;
326 }
327
334 [[nodiscard]] static std::string sanitize_string(std::string_view input) {
335 std::string result;
336 result.reserve(input.size());
337
338 for (char c : input) {
339 // Keep printable ASCII and common whitespace
340 if (c >= 0x20 || c == '\t' || c == '\n' || c == '\r') {
341 result += c;
342 }
343 }
344
345 return result;
346 }
347
357 [[nodiscard]] static size_t safe_buffer_size(
358 size_t requested_size,
359 size_t max_size = message_limits::MAX_MESSAGE_SIZE) noexcept {
360 return std::min(requested_size, max_size);
361 }
362};
