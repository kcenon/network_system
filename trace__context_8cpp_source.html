52// Thread-local storage for current trace context
53thread_local trace_context tls_current_context;
54
55// Hex character lookup table
56constexpr std::array<char, 16> hex_chars = {'0', '1', '2', '3', '4', '5', '6', '7',
57 '8', '9', 'a', 'b', 'c', 'd', 'e', 'f'};
58
59// Convert single hex character to value
60auto hex_char_to_value(char c) -> int
61{
62 if (c >= '0' && c <= '9')
63 {
64 return c - '0';
65 }
66 if (c >= 'a' && c <= 'f')
67 {
68 return c - 'a' + 10;
69 }
70 if (c >= 'A' && c <= 'F')
71 {
72 return c - 'A' + 10;
73 }
74 return -1;
75}

77// Check if trace ID is all zeros (invalid)
78auto is_zero_trace_id(const trace_id_t& id) -> bool
79{
80 return std::all_of(id.begin(), id.end(), [](uint8_t b) { return b == 0; });
81}
82
83// Check if span ID is all zeros (invalid)
84auto is_zero_span_id(const span_id_t& id) -> bool
85{
86 return std::all_of(id.begin(), id.end(), [](uint8_t b) { return b == 0; });
87}

89// Thread-safe random number generator
90auto get_random_bytes(uint8_t* buffer, size_t size) -> void
91{
92 static thread_local std::random_device rd;
93 static thread_local std::mt19937_64 gen(rd());
94 static thread_local std::uniform_int_distribution<uint64_t> dist;
95
96 size_t offset = 0;
97 while (offset < size)
98 {
99 uint64_t value = dist(gen);
100 size_t to_copy = std::min(sizeof(value), size - offset);
101 std::memcpy(buffer + offset, &value, to_copy);
102 offset += to_copy;
103 }
104}

133auto trace_context::create_span(std::string_view name) -> span
134{
135 // If there's a current context, create a child span
136 if (tls_current_context.is_valid())
137 {
138 return tls_current_context.create_child_span(name);
139 }
140
141 // Create a new root span with a fresh trace context
142 auto new_trace_id = generate_trace_id();
143 auto new_span_id = generate_span_id();
144
145 trace_context new_ctx(new_trace_id, new_span_id, trace_flags::sampled);
146 return span(name, new_ctx);
147}

149auto trace_context::create_child_span(std::string_view name) const -> span
150{
151 if (!is_valid())
152 {
153 // If parent context is invalid, create a new root span
154 auto new_trace_id = generate_trace_id();
155 auto new_span_id = generate_span_id();
156
157 trace_context new_ctx(new_trace_id, new_span_id, trace_flags::sampled);
158 return span(name, new_ctx);
159 }
160
161 // Create child span with same trace ID but new span ID
162 auto new_span_id = generate_span_id();
163 trace_context child_ctx(trace_id_, new_span_id, flags_, span_id_);
164
165 return span(name, child_ctx);
166}

168auto trace_context::to_traceparent() const -> std::string
169{
170 if (!is_valid())
171 {
172 return {};
173 }
174
175 // Format: {version}-{trace-id}-{parent-id}-{trace-flags}
176 // Example: 00-0af7651916cd43dd8448eb211c80319c-b7ad6b7169203331-01
177 std::ostringstream oss;
178 oss << "00-";
179 oss << trace_id_hex();
180 oss << "-";
181 oss << span_id_hex();
182 oss << "-";
183 oss << (is_sampled() ? "01" : "00");
184
185 return oss.str();
186}

201auto trace_context::from_traceparent(std::string_view traceparent) -> trace_context
202{
203 // Minimum length check: 00-{32}-{16}-{2} = 55 characters
204 if (traceparent.size() < 55)
205 {
206 return trace_context{};
207 }
208
209 // Parse version
210 if (traceparent[0] != '0' || traceparent[1] != '0' || traceparent[2] != '-')
211 {
212 return trace_context{};
213 }
214
215 // Parse trace ID (32 hex chars)
216 trace_id_t trace_id{};
217 if (!hex_to_bytes(traceparent.substr(3, 32), trace_id.data(), 16))
218 {
219 return trace_context{};
220 }
221
222 // Check separator
223 if (traceparent[35] != '-')
224 {
225 return trace_context{};
226 }
227
228 // Parse span ID (16 hex chars)
229 span_id_t span_id{};
230 if (!hex_to_bytes(traceparent.substr(36, 16), span_id.data(), 8))
231 {
232 return trace_context{};
233 }
234
235 // Check separator
236 if (traceparent[52] != '-')
237 {
238 return trace_context{};
239 }
240
241 // Parse flags (2 hex chars)
242 int high = hex_char_to_value(traceparent[53]);
243 int low = hex_char_to_value(traceparent[54]);
244 if (high < 0 || low < 0)
245 {
246 return trace_context{};
247 }
248
249 auto flags = static_cast<trace_flags>((high << 4) | low);
250
251 return trace_context(trace_id, span_id, flags);
252}

254auto trace_context::from_headers(
255 const std::vector<std::pair<std::string, std::string>>& headers) -> trace_context
256{
257 for (const auto& [name, value] : headers)
258 {
259 // Case-insensitive comparison
260 std::string lower_name = name;
261 std::transform(lower_name.begin(), lower_name.end(), lower_name.begin(),
262 [](unsigned char c) { return static_cast<char>(std::tolower(c)); });
263
264 if (lower_name == "traceparent")
265 {
266 auto ctx = from_traceparent(value);
267 if (ctx.is_valid())
268 {
269 return ctx;
270 }
271 }
272 }
273
274 return trace_context{};
275}

343auto bytes_to_hex(const uint8_t* data, size_t size) -> std::string
344{
345 std::string result;
346 result.reserve(size * 2);
347
348 for (size_t i = 0; i < size; ++i)
349 {
350 result.push_back(hex_chars[(data[i] >> 4) & 0x0F]);
351 result.push_back(hex_chars[data[i] & 0x0F]);
352 }
353
354 return result;
355}
356
357auto hex_to_bytes(std::string_view hex, uint8_t* out, size_t size) -> bool
358{
359 if (hex.size() != size * 2)
360 {
361 return false;
362 }
363
364 for (size_t i = 0; i < size; ++i)
365 {
366 int high = hex_char_to_value(hex[i * 2]);
367 int low = hex_char_to_value(hex[i * 2 + 1]);
368
369 if (high < 0 || low < 0)
370 {
371 return false;
372 }
373
374 out[i] = static_cast<uint8_t>((high << 4) | low);
375 }
376
377 return true;
378}
