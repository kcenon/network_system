59// Helper function to convert log level to string
60static const char* level_to_string(log_level level) {
61 switch (level) {
62 case log_level::trace: return "TRACE";
63 case log_level::debug: return "DEBUG";
64 case log_level::info: return "INFO ";
65 case log_level::warn: return "WARN ";
66 case log_level::error: return "ERROR";
67 case log_level::fatal: return "FATAL";
68 default: return "UNKN ";
69 }
70}

72// Helper function to get current timestamp (thread-safe)
73static std::string get_timestamp() {
74 auto now = std::chrono::system_clock::now();
75 auto time_t_val = std::chrono::system_clock::to_time_t(now);
76 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
77 now.time_since_epoch()) % 1000;
78
79 std::tm tm_buf{};
80#ifdef _WIN32
81 localtime_s(&tm_buf, &time_t_val); // Windows thread-safe version
82#else
83 localtime_r(&time_t_val, &tm_buf); // POSIX thread-safe version
84#endif
85
86 std::stringstream ss;
87 ss << std::put_time(&tm_buf, "%Y-%m-%d %H:%M:%S");
88 ss << '.' << std::setfill('0') << std::setw(3) << ms.count();
89 return ss.str();
90}

116void common_system_logger_adapter::log(log_level level, const std::string& message,
117 const std::string& file, int line,

119 auto logger = get_logger();

123 kcenon::common::interfaces::log_entry entry;

152class basic_logger::impl {
153public:
154 explicit impl(log_level min_level) : min_level_(static_cast<int>(min_level)) {}
155
156 void log(log_level level, const std::string& message) {
157 if (static_cast<int>(level) < min_level_) return;
158
159 std::lock_guard<std::mutex> lock(mutex_);
160
161 auto timestamp = get_timestamp();
162 std::ostringstream oss;
163 oss << "[" << timestamp << "] "
164 << "[" << level_to_string(level) << "] "
165 << "[network_system] "
166 << message;
167
168 auto* target = (level >= log_level::error) ? stderr : stdout;
169 const auto& record = oss.str();
170 std::fwrite(record.data(), 1, record.size(), target);
171 std::fputc('\n', target);
172 std::fflush(target);
173 }
174
175 void log(log_level level, const std::string& message,
176 const std::string& file, int line, const std::string& function) {
177 if (static_cast<int>(level) < min_level_) return;
178
179 std::lock_guard<std::mutex> lock(mutex_);
180
181 auto timestamp = get_timestamp();
182 std::ostringstream oss;
183 oss << "[" << timestamp << "] "
184 << "[" << level_to_string(level) << "] "
185 << "[network_system] "
186 << message
187 << " (" << file << ":" << line << " in " << function << ")";
188
189 auto* target = (level >= log_level::error) ? stderr : stdout;
190 const auto& record = oss.str();
191 std::fwrite(record.data(), 1, record.size(), target);
192 std::fputc('\n', target);
193 std::fflush(target);
194 }
195
196 bool is_level_enabled(log_level level) const {
197 return static_cast<int>(level) >= min_level_;
198 }
199
200 void flush() {
201 std::lock_guard<std::mutex> lock(mutex_);
202 std::fflush(stdout);
203 std::fflush(stderr);
204 }
205
206 void set_min_level(log_level level) {
207 min_level_ = static_cast<int>(level);
208 }
209
210 log_level get_min_level() const {
211 return static_cast<log_level>(min_level_.load());
212 }
213
214private:
215 mutable std::mutex mutex_;
216 std::atomic<int> min_level_;
217};

256class logger_integration_manager::impl {
257public:
258 impl() {
259#if KCENON_WITH_COMMON_SYSTEM
260 // Use common_system_logger_adapter by default
261 // It delegates to GlobalLoggerRegistry which provides NullLogger fallback
262 logger_ = std::make_shared<common_system_logger_adapter>();
263#else
264 // Use basic_logger for standalone operation
265 logger_ = std::make_shared<basic_logger>();
266#endif

269 void set_logger(std::shared_ptr<logger_interface> logger) {
270 std::lock_guard<std::mutex> lock(mutex_);
271 logger_ = logger;
272 }
273
274 std::shared_ptr<logger_interface> get_logger() {
275 std::lock_guard<std::mutex> lock(mutex_);
276 if (!logger_) {
277#if KCENON_WITH_COMMON_SYSTEM
278 logger_ = std::make_shared<common_system_logger_adapter>();
279#else
280 logger_ = std::make_shared<basic_logger>();
281#endif
282 }
283 return logger_;
284 }

287#if KCENON_WITH_COMMON_SYSTEM

290 if (common_logger) {

293#else
294 auto logger = get_logger();

298#endif
299 }
