55auto udp_facade::generate_client_id() -> std::string
56{

92auto udp_facade::create_client(const client_config& config) const
93 -> std::shared_ptr<interfaces::i_protocol_client>
94{
95 // Validate configuration
96 validate_client_config(config);
97
98 // Generate client ID if not provided
99 const std::string client_id = config.client_id.empty() ? generate_client_id() : config.client_id;
100
101 // Create UDP client adapter that wraps messaging_udp_client
102 auto client = std::make_shared<internal::adapters::udp_client_adapter>(client_id);
103
104 // Start the client and connect to target
105 auto result = client->start(config.host, config.port);
106 if (result.is_err())
107 {
108 throw std::runtime_error("udp_facade: failed to start client: " + result.error().message);
109 }
110
111 return client;
112}
