89namespace stream_flags
90{
91 constexpr uint8_t fin = 0x01;
92 constexpr uint8_t len = 0x02;
93 constexpr uint8_t off = 0x04;
94 constexpr uint8_t mask = 0x07;
95 constexpr uint8_t base = 0x08;
96} // namespace stream_flags
97
101[[nodiscard]] constexpr auto is_stream_frame(uint64_t type) noexcept -> bool
102{
103 return (type >= 0x08 && type <= 0x0f);
104}
105
109[[nodiscard]] constexpr auto get_stream_flags(uint64_t type) noexcept -> uint8_t
110{
111 return static_cast<uint8_t>(type & stream_flags::mask);
112}
113
117[[nodiscard]] constexpr auto make_stream_type(bool has_fin, bool has_length,
118 bool has_offset) noexcept -> uint8_t
119{
120 uint8_t type = stream_flags::base;
121 if (has_fin) type |= stream_flags::fin;
122 if (has_length) type |= stream_flags::len;
123 if (has_offset) type |= stream_flags::off;
124 return type;
125}
