121auto tcp_listener_adapter::send_to(std::string_view connection_id,
122 std::span<const std::byte> data) -> VoidResult
123{
124 core::tcp_server::session_ptr session;
125 {
126 std::lock_guard lock(sessions_mutex_);
127 auto it = sessions_.find(std::string(connection_id));
128 if (it == sessions_.end()) {
129 return error_void(
130 static_cast<int>(std::errc::no_such_device_or_address),
131 "Connection not found: " + std::string(connection_id)
132 );
133 }
134 session = it->second;
135 }
136
137 if (!session) {
138 return error_void(
139 static_cast<int>(std::errc::not_connected),
140 "Session is no longer valid"
141 );
142 }
143
144 std::vector<uint8_t> buffer(data.size());
145 std::memcpy(buffer.data(), data.data(), data.size());
146
147 return session->send(std::move(buffer));
148}

150auto tcp_listener_adapter::broadcast(std::span<const std::byte> data) -> VoidResult
151{
152 std::vector<core::tcp_server::session_ptr> sessions_copy;
153 {
154 std::lock_guard lock(sessions_mutex_);
155 sessions_copy.reserve(sessions_.size());
156 for (const auto& [id, session] : sessions_) {
157 if (session) {
158 sessions_copy.push_back(session);
159 }
160 }
161 }
162
163 if (sessions_copy.empty()) {
164 return ok(); // No connections, nothing to broadcast
165 }
166
167 std::vector<uint8_t> buffer(data.size());
168 std::memcpy(buffer.data(), data.data(), data.size());
169
170 bool any_success = false;
171 for (const auto& session : sessions_copy) {
172 std::vector<uint8_t> buffer_copy = buffer;
173 auto result = session->send(std::move(buffer_copy));
174 if (result.is_ok()) {
175 any_success = true;
176 }
177 }
178
179 if (any_success) {
180 return ok();
181 }
182
183 return error_void(
184 static_cast<int>(std::errc::io_error),
185 "Failed to send to any connection"
186 );
187}

189auto tcp_listener_adapter::close_connection(std::string_view connection_id) noexcept -> void
190{
191 core::tcp_server::session_ptr session;
192 {
193 std::lock_guard lock(sessions_mutex_);
194 auto it = sessions_.find(std::string(connection_id));
195 if (it != sessions_.end()) {
196 session = it->second;
197 if (session) {
198 session_to_id_.erase(session.get());
199 }
200 sessions_.erase(it);
201 }
202 }
203
204 // Session will be closed when its shared_ptr goes out of scope
205 // or we could explicitly close it if such method exists
206}

215auto tcp_listener_adapter::setup_internal_callbacks() -> void
216{
217 if (!server_) {
218 return;
219 }
220
221 // Bridge connection callback
222 server_->set_connection_callback([this](core::tcp_server::session_ptr session) {
223 if (!session) {
224 return;
225 }
226
227 std::string conn_id = generate_connection_id(session);
228
229 {
230 std::lock_guard lock(sessions_mutex_);
231 sessions_[conn_id] = session;
232 session_to_id_[session.get()] = conn_id;
233 }
234
235 // Call user callbacks
236 {
237 std::lock_guard lock(callbacks_mutex_);
238 if (callbacks_.on_accept) {
239 callbacks_.on_accept(conn_id);
240 }
241 }
242 });
243
244 // Bridge disconnection callback
245 server_->set_disconnection_callback([this](const std::string& session_id) {
246 std::string conn_id;
247 {
248 std::lock_guard lock(sessions_mutex_);
249 // Find the connection ID by session's internal ID
250 for (auto it = sessions_.begin(); it != sessions_.end(); ++it) {
251 if (it->second && std::string(it->second->id()) == session_id) {
252 conn_id = it->first;
253 if (it->second) {
254 session_to_id_.erase(it->second.get());
255 }
256 sessions_.erase(it);
257 break;
258 }
259 }
260 }
261
262 if (!conn_id.empty()) {
263 std::lock_guard lock(callbacks_mutex_);
264 if (callbacks_.on_disconnect) {
265 callbacks_.on_disconnect(conn_id);
266 }
267 }
268 });
269
270 // Bridge receive callback
271 server_->set_receive_callback([this](core::tcp_server::session_ptr session,
272 const std::vector<uint8_t>& data) {
273 if (!session) {
274 return;
275 }
276
277 std::string conn_id;
278 {
279 std::lock_guard lock(sessions_mutex_);
280 auto it = session_to_id_.find(session.get());
281 if (it != session_to_id_.end()) {
282 conn_id = it->second;
283 }
284 }
285
286 if (!conn_id.empty()) {
287 std::lock_guard lock(callbacks_mutex_);
288 if (callbacks_.on_data) {
289 std::span<const std::byte> byte_span(
290 reinterpret_cast<const std::byte*>(data.data()),
291 data.size()
292 );
293 callbacks_.on_data(conn_id, byte_span);
294 }
295 }
296 });
297
298 // Bridge error callback
299 server_->set_error_callback([this](core::tcp_server::session_ptr session,
300 std::error_code ec) {
301 if (!session) {
302 return;
303 }
304
305 std::string conn_id;
306 {
307 std::lock_guard lock(sessions_mutex_);
308 auto it = session_to_id_.find(session.get());
309 if (it != session_to_id_.end()) {
310 conn_id = it->second;
311 }
312 }
313
314 if (!conn_id.empty()) {
315 std::lock_guard lock(callbacks_mutex_);
316 if (callbacks_.on_error) {
317 callbacks_.on_error(conn_id, ec);
318 }
319 }
320 });
321}
