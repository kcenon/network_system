93auto tcp_listener_adapter::send_to(std::string_view connection_id,
94 std::span<const std::byte> data) -> VoidResult
95{
96 core::tcp_server::session_ptr session;
97 {
98 std::lock_guard lock(sessions_mutex_);
99 auto it = sessions_.find(std::string(connection_id));
100 if (it == sessions_.end()) {
101 return error_void(
102 static_cast<int>(std::errc::no_such_device_or_address),
103 "Connection not found: " + std::string(connection_id)
104 );
105 }
106 session = it->second;
107 }
108
109 if (!session) {
110 return error_void(
111 static_cast<int>(std::errc::not_connected),
112 "Session is no longer valid"
113 );
114 }
115
116 std::vector<uint8_t> buffer(data.size());
117 std::memcpy(buffer.data(), data.data(), data.size());
118
119 return session->send(std::move(buffer));
120}

122auto tcp_listener_adapter::broadcast(std::span<const std::byte> data) -> VoidResult
123{
124 std::vector<core::tcp_server::session_ptr> sessions_copy;
125 {
126 std::lock_guard lock(sessions_mutex_);
127 sessions_copy.reserve(sessions_.size());
128 for (const auto& [id, session] : sessions_) {
129 if (session) {
130 sessions_copy.push_back(session);
131 }
132 }
133 }
134
135 if (sessions_copy.empty()) {
136 return ok(); // No connections, nothing to broadcast
137 }
138
139 std::vector<uint8_t> buffer(data.size());
140 std::memcpy(buffer.data(), data.data(), data.size());
141
142 bool any_success = false;
143 for (const auto& session : sessions_copy) {
144 std::vector<uint8_t> buffer_copy = buffer;
145 auto result = session->send(std::move(buffer_copy));
146 if (result.is_ok()) {
147 any_success = true;
148 }
149 }
150
151 if (any_success) {
152 return ok();
153 }
154
155 return error_void(
156 static_cast<int>(std::errc::io_error),
157 "Failed to send to any connection"
158 );
159}

161auto tcp_listener_adapter::close_connection(std::string_view connection_id) noexcept -> void
162{
163 core::tcp_server::session_ptr session;
164 {
165 std::lock_guard lock(sessions_mutex_);
166 auto it = sessions_.find(std::string(connection_id));
167 if (it != sessions_.end()) {
168 session = it->second;
169 if (session) {
170 session_to_id_.erase(session.get());
171 }
172 sessions_.erase(it);
173 }
174 }
175
176 // Session will be closed when its shared_ptr goes out of scope
177 // or we could explicitly close it if such method exists
178}

187auto tcp_listener_adapter::setup_internal_callbacks() -> void
188{
189 if (!server_) {
190 return;
191 }
192
193 // Bridge connection callback
194 server_->set_connection_callback([this](core::tcp_server::session_ptr session) {
195 if (!session) {
196 return;
197 }
198
199 std::string conn_id = generate_connection_id(session);
200
201 {
202 std::lock_guard lock(sessions_mutex_);
203 sessions_[conn_id] = session;
204 session_to_id_[session.get()] = conn_id;
205 }
206
207 // Call user callbacks
208 {
209 std::lock_guard lock(callbacks_mutex_);
210 if (callbacks_.on_accept) {
211 callbacks_.on_accept(conn_id);
212 }
213 }
214 });
215
216 // Bridge disconnection callback
217 server_->set_disconnection_callback([this](const std::string& session_id) {
218 std::string conn_id;
219 {
220 std::lock_guard lock(sessions_mutex_);
221 // Find the connection ID by session's internal ID
222 for (auto it = sessions_.begin(); it != sessions_.end(); ++it) {
223 if (it->second && std::string(it->second->id()) == session_id) {
224 conn_id = it->first;
225 if (it->second) {
226 session_to_id_.erase(it->second.get());
227 }
228 sessions_.erase(it);
229 break;
230 }
231 }
232 }
233
234 if (!conn_id.empty()) {
235 std::lock_guard lock(callbacks_mutex_);
236 if (callbacks_.on_disconnect) {
237 callbacks_.on_disconnect(conn_id);
238 }
239 }
240 });
241
242 // Bridge receive callback
243 server_->set_receive_callback([this](core::tcp_server::session_ptr session,
244 const std::vector<uint8_t>& data) {
245 if (!session) {
246 return;
247 }
248
249 std::string conn_id;
250 {
251 std::lock_guard lock(sessions_mutex_);
252 auto it = session_to_id_.find(session.get());
253 if (it != session_to_id_.end()) {
254 conn_id = it->second;
255 }
256 }
257
258 if (!conn_id.empty()) {
259 std::lock_guard lock(callbacks_mutex_);
260 if (callbacks_.on_data) {
261 std::span<const std::byte> byte_span(
262 reinterpret_cast<const std::byte*>(data.data()),
263 data.size()
264 );
265 callbacks_.on_data(conn_id, byte_span);
266 }
267 }
268 });
269
270 // Bridge error callback
271 server_->set_error_callback([this](core::tcp_server::session_ptr session,
272 std::error_code ec) {
273 if (!session) {
274 return;
275 }
276
277 std::string conn_id;
278 {
279 std::lock_guard lock(sessions_mutex_);
280 auto it = session_to_id_.find(session.get());
281 if (it != session_to_id_.end()) {
282 conn_id = it->second;
283 }
284 }
285
286 if (!conn_id.empty()) {
287 std::lock_guard lock(callbacks_mutex_);
288 if (callbacks_.on_error) {
289 callbacks_.on_error(conn_id, ec);
290 }
291 }
292 });
293}
