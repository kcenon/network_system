85 ::kcenon::common::VoidResult execute() override {

102 }

110private:

132class network_to_common_thread_adapter
133 : public ::kcenon::common::interfaces::IExecutor {
134public:
140 explicit network_to_common_thread_adapter(
141 std::shared_ptr<thread_pool_interface> pool)
142 : pool_(std::move(pool)) {
143 if (!pool_) {
144 throw std::invalid_argument(
145 "network_to_common_thread_adapter requires non-null pool");
146 }
147 }
148
154 ::kcenon::common::Result<std::future<void>> execute(
155 std::unique_ptr<::kcenon::common::interfaces::IJob>&& job) override {
156 if (!pool_ || !pool_->is_running()) {
157 return ::kcenon::common::Result<std::future<void>>::err(
158 ::kcenon::common::error_codes::INVALID_ARGUMENT,
159 "Thread pool not running",
160 "network_to_common_thread_adapter");
161 }
162
163 try {
164 // Wrap the job in a shared_ptr for safe capture in lambda
165 auto shared_job = std::shared_ptr<::kcenon::common::interfaces::IJob>(
166 std::move(job));
167
168 auto future = pool_->submit([shared_job]() mutable {
169 auto result = shared_job->execute();
170 if (result.is_err()) {
171 throw std::runtime_error(result.error().message);
172 }
173 });
174
175 return ::kcenon::common::Result<std::future<void>>::ok(std::move(future));
176 } catch (const std::exception& e) {
177 return ::kcenon::common::Result<std::future<void>>::err(
178 ::kcenon::common::error_codes::INTERNAL_ERROR,
179 e.what(),
180 "network_to_common_thread_adapter");
181 }
182 }
183
190 ::kcenon::common::Result<std::future<void>> execute_delayed(
191 std::unique_ptr<::kcenon::common::interfaces::IJob>&& job,
192 std::chrono::milliseconds delay) override {
193 if (!pool_ || !pool_->is_running()) {
194 return ::kcenon::common::Result<std::future<void>>::err(
195 ::kcenon::common::error_codes::INVALID_ARGUMENT,
196 "Thread pool not running",
197 "network_to_common_thread_adapter");
198 }
199
200 try {
201 auto shared_job = std::shared_ptr<::kcenon::common::interfaces::IJob>(
202 std::move(job));
203
204 auto future = pool_->submit_delayed(
205 [shared_job]() mutable {
206 auto result = shared_job->execute();
207 if (result.is_err()) {
208 throw std::runtime_error(result.error().message);
209 }
210 },
211 delay);
212
213 return ::kcenon::common::Result<std::future<void>>::ok(std::move(future));
214 } catch (const std::exception& e) {
215 return ::kcenon::common::Result<std::future<void>>::err(
216 ::kcenon::common::error_codes::INTERNAL_ERROR,
217 e.what(),
218 "network_to_common_thread_adapter");
219 }
220 }
221
226 size_t worker_count() const override {
227 return pool_ ? pool_->worker_count() : 0;
228 }
229
234 bool is_running() const override {
235 return pool_ ? pool_->is_running() : false;
236 }
237
242 size_t pending_tasks() const override {
243 return pool_ ? pool_->pending_tasks() : 0;
244 }
245
253 void shutdown([[maybe_unused]] bool wait_for_completion = true) override {
254 // thread_pool_interface doesn't expose shutdown
255 // Lifecycle management should be done on the underlying pool directly
256 }
257
258private:
259 std::shared_ptr<thread_pool_interface> pool_;
260};

281class common_to_network_thread_adapter : public thread_pool_interface {
282public:
288 explicit common_to_network_thread_adapter(
289 std::shared_ptr<::kcenon::common::interfaces::IExecutor> executor)
290 : executor_(std::move(executor)) {
291 if (!executor_) {
292 throw std::invalid_argument(
293 "common_to_network_thread_adapter requires non-null executor");
294 }
295 }
296
302 std::future<void> submit(std::function<void()> task) override {
303 if (!executor_ || !executor_->is_running()) {
304 return make_error_future("Executor not running");
305 }
306
307 auto result = executor_->execute(
308 std::make_unique<function_job>(std::move(task)));
309
310 if (result.is_err()) {
311 return make_error_future(result.error().message);
312 }
313
314 return std::move(result.value());
315 }
316
323 std::future<void> submit_delayed(
324 std::function<void()> task,
325 std::chrono::milliseconds delay) override {
326 if (!executor_ || !executor_->is_running()) {
327 return make_error_future("Executor not running");
328 }
329
330 auto result = executor_->execute_delayed(
331 std::make_unique<function_job>(std::move(task)), delay);
332
333 if (result.is_err()) {
334 return make_error_future(result.error().message);
335 }
336
337 return std::move(result.value());
338 }
339
344 size_t worker_count() const override {
345 return executor_ ? executor_->worker_count() : 0;
346 }
347
352 bool is_running() const override {
353 return executor_ ? executor_->is_running() : false;
354 }
355
360 size_t pending_tasks() const override {
361 return executor_ ? executor_->pending_tasks() : 0;
362 }
363
368 void shutdown(bool wait_for_completion = true) {
369 if (executor_) {
370 executor_->shutdown(wait_for_completion);
371 }
372 }
373
374private:
375 static std::future<void> make_error_future(const std::string& message) {
376 std::promise<void> promise;
377 promise.set_exception(std::make_exception_ptr(std::runtime_error(message)));
378 return promise.get_future();
379 }
380
381 std::shared_ptr<::kcenon::common::interfaces::IExecutor> executor_;
382};
