47auto histogram_snapshot::to_prometheus(const std::string& name) const -> std::string
48{
49 std::ostringstream oss;
50 oss << std::fixed << std::setprecision(6);
51
52 // Build label string
53 std::string label_str;
54 if (!labels.empty())
55 {
56 std::ostringstream label_oss;
57 label_oss << "{";
58 bool first = true;
59 for (const auto& [key, value] : labels)
60 {
61 if (!first)
62 {
63 label_oss << ",";
64 }
65 label_oss << key << "=\"" << value << "\"";
66 first = false;
67 }
68 label_oss << "}";
69 label_str = label_oss.str();
70 }
71
72 // Output bucket counts (cumulative)
73 for (const auto& [boundary, bucket_count] : buckets)
74 {
75 oss << name << "_bucket{le=\"";
76 if (boundary == std::numeric_limits<double>::infinity())
77 {
78 oss << "+Inf";
79 }
80 else
81 {
82 oss << boundary;
83 }
84 oss << "\"";
85 if (!labels.empty())
86 {
87 oss << "," << label_str.substr(1, label_str.length() - 2);
88 }
89 oss << "} " << bucket_count << "\n";
90 }
91
92 // Output sum and count
93 oss << name << "_sum";
94 if (!labels.empty())
95 {
96 oss << label_str;
97 }
98 oss << " " << sum << "\n";
99
100 oss << name << "_count";
101 if (!labels.empty())
102 {
103 oss << label_str;
104 }
105 oss << " " << count << "\n";
106
107 return oss.str();
108}

181histogram::histogram(histogram_config cfg)
182{
183 if (cfg.bucket_boundaries.empty())
184 {
185 cfg = histogram_config::default_latency_config();
186 }
187
188 boundaries_ = std::move(cfg.bucket_boundaries);
189 std::sort(boundaries_.begin(), boundaries_.end());
190
191 // Add +Inf bucket
192 boundaries_.push_back(std::numeric_limits<double>::infinity());
193
194 // Initialize bucket counts
195 bucket_count_ = boundaries_.size();
196 bucket_counts_ = std::make_unique<std::atomic<uint64_t>[]>(bucket_count_);
197 for (size_t i = 0; i < bucket_count_; ++i)
198 {
199 bucket_counts_[i].store(0, std::memory_order_relaxed);
200 }
201}

231void histogram::record(double value)
232{

234 bucket_counts_[bucket_idx].fetch_add(1, std::memory_order_relaxed);

236 count_.fetch_add(1, std::memory_order_relaxed);
237 add_to_sum(value);

272auto histogram::percentile(double p) const -> double
273{
274 std::lock_guard<std::mutex> lock(mutex_);
275
276 uint64_t total = count_.load(std::memory_order_relaxed);
277 if (total == 0)
278 {
279 return 0.0;
280 }
281
282 // Clamp percentile to valid range
283 p = std::clamp(p, 0.0, 1.0);
284
285 // Target count for this percentile
286 double target = p * static_cast<double>(total);
287
288 // Calculate cumulative counts
289 std::vector<uint64_t> cumulative;
290 cumulative.reserve(bucket_count_);
291 uint64_t running_sum = 0;
292 for (size_t i = 0; i < bucket_count_; ++i)
293 {
294 running_sum += bucket_counts_[i].load(std::memory_order_relaxed);
295 cumulative.push_back(running_sum);
296 }
297
298 // Find the bucket containing the target count
299 size_t bucket_idx = 0;
300 for (size_t i = 0; i < cumulative.size(); ++i)
301 {
302 if (static_cast<double>(cumulative[i]) >= target)
303 {
304 bucket_idx = i;
305 break;
306 }
307 }
308
309 // Linear interpolation within the bucket
310 double lower_bound = (bucket_idx == 0) ? 0.0 : boundaries_[bucket_idx - 1];
311 double upper_bound = boundaries_[bucket_idx];
312
313 // Handle infinity bucket
314 if (std::isinf(upper_bound))
315 {
316 return lower_bound;
317 }
318
319 uint64_t lower_count = (bucket_idx == 0) ? 0 : cumulative[bucket_idx - 1];
320 uint64_t upper_count = cumulative[bucket_idx];
321
322 if (upper_count == lower_count)
323 {
324 return lower_bound;
325 }
326
327 double fraction = (target - static_cast<double>(lower_count))
328 / (static_cast<double>(upper_count) - static_cast<double>(lower_count));
329
330 return lower_bound + (fraction * (upper_bound - lower_bound));
331}

333auto histogram::buckets() const -> std::vector<std::pair<double, uint64_t>>
334{
335 std::lock_guard<std::mutex> lock(mutex_);
336
337 std::vector<std::pair<double, uint64_t>> result;
338 result.reserve(bucket_count_);
339
340 uint64_t cumulative = 0;
341 for (size_t i = 0; i < bucket_count_; ++i)
342 {
343 cumulative += bucket_counts_[i].load(std::memory_order_relaxed);
344 result.emplace_back(boundaries_[i], cumulative);
345 }
346
347 return result;
348}

350auto histogram::snapshot(const std::map<std::string, std::string>& labels) const
351 -> histogram_snapshot
352{
353 histogram_snapshot snap;
354 snap.count = count();
355 snap.sum = sum();
356 snap.min_value = min();
357 snap.max_value = max();
358 snap.labels = labels;
359
360 // Calculate percentiles
361 snap.percentiles[0.5] = p50();
362 snap.percentiles[0.9] = percentile(0.9);
363 snap.percentiles[0.95] = p95();
364 snap.percentiles[0.99] = p99();
365 snap.percentiles[0.999] = p999();
366
367 // Get buckets
368 snap.buckets = buckets();
369
370 return snap;
371}

373void histogram::reset()
374{
375 std::lock_guard<std::mutex> lock(mutex_);
376
377 count_.store(0, std::memory_order_relaxed);
378 sum_.store(0.0, std::memory_order_relaxed);
379 min_.store(std::numeric_limits<double>::infinity(), std::memory_order_relaxed);
380 max_.store(-std::numeric_limits<double>::infinity(), std::memory_order_relaxed);
381
382 for (size_t i = 0; i < bucket_count_; ++i)
383 {
384 bucket_counts_[i].store(0, std::memory_order_relaxed);
385 }
386}

388auto histogram::find_bucket(double value) const -> size_t
389{
390 auto it = std::lower_bound(boundaries_.begin(), boundaries_.end(), value);
391 if (it == boundaries_.end())
392 {
393 return boundaries_.size() - 1;
394 }
395 return static_cast<size_t>(std::distance(boundaries_.begin(), it));
396}
