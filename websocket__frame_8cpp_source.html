61 auto to_network_u16(uint16_t value) -> std::array<uint8_t, 2>
62 {
63 return {static_cast<uint8_t>((value >> 8) & 0xFF),

106 auto websocket_frame::calculate_header_size(uint64_t payload_len, bool mask) -> size_t
107 {
108 size_t header_size = MIN_HEADER_SIZE;
109
110 if (payload_len >= 65536)
111 {
112 header_size += 8; // 64-bit length field
113 }
114 else if (payload_len >= 126)
115 {
116 header_size += 2; // 16-bit length field
117 }
118
119 if (mask)
120 {
121 header_size += MASKING_KEY_SIZE;
122 }
123
124 return header_size;
125 }

127 auto websocket_frame::encode_frame(ws_opcode opcode, std::vector<uint8_t>&& payload,
128 bool fin, bool mask) -> std::vector<uint8_t>
129 {
130 const uint64_t payload_len = payload.size();
131 const size_t header_size = calculate_header_size(payload_len, mask);
132
133 std::vector<uint8_t> frame;
134 frame.reserve(header_size + payload_len);
135
136 // Byte 0: FIN, RSV1-3, Opcode
137 uint8_t byte0 = static_cast<uint8_t>(opcode);
138 if (fin)
139 {
140 byte0 |= FIN_BIT;
141 }
142 frame.push_back(byte0);
143
144 // Byte 1: Mask, Payload length
145 uint8_t byte1 = 0;
146 if (mask)
147 {
148 byte1 |= MASK_BIT;
149 }
150
151 if (payload_len < 126)
152 {
153 byte1 |= static_cast<uint8_t>(payload_len);
154 frame.push_back(byte1);
155 }
156 else if (payload_len < 65536)
157 {
158 byte1 |= PAYLOAD_LEN_16BIT;
159 frame.push_back(byte1);
160
161 // Extended payload length (16-bit)
162 auto len_bytes = to_network_u16(static_cast<uint16_t>(payload_len));
163 frame.insert(frame.end(), len_bytes.begin(), len_bytes.end());
164 }
165 else
166 {
167 byte1 |= PAYLOAD_LEN_64BIT;
168 frame.push_back(byte1);
169
170 // Extended payload length (64-bit)
171 auto len_bytes = to_network_u64(payload_len);
172 frame.insert(frame.end(), len_bytes.begin(), len_bytes.end());
173 }
174
175 // Masking key (if mask is enabled)
176 std::array<uint8_t, 4> masking_key{};
177 if (mask)
178 {
179 masking_key = generate_mask();
180 frame.insert(frame.end(), masking_key.begin(), masking_key.end());
181
182 // Apply masking to payload
183 apply_mask(payload, masking_key);
184 }
185
186 // Append payload
187 frame.insert(frame.end(), payload.begin(), payload.end());
188
189 return frame;
190 }

192 auto websocket_frame::decode_header(const std::vector<uint8_t>& data)
193 -> std::optional<ws_frame_header>
194 {
195 if (data.size() < MIN_HEADER_SIZE)
196 {
197 return std::nullopt;
198 }
199
200 ws_frame_header header{};
201
202 // Parse byte 0: FIN, RSV1-3, Opcode
203 const uint8_t byte0 = data[0];
204 header.fin = (byte0 & FIN_BIT) != 0;
205 header.rsv1 = (byte0 & RSV1_BIT) != 0;
206 header.rsv2 = (byte0 & RSV2_BIT) != 0;
207 header.rsv3 = (byte0 & RSV3_BIT) != 0;
208 header.opcode = static_cast<ws_opcode>(byte0 & OPCODE_MASK);
209
210 // Parse byte 1: Mask, Payload length
211 const uint8_t byte1 = data[1];
212 header.mask = (byte1 & MASK_BIT) != 0;
213 uint8_t payload_len = byte1 & PAYLOAD_LEN_MASK;
214
215 size_t offset = MIN_HEADER_SIZE;
216
217 // Extended payload length
218 if (payload_len == PAYLOAD_LEN_16BIT)
219 {
220 if (data.size() < offset + 2)
221 {
222 return std::nullopt;
223 }
224 header.payload_len = from_network_u16(&data[offset]);
225 offset += 2;
226 }
227 else if (payload_len == PAYLOAD_LEN_64BIT)
228 {
229 if (data.size() < offset + 8)
230 {
231 return std::nullopt;
232 }
233 header.payload_len = from_network_u64(&data[offset]);
234 offset += 8;
235 }
236 else
237 {
238 header.payload_len = payload_len;
239 }
240
241 // Masking key
242 if (header.mask)
243 {
244 if (data.size() < offset + MASKING_KEY_SIZE)
245 {
246 return std::nullopt;
247 }
248 std::copy_n(data.begin() + offset, MASKING_KEY_SIZE, header.masking_key.begin());
249 }
250
251 return header;
252 }

254 auto websocket_frame::decode_payload(const ws_frame_header& header,
255 const std::vector<uint8_t>& data)
256 -> std::vector<uint8_t>
257 {
258 const size_t header_size = calculate_header_size(header.payload_len, header.mask);
259
260 if (data.size() < header_size + header.payload_len)
261 {
262 return {};
263 }
264
265 std::vector<uint8_t> payload(data.begin() + header_size,
266 data.begin() + header_size + header.payload_len);
267
268 // Unmask if necessary
269 if (header.mask)
270 {
271 apply_mask(payload, header.masking_key);
272 }
273
274 return payload;
275 }

277 auto websocket_frame::apply_mask(std::vector<uint8_t>& data,
278 const std::array<uint8_t, 4>& mask) -> void
279 {
280 for (size_t i = 0; i < data.size(); ++i)
281 {
282 data[i] ^= mask[i % 4];
283 }
284 }

286 auto websocket_frame::generate_mask() -> std::array<uint8_t, 4>
287 {
288 static thread_local std::random_device rd;
289 static thread_local std::mt19937 gen(rd());
290 static thread_local std::uniform_int_distribution<uint32_t> dis;
291
292 const uint32_t random_value = dis(gen);
293 return {static_cast<uint8_t>((random_value >> 24) & 0xFF),
294 static_cast<uint8_t>((random_value >> 16) & 0xFF),
295 static_cast<uint8_t>((random_value >> 8) & 0xFF),
296 static_cast<uint8_t>(random_value & 0xFF)};
297 }
