74{

147auto http_server_adapter::start(uint16_t port) -> VoidResult
148{
149 if (!server_)
150 {
151 return error_void(
152 error_codes::common_errors::internal_error,
153 "Server not initialized",
154 "http_server_adapter::start"
155 );
156 }
157
158 auto result = server_->start(port);
159 if (result.is_ok())
160 {
161 is_running_.store(true, std::memory_order_release);
162 }
163 return result;
164}

166auto http_server_adapter::stop() -> VoidResult
167{
168 if (!server_)
169 {
170 return error_void(
171 error_codes::common_errors::internal_error,
172 "Server not initialized",
173 "http_server_adapter::stop"
174 );
175 }
176
177 auto result = server_->stop();
178 is_running_.store(false, std::memory_order_release);
179
180 // Clear tracked sessions on stop
181 {
182 std::lock_guard<std::mutex> lock(sessions_mutex_);
183 sessions_.clear();
184 }
185
186 return result;
187}

223auto http_server_adapter::setup_internal_routes() -> void
224{
225 if (!server_)
226 {
227 return;
228 }
229
230 // Register a catch-all POST handler for unified protocol interface
231 // The i_protocol_client.send() performs HTTP POST, so we handle POST requests
232 server_->post("/:path", [this](const core::http_request_context& ctx) {
233 // Create a session for this request
234 const auto request_id = request_counter_.fetch_add(1, std::memory_order_relaxed);
235 const auto session_id = make_session_id("client", 0, request_id);
236 auto session = std::make_shared<http_request_session>(
237 session_id, "client", 0, server_);
238
239 // Track the session
240 {
241 std::lock_guard<std::mutex> lock(sessions_mutex_);
242 sessions_[session_id] = session;
243 }
244
245 // Notify connection callback
246 {
247 connection_callback_t callback_copy;
248 {
249 std::lock_guard<std::mutex> lock(callbacks_mutex_);
250 callback_copy = connection_callback_;
251 }
252 if (callback_copy)
253 {
254 callback_copy(session);
255 }
256 }
257
258 // Notify receive callback with request body
259 {
260 receive_callback_t callback_copy;
261 {
262 std::lock_guard<std::mutex> lock(callbacks_mutex_);
263 callback_copy = receive_callback_;
264 }
265 if (callback_copy)
266 {
267 callback_copy(session_id, ctx.request.body);
268 }
269 }
270
271 // Build response from session data
272 network::internal::http_response response;
273 response.status_code = 200;
274 response.body = session->get_response_data();
275 response.set_header("Content-Type", "application/octet-stream");
276
277 // Notify disconnection callback
278 {
279 disconnection_callback_t callback_copy;
280 {
281 std::lock_guard<std::mutex> lock(callbacks_mutex_);
282 callback_copy = disconnection_callback_;
283 }
284 if (callback_copy)
285 {
286 callback_copy(session_id);
287 }
288 }
289
290 // Remove session
291 remove_session(session_id);
292
293 return response;
294 });
295
296 // Also register root POST handler
297 server_->post("/", [this](const core::http_request_context& ctx) {
298 const auto request_id = request_counter_.fetch_add(1, std::memory_order_relaxed);
299 const auto session_id = make_session_id("client", 0, request_id);
300 auto session = std::make_shared<http_request_session>(
301 session_id, "client", 0, server_);
302
303 {
304 std::lock_guard<std::mutex> lock(sessions_mutex_);
305 sessions_[session_id] = session;
306 }
307
308 {
309 connection_callback_t callback_copy;
310 {
311 std::lock_guard<std::mutex> lock(callbacks_mutex_);
312 callback_copy = connection_callback_;
313 }
314 if (callback_copy)
315 {
316 callback_copy(session);
317 }
318 }
319
320 {
321 receive_callback_t callback_copy;
322 {
323 std::lock_guard<std::mutex> lock(callbacks_mutex_);
324 callback_copy = receive_callback_;
325 }
326 if (callback_copy)
327 {
328 callback_copy(session_id, ctx.request.body);
329 }
330 }
331
332 network::internal::http_response response;
333 response.status_code = 200;
334 response.body = session->get_response_data();
335 response.set_header("Content-Type", "application/octet-stream");
336
337 {
338 disconnection_callback_t callback_copy;
339 {
340 std::lock_guard<std::mutex> lock(callbacks_mutex_);
341 callback_copy = disconnection_callback_;
342 }
343 if (callback_copy)
344 {
345 callback_copy(session_id);
346 }
347 }
348
349 remove_session(session_id);
350
351 return response;
352 });
353}

357 uint16_t port,

365auto http_server_adapter::get_or_create_session(
366 const std::string& address,
367 uint16_t port) -> std::shared_ptr<http_request_session>
368{
369 const auto request_id = request_counter_.fetch_add(1, std::memory_order_relaxed);
370 const auto session_id = make_session_id(address, port, request_id);
371
372 std::lock_guard<std::mutex> lock(sessions_mutex_);
373
374 auto session = std::make_shared<http_request_session>(
375 session_id, address, port, server_);
376 sessions_[session_id] = session;
377
378 return session;
379}
380
381auto http_server_adapter::remove_session(const std::string& session_id) -> void
382{
383 std::lock_guard<std::mutex> lock(sessions_mutex_);
384 sessions_.erase(session_id);
385}
