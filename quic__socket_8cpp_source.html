163auto quic_socket::connect(const asio::ip::udp::endpoint& endpoint,
164 const std::string& server_name) -> VoidResult
165{
166 if (role_ != quic_role::client)
167 {
168 return error_void(
169 error_codes::common_errors::invalid_argument,
170 "connect() can only be called on client sockets",
171 "quic_socket");
172 }
173
174 if (state_.load() != quic_connection_state::idle)
175 {
176 return error_void(
177 error_codes::common_errors::invalid_argument,
178 "Connection already in progress or established",
179 "quic_socket");
180 }
181
182 remote_endpoint_ = endpoint;
183
184 // Initialize client-side crypto
185 auto init_result = crypto_.init_client(
186 server_name.empty() ? endpoint.address().to_string() : server_name);
187 if (init_result.is_err())
188 {
189 return error_void(
190 error_codes::network_system::connection_failed,
191 "Failed to initialize TLS client",
192 "quic_socket",
193 init_result.error().message);
194 }
195
196 // Generate initial secrets from destination connection ID
197 // For client, we use a random connection ID as the destination
198 remote_conn_id_ = generate_connection_id();
199
200 auto derive_result = crypto_.derive_initial_secrets(remote_conn_id_);
201 if (derive_result.is_err())
202 {
203 return error_void(
204 error_codes::network_system::connection_failed,
205 "Failed to derive initial secrets",
206 "quic_socket",
207 derive_result.error().message);
208 }
209
210 transition_state(quic_connection_state::handshake_start);
211
212 // Start handshake - generate ClientHello
213 auto handshake_result = crypto_.start_handshake();
214 if (handshake_result.is_err())
215 {
216 transition_state(quic_connection_state::closed);
217 return error_void(
218 error_codes::network_system::connection_failed,
219 "Failed to start TLS handshake",
220 "quic_socket",
221 handshake_result.error().message);
222 }
223
224 // Queue the CRYPTO data for sending
225 if (!handshake_result.value().empty())
226 {
227 queue_crypto_data(std::move(handshake_result.value()));
228 }
229
230 transition_state(quic_connection_state::handshake);
231
232 // Start receiving
233 start_receive();
234
235 // Send initial packet
236 send_pending_packets();
237
238 return ok();
239}

241auto quic_socket::accept(const std::string& cert_file,
242 const std::string& key_file) -> VoidResult
243{
244 if (role_ != quic_role::server)
245 {
246 return error_void(
247 error_codes::common_errors::invalid_argument,
248 "accept() can only be called on server sockets",
249 "quic_socket");
250 }
251
252 if (state_.load() != quic_connection_state::idle)
253 {
254 return error_void(
255 error_codes::common_errors::invalid_argument,
256 "Connection already in progress",
257 "quic_socket");
258 }
259
260 // Initialize server-side crypto
261 auto init_result = crypto_.init_server(cert_file, key_file);
262 if (init_result.is_err())
263 {
264 return error_void(
265 error_codes::network_system::connection_failed,
266 "Failed to initialize TLS server",
267 "quic_socket",
268 init_result.error().message);
269 }
270
271 transition_state(quic_connection_state::handshake_start);
272
273 // Start receiving - server waits for Initial packet from client
274 start_receive();
275
276 return ok();
277}

279auto quic_socket::close(uint64_t error_code, const std::string& reason) -> VoidResult
280{
281 auto current_state = state_.load();
282 if (current_state == quic_connection_state::closed ||
283 current_state == quic_connection_state::closing ||
284 current_state == quic_connection_state::draining)
285 {
286 return ok(); // Already closing/closed
287 }
288
289 transition_state(quic_connection_state::closing);
290
291 // Build CONNECTION_CLOSE frame
292 connection_close_frame close_frame;
293 close_frame.error_code = error_code;
294 close_frame.reason_phrase = reason;
295 close_frame.is_application_error = (error_code != 0);
296
297 // Send CONNECTION_CLOSE
298 std::vector<frame> frames;
299 frames.push_back(close_frame);
300
301 auto level = handshake_complete_.load()
302 ? encryption_level::application
303 : encryption_level::initial;
304
305 auto send_result = send_packet(level, std::move(frames));
306 if (send_result.is_err())
307 {
308 // Log error but continue with close
309 }
310
311 transition_state(quic_connection_state::draining);
312
313 // Set a timer for draining period (3 * PTO), then fully close
314 idle_timer_.expires_after(std::chrono::milliseconds(300));
315 idle_timer_.async_wait(
316 [self = shared_from_this()](const std::error_code& ec)
317 {
318 if (!ec)
319 {
320 self->transition_state(quic_connection_state::closed);
321 self->stop_receive();
322 }
323 });
324
325 return ok();
326}

369auto quic_socket::create_stream(bool unidirectional) -> Result<uint64_t>
370{
371 if (!is_connected())
372 {
373 return error<uint64_t>(
374 error_codes::network_system::connection_failed,
375 "Connection not established",
376 "quic_socket");
377 }
378
379 std::lock_guard<std::mutex> lock(state_mutex_);
380
381 // Stream ID encoding (RFC 9000 Section 2.1):
382 // - Bits 0-1: Type (0=client-initiated bidi, 1=server-initiated bidi,
383 // 2=client-initiated uni, 3=server-initiated uni)
384 // - Bits 2+: Sequence number
385
386 uint64_t type_bits = 0;
387 if (role_ == quic_role::server)
388 {
389 type_bits |= 0x01; // Server-initiated
390 }
391 if (unidirectional)
392 {
393 type_bits |= 0x02; // Unidirectional
394 }
395
396 uint64_t stream_id = (next_stream_id_ << 2) | type_bits;
397 next_stream_id_++;
398
399 // Initialize the stream's pending data queue
400 pending_stream_data_[stream_id] = {};
401
402 return ok(std::move(stream_id));
403}
404
405auto quic_socket::close_stream(uint64_t stream_id) -> VoidResult
406{
407 std::lock_guard<std::mutex> lock(state_mutex_);
408
409 auto it = pending_stream_data_.find(stream_id);
410 if (it == pending_stream_data_.end())
411 {
412 return error_void(
413 error_codes::common_errors::not_found,
414 "Stream not found",
415 "quic_socket");
416 }
417
418 // Send FIN on the stream
419 it->second.push_back({{}, true});
420
421 send_pending_packets();
422
423 return ok();
424}

469auto quic_socket::do_receive() -> void
470{
471 if (!is_receiving_.load())
472 {
473 return;
474 }
475
476 auto self = shared_from_this();
477 udp_socket_.async_receive_from(
478 asio::buffer(recv_buffer_),
479 remote_endpoint_,
480 [this, self](std::error_code ec, std::size_t bytes_transferred)
481 {
482 if (!is_receiving_.load())
483 {
484 return;
485 }
486
487 if (ec)
488 {
489 if (ec != asio::error::operation_aborted)
490 {
491 std::lock_guard<std::mutex> lock(callback_mutex_);
492 if (error_cb_)
493 {
494 error_cb_(ec);
495 }
496 }
497 return;
498 }
499
500 if (bytes_transferred > 0)
501 {
502 handle_packet(std::span(recv_buffer_.data(), bytes_transferred));
503 }
504
505 // Continue receiving
506 if (is_receiving_.load())
507 {
508 do_receive();
509 }
510 });
511}

513auto quic_socket::handle_packet(std::span<const uint8_t> data) -> void
514{
515 // Parse packet header
516 auto header_result = packet_parser::parse_header(data);
517 if (header_result.is_err())
518 {
519 // Invalid packet - silently ignore
520 return;
521 }
522
523 auto& [header, header_length] = header_result.value();
524
525 // Determine encryption level
526 auto level = determine_encryption_level(header);
527
528 // For server, derive initial secrets on first Initial packet
529 if (role_ == quic_role::server &&
530 state_.load() == quic_connection_state::handshake_start)
531 {
532 if (std::holds_alternative<long_header>(header))
533 {
534 const auto& lh = std::get<long_header>(header);
535 if (lh.type() == packet_type::initial)
536 {
537 // Use client's DCID as our remote connection ID
538 remote_conn_id_ = lh.src_conn_id;
539
540 // Derive initial secrets from client's DCID
541 auto derive_result = crypto_.derive_initial_secrets(lh.dest_conn_id);
542 if (derive_result.is_err())
543 {
544 return;
545 }
546
547 transition_state(quic_connection_state::handshake);
548 }
549 }
550 }
551
552 // Get read keys for this level
553 auto keys_result = crypto_.get_read_keys(level);
554 if (keys_result.is_err())
555 {
556 // Keys not available yet - queue packet for later processing
557 return;
558 }
559
560 // Unprotect (decrypt) the packet
561 // First we need to get the packet number offset
562 size_t pn_offset = header_length;
563 if (std::holds_alternative<long_header>(header))
564 {
565 // For long headers, packet number is after the header
566 // (already accounted for in header_length for Initial/Handshake)
567 }
568
569 // Determine packet number length from header (after removing header protection)
570 size_t sample_offset = pn_offset + 4; // Sample is 4 bytes after PN start
571 if (sample_offset + hp_sample_size > data.size())
572 {
573 return; // Not enough data for sample
574 }
575
576 std::span<const uint8_t> sample(data.data() + sample_offset, hp_sample_size);
577
578 // Create mutable copy for header unprotection
579 std::vector<uint8_t> packet_copy(data.begin(), data.end());
580
581 auto unprotect_header_result = packet_protection::unprotect_header(
582 keys_result.value(),
583 std::span(packet_copy.data(), header_length + 4), // Include space for max PN
584 pn_offset,
585 sample);
586
587 if (unprotect_header_result.is_err())
588 {
589 return;
590 }
591
592 auto& [first_byte, pn_length] = unprotect_header_result.value();
593
594 // Extract packet number
595 uint64_t truncated_pn = 0;
596 for (size_t i = 0; i < pn_length; ++i)
597 {
598 truncated_pn = (truncated_pn << 8) | packet_copy[pn_offset + i];
599 }
600
601 // Decode full packet number
602 auto level_idx = static_cast<size_t>(level);
603 uint64_t full_pn = packet_number::decode(
604 truncated_pn, pn_length, largest_received_pn_[level_idx]);
605
606 // Update largest received
607 if (full_pn > largest_received_pn_[level_idx])
608 {
609 largest_received_pn_[level_idx] = full_pn;
610 }
611
612 // Decrypt packet payload
613 size_t payload_offset = pn_offset + pn_length;
614 auto unprotect_result = packet_protection::unprotect(
615 keys_result.value(),
616 std::span(packet_copy),
617 payload_offset,
618 full_pn);
619
620 if (unprotect_result.is_err())
621 {
622 return;
623 }
624
625 auto& [unprotected_header, payload] = unprotect_result.value();
626
627 // Parse frames from payload
628 auto frames_result = frame_parser::parse_all(payload);
629 if (frames_result.is_err())
630 {
631 return;
632 }
633
634 // Process each frame
635 for (const auto& f : frames_result.value())
636 {
637 process_frame(f);
638 }
639
640 // Send any pending responses
641 send_pending_packets();
642}

644auto quic_socket::process_frame(const frame& f) -> void
645{
646 std::visit([this](auto&& arg) {
647 using T = std::decay_t<decltype(arg)>;
648
649 if constexpr (std::is_same_v<T, crypto_frame>)
650 {
651 process_crypto_frame(arg);
652 }
653 else if constexpr (std::is_same_v<T, stream_frame>)
654 {
655 process_stream_frame(arg);
656 }
657 else if constexpr (std::is_same_v<T, ack_frame>)
658 {
659 process_ack_frame(arg);
660 }
661 else if constexpr (std::is_same_v<T, connection_close_frame>)
662 {
663 process_connection_close_frame(arg);
664 }
665 else if constexpr (std::is_same_v<T, handshake_done_frame>)
666 {
667 process_handshake_done_frame();
668 }
669 else if constexpr (std::is_same_v<T, ping_frame>)
670 {
671 // PING requires ACK - will be sent with next packet
672 }
673 else if constexpr (std::is_same_v<T, padding_frame>)
674 {
675 // PADDING is ignored
676 }
677 // Other frame types can be added as needed
678 }, f);
679}

681auto quic_socket::process_crypto_frame(const crypto_frame& f) -> void
682{
683 auto level = crypto_.current_level();
684
685 // Process the crypto data through TLS
686 auto response_result = crypto_.process_crypto_data(level, f.data);
687 if (response_result.is_err())
688 {
689 return;
690 }
691
692 // Queue any response crypto data
693 if (!response_result.value().empty())
694 {
695 queue_crypto_data(std::move(response_result.value()));
696 }
697
698 // Check if handshake is now complete
699 if (crypto_.is_handshake_complete() && !handshake_complete_.load())
700 {
701 handshake_complete_.store(true);
702
703 // For client, transition to connected
704 // For server, we send HANDSHAKE_DONE
705 if (role_ == quic_role::client)
706 {
707 transition_state(quic_connection_state::connected);
708
709 std::lock_guard<std::mutex> lock(callback_mutex_);
710 if (connected_cb_)
711 {
712 connected_cb_();
713 }
714 }
715 else
716 {
717 // Server sends HANDSHAKE_DONE
718 std::vector<frame> frames;
719 frames.push_back(handshake_done_frame{});
720
721 (void)send_packet(encryption_level::application, std::move(frames));
722
723 transition_state(quic_connection_state::connected);
724
725 std::lock_guard<std::mutex> lock(callback_mutex_);
726 if (connected_cb_)
727 {
728 connected_cb_();
729 }
730 }
731 }
732}

750auto quic_socket::process_connection_close_frame(const connection_close_frame& f) -> void
751{
752 transition_state(quic_connection_state::draining);
753
754 std::lock_guard<std::mutex> lock(callback_mutex_);
755 if (close_cb_)
756 {
757 close_cb_(f.error_code, f.reason_phrase);
758 }
759
760 // Enter draining period
761 idle_timer_.expires_after(std::chrono::milliseconds(300));
762 idle_timer_.async_wait(
763 [self = shared_from_this()](const std::error_code& ec)
764 {
765 if (!ec)
766 {
767 self->transition_state(quic_connection_state::closed);
768 self->stop_receive();
769 }
770 });
771}

773auto quic_socket::process_handshake_done_frame() -> void
774{
775 // Client receives HANDSHAKE_DONE
776 if (role_ == quic_role::client && !handshake_complete_.load())
777 {
778 handshake_complete_.store(true);
779 transition_state(quic_connection_state::connected);
780
781 std::lock_guard<std::mutex> lock(callback_mutex_);
782 if (connected_cb_)
783 {
784 connected_cb_();
785 }

788
789auto quic_socket::send_pending_packets() -> void
790{
791 auto current_state = state_.load();
792 if (current_state == quic_connection_state::closed ||
793 current_state == quic_connection_state::idle)
794 {
795 return;
796 }
797
798 // Determine which encryption level to use
799 encryption_level level = crypto_.current_level();
800
801 std::vector<frame> frames;
802
803 // Add pending CRYPTO data
804 {
805 std::lock_guard<std::mutex> lock(state_mutex_);
806 auto level_idx = static_cast<size_t>(level);
807 while (!pending_crypto_data_[level_idx].empty())
808 {
809 auto& data = pending_crypto_data_[level_idx].front();
810 crypto_frame cf;
811 cf.offset = 0; // Simplified - full impl needs offset tracking
812 cf.data = std::move(data);
813 frames.push_back(std::move(cf));
814 pending_crypto_data_[level_idx].pop_front();
815 }
816 }
817
818 // Add pending STREAM data (only if connected)
819 if (is_connected())
820 {
821 std::lock_guard<std::mutex> lock(state_mutex_);
822 for (auto& [stream_id, queue] : pending_stream_data_)
823 {
824 while (!queue.empty())
825 {
826 auto& [data, fin] = queue.front();
827 stream_frame sf;
828 sf.stream_id = stream_id;
829 sf.offset = 0; // Simplified - full impl needs offset tracking
830 sf.data = std::move(data);
831 sf.fin = fin;
832 frames.push_back(std::move(sf));
833 queue.pop_front();
834 }
835 }
836 }
837
838 if (!frames.empty())
839 {
840 (void)send_packet(level, std::move(frames));
841 }
842}

844auto quic_socket::send_packet(encryption_level level,
845 std::vector<frame>&& frames) -> VoidResult
846{
847 // Get write keys
848 auto keys_result = crypto_.get_write_keys(level);
849 if (keys_result.is_err())
850 {
851 return error_void(
852 error_codes::common_errors::not_initialized,
853 "Write keys not available",
854 "quic_socket");
855 }
856
857 // Build frames payload
858 std::vector<uint8_t> payload;
859 for (const auto& f : frames)
860 {
861 auto frame_bytes = frame_builder::build(f);
862 payload.insert(payload.end(), frame_bytes.begin(), frame_bytes.end());
863 }
864
865 // Get next packet number
866 auto level_idx = static_cast<size_t>(level);
867 uint64_t pn = next_packet_number_[level_idx]++;
868
869 // Build header
870 std::vector<uint8_t> header;
871 if (level == encryption_level::initial)
872 {
873 header = packet_builder::build_initial(
874 remote_conn_id_, local_conn_id_, {}, pn);
875 }
876 else if (level == encryption_level::handshake)
877 {
878 header = packet_builder::build_handshake(
879 remote_conn_id_, local_conn_id_, pn);
880 }
881 else
882 {
883 header = packet_builder::build_short(
884 remote_conn_id_, pn, crypto_.key_phase());
885 }
886
887 // Protect (encrypt) the packet
888 auto protect_result = packet_protection::protect(
889 keys_result.value(), header, payload, pn);
890
891 if (protect_result.is_err())
892 {
893 return error_void(
894 error_codes::common_errors::internal_error,
895 "Failed to protect packet",
896 "quic_socket",
897 protect_result.error().message);
898 }
899
900 // Send the packet
901 auto& protected_packet = protect_result.value();
902
903 auto self = shared_from_this();
904 auto buffer = std::make_shared<std::vector<uint8_t>>(std::move(protected_packet));
905
906 udp_socket_.async_send_to(
907 asio::buffer(*buffer),
908 remote_endpoint_,
909 [self, buffer](std::error_code ec, std::size_t /*bytes_sent*/)
910 {
911 if (ec && ec != asio::error::operation_aborted)
912 {
913 std::lock_guard<std::mutex> lock(self->callback_mutex_);
914 if (self->error_cb_)
915 {
916 self->error_cb_(ec);
917 }
918 }
919 });
920
921 return ok();
922}

933auto quic_socket::determine_encryption_level(const packet_header& header) const noexcept
934 -> encryption_level
935{
936 if (std::holds_alternative<long_header>(header))
937 {
938 const auto& lh = std::get<long_header>(header);
939 switch (lh.type())
940 {
941 case packet_type::initial:
942 return encryption_level::initial;
943 case packet_type::zero_rtt:
944 return encryption_level::zero_rtt;
945 case packet_type::handshake:
946 return encryption_level::handshake;
947 default:
948 return encryption_level::initial;
949 }
950 }
951 else
952 {
953 // Short header = 1-RTT = application level
954 return encryption_level::application;
955 }
956}
