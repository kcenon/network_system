43connection_id::connection_id(std::span<const uint8_t> data)
44{

50auto connection_id::generate(size_t length) -> connection_id
51{
52 connection_id cid;
53
54 // Clamp length to valid range
55 size_t actual_length = std::min(length, max_length);
56 if (actual_length == 0)
57 {
58 actual_length = 1; // Minimum 1 byte for non-empty CID
59 }
60
61 // Use random_device for cryptographically secure random bytes
62 std::random_device rd;
63 std::mt19937_64 gen(rd());
64 std::uniform_int_distribution<uint16_t> dist(0, 255);
65
66 for (size_t i = 0; i < actual_length; ++i)
67 {
68 cid.data_[i] = static_cast<uint8_t>(dist(gen));
69 }
70 cid.length_ = static_cast<uint8_t>(actual_length);
71
72 return cid;
73}

105auto connection_id::operator<(const connection_id& other) const noexcept -> bool
106{
107 // Compare by length first, then by content
108 if (length_ != other.length_)
109 {
110 return length_ < other.length_;
111 }
112 return std::lexicographical_compare(
113 data_.begin(), data_.begin() + length_,
114 other.data_.begin(), other.data_.begin() + other.length_);
115}

117auto connection_id::to_string() const -> std::string
118{
119 if (length_ == 0)
120 {
121 return "<empty>";
122 }
123
124 std::ostringstream oss;
125 oss << std::hex << std::setfill('0');
126 for (size_t i = 0; i < length_; ++i)
127 {
128 oss << std::setw(2) << static_cast<unsigned>(data_[i]);
129 }
130 return oss.str();
131}
