64auto rtt_estimator::update(std::chrono::microseconds latest_rtt,
65 std::chrono::microseconds ack_delay,
66 bool is_handshake_confirmed) -> void
67{
68 latest_rtt_ = latest_rtt;
69
70 // Update min_rtt (RFC 9002 Section 5.2)
71 if (latest_rtt < min_rtt_)
72 {
73 min_rtt_ = latest_rtt;
74 }
75
76 // Adjust RTT sample by subtracting ACK delay (RFC 9002 Section 5.3)
77 auto adjusted_rtt = latest_rtt;
78 if (is_handshake_confirmed)
79 {
80 // Only subtract ACK delay for handshake-confirmed samples
81 // and only up to max_ack_delay
82 auto effective_ack_delay = std::min(ack_delay, max_ack_delay_);
83
84 // Don't adjust RTT below min_rtt
85 if (adjusted_rtt > min_rtt_ + effective_ack_delay)
86 {
87 adjusted_rtt -= effective_ack_delay;
88 }
89 else if (adjusted_rtt > min_rtt_)
90 {
91 adjusted_rtt = min_rtt_;
92 }
93 }
94
95 if (first_sample_)
96 {
97 // First RTT sample (RFC 9002 Section 5.3)
98 smoothed_rtt_ = adjusted_rtt;
99 rttvar_ = adjusted_rtt / 2;
100 first_sample_ = false;
101 }
102 else
103 {
104 // Subsequent RTT samples (RFC 9002 Section 5.3)
105 // rttvar = 3/4 * rttvar + 1/4 * |smoothed_rtt - adjusted_rtt|
106 auto rtt_diff = smoothed_rtt_ > adjusted_rtt
107 ? smoothed_rtt_ - adjusted_rtt
108 : adjusted_rtt - smoothed_rtt_;
109
110 rttvar_ = std::chrono::microseconds{
111 (3 * rttvar_.count() + rtt_diff.count()) / 4};
112
113 // smoothed_rtt = 7/8 * smoothed_rtt + 1/8 * adjusted_rtt
114 smoothed_rtt_ = std::chrono::microseconds{
115 (7 * smoothed_rtt_.count() + adjusted_rtt.count()) / 8};
116 }
117}

119auto rtt_estimator::pto() const noexcept -> std::chrono::microseconds
120{
121 // PTO = smoothed_rtt + max(4*rttvar, kGranularity) + max_ack_delay
122 // (RFC 9002 Section 6.2.1)
123 auto granularity_us = std::chrono::duration_cast<std::chrono::microseconds>(kGranularity);
124 auto four_rttvar = std::chrono::microseconds{4 * rttvar_.count()};
125 auto pto_value = smoothed_rtt_ + std::max(four_rttvar, granularity_us) + max_ack_delay_;
126
127 return pto_value;
128}
