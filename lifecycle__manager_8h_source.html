200 [[nodiscard]] auto prepare_stop() -> bool
201 {
202 // Check if already stopping
203 bool expected = false;
204 if (!stop_initiated_.compare_exchange_strong(expected, true,
205 std::memory_order_acq_rel))
206 {
207 return false; // Already stopping
208 }
209
210 // Check if running
211 if (!is_running_.load(std::memory_order_acquire))
212 {
213 stop_initiated_.store(false, std::memory_order_release);
214 return false; // Not running
215 }
216
217 // Create synchronization primitives
218 stop_promise_.emplace();
219 stop_future_ = stop_promise_->get_future();
220 return true;
221 }
222
228 auto reset() -> void
229 {
230 is_running_.store(false, std::memory_order_release);
231 stop_initiated_.store(false, std::memory_order_release);
232 stop_promise_.reset();
233 stop_future_ = std::future<void>{};
234 }
235
236 private:
237 std::atomic<bool> is_running_{false};
238 std::atomic<bool> stop_initiated_{false};
239 std::optional<std::promise<void>> stop_promise_;
240 std::future<void> stop_future_;
