84auto messaging_quic_server::start_server(unsigned short port,
85 const quic_server_config& config)
86 -> VoidResult
87{
88 if (lifecycle_.is_running())
89 {
90 return error_void(
91 error_codes::network_system::server_already_running,
92 "QUIC server is already running",
93 "messaging_quic_server::start_server");
94 }
95
96 config_ = config;
97 lifecycle_.set_running();
98
99 auto result = do_start_impl(port);
100 if (result.is_err())
101 {
102 lifecycle_.mark_stopped();
103 }
104
105 return result;
106}

108auto messaging_quic_server::stop_server() -> VoidResult
109{
110 if (!lifecycle_.is_running())
111 {
112 return error_void(

114 "QUIC server is not running",
115 "messaging_quic_server::stop_server");

147auto messaging_quic_server::do_start_impl(unsigned short port) -> VoidResult
148{
149 try
150 {
151 // Create io_context
152 io_context_ = std::make_unique<asio::io_context>();
153 work_guard_ = std::make_unique<
154 asio::executor_work_guard<asio::io_context::executor_type>>(
155 asio::make_work_guard(*io_context_));
156
157 // Create UDP socket
158 udp_socket_ = std::make_unique<asio::ip::udp::socket>(
159 *io_context_,
160 asio::ip::udp::endpoint(asio::ip::udp::v4(), port));
161
162 // Create cleanup timer
163 cleanup_timer_ = std::make_unique<asio::steady_timer>(*io_context_);
164
165 // Start receiving packets
166 start_receive();
167
168 // Start periodic cleanup timer
169 start_cleanup_timer();
170
171 // Get thread pool from network context
172 thread_pool_ = network_context::instance().get_thread_pool();
173 if (!thread_pool_)
174 {
175 thread_pool_ = std::make_shared<integration::basic_thread_pool>(2);
176 }
177
178 // Submit io_context run task to thread pool
179 io_context_future_ = thread_pool_->submit(
180 [this]()
181 {
182 try
183 {
184 NETWORK_LOG_INFO(
185 "[messaging_quic_server] Starting io_context on thread pool");
186 io_context_->run();
187 NETWORK_LOG_INFO(
188 "[messaging_quic_server] io_context stopped");
189 }
190 catch (const std::exception& e)
191 {
192 NETWORK_LOG_ERROR(
193 "[messaging_quic_server] Exception in io_context: "
194 + std::string(e.what()));
195 }
196 });
197
198 NETWORK_LOG_INFO("[messaging_quic_server] Started listening on port "
199 + std::to_string(port));
200 return ok();
201 }
202 catch (const std::system_error& e)
203 {
204 if (e.code() == asio::error::address_in_use ||
205 e.code() == std::errc::address_in_use)
206 {
207 return error_void(error_codes::network_system::bind_failed,
208 "Failed to bind to port: address already in use",
209 "messaging_quic_server::do_start_impl",
210 "Port: " + std::to_string(port));
211 }
212 else if (e.code() == asio::error::access_denied ||
213 e.code() == std::errc::permission_denied)
214 {
215 return error_void(error_codes::network_system::bind_failed,
216 "Failed to bind to port: permission denied",
217 "messaging_quic_server::do_start_impl",
218 "Port: " + std::to_string(port));
219 }
220
221 return error_void(error_codes::common_errors::internal_error,
222 "Failed to start server: " + std::string(e.what()),
223 "messaging_quic_server::do_start_impl",
224 "Port: " + std::to_string(port));
225 }
226 catch (const std::exception& e)
227 {
228 return error_void(error_codes::common_errors::internal_error,
229 "Failed to start server: " + std::string(e.what()),
230 "messaging_quic_server::do_start_impl",
231 "Port: " + std::to_string(port));
232 }
233}

235auto messaging_quic_server::do_stop_impl() -> VoidResult
236{
237 try
238 {
239 // Cancel cleanup timer
240 if (cleanup_timer_)
241 {
242 cleanup_timer_->cancel();
243 }
244
245 // Close UDP socket
246 if (udp_socket_)
247 {
248 asio::error_code ec;
249 udp_socket_->cancel(ec);
250 if (udp_socket_->is_open())
251 {
252 udp_socket_->close(ec);
253 }
254 }
255
256 // Stop all sessions
257 disconnect_all(0);
258
259 // Release work guard
260 if (work_guard_)
261 {
262 work_guard_.reset();
263 }
264
265 // Stop io_context
266 if (io_context_)
267 {
268 io_context_->stop();
269 }
270
271 // Wait for io_context task
272 if (io_context_future_.valid())
273 {
274 io_context_future_.wait();
275 }
276
277 // Release resources
278 udp_socket_.reset();
279 cleanup_timer_.reset();
280 thread_pool_.reset();
281 io_context_.reset();
282
283 NETWORK_LOG_INFO("[messaging_quic_server] Stopped.");
284 return ok();
285 }
286 catch (const std::exception& e)
287 {
288 return error_void(error_codes::common_errors::internal_error,
289 "Failed to stop server: " + std::string(e.what()),
290 "messaging_quic_server::do_stop_impl",
291 "Server ID: " + server_id_);
292 }

326auto messaging_quic_server::disconnect_session(const std::string& session_id,
327 uint64_t error_code)
328 -> VoidResult
329{
330 std::shared_ptr<session::quic_session> session;
331 {
332 std::unique_lock<std::shared_mutex> lock(sessions_mutex_);
333 auto it = sessions_.find(session_id);
334 if (it == sessions_.end())
335 {
336 return error_void(error_codes::common_errors::not_found,
337 "Session not found",
338 "messaging_quic_server::disconnect_session",
339 "Session ID: " + session_id);
340 }
341 session = it->second;
342 sessions_.erase(it);
343 }
344
345 if (session)
346 {
347 return session->close(error_code);
348 }
349 return ok();
350}
351
352auto messaging_quic_server::disconnect_all(uint64_t error_code) -> void
353{
354 std::vector<std::shared_ptr<session::quic_session>> sessions_to_close;
355 {
356 std::unique_lock<std::shared_mutex> lock(sessions_mutex_);
357 sessions_to_close.reserve(sessions_.size());
358 for (auto& [id, session] : sessions_)
359 {
360 sessions_to_close.push_back(session);
361 }
362 sessions_.clear();
363 }
364
365 for (auto& session : sessions_to_close)
366 {
367 if (session)
368 {
369 auto result = session->close(error_code);
370 (void)result;
371 }
372 }

375auto messaging_quic_server::broadcast(std::vector<uint8_t>&& data)
376 -> VoidResult
377{
378 auto sessions_list = sessions();
379 for (auto& session : sessions_list)
380 {
381 if (session && session->is_active())
382 {
383 std::vector<uint8_t> data_copy(data);
384 auto result = session->send(std::move(data_copy));
385 if (result.is_err())
386 {
387 NETWORK_LOG_WARN("[messaging_quic_server] Failed to send to session "
388 + session->session_id() + ": "
389 + result.error().message);
390 }
391 }
392 }
393 return ok();
394}

431auto messaging_quic_server::start_receive() -> void
432{
433 if (!is_running() || !udp_socket_)
434 {
435 return;
436 }
437
438 auto self = shared_from_this();
439 udp_socket_->async_receive_from(
440 asio::buffer(recv_buffer_),
441 recv_endpoint_,
442 [this, self](std::error_code ec, std::size_t bytes_received)
443 {
444 if (!is_running())
445 {
446 return;
447 }
448
449 if (ec)
450 {
451 if (ec != asio::error::operation_aborted)
452 {
453 NETWORK_LOG_ERROR(
454 "[messaging_quic_server] Receive error: " + ec.message());
455
456 invoke_error_callback(ec);
457 }
458 return;
459 }
460
461 // Handle the received packet
462 std::span<const uint8_t> packet_data(recv_buffer_.data(),
463 bytes_received);
464 handle_packet(packet_data, recv_endpoint_);
465
466 // Continue receiving
467 start_receive();
468 });
469}

471auto messaging_quic_server::handle_packet(std::span<const uint8_t> data,
472 const asio::ip::udp::endpoint& from)
473 -> void
474{
475 if (data.empty())
476 {
477 return;
478 }
479
480 // Parse packet header to get destination connection ID
481 auto header_result = protocols::quic::packet_parser::parse_header(data);
482 if (header_result.is_err())
483 {
484 NETWORK_LOG_DEBUG("[messaging_quic_server] Invalid packet from "
485 + from.address().to_string());
486 return;
487 }
488
489 // Extract destination connection ID from header
490 protocols::quic::connection_id dcid;
491 std::visit(
492 [&dcid](auto&& hdr)
493 {
494 dcid = hdr.dest_conn_id;
495 },
496 header_result.value().first);
497
498 // Find or create session for this connection
499 auto session = find_or_create_session(dcid, from);
500 if (!session)
501 {
502 NETWORK_LOG_DEBUG(
503 "[messaging_quic_server] Could not find or create session for packet");
504 return;
505 }
506
507 // Delegate packet handling to session
508 session->handle_packet(data);
509}

511auto messaging_quic_server::find_or_create_session(
512 const protocols::quic::connection_id& dcid,
513 const asio::ip::udp::endpoint& endpoint)
514 -> std::shared_ptr<session::quic_session>
515{
516 // First, check existing sessions
517 {
518 std::shared_lock<std::shared_mutex> lock(sessions_mutex_);
519 for (const auto& [id, session] : sessions_)
520 {
521 if (session->matches_connection_id(dcid))
522 {
523 return session;
524 }
525 }
526 }
527
528 // Check connection limit
529 if (session_count() >= config_.max_connections)
530 {
531 NETWORK_LOG_WARN(
532 "[messaging_quic_server] Connection limit reached, rejecting new connection");
533 return nullptr;
534 }
535
536 // Create new session for this connection
537 auto session_id = generate_session_id();
538
539 // Create QUIC socket for the new session
540 asio::ip::udp::socket session_socket(
541 *io_context_, asio::ip::udp::endpoint(asio::ip::udp::v4(), 0));
542 session_socket.connect(endpoint);
543
544 auto quic_socket = std::make_shared<internal::quic_socket>(
545 std::move(session_socket), internal::quic_role::server);
546
547 // Accept the connection with TLS config
548 if (!config_.cert_file.empty() && !config_.key_file.empty())
549 {
550 auto accept_result =
551 quic_socket->accept(config_.cert_file, config_.key_file);
552 if (accept_result.is_err())
553 {
554 NETWORK_LOG_ERROR(
555 "[messaging_quic_server] Failed to accept connection: "
556 + accept_result.error().message);
557 return nullptr;
558 }
559 }
560
561 auto session =
562 std::make_shared<session::quic_session>(quic_socket, session_id);
563
564 // Set up session callbacks
565 auto self = weak_from_this();
566
567 session->set_receive_callback(
568 [self, session_id](const std::vector<uint8_t>& data)
569 {
570 if (auto server = self.lock())
571 {
572 auto sess = server->get_session(session_id);
573 if (sess)
574 {
575 server->invoke_receive_callback(sess, data);
576 }
577 }
578 });
579
580 session->set_stream_receive_callback(
581 [self, session_id](uint64_t stream_id,
582 const std::vector<uint8_t>& data,
583 bool fin)
584 {
585 if (auto server = self.lock())
586 {
587 auto sess = server->get_session(session_id);
588 if (sess)
589 {
590 server->invoke_stream_receive_callback(sess, stream_id, data, fin);
591 }
592 }
593 });
594
595 session->set_close_callback(
596 [self, session_id]()
597 {
598 if (auto server = self.lock())
599 {
600 server->on_session_close(session_id);
601 }
602 });
603
604 // Add session to map
605 {
606 std::unique_lock<std::shared_mutex> lock(sessions_mutex_);
607 sessions_[session_id] = session;
608 }
609
610 // Start the session
611 session->start_session();
612
613 // Report metrics
614 metrics::metric_reporter::report_connection_accepted();
615 metrics::metric_reporter::report_active_connections(session_count());
616
617 // Invoke connection callback
618 invoke_connection_callback(session);
619
620#if KCENON_WITH_COMMON_SYSTEM
621 if (monitor_)
622 {
623 monitor_->record_metric("active_connections",
624 static_cast<double>(session_count()));
625 }
626#endif
627
628 NETWORK_LOG_INFO("[messaging_quic_server] New session created: "
629 + session_id + " from " + endpoint.address().to_string());
630
631 return session;
632}

635{

642auto messaging_quic_server::on_session_close(const std::string& session_id)
643 -> void
644{
645 std::shared_ptr<session::quic_session> session;
646 {
647 std::unique_lock<std::shared_mutex> lock(sessions_mutex_);
648 auto it = sessions_.find(session_id);
649 if (it != sessions_.end())
650 {
651 session = it->second;
652 sessions_.erase(it);
653 }
654 }
655
656 if (session)
657 {
658 // Report metrics
659 metrics::metric_reporter::report_active_connections(session_count());
660
661 // Invoke disconnection callback
662 invoke_disconnection_callback(session);
663
664#if KCENON_WITH_COMMON_SYSTEM
665 if (monitor_)
666 {
667 monitor_->record_metric("active_connections",
668 static_cast<double>(session_count()));
669 }
670#endif
671
672 NETWORK_LOG_INFO("[messaging_quic_server] Session closed: "
673 + session_id);
674 }
675}

677auto messaging_quic_server::start_cleanup_timer() -> void
678{
679 if (!cleanup_timer_ || !is_running())
680 {
681 return;
682 }
683
684 // Schedule cleanup every 30 seconds
685 cleanup_timer_->expires_after(std::chrono::seconds(30));
686
687 auto self = shared_from_this();
688 cleanup_timer_->async_wait(
689 [this, self](const std::error_code& ec)
690 {
691 if (!ec && is_running())
692 {
693 cleanup_dead_sessions();
694 start_cleanup_timer(); // Reschedule
695 }
696 });
697}
698
699auto messaging_quic_server::cleanup_dead_sessions() -> void
700{
701 std::vector<std::string> dead_session_ids;
702
703 {
704 std::shared_lock<std::shared_mutex> lock(sessions_mutex_);
705 for (const auto& [id, session] : sessions_)
706 {
707 if (!session || !session->is_active())
708 {
709 dead_session_ids.push_back(id);
710 }
711 }
712 }
713
714 for (const auto& id : dead_session_ids)
715 {
716 on_session_close(id);
717 }
718
719 if (!dead_session_ids.empty())
720 {
721 NETWORK_LOG_DEBUG(
722 "[messaging_quic_server] Cleaned up "
723 + std::to_string(dead_session_ids.size())
724 + " dead sessions. Active: " + std::to_string(session_count()));
725 }
726}
