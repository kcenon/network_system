103VoidResult messaging_bridge::initialize(const BridgeConfig& config) {
104 std::lock_guard<std::mutex> lock(pimpl_->metrics_mutex_);
105
106 if (pimpl_->initialized_.load()) {
107 return error_void(
108 error_codes::common_errors::already_exists,
109 "messaging_bridge already initialized",
110 "messaging_bridge::initialize");
111 }
112
113 // Process configuration properties
114 auto enabled_it = config.properties.find("enabled");
115 if (enabled_it != config.properties.end() && enabled_it->second == "false") {
116 return error_void(
117 error_codes::common_errors::invalid_argument,
118 "Bridge is disabled in configuration",
119 "messaging_bridge::initialize");
120 }
121
122 // Initialize bridge metrics
123 pimpl_->bridge_metrics_.is_healthy = true;
124 pimpl_->bridge_metrics_.last_activity = std::chrono::steady_clock::now();
125 pimpl_->bridge_metrics_.custom_metrics.clear();
126
127 pimpl_->initialized_.store(true);
128 return ok();
129}
130
131VoidResult messaging_bridge::shutdown() {
132 std::lock_guard<std::mutex> lock(pimpl_->metrics_mutex_);
133
134 if (!pimpl_->initialized_.load()) {
135 return ok(); // Already shut down, idempotent
136 }
137
138 pimpl_->initialized_.store(false);
139 pimpl_->bridge_metrics_.is_healthy = false;
140
141 return ok();
142}
143
144bool messaging_bridge::is_initialized() const {

146}

148BridgeMetrics messaging_bridge::get_metrics() const {
149 std::lock_guard<std::mutex> lock(pimpl_->metrics_mutex_);
150
151 // Update bridge metrics from performance metrics
152 pimpl_->bridge_metrics_.is_healthy = pimpl_->initialized_.load();
153 pimpl_->bridge_metrics_.last_activity = std::chrono::steady_clock::now();
154 pimpl_->bridge_metrics_.custom_metrics["messages_sent"] = static_cast<double>(pimpl_->metrics_.messages_sent);
155 pimpl_->bridge_metrics_.custom_metrics["messages_received"] = static_cast<double>(pimpl_->metrics_.messages_received);
156 pimpl_->bridge_metrics_.custom_metrics["bytes_sent"] = static_cast<double>(pimpl_->metrics_.bytes_sent);
157 pimpl_->bridge_metrics_.custom_metrics["bytes_received"] = static_cast<double>(pimpl_->metrics_.bytes_received);
158 pimpl_->bridge_metrics_.custom_metrics["connections_active"] = static_cast<double>(pimpl_->metrics_.connections_active);
159 pimpl_->bridge_metrics_.custom_metrics["avg_latency_ms"] = static_cast<double>(pimpl_->metrics_.avg_latency.count());
160
161 return pimpl_->bridge_metrics_;
162}
