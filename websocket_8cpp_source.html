48auto generate_unique_id(std::string_view prefix) -> std::string
49{
50 static std::atomic<uint64_t> counter{0};
51 auto now = std::chrono::steady_clock::now().time_since_epoch().count();
52 auto count = counter.fetch_add(1);
53
54 std::ostringstream oss;
55 oss << prefix << "-" << now << "-" << count;
56 return oss.str();
57}

61auto create_connection(std::string_view id) -> std::unique_ptr<unified::i_connection>
62{
63 std::string connection_id = id.empty() ? generate_unique_id("ws-conn") : std::string(id);
64 return std::make_unique<unified::adapters::ws_connection_adapter>(connection_id);
65}
