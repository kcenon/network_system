56auto get_frame_type(const frame& f) -> frame_type
57{
58 return std::visit([](const auto& fr) -> frame_type {
59 using T = std::decay_t<decltype(fr)>;
60 if constexpr (std::is_same_v<T, padding_frame>)
61 return frame_type::padding;
62 else if constexpr (std::is_same_v<T, ping_frame>)
63 return frame_type::ping;
64 else if constexpr (std::is_same_v<T, ack_frame>)
65 return fr.ecn ? frame_type::ack_ecn : frame_type::ack;
66 else if constexpr (std::is_same_v<T, reset_stream_frame>)
67 return frame_type::reset_stream;
68 else if constexpr (std::is_same_v<T, stop_sending_frame>)
69 return frame_type::stop_sending;
70 else if constexpr (std::is_same_v<T, crypto_frame>)
71 return frame_type::crypto;
72 else if constexpr (std::is_same_v<T, new_token_frame>)
73 return frame_type::new_token;
74 else if constexpr (std::is_same_v<T, stream_frame>)
75 return frame_type::stream_base;
76 else if constexpr (std::is_same_v<T, max_data_frame>)
77 return frame_type::max_data;
78 else if constexpr (std::is_same_v<T, max_stream_data_frame>)
79 return frame_type::max_stream_data;
80 else if constexpr (std::is_same_v<T, max_streams_frame>)
81 return fr.bidirectional ? frame_type::max_streams_bidi
82 : frame_type::max_streams_uni;
83 else if constexpr (std::is_same_v<T, data_blocked_frame>)
84 return frame_type::data_blocked;
85 else if constexpr (std::is_same_v<T, stream_data_blocked_frame>)
86 return frame_type::stream_data_blocked;
87 else if constexpr (std::is_same_v<T, streams_blocked_frame>)
88 return fr.bidirectional ? frame_type::streams_blocked_bidi
89 : frame_type::streams_blocked_uni;
90 else if constexpr (std::is_same_v<T, new_connection_id_frame>)
91 return frame_type::new_connection_id;
92 else if constexpr (std::is_same_v<T, retire_connection_id_frame>)
93 return frame_type::retire_connection_id;
94 else if constexpr (std::is_same_v<T, path_challenge_frame>)
95 return frame_type::path_challenge;
96 else if constexpr (std::is_same_v<T, path_response_frame>)
97 return frame_type::path_response;
98 else if constexpr (std::is_same_v<T, connection_close_frame>)
99 return fr.is_application_error ? frame_type::connection_close_app
100 : frame_type::connection_close;
101 else if constexpr (std::is_same_v<T, handshake_done_frame>)
102 return frame_type::handshake_done;
103 else
104 return frame_type::padding;
105 }, f);
106}

150auto frame_parser::parse(std::span<const uint8_t> data)
151 -> Result<std::pair<frame, size_t>>
152{
153 if (data.empty())
154 {
155 return make_error<std::pair<frame, size_t>>(
156 error_codes::common_errors::invalid_argument,
157 "Empty frame data");
158 }
159
160 // Parse frame type
161 auto type_result = varint::decode(data);
162 if (type_result.is_err())
163 {
164 return make_error<std::pair<frame, size_t>>(
165 error_codes::common_errors::invalid_argument,
166 "Failed to decode frame type");
167 }
168
169 auto [type_value, type_len] = type_result.value();
170 auto remaining = data.subspan(type_len);
171
172 // Check for STREAM frames (0x08-0x0f)
173 if (is_stream_frame(type_value))
174 {
175 auto result = parse_stream(remaining, get_stream_flags(type_value));
176 if (result.is_err()) return result;
177 auto& [f, consumed] = result.value();
178 return ok(std::make_pair(std::move(f), type_len + consumed));
179 }
180
181 // Dispatch based on frame type
182 switch (static_cast<frame_type>(type_value))
183 {
184 case frame_type::padding:
185 return parse_padding(data);
186
187 case frame_type::ping:
188 {
189 auto result = parse_ping(remaining);
190 if (result.is_err()) return result;

195 case frame_type::ack:

371auto frame_parser::parse_all(std::span<const uint8_t> data)
372 -> Result<std::vector<frame>>
373{
374 std::vector<frame> frames;
375 size_t offset = 0;
376
377 while (offset < data.size())
378 {
379 auto result = parse(data.subspan(offset));
380 if (result.is_err())
381 {
382 return make_error<std::vector<frame>>(
383 error_codes::common_errors::invalid_argument,
384 "Failed to parse frame at offset " + std::to_string(offset));
385 }
386
387 auto& [f, consumed] = result.value();
388 frames.push_back(std::move(f));
389 offset += consumed;
390 }
391
392 return ok(std::move(frames));
393}

395auto frame_parser::parse_padding(std::span<const uint8_t> data)
396 -> Result<std::pair<frame, size_t>>
397{
398 // Count consecutive padding bytes
399 size_t count = 0;
400 while (count < data.size() && data[count] == 0x00)
401 {
402 ++count;
403 }
404
405 if (count == 0)
406 {
407 count = 1; // At least one padding byte was consumed (the type)
408 }
409
410 padding_frame f;
411 f.count = count;
412 return ok(std::make_pair(frame{f}, count));
413}

422auto frame_parser::parse_ack(std::span<const uint8_t> data, bool has_ecn)
423 -> Result<std::pair<frame, size_t>>
424{
425 size_t offset = 0;
426 ack_frame f;
427
428 // Largest Acknowledged
429 auto largest = varint::decode(data.subspan(offset));
430 if (largest.is_err()) return make_error<std::pair<frame, size_t>>(
431 error_codes::common_errors::invalid_argument, "Failed to parse largest acknowledged");
432 f.largest_acknowledged = largest.value().first;
433 offset += largest.value().second;
434
435 // ACK Delay
436 auto delay = varint::decode(data.subspan(offset));
437 if (delay.is_err()) return make_error<std::pair<frame, size_t>>(
438 error_codes::common_errors::invalid_argument, "Failed to parse ack delay");
439 f.ack_delay = delay.value().first;
440 offset += delay.value().second;
441
442 // ACK Range Count
443 auto range_count = varint::decode(data.subspan(offset));
444 if (range_count.is_err()) return make_error<std::pair<frame, size_t>>(
445 error_codes::common_errors::invalid_argument, "Failed to parse ack range count");
446 offset += range_count.value().second;
447
448 // First ACK Range
449 auto first_range = varint::decode(data.subspan(offset));
450 if (first_range.is_err()) return make_error<std::pair<frame, size_t>>(
451 error_codes::common_errors::invalid_argument, "Failed to parse first ack range");
452 offset += first_range.value().second;
453
454 // The first range implicitly starts at largest_acknowledged
455 // Additional ranges
456 for (uint64_t i = 0; i < range_count.value().first; ++i)
457 {
458 ack_range range;
459
460 auto gap = varint::decode(data.subspan(offset));
461 if (gap.is_err()) return make_error<std::pair<frame, size_t>>(
462 error_codes::common_errors::invalid_argument, "Failed to parse ack gap");
463 range.gap = gap.value().first;
464 offset += gap.value().second;
465
466 auto length = varint::decode(data.subspan(offset));
467 if (length.is_err()) return make_error<std::pair<frame, size_t>>(
468 error_codes::common_errors::invalid_argument, "Failed to parse ack range length");
469 range.length = length.value().first;
470 offset += length.value().second;
471
472 f.ranges.push_back(range);
473 }
474
475 // ECN Counts (if present)
476 if (has_ecn)
477 {
478 ecn_counts ecn;
479
480 auto ect0 = varint::decode(data.subspan(offset));
481 if (ect0.is_err()) return make_error<std::pair<frame, size_t>>(
482 error_codes::common_errors::invalid_argument, "Failed to parse ECT(0) count");
483 ecn.ect0 = ect0.value().first;
484 offset += ect0.value().second;
485
486 auto ect1 = varint::decode(data.subspan(offset));
487 if (ect1.is_err()) return make_error<std::pair<frame, size_t>>(
488 error_codes::common_errors::invalid_argument, "Failed to parse ECT(1) count");
489 ecn.ect1 = ect1.value().first;
490 offset += ect1.value().second;
491
492 auto ecn_ce = varint::decode(data.subspan(offset));
493 if (ecn_ce.is_err()) return make_error<std::pair<frame, size_t>>(
494 error_codes::common_errors::invalid_argument, "Failed to parse ECN-CE count");
495 ecn.ecn_ce = ecn_ce.value().first;
496 offset += ecn_ce.value().second;
497
498 f.ecn = ecn;
499 }
500
501 return ok(std::make_pair(frame{f}, offset));
502}

552auto frame_parser::parse_crypto(std::span<const uint8_t> data)
553 -> Result<std::pair<frame, size_t>>
554{
555 size_t offset = 0;
556 crypto_frame f;
557
558 auto crypto_offset = varint::decode(data.subspan(offset));
559 if (crypto_offset.is_err()) return make_error<std::pair<frame, size_t>>(
560 error_codes::common_errors::invalid_argument, "Failed to parse crypto offset");
561 f.offset = crypto_offset.value().first;
562 offset += crypto_offset.value().second;
563
564 auto length = varint::decode(data.subspan(offset));
565 if (length.is_err()) return make_error<std::pair<frame, size_t>>(
566 error_codes::common_errors::invalid_argument, "Failed to parse crypto length");
567 offset += length.value().second;
568
569 if (data.size() - offset < length.value().first)
570 {
571 return make_error<std::pair<frame, size_t>>(
572 error_codes::common_errors::invalid_argument, "Insufficient crypto data");
573 }
574
575 f.data.assign(data.begin() + offset, data.begin() + offset + length.value().first);
576 offset += length.value().first;
577
578 return ok(std::make_pair(frame{f}, offset));
579}

604auto frame_parser::parse_stream(std::span<const uint8_t> data, uint8_t flags)
605 -> Result<std::pair<frame, size_t>>
606{
607 size_t offset = 0;
608 stream_frame f;
609
610 // Stream ID
611 auto stream_id = varint::decode(data.subspan(offset));
612 if (stream_id.is_err()) return make_error<std::pair<frame, size_t>>(
613 error_codes::common_errors::invalid_argument, "Failed to parse stream id");
614 f.stream_id = stream_id.value().first;
615 offset += stream_id.value().second;
616
617 // Offset (if OFF bit set)
618 if (flags & stream_flags::off)
619 {
620 auto stream_offset = varint::decode(data.subspan(offset));
621 if (stream_offset.is_err()) return make_error<std::pair<frame, size_t>>(
622 error_codes::common_errors::invalid_argument, "Failed to parse stream offset");
623 f.offset = stream_offset.value().first;
624 offset += stream_offset.value().second;
625 }
626
627 // Length (if LEN bit set)
628 uint64_t length = 0;
629 if (flags & stream_flags::len)
630 {
631 auto len_result = varint::decode(data.subspan(offset));
632 if (len_result.is_err()) return make_error<std::pair<frame, size_t>>(

762 auto seq = varint::decode(data.subspan(offset));
763 if (seq.is_err()) return make_error<std::pair<frame, size_t>>(
764 error_codes::common_errors::invalid_argument, "Failed to parse sequence number");
765 f.sequence_number = seq.value().first;

769 auto retire = varint::decode(data.subspan(offset));
770 if (retire.is_err()) return make_error<std::pair<frame, size_t>>(
771 error_codes::common_errors::invalid_argument, "Failed to parse retire prior to");

790 {
791 return make_error<std::pair<frame, size_t>>(

802 }
803 std::copy(data.begin() + offset, data.begin() + offset + 16,
804 f.stateless_reset_token.begin());

920{

937 else if constexpr (std::is_same_v<T, stream_frame>)
938 return build_stream(fr);
939 else if constexpr (std::is_same_v<T, max_data_frame>)
940 return build_max_data(fr);
941 else if constexpr (std::is_same_v<T, max_stream_data_frame>)

978auto frame_builder::build_ack(const ack_frame& f) -> std::vector<uint8_t>
979{
980 std::vector<uint8_t> buffer;
981
982 // Frame type
983 append_varint(buffer, f.ecn ? static_cast<uint64_t>(frame_type::ack_ecn)
984 : static_cast<uint64_t>(frame_type::ack));
985
986 // Largest Acknowledged
987 append_varint(buffer, f.largest_acknowledged);
988
989 // ACK Delay
990 append_varint(buffer, f.ack_delay);
991
992 // ACK Range Count
993 append_varint(buffer, f.ranges.size());
994
995 // First ACK Range (largest_ack - smallest_ack in first range)
996 // For simplicity, we'll use 0 if no ranges are specified
997 uint64_t first_range = 0;
998 if (!f.ranges.empty())
999 {
1000 first_range = f.ranges[0].length;
1001 }
1002 append_varint(buffer, first_range);
1003
1004 // Additional ranges (skip first)
1005 for (size_t i = 1; i < f.ranges.size(); ++i)
1006 {
1007 append_varint(buffer, f.ranges[i].gap);
1008 append_varint(buffer, f.ranges[i].length);
1009 }
1010
1011 // ECN Counts
1012 if (f.ecn)
1013 {
1014 append_varint(buffer, f.ecn->ect0);
1015 append_varint(buffer, f.ecn->ect1);
1016 append_varint(buffer, f.ecn->ecn_ce);
1017 }
1018
1019 return buffer;
