56void connection_id_manager::set_initial_peer_cid(const connection_id& cid)
57{
58 // Initial CID has sequence number 0 and no stateless reset token
59 connection_id_entry entry;
60 entry.cid = cid;
61 entry.sequence_number = 0;
62 entry.stateless_reset_token.fill(0);
63 entry.retired = false;
64
65 peer_cids_.clear();
66 peer_cids_.push_back(entry);
67 active_index_ = 0;
68}

70auto connection_id_manager::add_peer_cid(const connection_id& cid,
71 uint64_t sequence,
72 uint64_t retire_prior_to,
73 const std::array<uint8_t, 16>& reset_token)
74 -> VoidResult
75{
76 // RFC 9000 Section 19.15: retire_prior_to must be <= sequence
77 if (retire_prior_to > sequence)
78 {
79 return error_void(cid_manager_error::invalid_retire_prior_to,
80 "retire_prior_to exceeds sequence number",
81 "connection_id_manager");
82 }
83
84 // Check for duplicate sequence number
85 auto it = find_by_sequence(sequence);
86 if (it != peer_cids_.end())
87 {
88 // RFC 9000: If a NEW_CONNECTION_ID frame is received with a sequence
89 // number already seen, it must be ignored if identical, or a protocol
90 // error if different.
91 if (it->cid == cid && it->stateless_reset_token == reset_token)
92 {
93 // Identical frame, ignore
94 return ok();
95 }
96 return error_void(cid_manager_error::duplicate_sequence,
97 "Duplicate sequence number with different CID",
98 "connection_id_manager");
99 }
100
101 // Check active CID limit (count non-retired CIDs)
102 auto active_count = count_active_cids();
103 if (active_count >= active_cid_limit_)
104 {
105 return error_void(cid_manager_error::active_cid_limit_exceeded,
106 "Active connection ID limit exceeded",
107 "connection_id_manager");
108 }
109
110 // Add the new CID
111 connection_id_entry entry;
112 entry.cid = cid;
113 entry.sequence_number = sequence;
114 entry.stateless_reset_token = reset_token;
115 entry.retired = false;
116
117 peer_cids_.push_back(entry);
118
119 // Process retire_prior_to
120 if (retire_prior_to > largest_retire_prior_to_)
121 {
122 retire_cids_prior_to(retire_prior_to);
123 largest_retire_prior_to_ = retire_prior_to;
124 }
125
126 return ok();
127}

129auto connection_id_manager::get_active_peer_cid() const -> const connection_id&
130{
131 if (peer_cids_.empty() || active_index_ >= peer_cids_.size())

139auto connection_id_manager::rotate_peer_cid() -> VoidResult
140{
141 // Find an unused, non-retired CID
142 for (size_t i = 0; i < peer_cids_.size(); ++i)
143 {
144 if (i != active_index_ && !peer_cids_[i].retired)
145 {
146 // Mark the old CID for retirement
147 if (active_index_ < peer_cids_.size())
148 {
149 auto old_seq = peer_cids_[active_index_].sequence_number;
150 (void)retire_peer_cid(old_seq);
151 }
152
153 active_index_ = i;
154 return ok();
155 }
156 }
157
158 return error_void(cid_manager_error::no_available_cid,
159 "No available connection ID for rotation",
160 "connection_id_manager");
161}

163auto connection_id_manager::available_peer_cids() const -> size_t
164{
165 size_t count = 0;

190// Retirement Management

192
193void connection_id_manager::retire_cids_prior_to(uint64_t prior_to)
194{
195 for (auto& entry : peer_cids_)
196 {
197 if (!entry.retired && entry.sequence_number < prior_to)
198 {
199 entry.retired = true;
200
201 // Queue RETIRE_CONNECTION_ID frame
202 retire_connection_id_frame frame;
203 frame.sequence_number = entry.sequence_number;
204 pending_retire_frames_.push_back(frame);
205
206 // If this was the active CID, we need to switch
207 if (&entry == &peer_cids_[active_index_])
208 {
209 // Find a new active CID
210 for (size_t i = 0; i < peer_cids_.size(); ++i)
211 {
212 if (!peer_cids_[i].retired)
213 {
214 active_index_ = i;
215 break;
216 }
217 }
218 }
219 }
220 }
221}

234auto connection_id_manager::retire_peer_cid(uint64_t sequence) -> VoidResult
235{
236 auto it = find_by_sequence(sequence);
237 if (it == peer_cids_.end())
238 {
239 return error_void(cid_manager_error::cid_not_found,
240 "Connection ID sequence not found",
241 "connection_id_manager");
242 }
243
244 if (!it->retired)
245 {
246 it->retired = true;
247
248 // Queue RETIRE_CONNECTION_ID frame
249 retire_connection_id_frame frame;
250 frame.sequence_number = sequence;
251 pending_retire_frames_.push_back(frame);
252 }
253
254 return ok();
255}

261auto connection_id_manager::has_peer_cid(const connection_id& cid) const -> bool
262{
263 return std::any_of(peer_cids_.begin(), peer_cids_.end(),
264 [&cid](const connection_id_entry& entry)
265 { return entry.cid == cid && !entry.retired; });
266}

272auto connection_id_manager::find_by_sequence(uint64_t sequence)
273 -> std::vector<connection_id_entry>::iterator
274{
