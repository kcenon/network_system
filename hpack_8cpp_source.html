120 auto static_table::find(std::string_view name, std::string_view value)
121 -> size_t
122 {
123 for (size_t i = 1; i <= static_table_size; ++i)
124 {
125 const auto& entry = static_table_entries[i];
126 if (entry.name == name)
127 {
128 if (value.empty() || entry.value == value)
129 {
130 return i;
131 }
132 }
133 }
134 return 0;
135 }

224 auto hpack_encoder::encode(const std::vector<http_header>& headers)
225 -> std::vector<uint8_t>
226 {
227 std::vector<uint8_t> result;
228
229 for (const auto& header : headers)
230 {
231 // Try to find in static table first
232 size_t static_index = static_table::find(header.name, header.value);
233 if (static_index > 0)
234 {
235 // Indexed header field representation
236 auto encoded = encode_indexed(static_index);
237 result.insert(result.end(), encoded.begin(), encoded.end());
238 continue;
239 }
240
241 // Try to find in dynamic table
242 auto dynamic_index = table_.find(header.name, header.value);
243 if (dynamic_index.has_value())
244 {
245 // Indexed header field representation
246 size_t index = static_table::size() + 1 + dynamic_index.value();
247 auto encoded = encode_indexed(index);
248 result.insert(result.end(), encoded.begin(), encoded.end());
249 continue;
250 }
251
252 // Check if name is in static table
253 size_t name_index = static_table::find(header.name, "");
254 if (name_index > 0)
255 {
256 // Literal with incremental indexing - indexed name
257 auto encoded = encode_literal_with_indexing(name_index, header.value);
258 result.insert(result.end(), encoded.begin(), encoded.end());
259 table_.insert(header.name, header.value);
260 }
261 else
262 {
263 // Check if name is in dynamic table
264 auto dynamic_name_index = table_.find(header.name, "");
265 if (dynamic_name_index.has_value())
266 {
267 size_t index = static_table::size() + 1 + dynamic_name_index.value();
268 auto encoded = encode_literal_with_indexing(index, header.value);
269 result.insert(result.end(), encoded.begin(), encoded.end());
270 table_.insert(header.name, header.value);
271 }
272 else
273 {
274 // Literal with incremental indexing - new name
275 auto encoded = encode_literal_with_indexing(header.name, header.value);
276 result.insert(result.end(), encoded.begin(), encoded.end());
277 table_.insert(header.name, header.value);
278 }
279 }
280 }
281
282 return result;
283 }

295 auto hpack_encoder::encode_integer(uint64_t value, uint8_t prefix_bits)
296 -> std::vector<uint8_t>
297 {
298 std::vector<uint8_t> result;
299 uint8_t max_prefix = (1 << prefix_bits) - 1;
300
301 if (value < max_prefix)
302 {
303 result.push_back(static_cast<uint8_t>(value));
304 }
305 else
306 {
307 result.push_back(max_prefix);
308 value -= max_prefix;
309
310 while (value >= 128)
311 {
312 result.push_back(static_cast<uint8_t>((value % 128) + 128));
313 value /= 128;
314 }
315 result.push_back(static_cast<uint8_t>(value));
316 }
317
318 return result;
319 }

321 auto hpack_encoder::encode_string(std::string_view str, bool huffman)
322 -> std::vector<uint8_t>
323 {
324 std::vector<uint8_t> result;
325
326 // String length (with Huffman bit)
327 auto length_bytes = encode_integer(str.size(), 7);
328 if (!huffman)
329 {
330 result.insert(result.end(), length_bytes.begin(), length_bytes.end());
331 }
332 else
333 {
334 // Huffman encoding not implemented yet
335 // Set H bit in first byte
336 length_bytes[0] |= 0x80;
337 result.insert(result.end(), length_bytes.begin(), length_bytes.end());
338 }
339
340 // String data
341 result.insert(result.end(), str.begin(), str.end());
342
343 return result;
344 }

435 auto hpack_decoder::decode(std::span<const uint8_t> data)
436 -> Result<std::vector<http_header>>
437 {
438 std::vector<http_header> headers;
439 auto remaining = data;
440
441 while (!remaining.empty())
442 {
443 uint8_t first_byte = remaining[0];
444
445 if (first_byte & 0x80)
446 {
447 // Indexed header field representation
448 auto index_result = decode_integer(remaining, 7);
449 if (index_result.is_err())
450 {
451 return index_result.error();
452 }
453 size_t index = index_result.value();
454
455 auto header_result = get_indexed_header(index);
456 if (header_result.is_err())
457 {
458 return header_result.error();
459 }
460
461 headers.push_back(header_result.value());
462 }
463 else if (first_byte & 0x40)
464 {
465 // Literal with incremental indexing
466 auto name_index_result = decode_integer(remaining, 6);
467 if (name_index_result.is_err())
468 {
469 return name_index_result.error();
470 }
471 size_t name_index = name_index_result.value();
472
473 std::string name;
474 if (name_index == 0)
475 {
476 // New name
477 auto name_result = decode_string(remaining);
478 if (name_result.is_err())
479 {
480 return name_result.error();
481 }
482 name = name_result.value();
483 }
484 else
485 {
486 // Indexed name
487 auto header_result = get_indexed_header(name_index);
488 if (header_result.is_err())
489 {
490 return header_result.error();
491 }
492 name = header_result.value().name;
493 }
494
495 // Decode value
496 auto value_result = decode_string(remaining);
497 if (value_result.is_err())
498 {
499 return value_result.error();
500 }
501 std::string value = value_result.value();
502
503 headers.emplace_back(name, value);
504 table_.insert(name, value);
505 }
506 else
507 {
508 // Literal without indexing or never indexed
509 uint8_t prefix_bits = (first_byte & 0x10) ? 4 : 4;
510
511 auto name_index_result = decode_integer(remaining, prefix_bits);
512 if (name_index_result.is_err())
513 {
514 return name_index_result.error();
515 }
516 size_t name_index = name_index_result.value();
517
518 std::string name;
519 if (name_index == 0)
520 {
521 // New name
522 auto name_result = decode_string(remaining);
523 if (name_result.is_err())
524 {
525 return name_result.error();
526 }
527 name = name_result.value();
528 }
529 else
530 {
531 // Indexed name
532 auto header_result = get_indexed_header(name_index);
533 if (header_result.is_err())
534 {
535 return header_result.error();
536 }
537 name = header_result.value().name;
538 }
539
540 // Decode value
541 auto value_result = decode_string(remaining);
542 if (value_result.is_err())
543 {
544 return value_result.error();
545 }
546
547 headers.emplace_back(name, value_result.value());
548 }
549 }
550
551 return headers;
552 }

564 auto hpack_decoder::decode_integer(std::span<const uint8_t>& data,
565 uint8_t prefix_bits)
566 -> Result<uint64_t>
567 {
568 if (data.empty())
569 {
570 return error_info(100, "Insufficient data for integer", "hpack");
571 }
572
573 uint8_t prefix_mask = (1 << prefix_bits) - 1;
574 uint64_t value = data[0] & prefix_mask;
575 data = data.subspan(1);
576
577 if (value < prefix_mask)
578 {
579 return value;
580 }
581
582 // Multi-byte integer
583 uint64_t m = 0;
584 do
585 {
586 if (data.empty())
587 {
588 return error_info(101, "Incomplete integer encoding", "hpack");
589 }
590
591 uint8_t byte = data[0];
592 data = data.subspan(1);
593
594 value += (byte & 0x7F) << m;
595 m += 7;
596
597 if (m >= 64)
598 {
599 return error_info(102, "Integer overflow", "hpack");
600 }
601
602 if ((byte & 0x80) == 0)
603 {
604 break;
605 }
606 } while (true);
607
608 return value;
609 }

611 auto hpack_decoder::decode_string(std::span<const uint8_t>& data)
612 -> Result<std::string>
613 {
614 if (data.empty())
615 {
616 return error_info(103, "Insufficient data for string", "hpack");
617 }
618
619 bool huffman = (data[0] & 0x80) != 0;
620
621 auto length_result = decode_integer(data, 7);
622 if (length_result.is_err())
623 {
624 return length_result.error();
625 }
626
627 size_t length = length_result.value();
628
629 if (data.size() < length)
630 {
631 return error_info(104, "Insufficient data for string value", "hpack");
632 }
633
634 std::string result;
635 if (huffman)
636 {
637 // Huffman decoding not implemented yet - just return raw for now
638 result.assign(data.begin(), data.begin() + length);
639 }
640 else
641 {
642 result.assign(data.begin(), data.begin() + length);
643 }
644
645 data = data.subspan(length);
646 return result;
647 }

649 auto hpack_decoder::get_indexed_header(size_t index) const
650 -> Result<http_header>
651 {
652 if (index == 0)
653 {
654 return error_info(105, "Invalid index 0", "hpack");
655 }
656
657 // Check static table
658 if (index <= static_table::size())
659 {
660 auto header = static_table::get(index);
661 if (header.has_value())
662 {
663 return std::move(header.value());
664 }
665 return error_info(106, "Invalid static table index", "hpack");
666 }
667
668 // Check dynamic table
669 size_t dynamic_index = index - static_table::size() - 1;
670 auto header = table_.get(dynamic_index);
671 if (header.has_value())
672 {
673 return std::move(header.value());
674 }
675
676 return error_info(107, "Invalid dynamic table index", "hpack");
677 }
