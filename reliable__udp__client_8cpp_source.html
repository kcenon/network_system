50 // Packet header structure (12 bytes)
51 struct packet_header
52 {
53 uint32_t sequence_number; // Packet sequence number
54 uint32_t ack_number; // Acknowledgment number
55 uint16_t flags; // Control flags
56 uint16_t data_length; // Payload length
57
58 static constexpr uint16_t FLAG_ACK = 0x01; // Acknowledgment packet
59 static constexpr uint16_t FLAG_DATA = 0x02; // Data packet
60 static constexpr uint16_t FLAG_SYN = 0x04; // Connection start
61 static constexpr uint16_t FLAG_FIN = 0x08; // Connection end
62
63 auto serialize() const -> std::vector<uint8_t>
64 {
65 std::vector<uint8_t> buffer(sizeof(packet_header));
66 std::memcpy(buffer.data(), this, sizeof(packet_header));
67 return buffer;
68 }
69
70 static auto deserialize(const std::vector<uint8_t>& buffer) -> packet_header
71 {
72 packet_header header{};
73 if (buffer.size() >= sizeof(packet_header))
74 {
75 std::memcpy(&header, buffer.data(), sizeof(packet_header));
76 }
77 return header;
78 }
79 };

108 auto start_client(std::string_view host, uint16_t port) -> VoidResult
109 {
110 std::lock_guard<std::mutex> lock(state_mutex_);
111
112 if (is_running_.load())
113 {
114 return error_void(error_codes::common_errors::already_exists,
115 "Client is already running");
116 }
117
118 // Create underlying UDP client
119 udp_client_ = std::make_shared<messaging_udp_client>(client_id_);
120
121 // Set receive callback
122 udp_client_->set_receive_callback(
123 [this, self = shared_from_this()](const std::vector<uint8_t>& data,
124 const asio::ip::udp::endpoint&) {
125 handle_received_packet(data);
126 });
127
128 // Set error callback
129 udp_client_->set_error_callback([this, self = shared_from_this()](
130 std::error_code ec) { handle_error(ec); });
131
132 // Start UDP client
133 auto result = udp_client_->start_client(host, port);
134 if (result.is_err())
135 {
136 return result;
137 }
138
139 is_running_.store(true);
140
141 // Start retransmission timer for reliable modes
142 if (mode_ != reliability_mode::unreliable)
143 {
144 start_retransmission_timer();
145 }
146
147 NETWORK_LOG_INFO("[reliable_udp_client::" + client_id_ + "] Started successfully");
148 return ok();
149 }

264 // Unreliable send (pure UDP)
265 auto send_unreliable(std::vector<uint8_t>&& data) -> VoidResult
266 {
267 packet_header header{};
268 header.sequence_number = 0;
269 header.ack_number = 0;
270 header.flags = packet_header::FLAG_DATA;
271 header.data_length = static_cast<uint16_t>(data.size());
272
273 auto packet = create_packet(header, data);
274
275 return udp_client_->send(
276 std::move(packet), [this](std::error_code ec, std::size_t) {
277 if (!ec)
278 {
279 std::lock_guard<std::mutex> lock(stats_mutex_);
280 stats_.packets_sent++;
281 }
282 });
283 }

285 // Reliable send (with ACK and retransmission)
286 auto send_reliable(std::vector<uint8_t>&& data) -> VoidResult
287 {
288 std::lock_guard<std::mutex> lock(pending_mutex_);
289
290 // Check congestion window
291 if (pending_packets_.size() >= congestion_window_)
292 {
293 return error_void(error_codes::common_errors::internal_error,
294 "Congestion window full");
295 }
296
297 uint32_t seq = next_sequence_++;
298
299 packet_header header{};
300 header.sequence_number = seq;
301 header.ack_number = 0;
302 header.flags = packet_header::FLAG_DATA;
303 header.data_length = static_cast<uint16_t>(data.size());
304
305 auto packet = create_packet(header, data);
306
307 // Store for retransmission
308 packet_info info{};
309 info.data = packet;
310 info.send_time = std::chrono::steady_clock::now();
311 info.retransmit_count = 0;
312 info.sequence_number = seq;
313 pending_packets_[seq] = std::move(info);
314
315 // Send packet
316 return udp_client_->send(
317 std::move(packet), [this, seq](std::error_code ec, std::size_t) {
318 if (!ec)
319 {
320 std::lock_guard<std::mutex> lock(stats_mutex_);
321 stats_.packets_sent++;
322 }
323 else
324 {
325 // Remove from pending if send failed
326 std::lock_guard<std::mutex> pending_lock(pending_mutex_);
327 pending_packets_.erase(seq);
328 }
329 });
330 }
331
332 // Sequenced send (drop old packets)
333 auto send_sequenced(std::vector<uint8_t>&& data) -> VoidResult
334 {
335 uint32_t seq = next_sequence_++;
336
337 packet_header header{};
338 header.sequence_number = seq;
339 header.ack_number = 0;
340 header.flags = packet_header::FLAG_DATA;
341 header.data_length = static_cast<uint16_t>(data.size());
342
343 auto packet = create_packet(header, data);
344
345 return udp_client_->send(
346 std::move(packet), [this](std::error_code ec, std::size_t) {
347 if (!ec)
348 {
349 std::lock_guard<std::mutex> lock(stats_mutex_);
350 stats_.packets_sent++;
351 }
352 });
353 }

355 // Create packet with header + payload
356 auto create_packet(const packet_header& header, const std::vector<uint8_t>& payload)
357 -> std::vector<uint8_t>
358 {
359 std::vector<uint8_t> packet(sizeof(packet_header) + payload.size());
360 std::memcpy(packet.data(), &header, sizeof(packet_header));
361 if (!payload.empty())
362 {
363 std::memcpy(packet.data() + sizeof(packet_header), payload.data(),
364 payload.size());
365 }
366 return packet;
367 }
368
369 // Handle received packet
370 auto handle_received_packet(const std::vector<uint8_t>& data) -> void
371 {
372 if (data.size() < sizeof(packet_header))
373 {
374 NETWORK_LOG_WARN("[reliable_udp_client::" + client_id_ +
375 "] Received invalid packet (too small)");
376 return;
377 }
378
379 packet_header header{};
380 std::memcpy(&header, data.data(), sizeof(packet_header));
381
382 // Handle ACK packet
383 if (header.flags & packet_header::FLAG_ACK)
384 {
385 handle_ack(header.ack_number);
386 return;
387 }
388
389 // Handle data packet
390 if (header.flags & packet_header::FLAG_DATA)
391 {
392 // Send ACK for reliable modes
393 if (mode_ == reliability_mode::reliable_ordered ||
394 mode_ == reliability_mode::reliable_unordered)
395 {
396 send_ack(header.sequence_number);
397 }
398
399 // Extract payload
400 std::vector<uint8_t> payload(data.begin() + sizeof(packet_header), data.end());
401
402 // Process based on mode
403 switch (mode_)
404 {
405 case reliability_mode::unreliable:
406 deliver_to_application(std::move(payload));
407 break;
408
409 case reliability_mode::reliable_ordered:
410 handle_ordered_delivery(header.sequence_number, std::move(payload));
411 break;
412
413 case reliability_mode::reliable_unordered:
414 deliver_to_application(std::move(payload));
415 break;
416
417 case reliability_mode::sequenced:
418 handle_sequenced_delivery(header.sequence_number, std::move(payload));
419 break;
420 }
421
422 {
423 std::lock_guard<std::mutex> lock(stats_mutex_);
424 stats_.packets_received++;
425 }
426 }
427 }
428
429 // Handle ACK
430 auto handle_ack(uint32_t ack_number) -> void
431 {
432 std::lock_guard<std::mutex> lock(pending_mutex_);
433
434 auto it = pending_packets_.find(ack_number);
435 if (it != pending_packets_.end())
436 {
437 // Calculate RTT
438 auto now = std::chrono::steady_clock::now();
439 auto rtt = std::chrono::duration_cast<std::chrono::milliseconds>(
440 now - it->second.send_time)
441 .count();
442
443 // Update stats
444 {
445 std::lock_guard<std::mutex> stats_lock(stats_mutex_);
446 stats_.acks_received++;
447
448 // Update average RTT (exponential moving average)
449 if (stats_.average_rtt_ms == 0.0)
450 {
451 stats_.average_rtt_ms = static_cast<double>(rtt);
452 }
453 else
454 {
455 stats_.average_rtt_ms = 0.875 * stats_.average_rtt_ms + 0.125 * rtt;
456 }
457 }
458
459 // Remove from pending
460 pending_packets_.erase(it);
461
462 NETWORK_LOG_TRACE("[reliable_udp_client::" + client_id_ + "] Received ACK for seq=" +
463 std::to_string(ack_number) + ", RTT=" + std::to_string(rtt) + "ms");
464 }
465 }
466
467 // Send ACK
468 auto send_ack(uint32_t sequence_number) -> void
469 {
470 packet_header header{};
471 header.sequence_number = 0;
472 header.ack_number = sequence_number;
473 header.flags = packet_header::FLAG_ACK;
474 header.data_length = 0;
475
476 auto packet = create_packet(header, {});
477
478 udp_client_->send(std::move(packet), [this](std::error_code ec, std::size_t) {
479 if (!ec)
480 {
481 std::lock_guard<std::mutex> lock(stats_mutex_);
482 stats_.acks_sent++;
483 }
484 });
485 }

487 // Handle ordered delivery
488 auto handle_ordered_delivery(uint32_t sequence_number, std::vector<uint8_t>&& payload)
489 -> void
490 {
491 std::lock_guard<std::mutex> lock(receive_mutex_);
492
493 if (sequence_number == expected_sequence_)
494 {
495 // Deliver in-order packet
496 deliver_to_application(std::move(payload));
497 expected_sequence_++;
498
499 // Deliver any buffered packets that are now in order
500 while (true)
501 {
502 auto it = receive_buffer_.find(expected_sequence_);
503 if (it == receive_buffer_.end())
504 {
505 break;
506 }
507
508 deliver_to_application(std::move(it->second));
509 receive_buffer_.erase(it);
510 expected_sequence_++;
511 }
512 }
513 else if (sequence_number > expected_sequence_)
514 {
515 // Buffer out-of-order packet
516 receive_buffer_[sequence_number] = std::move(payload);
517
518 NETWORK_LOG_TRACE("[reliable_udp_client::" + client_id_ +
519 "] Buffered out-of-order packet seq=" +
520 std::to_string(sequence_number) +
521 " (expected=" + std::to_string(expected_sequence_) + ")");
522 }
523 else
524 {
525 // Duplicate or old packet, ignore
526 NETWORK_LOG_TRACE("[reliable_udp_client::" + client_id_ +
527 "] Dropped old packet seq=" + std::to_string(sequence_number));
528 }
529 }
530
531 // Handle sequenced delivery
532 auto handle_sequenced_delivery(uint32_t sequence_number, std::vector<uint8_t>&& payload)
533 -> void
534 {
535 if (sequence_number >= expected_sequence_)
536 {
537 expected_sequence_ = sequence_number + 1;
538 deliver_to_application(std::move(payload));
539 }
540 else
541 {
542 // Drop old packet
543 std::lock_guard<std::mutex> lock(stats_mutex_);
544 stats_.packets_dropped++;
545
546 NETWORK_LOG_TRACE("[reliable_udp_client::" + client_id_ +
547 "] Dropped old packet in sequenced mode seq=" +
548 std::to_string(sequence_number));
549 }
550 }
551
552 // Deliver packet to application
553 auto deliver_to_application(std::vector<uint8_t>&& payload) -> void
554 {
555 std::lock_guard<std::mutex> lock(callback_mutex_);
556 if (receive_callback_)
557 {
558 receive_callback_(payload);
559 }
560 }

562 // Handle error
563 auto handle_error(std::error_code ec) -> void
564 {
565 NETWORK_LOG_ERROR("[reliable_udp_client::" + client_id_ + "] Error: " + ec.message());
566
567 std::lock_guard<std::mutex> lock(callback_mutex_);
568 if (error_callback_)
569 {
570 error_callback_(ec);
571 }
572 }

574 // Start retransmission timer
575 auto start_retransmission_timer() -> void
576 {
577 thread_pool_ = network_context::instance().get_thread_pool();
578 if (!thread_pool_) {
579 NETWORK_LOG_ERROR("[reliable_udp_client::" + client_id_ +
580 "] Failed to get thread pool for retransmission timer");
581 return;
582 }
583
584 stop_retransmission_.store(false);
585 retransmission_future_ = thread_pool_->submit([this]() {
586 while (is_running_.load() && !stop_retransmission_.load())
587 {
588 std::this_thread::sleep_for(
589 std::chrono::milliseconds(retransmission_timeout_ms_));
590 if (!stop_retransmission_.load()) {
591 check_and_retransmit();
592 }
593 }
594 });
595 }
596
597 // Stop retransmission timer
598 auto stop_retransmission_timer() -> void
599 {
600 stop_retransmission_.store(true);
601 if (retransmission_future_.valid())
602 {
603 try {
604 retransmission_future_.wait();
605 } catch (...) {
606 NETWORK_LOG_ERROR("[reliable_udp_client::" + client_id_ +
607 "] Exception while waiting for retransmission timer to stop");
608 }
609 }
610 thread_pool_.reset();
611 }
612
613 // Check for packets needing retransmission
614 auto check_and_retransmit() -> void
615 {
616 std::lock_guard<std::mutex> lock(pending_mutex_);
617
618 auto now = std::chrono::steady_clock::now();
619
620 for (auto it = pending_packets_.begin(); it != pending_packets_.end();)
621 {
622 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
623 now - it->second.send_time)
624 .count();
625
626 if (elapsed >= retransmission_timeout_ms_)
627 {
628 // Check max retries
629 if (it->second.retransmit_count >= max_retries_)
630 {
631 NETWORK_LOG_WARN("[reliable_udp_client::" + client_id_ +
632 "] Packet seq=" + std::to_string(it->first) +
633 " exceeded max retries, dropping");
634
635 {
636 std::lock_guard<std::mutex> stats_lock(stats_mutex_);
637 stats_.packets_dropped++;
638 }
639
640 it = pending_packets_.erase(it);
641 continue;
642 }
643
644 // Retransmit
645 auto packet_copy = it->second.data;
646 it->second.send_time = now;
647 it->second.retransmit_count++;
648
649 {
650 std::lock_guard<std::mutex> stats_lock(stats_mutex_);
651 stats_.packets_retransmitted++;
652 }
653
654 NETWORK_LOG_TRACE("[reliable_udp_client::" + client_id_ +
655 "] Retransmitting seq=" + std::to_string(it->first) +
656 " (attempt " + std::to_string(it->second.retransmit_count) +
657 ")");
658
659 udp_client_->send(std::move(packet_copy),
660 [](std::error_code, std::size_t) {});
661 }
662
663 ++it;
664 }
665 }
