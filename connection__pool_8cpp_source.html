49 connection_pool::~connection_pool() noexcept
50 {
51 try
52 {
53 // Set shutdown flag
54 is_shutdown_.store(true);
55
56 // Notify all waiting threads
57 cv_.notify_all();
58
59 // Stop all connections
60 std::lock_guard<std::mutex> lock(mutex_);
61 while (!available_.empty())
62 {
63 auto client = std::move(available_.front());
64 available_.pop();
65 if (client)
66 {
67 (void)client->stop_client();
68 }
69 }
70 }
71 catch (...)
72 {
73 // Destructor must not throw
74 }
75 }

77 auto connection_pool::initialize() -> VoidResult
78 {
79 NETWORK_LOG_INFO("[connection_pool] Initializing pool with " +
80 std::to_string(pool_size_) + " connections to " + host_ + ":" +
81 std::to_string(port_));
82
83 // Zero-size pool requires no connections — succeed immediately
84 if (pool_size_ == 0)
85 {
86 NETWORK_LOG_INFO("[connection_pool] Pool size is 0, no connections to create");
87 return ok();
88 }
89
90 // Launch all connections in parallel using std::async
91 struct connect_result
92 {
93 std::unique_ptr<messaging_client> client;
94 bool success = false;
95 std::string error_message;
96 };
97
98 std::vector<std::future<connect_result>> futures;
99 futures.reserve(pool_size_);
100
101 for (size_t i = 0; i < pool_size_; ++i)
102 {
103 futures.push_back(std::async(std::launch::async,
104 [this, i]() -> connect_result {
105 connect_result cr;
106 cr.client = std::make_unique<messaging_client>(
107 "pool_client_" + std::to_string(i));
108
109 auto result = cr.client->start_client(host_, port_);
110 if (result.is_err())
111 {
112 cr.success = false;
113 cr.error_message = result.error().message;
114 cr.client.reset();
115 }
116 else
117 {
118 cr.success = true;
119 }
120 return cr;
121 }));
122 }
123
124 // Collect results
125 size_t success_count = 0;
126 size_t fail_count = 0;
127 {
128 std::lock_guard<std::mutex> lock(mutex_);
129 for (size_t i = 0; i < futures.size(); ++i)
130 {
131 auto cr = futures[i].get();
132 if (cr.success && cr.client)
133 {
134 available_.push(std::move(cr.client));
135 ++success_count;
136 }
137 else
138 {
139 ++fail_count;
140 NETWORK_LOG_WARN(
141 "[connection_pool] Connection " + std::to_string(i) +
142 " failed: " + cr.error_message);
143 }
144 }
145 }
146
147 if (success_count == 0)
148 {
149 return error_void(
150 -1,
151 "All " + std::to_string(pool_size_) +
152 " pool connections failed",
153 "connection_pool::initialize",
154 "Host: " + host_ + ", Port: " + std::to_string(port_)
155 );
156 }
157
158 if (fail_count > 0)
159 {
160 NETWORK_LOG_WARN("[connection_pool] Initialized with " +
161 std::to_string(success_count) + "/" +
162 std::to_string(pool_size_) +
163 " connections (" + std::to_string(fail_count) + " failed)");
164 }
165 else
166 {
167 NETWORK_LOG_INFO("[connection_pool] Successfully initialized " +
168 std::to_string(success_count) + " connections");
169 }
170
171 return ok();
172 }

174 auto connection_pool::acquire() -> std::unique_ptr<messaging_client>
175 {
176 std::unique_lock<std::mutex> lock(mutex_);
177
178 // Wait until a connection is available, shutdown, or timeout
179 if (acquire_timeout_.count() > 0)
180 {
181 if (!cv_.wait_for(lock, acquire_timeout_, [this] {
182 return !available_.empty() || is_shutdown_.load();
183 }))
184 {
185 NETWORK_LOG_WARN(
186 "[connection_pool] Acquire timed out after " +
187 std::to_string(acquire_timeout_.count()) +
188 "s. Active: " +
189 std::to_string(active_count_.load()) + "/" +
190 std::to_string(pool_size_));
191 return nullptr;
192 }
193 }
194 else
195 {
196 cv_.wait(lock, [this] {
197 return !available_.empty() || is_shutdown_.load();
198 });
199 }
200
201 // Check if shutdown
202 if (is_shutdown_.load())
203 {
204 return nullptr;
205 }
206
207 // Get connection from queue
208 auto client = std::move(available_.front());
209 available_.pop();
210 active_count_.fetch_add(1, std::memory_order_relaxed);
211
212 NETWORK_LOG_DEBUG("[connection_pool] Acquired connection. Active: " +
213 std::to_string(active_count_.load()) + "/" +
214 std::to_string(pool_size_));
215
216 return client;
217 }

219 auto connection_pool::release(std::unique_ptr<messaging_client> client)
220 -> void
221 {
222 if (!client)
223 {
224 return;
225 }
226
227 // Check if client is still connected, reconnect if necessary
228 if (!client->is_connected())
229 {
230 NETWORK_LOG_WARN("[connection_pool] Connection lost, reconnecting...");
231
232 auto result = client->start_client(host_, port_);
233 if (result.is_err())
234 {
235 NETWORK_LOG_ERROR("[connection_pool] Failed to reconnect: " +
236 result.error().message);
237 // Drop this connection and decrement active count
238 active_count_.fetch_sub(1, std::memory_order_relaxed);
239 return;
240 }
241 }
242
243 // Return to pool
244 {
245 std::lock_guard<std::mutex> lock(mutex_);
246 available_.push(std::move(client));
247 }
248
249 active_count_.fetch_sub(1, std::memory_order_relaxed);
250
251 // Notify one waiting thread
252 cv_.notify_one();
253
254 NETWORK_LOG_DEBUG("[connection_pool] Released connection. Active: " +
255 std::to_string(active_count_.load()) + "/" +
256 std::to_string(pool_size_));
257 }
