25auto udp_listener_adapter::start(const endpoint_info& bind_address) -> VoidResult
26{
27 if (!server_) {
28 return error_void(
29 static_cast<int>(std::errc::invalid_argument),
30 "Server is not initialized"
31 );
32 }
33
34 if (server_->is_running()) {
35 return error_void(
36 static_cast<int>(std::errc::already_connected),
37 "Already listening"
38 );
39 }
40
41 {
42 std::lock_guard lock(endpoint_mutex_);
43 local_endpoint_ = bind_address;
44 }
45
46 // Clear known endpoints when starting fresh
47 {
48 std::lock_guard lock(connections_mutex_);
49 known_endpoints_.clear();
50 }
51
52 return server_->start_server(bind_address.port);
53}

60auto udp_listener_adapter::stop() noexcept -> void
61{
62 if (server_ && server_->is_running()) {
63 // Notify disconnection for all known endpoints
64 {
65 std::lock_guard cb_lock(callbacks_mutex_);
66 std::lock_guard conn_lock(connections_mutex_);
67
68 if (callbacks_.on_disconnect) {
69 for (const auto& [conn_id, _] : known_endpoints_) {
70 callbacks_.on_disconnect(conn_id);
71 }
72 }
73 known_endpoints_.clear();
74 }
75
76 (void)server_->stop_server();
77 }
78}

112auto udp_listener_adapter::send_to(std::string_view connection_id,
113 std::span<const std::byte> data) -> VoidResult
114{
115 if (!server_ || !server_->is_running()) {
116 return error_void(
117 static_cast<int>(std::errc::not_connected),
118 "Server is not running"
119 );
120 }
121
122 auto ep_opt = parse_connection_id(connection_id);
123 if (!ep_opt) {
124 return error_void(
125 static_cast<int>(std::errc::invalid_argument),
126 "Invalid connection ID format"
127 );
128 }
129
130 interfaces::i_udp_server::endpoint_info target_ep;
131 target_ep.address = ep_opt->host;
132 target_ep.port = ep_opt->port;
133
134 std::vector<uint8_t> buffer(data.size());
135 std::memcpy(buffer.data(), data.data(), data.size());
136
137 return server_->send_to(target_ep, std::move(buffer));
138}

140auto udp_listener_adapter::broadcast(std::span<const std::byte> data) -> VoidResult
141{
142 if (!server_ || !server_->is_running()) {
143 return error_void(
144 static_cast<int>(std::errc::not_connected),
145 "Server is not running"
146 );
147 }
148
149 std::vector<std::pair<std::string, endpoint_info>> endpoints_copy;
150 {
151 std::lock_guard lock(connections_mutex_);
152 endpoints_copy.reserve(known_endpoints_.size());
153 for (const auto& [id, ep] : known_endpoints_) {
154 endpoints_copy.emplace_back(id, ep);
155 }
156 }
157
158 if (endpoints_copy.empty()) {
159 return ok(); // No endpoints to broadcast to
160 }
161
162 VoidResult last_result = ok();
163 for (const auto& [conn_id, ep] : endpoints_copy) {
164 auto result = send_to(conn_id, data);
165 if (!result.is_ok()) {
166 last_result = result; // Return last error if any failed
167 }
168 }
169
170 return last_result;
171}

174{

195auto udp_listener_adapter::setup_internal_callbacks() -> void
196{
197 if (!server_) {
198 return;
199 }
200
201 // Bridge receive callback - also handles "connection" tracking for UDP
202 server_->set_receive_callback(
203 [this](const std::vector<uint8_t>& data,
204 const interfaces::i_udp_server::endpoint_info& sender) {
205 std::string conn_id = make_connection_id(sender.address, sender.port);
206 bool is_new_endpoint = false;
207
208 {
209 std::lock_guard lock(connections_mutex_);
210 if (known_endpoints_.find(conn_id) == known_endpoints_.end()) {
211 known_endpoints_[conn_id] = endpoint_info{sender.address, sender.port};
212 is_new_endpoint = true;
213 }
214 }
215
216 // If this is a new endpoint, trigger on_accept
217 if (is_new_endpoint) {
218 std::lock_guard lock(callbacks_mutex_);
219 if (callbacks_.on_accept) {
220 callbacks_.on_accept(conn_id);
221 }
222 // Note: accept_callback_ with unique_ptr<i_connection> doesn't make
223 // sense for UDP since there's no real connection object to hand over
224 }
225
226 // Forward data to callback
227 {
228 std::lock_guard lock(callbacks_mutex_);
229 if (callbacks_.on_data) {
230 std::span<const std::byte> byte_span(
231 reinterpret_cast<const std::byte*>(data.data()),
232 data.size()
233 );
234 callbacks_.on_data(conn_id, byte_span);
235 }
236 }
237 });
238
239 // Bridge error callback
240 server_->set_error_callback([this](std::error_code ec) {
241 std::lock_guard lock(callbacks_mutex_);
242 if (callbacks_.on_error) {
243 // For server-level errors, use empty connection ID
244 callbacks_.on_error("", ec);
245 }
246 });
247}

256auto udp_listener_adapter::parse_connection_id(std::string_view connection_id) -> std::optional<endpoint_info>
257{
258 std::string conn_str(connection_id);
259 auto colon_pos = conn_str.rfind(':');
260
261 if (colon_pos == std::string::npos) {
262 return std::nullopt;
263 }
264
265 std::string host = conn_str.substr(0, colon_pos);
266 std::string port_str = conn_str.substr(colon_pos + 1);
267
268 uint16_t port = 0;
269 try {
270 port = static_cast<uint16_t>(std::stoi(port_str));
271 } catch (...) {
272 return std::nullopt;
273 }
274
275 return endpoint_info{host, port};
276}
