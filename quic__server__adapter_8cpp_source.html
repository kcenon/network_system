55quic_session_wrapper::quic_session_wrapper(
56 std::string_view session_id,
57 std::shared_ptr<session::quic_session> session)

59 , session_(std::move(session))
60{
61}
62
63auto quic_session_wrapper::id() const -> std::string_view
64{
65 return session_id_;
66}
67
68auto quic_session_wrapper::is_connected() const -> bool
69{
70 return is_connected_.load(std::memory_order_acquire) &&
71 session_ && session_->is_active();
72}
73
74auto quic_session_wrapper::send(std::vector<uint8_t>&& data) -> VoidResult
75{
76 if (!is_connected_.load(std::memory_order_acquire))
77 {
78 return error_void(
79 error_codes::common_errors::invalid_argument,
80 "Session is not connected",
81 "quic_session_wrapper::send"
82 );
83 }
84
85 if (!session_)
86 {
87 return error_void(
88 error_codes::common_errors::internal_error,
89 "Session not initialized",
90 "quic_session_wrapper::send"
91 );
92 }
93
94 // Send using the interface send method
95 return session_->send(std::move(data));
96}
97
98auto quic_session_wrapper::close() -> void
99{
100 is_connected_.store(false, std::memory_order_release);
101 if (session_)
102 {
103 (void)session_->close(0); // 0 = no error
104 }
105}

186auto quic_server_adapter::start(uint16_t port) -> VoidResult
187{
188 if (!server_)
189 {
190 return error_void(
191 error_codes::common_errors::internal_error,
192 "Server not initialized",
193 "quic_server_adapter::start"
194 );
195 }
196
197 // Build configuration
198 core::quic_server_config config;
199 config.cert_file = cert_path_;
200 config.key_file = key_path_;
201 config.ca_cert_file = ca_cert_path_;
202 config.require_client_cert = require_client_cert_;
203 config.alpn_protocols = alpn_protocols_;
204 config.max_idle_timeout_ms = max_idle_timeout_ms_;
205 config.max_connections = max_connections_;
206
207 auto result = server_->start_server(port, config);
208 if (result.is_ok())
209 {
210 is_running_.store(true, std::memory_order_release);
211 }
212 return result;
213}

215auto quic_server_adapter::stop() -> VoidResult
216{
217 if (!server_)
218 {
219 return error_void(
220 error_codes::common_errors::internal_error,
221 "Server not initialized",
222 "quic_server_adapter::stop"
223 );
224 }
225
226 auto result = server_->stop_server();
227 is_running_.store(false, std::memory_order_release);
228
229 // Clear tracked sessions on stop
230 {
231 std::lock_guard<std::mutex> lock(sessions_mutex_);
232 sessions_.clear();
233 }
234
235 return result;
236}

272auto quic_server_adapter::setup_internal_callbacks() -> void
273{
274 if (!server_)
275 {
276 return;
277 }
278
279 // Bridge connection callback using the legacy callback type (takes quic_session directly)
280 core::messaging_quic_server::connection_callback_t conn_cb =
281 [this](std::shared_ptr<session::quic_session> quic_sess) {
282 if (!quic_sess)
283 {
284 return;
285 }
286
287 auto wrapper = get_or_create_wrapper(quic_sess);
288
289 connection_callback_t callback_copy;
290 {
291 std::lock_guard<std::mutex> lock(callbacks_mutex_);
292 callback_copy = connection_callback_;
293 }
294
295 if (callback_copy)
296 {
297 callback_copy(wrapper);
298 }
299 };
300 server_->set_connection_callback(conn_cb);
301
302 // Bridge disconnection callback
303 server_->set_disconnection_callback(
304 [this](std::string_view session_id) {
305 disconnection_callback_t callback_copy;
306 {
307 std::lock_guard<std::mutex> lock(callbacks_mutex_);
308 callback_copy = disconnection_callback_;
309 }
310
311 if (callback_copy)
312 {
313 callback_copy(session_id);
314 }
315
316 remove_wrapper(std::string(session_id));
317 });
318
319 // Bridge receive callback
320 server_->set_receive_callback(
321 [this](std::string_view session_id, const std::vector<uint8_t>& data) {
322 receive_callback_t callback_copy;
323 {
324 std::lock_guard<std::mutex> lock(callbacks_mutex_);
325 callback_copy = receive_callback_;
326 }
327
328 if (callback_copy)
329 {
330 callback_copy(session_id, data);
331 }
332 });
333
334 // Bridge error callback
335 server_->set_error_callback(
336 [this](std::string_view session_id, std::error_code ec) {
337 error_callback_t callback_copy;
338 {
339 std::lock_guard<std::mutex> lock(callbacks_mutex_);
340 callback_copy = error_callback_;
341 }
342
343 if (callback_copy)
344 {
345 callback_copy(session_id, ec);
346 }
347 });
348}

350auto quic_server_adapter::get_or_create_wrapper(std::shared_ptr<session::quic_session> session)
351 -> std::shared_ptr<quic_session_wrapper>
352{
353 const std::string session_id = session->session_id();
354
355 std::lock_guard<std::mutex> lock(sessions_mutex_);
356
357 auto it = sessions_.find(session_id);
358 if (it != sessions_.end())
359 {
360 return it->second;
361 }
362
363 auto wrapper = std::make_shared<quic_session_wrapper>(session_id, session);
364 sessions_[session_id] = wrapper;
365 return wrapper;
366}
367
368auto quic_server_adapter::remove_wrapper(const std::string& session_id) -> void
369{
370 std::lock_guard<std::mutex> lock(sessions_mutex_);
371 sessions_.erase(session_id);
372}
