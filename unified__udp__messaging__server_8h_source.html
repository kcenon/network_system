272 struct endpoint_hash
273 {
274 std::size_t operator()(const asio::ip::udp::endpoint& ep) const
275 {
276 auto addr_hash = std::hash<std::string>{}(ep.address().to_string());
277 auto port_hash = std::hash<unsigned short>{}(ep.port());
278 return addr_hash ^ (port_hash << 1);
279 }
