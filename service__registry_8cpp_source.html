61auto parse_method_path(std::string_view full_path)
62 -> std::optional<std::pair<std::string, std::string>>
63{
64 // Expected format: "/package.Service/Method"
65 if (full_path.empty() || full_path[0] != '/')
66 {
67 return std::nullopt;
68 }
69
70 // Remove leading slash
71 auto path = full_path.substr(1);
72
73 // Find the separator between service and method
74 auto pos = path.rfind('/');
75 if (pos == std::string_view::npos || pos == 0 || pos == path.size() - 1)
76 {
77 return std::nullopt;
78 }
79
80 std::string service_name(path.substr(0, pos));
81 std::string method_name(path.substr(pos + 1));
82
83 return std::make_pair(std::move(service_name), std::move(method_name));
84}
85
86auto build_method_path(std::string_view service_name,
87 std::string_view method_name) -> std::string
88{
89 std::string result;
90 result.reserve(1 + service_name.size() + 1 + method_name.size());
91 result += '/';
92 result += service_name;
93 result += '/';
94 result += method_name;
95 return result;
96}

125 auto register_unary_method(
126 const std::string& method_name,
127 unary_handler handler,
128 const std::string& input_type,
129 const std::string& output_type) -> VoidResult
130 {
131 std::lock_guard<std::mutex> lock(mutex_);
132
133 if (unary_handlers_.find(method_name) != unary_handlers_.end())
134 {
135 return error_void(
136 error_codes::common_errors::already_exists,
137 "Method already registered: " + method_name,
138 "generic_service");

140
141 // Create method descriptor

143 method.name = method_name;

149 descriptor_.methods.push_back(std::move(method));
150 unary_handlers_[method_name] = std::move(handler);

152 return ok();
153 }

474class service_registry::impl
475{
476public:
477 explicit impl(const registry_config& config)
478 : config_(config)

481
482 auto register_service(service_base* service) -> VoidResult
483 {
484 if (service == nullptr)
485 {
486 return error_void(
487 error_codes::common_errors::invalid_argument,
488 "Service cannot be null",
489 "service_registry");
490 }
491
492 std::lock_guard<std::shared_mutex> lock(mutex_);
493
494 const auto& desc = service->descriptor();
495 auto full_name = desc.full_name();
496
497 if (services_.find(full_name) != services_.end())
498 {
499 return error_void(
500 error_codes::common_errors::already_exists,
501 "Service already registered: " + full_name,
502 "service_registry");
503 }
504
505 services_[full_name] = service;
506
507 // Set initial health status
508 if (config_.enable_health_check)
509 {
510 health_status_[full_name] = true;
511 }
512
513 return ok();
514 }
515
516 auto unregister_service(const std::string& service_name) -> VoidResult
517 {
518 std::lock_guard<std::shared_mutex> lock(mutex_);
519
520 auto it = services_.find(service_name);
521 if (it == services_.end())
522 {
523 return error_void(
524 error_codes::common_errors::not_found,
525 "Service not found: " + service_name,
526 "service_registry");
527 }
528
529 services_.erase(it);
530 health_status_.erase(service_name);
531
532 return ok();
533 }
534
535 auto find_service(const std::string& service_name) const -> service_base*
536 {
537 std::shared_lock<std::shared_mutex> lock(mutex_);
538
539 auto it = services_.find(service_name);
540 return it != services_.end() ? it->second : nullptr;
541 }
542
543 auto services() const -> std::vector<service_base*>
544 {
545 std::shared_lock<std::shared_mutex> lock(mutex_);
546
547 std::vector<service_base*> result;
548 result.reserve(services_.size());
549 for (const auto& [name, service] : services_)
550 {
551 result.push_back(service);
552 }
553 return result;
554 }
555
556 auto service_names() const -> std::vector<std::string>
557 {
558 std::shared_lock<std::shared_mutex> lock(mutex_);
559
560 std::vector<std::string> result;
561 result.reserve(services_.size());
562 for (const auto& [name, service] : services_)
563 {
564 result.push_back(name);
565 }
566 return result;
567 }
568
569 auto find_method(const std::string& full_method_path) const
570 -> std::optional<std::pair<service_base*, const method_descriptor*>>
571 {
572 auto parsed = parse_method_path(full_method_path);
573 if (!parsed)
574 {
575 return std::nullopt;
576 }
577
578 const auto& [service_name, method_name] = *parsed;
579
580 std::shared_lock<std::shared_mutex> lock(mutex_);
581
582 auto it = services_.find(service_name);
583 if (it == services_.end())
584 {
585 return std::nullopt;
586 }
587
588 auto* service = it->second;
589 const auto* method = service->descriptor().find_method(method_name);
590 if (method == nullptr)
591 {
592 return std::nullopt;
593 }
594
595 return std::make_pair(service, method);
596 }
597
598 auto is_reflection_enabled() const -> bool
599 {
600 return config_.enable_reflection;
601 }
602
603 auto configure_server(grpc_server& server) -> VoidResult
604 {
605 std::shared_lock<std::shared_mutex> lock(mutex_);
606
607 for (const auto& [name, service] : services_)
608 {
609 // Check if service is a generic_service with handlers
610 auto* generic = dynamic_cast<generic_service*>(service);
611 if (generic != nullptr)
612 {
613 const auto& desc = generic->descriptor();
614 for (const auto& method : desc.methods)
615 {
616 VoidResult result = ok();
617 switch (method.type)
618 {
619 case method_type::unary:
620 {
621 auto* handler = generic->get_unary_handler(method.name);
622 if (handler != nullptr)
623 {
624 result = server.register_unary_method(
625 method.full_name, *handler);
626 }
627 break;
628 }
629 case method_type::server_streaming:
630 {
631 auto* handler = generic->get_server_streaming_handler(method.name);
632 if (handler != nullptr)
633 {
634 result = server.register_server_streaming_method(
635 method.full_name, *handler);
636 }
637 break;
638 }
639 case method_type::client_streaming:
640 {
641 auto* handler = generic->get_client_streaming_handler(method.name);
642 if (handler != nullptr)
643 {
644 result = server.register_client_streaming_method(
645 method.full_name, *handler);
646 }
647 break;
648 }
649 case method_type::bidi_streaming:
650 {
651 auto* handler = generic->get_bidi_streaming_handler(method.name);
652 if (handler != nullptr)
653 {
654 result = server.register_bidi_streaming_method(
655 method.full_name, *handler);
656 }
657 break;
658 }
659 }
660
661 if (result.is_err())
662 {
663 return result;
664 }
665 }
666 }
667 else
668 {
669 // For non-generic services, use register_service
670 auto result = server.register_service(
671 dynamic_cast<grpc_service*>(service));
672 if (result.is_err())
673 {
674 return result;
675 }
676 }
677 }
678
679 return ok();
680 }

682#if NETWORK_GRPC_OFFICIAL
683 auto configure_server_builder(::grpc::ServerBuilder& builder) -> VoidResult
684 {
685 std::shared_lock<std::shared_mutex> lock(mutex_);
686
687 for (const auto& [name, service] : services_)
688 {
689 auto* grpc_svc = service->grpc_service();

693 }
694 }
695
696 if (config_.enable_reflection)
697 {
698 enable_reflection(builder);
699 }
700
701 return ok();
702 }

707 return ok();

711 auto set_service_health(const std::string& service_name, bool serving)
712 -> VoidResult
713 {
714 std::lock_guard<std::shared_mutex> lock(mutex_);
715
716 if (!service_name.empty() &&
717 services_.find(service_name) == services_.end())
718 {
719 return error_void(
720 error_codes::common_errors::not_found,
721 "Service not found: " + service_name,
722 "service_registry");
723 }
724
725 health_status_[service_name] = serving;
726 return ok();
727 }
728
729 auto get_service_health(const std::string& service_name) const -> bool
730 {
731 std::shared_lock<std::shared_mutex> lock(mutex_);
732
733 auto it = health_status_.find(service_name);
734 return it != health_status_.end() ? it->second : false;
735 }
