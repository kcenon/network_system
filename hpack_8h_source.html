51 struct http_header
52 {
53 std::string name;
54 std::string value;
55
56 http_header() = default;
57 http_header(std::string n, std::string v)
58 : name(std::move(n)), value(std::move(v)) {}
59
60 auto size() const -> size_t
61 {
62 return name.size() + value.size() + 32; // RFC 7541: 32 byte overhead
63 }
64
65 bool operator==(const http_header& other) const
66 {
67 return name == other.name && value == other.value;
68 }
69 };
