73memory_snapshot memory_profiler::snapshot() {
74 memory_snapshot snap;
75 snap.timestamp = std::chrono::system_clock::now();
76 std::uint64_t rss=0, vms=0;
77 query_process_memory(rss, vms);
78 snap.resident_bytes = rss;
79 snap.virtual_bytes = vms;
80
81 std::lock_guard<std::mutex> lock(mutex_);
82 history_.push_back(snap);
83 if (history_.size() > max_history_) {
84 history_.erase(history_.begin(), history_.begin() + (history_.size() - max_history_));
85 }
86 return snap;
87}

89std::vector<memory_snapshot> memory_profiler::get_history(std::size_t max_count) const {
90 std::lock_guard<std::mutex> lock(mutex_);
91 if (max_count >= history_.size()) return history_;
92 return std::vector<memory_snapshot>(history_.end() - max_count, history_.end());
93}

100std::string memory_profiler::to_tsv() const {
101 std::ostringstream oss;
102 std::lock_guard<std::mutex> lock(mutex_);
103 for (const auto& s : history_) {
104 auto t = std::chrono::system_clock::to_time_t(s.timestamp);
105 oss << std::put_time(std::localtime(&t), "%Y-%m-%d %H:%M:%S")
106 << '\t' << s.resident_bytes
107 << '\t' << s.virtual_bytes
108 << '\n';
109 }
110 return oss.str();
111}

113void memory_profiler::schedule_next_sample() {
114#ifdef NETWORK_ENABLE_MEMORY_PROFILER
115 if (!running_.load()) return;
116
117 kcenon::network::integration::thread_integration_manager::instance()
118 .submit_delayed_task(
119 [this]() {
120 if (running_.load()) {
121 snapshot();
122 schedule_next_sample();
123 }
124 },
125 sampling_interval_);
126#endif
127}

129bool memory_profiler::query_process_memory(std::uint64_t& rss, std::uint64_t& vms) {
130#if defined(__APPLE__)
131 task_basic_info_data_t info;
132 mach_msg_type_number_t count = TASK_BASIC_INFO_COUNT;
133 if (task_info(mach_task_self(), TASK_BASIC_INFO, (task_info_t)&info, &count) != KERN_SUCCESS) {
134 return false;
135 }
136 rss = static_cast<std::uint64_t>(info.resident_size);
137 vms = static_cast<std::uint64_t>(info.virtual_size);
138 return true;
139#elif defined(__linux__)
140 FILE* f = std::fopen("/proc/self/statm", "r");
141 if (!f) return false;
142 long pages_res=0, pages_total=0;
143 int rc = std::fscanf(f, "%ld %ld", &pages_total, &pages_res);
144 std::fclose(f);
145 if (rc != 2) return false;
146 long page_size = sysconf(_SC_PAGESIZE);
147 vms = static_cast<std::uint64_t>(pages_total) * static_cast<std::uint64_t>(page_size);
148 rss = static_cast<std::uint64_t>(pages_res) * static_cast<std::uint64_t>(page_size);
149 return true;
150#elif defined(_WIN32)
151 PROCESS_MEMORY_COUNTERS_EX pmc{};
152 if (GetProcessMemoryInfo(GetCurrentProcess(), (PROCESS_MEMORY_COUNTERS*)&pmc, sizeof(pmc))) {
153 rss = static_cast<std::uint64_t>(pmc.WorkingSetSize);
154 vms = static_cast<std::uint64_t>(pmc.PrivateUsage);
155 return true;
156 }
157 return false;
158#else
159 // Unknown platform
160 rss = vms = 0;
161 return false;
162#endif
163}
