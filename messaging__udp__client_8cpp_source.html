59 auto messaging_udp_client::start_client(std::string_view host, uint16_t port) -> VoidResult
60 {
61 if (lifecycle_.is_running())
62 {
63 return error_void(
64 error_codes::common_errors::already_exists,
65 "UDP client is already running",
66 "messaging_udp_client::start_client");
67 }
68
69 if (host.empty())
70 {
71 return error_void(
72 error_codes::common_errors::invalid_argument,
73 "Host cannot be empty",
74 "messaging_udp_client::start_client");
75 }
76
77 // Mark as running before starting
78 lifecycle_.set_running();
79
80 auto result = do_start_impl(host, port);
81 if (result.is_err())
82 {
83 lifecycle_.mark_stopped();
84 }
85
86 return result;
87 }
88
89 auto messaging_udp_client::stop_client() -> VoidResult
90 {
91 if (!lifecycle_.prepare_stop())
92 {
93 return ok(); // Not running or already stopping
94 }
95
96 auto result = do_stop_impl();
97
98 lifecycle_.mark_stopped();
99
100 return result;
101 }

136 auto messaging_udp_client::send(
137 std::vector<uint8_t>&& data,
138 interfaces::i_udp_client::send_callback_t handler) -> VoidResult
139 {
140 if (!lifecycle_.is_running())
141 {
142 return error_void(
143 error_codes::common_errors::internal_error,
144 "UDP client is not running",
145 "messaging_udp_client::send",
146 ""
147 );
148 }
149
150 std::lock_guard<std::mutex> socket_lock(socket_mutex_);
151 if (!socket_)
152 {
153 return error_void(
154 error_codes::common_errors::internal_error,
155 "Socket not available",
156 "messaging_udp_client::send",
157 ""
158 );
159 }
160
161 std::lock_guard<std::mutex> endpoint_lock(endpoint_mutex_);
162 socket_->async_send_to(std::move(data), target_endpoint_, std::move(handler));
163
164 return ok();
165 }

167 auto messaging_udp_client::set_target(std::string_view host, uint16_t port) -> VoidResult
168 {
169 if (!lifecycle_.is_running())
170 {
171 return error_void(
172 error_codes::common_errors::internal_error,
173 "UDP client is not running",
174 "messaging_udp_client::set_target",
175 ""
176 );
177 }
178
179 try
180 {
181 // Resolve new target endpoint
182 udp::resolver resolver(*io_context_);
183 auto endpoints = resolver.resolve(udp::v4(), std::string(host), std::to_string(port));
184
185 if (endpoints.empty())
186 {
187 return error_void(
188 error_codes::common_errors::internal_error,
189 "Failed to resolve host",
190 "messaging_udp_client::set_target",
191 "Host: " + std::string(host)
192 );
193 }
194
195 std::lock_guard<std::mutex> lock(endpoint_mutex_);
196 target_endpoint_ = *endpoints.begin();
197
198 NETWORK_LOG_INFO("Target updated to " + std::string(host) + ":" + std::to_string(port));
199
200 return ok();
201 }
202 catch (const std::exception& e)
203 {
204 return error_void(
205 error_codes::common_errors::internal_error,
206 std::string("Failed to set target: ") + e.what(),
207 "messaging_udp_client::set_target",
208 "Host: " + std::string(host) + ":" + std::to_string(port)
209 );
210 }

213 auto messaging_udp_client::set_receive_callback(
214 interfaces::i_udp_client::receive_callback_t callback) -> void
215 {
216 if (!callback)
217 {
218 // Clear the callback
219 callbacks_.set<to_index(callback_index::receive)>(nullptr);
220 return;
221 }
222
223 // Adapt the interface callback to the internal callback type
224 // Convert asio::ip::udp::endpoint to endpoint_info
225 callbacks_.set<to_index(callback_index::receive)>(
226 [callback = std::move(callback)](
227 const std::vector<uint8_t>& data,
228 const asio::ip::udp::endpoint& endpoint)
229 {
230 interfaces::i_udp_client::endpoint_info info;
231 info.address = endpoint.address().to_string();
232 info.port = endpoint.port();
233 callback(data, info);
234 });
235 }

251 auto messaging_udp_client::do_start_impl(std::string_view host, uint16_t port) -> VoidResult
252 {
253 try
254 {
255 // Create io_context
256 io_context_ = std::make_unique<asio::io_context>();
257
258 // Resolve target endpoint
259 udp::resolver resolver(*io_context_);
260 auto endpoints = resolver.resolve(udp::v4(), std::string(host), std::to_string(port));
261
262 if (endpoints.empty())
263 {
264 return error_void(
265 error_codes::common_errors::internal_error,
266 "Failed to resolve host",
267 "messaging_udp_client::do_start_impl",
268 "Host: " + std::string(host)
269 );
270 }
271
272 {
273 std::lock_guard<std::mutex> lock(endpoint_mutex_);
274 target_endpoint_ = *endpoints.begin();
275 }
276
277 // Create UDP socket (bind to any port)
278 asio::ip::udp::socket raw_socket(*io_context_, udp::endpoint(udp::v4(), 0));
279
280 // Wrap in our udp_socket
281 socket_ = std::make_shared<internal::udp_socket>(std::move(raw_socket));
282
283 // Set callbacks to socket
284 auto receive_cb = get_receive_callback();
285 if (receive_cb)
286 {
287 socket_->set_receive_callback(std::move(receive_cb));
288 }
289
290 auto error_cb = get_error_callback();
291 if (error_cb)
292 {
293 socket_->set_error_callback(std::move(error_cb));
294 }
295
296 // Start receiving
297 socket_->start_receive();
298
299 // Get thread pool from network context
300 thread_pool_ = network_context::instance().get_thread_pool();
301 if (!thread_pool_) {
302 // Fallback: create a temporary thread pool if network_context is not initialized
303 NETWORK_LOG_WARN("[messaging_udp_client] network_context not initialized, creating temporary thread pool");
304 thread_pool_ = std::make_shared<integration::basic_thread_pool>(std::thread::hardware_concurrency());
305 }
306
307 // Start io_context on thread pool
308 io_context_future_ = thread_pool_->submit(
309 [this]()
310 {
311 try
312 {
313 NETWORK_LOG_DEBUG("[messaging_udp_client] io_context started");
314 io_context_->run();
315 NETWORK_LOG_DEBUG("[messaging_udp_client] io_context stopped");
316 }
317 catch (const std::exception& e)
318 {
319 NETWORK_LOG_ERROR(std::string("Worker thread exception: ") + e.what());
320 }
321 });
322
323 NETWORK_LOG_INFO("UDP client started targeting " + std::string(host) + ":" + std::to_string(port));
324
325 return ok();
326 }
327 catch (const std::system_error& e)
328 {
329 return error_void(
330 error_codes::common_errors::internal_error,
331 std::string("Failed to create UDP socket: ") + e.what(),
332 "messaging_udp_client::do_start_impl",
333 "Host: " + std::string(host) + ":" + std::to_string(port)
334 );
335 }
336 catch (const std::exception& e)
337 {
338 return error_void(
339 error_codes::common_errors::internal_error,
340 std::string("Failed to start UDP client: ") + e.what(),
341 "messaging_udp_client::do_start_impl",
342 "Host: " + std::string(host) + ":" + std::to_string(port)
343 );
344 }
345 }

347 auto messaging_udp_client::do_stop_impl() -> VoidResult
348 {
349 try
350 {
351 // Stop receiving
352 if (socket_)
353 {
354 socket_->stop_receive();
355 }
356
357 // Stop io_context
358 if (io_context_)
359 {
360 io_context_->stop();
361 }
362
363 // Wait for io_context task to complete
364 if (io_context_future_.valid())
365 {
366 try {
367 io_context_future_.wait();
368 } catch (const std::exception& e) {
369 NETWORK_LOG_ERROR("[messaging_udp_client] Exception while waiting for io_context: " +
370 std::string(e.what()));
371 }
372 }
373
374 // Clean up
375 socket_.reset();
376 io_context_.reset();
377
378 NETWORK_LOG_INFO("UDP client stopped");
379
380 return ok();
381 }
382 catch (const std::exception& e)
383 {
384 return error_void(
385 error_codes::common_errors::internal_error,
386 std::string("Failed to stop UDP client: ") + e.what(),
387 "messaging_udp_client::do_stop_impl",
388 ""
389 );
390 }
391 }
