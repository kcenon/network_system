68auto vector_to_byte_buffer(const std::vector<uint8_t>& data) -> ::grpc::ByteBuffer
69{
70 ::grpc::Slice slice(data.data(), data.size());
71 return ::grpc::ByteBuffer(&slice, 1);
72}
73
74auto byte_buffer_to_vector(const ::grpc::ByteBuffer& buffer) -> std::vector<uint8_t>
75{
76 std::vector<::grpc::Slice> slices;
77 buffer.Dump(&slices);
78
79 std::vector<uint8_t> result;
80 result.reserve(buffer.Length());
81
82 for (const auto& slice : slices)
83 {
84 const auto* begin = reinterpret_cast<const uint8_t*>(slice.begin());
85 result.insert(result.end(), begin, begin + slice.size());
86 }
87
88 return result;
89}

374 auto connect() -> VoidResult
375 {
376 auto span = tracing::is_tracing_enabled()
377 ? std::make_optional(tracing::trace_context::create_span("grpc.client.connect"))
378 : std::nullopt;
379 if (span)
380 {
381 span->set_attribute("rpc.system", "grpc")
382 .set_attribute("rpc.grpc.target", target_)
383 .set_attribute("net.transport", "tcp")
384 .set_attribute("rpc.grpc.use_tls", config_.use_tls);
385 }
386
387 std::lock_guard<std::mutex> lock(mutex_);
388
389 if (connected_.load())
390 {
391 if (span)
392 {
393 span->set_attribute("grpc.client.already_connected", true);
394 }
395 return kcenon::network::protocols::grpc::ok();
396 }
397
398 // Create channel credentials
399 channel_credentials_config creds_config;
400 creds_config.insecure = !config_.use_tls;
401
402 if (config_.use_tls)
403 {
404 creds_config.root_certificates = config_.root_certificates;
405 creds_config.client_certificate = config_.client_certificate;
406 creds_config.client_key = config_.client_key;
407 }
408
409 channel_ = create_channel(target_, creds_config);
410
411 if (!channel_)
412 {
413 if (span)
414 {
415 span->set_error("Failed to create gRPC channel");
416 }
417 return error_void(
418 error_codes::network_system::connection_failed,
419 "Failed to create gRPC channel",
420 "grpc::client");
421 }
422
423 // Wait for channel to be ready
424 if (!wait_for_channel_ready(channel_, config_.default_timeout))
425 {
426 if (span)
427 {
428 span->set_error("Failed to connect to gRPC server");
429 }
430 return error_void(
431 error_codes::network_system::connection_failed,
432 "Failed to connect to gRPC server",
433 "grpc::client",
434 target_);
435 }
436
437 stub_ = std::make_unique<::grpc::GenericStub>(channel_);
438 connected_.store(true);
439
440 return kcenon::network::protocols::grpc::ok();
441 }
442
443 auto disconnect() -> void
444 {
445 std::lock_guard<std::mutex> lock(mutex_);
446
447 stub_.reset();
448 channel_.reset();
449 connected_.store(false);
450 }

478 auto call_raw(const std::string& method,
479 const std::vector<uint8_t>& request,
480 const call_options& options) -> Result<grpc_message>
481 {
482 auto span = tracing::is_tracing_enabled()
483 ? std::make_optional(tracing::trace_context::create_span("grpc.client.call"))
484 : std::nullopt;
485 if (span)
486 {
487 span->set_attribute("rpc.system", "grpc")
488 .set_attribute("rpc.method", method)
489 .set_attribute("rpc.grpc.target", target_)
490 .set_attribute("rpc.request.size", static_cast<int64_t>(request.size()));
491 }
492
493 if (!is_connected())
494 {
495 if (span)
496 {
497 span->set_error("Not connected to server");
498 }
499 return error<grpc_message>(
500 error_codes::network_system::connection_failed,
501 "Not connected to server",
502 "grpc::client");
503 }
504
505 ::grpc::ClientContext ctx;
506
507 // Set deadline if provided
508 if (options.deadline.has_value())
509 {
510 set_deadline(&ctx, options.deadline.value());
511 }
512 else
513 {
514 // Use default timeout
515 set_timeout(&ctx, config_.default_timeout);
516 }
517
518 // Add metadata
519 for (const auto& [key, value] : options.metadata)
520 {
521 ctx.AddMetadata(key, value);
522 }
523
524 // Set wait for ready
525 if (options.wait_for_ready)
526 {
527 ctx.set_wait_for_ready(true);
528 }
529
530 // Prepare request
531 ::grpc::ByteBuffer request_buffer = detail::vector_to_byte_buffer(request);
532 ::grpc::ByteBuffer response_buffer;
533
534 // Make unary call
535 ::grpc::Status status = stub_->UnaryCall(&ctx, method, request_buffer, &response_buffer);
536
537 if (!status.ok())
538 {
539 auto grpc_st = from_grpc_status(status);
540 if (span)
541 {
542 span->set_error(grpc_st.message)
543 .set_attribute("rpc.grpc.status_code", static_cast<int64_t>(grpc_st.code));
544 }
545 return error<grpc_message>(
546 static_cast<int>(grpc_st.code),
547 grpc_st.message,
548 "grpc::client");
549 }
550
551 auto response_data = detail::byte_buffer_to_vector(response_buffer);
552 if (span)
553 {
554 span->set_attribute("rpc.response.size", static_cast<int64_t>(response_data.size()))
555 .set_attribute("rpc.grpc.status_code", static_cast<int64_t>(0));
556 }
557 return kcenon::network::protocols::grpc::ok(grpc_message{std::move(response_data)});
558 }

560 auto call_raw_async(const std::string& method,
561 const std::vector<uint8_t>& request,
562 std::function<void(Result<grpc_message>)> callback,
563 const call_options& options) -> void
564 {
565 // Create async call in separate thread
566 std::thread([this, method, request, callback, options]() {
567 auto result = call_raw(method, request, options);
568 if (callback)
569 {
570 callback(std::move(result));
571 }
572 }).detach();
573 }

667// Server stream reader implementation
668class server_stream_reader_impl : public grpc_client::server_stream_reader
669{
670public:
671 server_stream_reader_impl(std::shared_ptr<http2::http2_client> http2_client,
672 uint32_t stream_id)
673 : http2_client_(std::move(http2_client))
674 , stream_id_(stream_id)
675 , has_more_(true)
676 {
677 }
678
679 auto read() -> Result<grpc_message> override
680 {
681 std::unique_lock<std::mutex> lock(mutex_);
682
683 // Wait for data or end of stream
684 cv_.wait(lock, [this] { return !buffer_.empty() || !has_more_; });
685
686 if (buffer_.empty())
687 {
688 if (!has_more_)
689 {
690 return error<grpc_message>(
691 static_cast<int>(status_code::ok),
692 "End of stream",
693 "server_stream_reader::read");
694 }
695 }
696
697 // Parse gRPC message from buffer
698 auto parse_result = grpc_message::parse(buffer_);
699 if (parse_result.is_err())
700 {
701 return parse_result;
702 }
703
704 // Remove parsed data from buffer
705 auto& msg = parse_result.value();
706 size_t msg_size = 5 + msg.data.size(); // 1 byte compression + 4 bytes length + data
707 if (buffer_.size() >= msg_size)
708 {
709 buffer_.erase(buffer_.begin(), buffer_.begin() + static_cast<ptrdiff_t>(msg_size));
710 }
711
712 return ok(std::move(msg));
713 }
714
715 auto has_more() const -> bool override
716 {
717 std::lock_guard<std::mutex> lock(mutex_);
718 return has_more_ || !buffer_.empty();
719 }
720
721 auto finish() -> grpc_status override
722 {
723 std::unique_lock<std::mutex> lock(mutex_);
724 cv_.wait(lock, [this] { return !has_more_; });
725 return final_status_;
726 }
727
728 void on_data(const std::vector<uint8_t>& data)
729 {
730 std::lock_guard<std::mutex> lock(mutex_);
731 buffer_.insert(buffer_.end(), data.begin(), data.end());
732 cv_.notify_one();
733 }
734
735 void on_headers(const std::vector<http2::http_header>& headers)
736 {
737 std::lock_guard<std::mutex> lock(mutex_);
738 for (const auto& h : headers)
739 {
740 if (h.name == "grpc-status")
741 {
742 try { final_status_.code = static_cast<status_code>(std::stoi(h.value)); }
743 catch (...) {}
744 }
745 else if (h.name == "grpc-message")
746 {
747 final_status_.message = h.value;
748 }
749 }
750 }
751
752 void on_complete(int status_code)
753 {
754 std::lock_guard<std::mutex> lock(mutex_);
755 has_more_ = false;
756 if (status_code != 200)
757 {
758 final_status_.code = grpc::status_code::unavailable;
759 final_status_.message = "HTTP error: " + std::to_string(status_code);
760 }
761 cv_.notify_all();
762 }
763
764private:
765 std::shared_ptr<http2::http2_client> http2_client_;
766 [[maybe_unused]] uint32_t stream_id_; // Reserved for future stream operations
767 mutable std::mutex mutex_;
768 std::condition_variable cv_;
769 std::vector<uint8_t> buffer_;
770 bool has_more_;
771 grpc_status final_status_;
772};

827 // Wait for response
828 std::unique_lock<std::mutex> lock(mutex_);
829 cv_.wait(lock, [this] { return response_ready_; });

831 if (!response_.data.empty() || final_status_.code == status_code::ok)
832 {
833 return ok(std::move(response_));
834 }
835
836 return error<grpc_message>(
837 static_cast<int>(final_status_.code),
838 final_status_.message,
839 "client_stream_writer::finish");
840 }

842 void on_data(const std::vector<uint8_t>& data)
843 {
844 std::lock_guard<std::mutex> lock(mutex_);
845 response_buffer_.insert(response_buffer_.end(), data.begin(), data.end());
846 }

849 {
850 std::lock_guard<std::mutex> lock(mutex_);
851 for (const auto& h : headers)
852 {
853 if (h.name == "grpc-status")
854 {
855 try { final_status_.code = static_cast<status_code>(std::stoi(h.value)); }
856 catch (...) {}
857 }
858 else if (h.name == "grpc-message")
859 {
860 final_status_.message = h.value;
861 }
862 }
863 }
864
865 void on_complete(int status_code)
866 {
867 std::lock_guard<std::mutex> lock(mutex_);
868

871 final_status_.code = grpc::status_code::unavailable;
872 final_status_.message = "HTTP error: " + std::to_string(status_code);
873 }
874
875 // Parse response
876 if (!response_buffer_.empty())
877 {
878 auto parse_result = grpc_message::parse(response_buffer_);
879 if (parse_result.is_ok())
880 {
881 response_ = std::move(parse_result.value());
882 }
883 }
884
885 response_ready_ = true;
886 cv_.notify_all();
887 }

891 uint32_t stream_id_;
892 bool writes_done_;
893 mutable std::mutex mutex_;
894 std::condition_variable cv_;
895 std::vector<uint8_t> response_buffer_;
896 grpc_message response_;
897 grpc_status final_status_;
898 bool response_ready_ = false;

1053 : std::nullopt;

1055 {

1058 .set_attribute("net.transport", "tcp");
1059 }

1063 if (connected_.load())

1066 {
1067 span->set_attribute("grpc.client.already_connected", true);
1068 }

1070 }
1071
1072 // Parse target address
1073 auto colon_pos = target_.find(':');
1074 if (colon_pos == std::string::npos)
1075 {
1076 if (span)
1077 {
1078 span->set_error("Invalid target address format");
1079 }
1080 return error_void(
1081 error_codes::common_errors::invalid_argument,
1082 "Invalid target address format",
1083 "grpc::client",

1087 host_ = target_.substr(0, colon_pos);
1088 auto port_str = target_.substr(colon_pos + 1);

1092 port_str.data(),
1093 port_str.data() + port_str.size(),
1094 port);
1095
1096 if (ec != std::errc())
1097 {
1098 if (span)

1100 span->set_error("Invalid port number");

1102 return error_void(

1104 "Invalid port number",

1111 .set_attribute("net.peer.port", static_cast<int64_t>(port));
1112 }
1113
1114 // Create HTTP/2 client
1115 http2_client_ = std::make_shared<http2::http2_client>("grpc-client");

1117
1118 // Connect using HTTP/2
1119 auto result = http2_client_->connect(host_, port);
1120 if (result.is_err())
1121 {
1122 const auto& err = result.error();
1123 if (span)
1124 {
1125 span->set_error(err.message);
1126 }

1174 auto call_raw(const std::string& method,
1175 const std::vector<uint8_t>& request,
1176 const call_options& options) -> Result<grpc_message>
1177 {
1178 auto span = tracing::is_tracing_enabled()
1179 ? std::make_optional(tracing::trace_context::create_span("grpc.client.call"))
1180 : std::nullopt;
1181 if (span)
1182 {
1183 span->set_attribute("rpc.system", "grpc")
1184 .set_attribute("rpc.method", method)
1185 .set_attribute("rpc.grpc.target", target_)
1186 .set_attribute("rpc.request.size", static_cast<int64_t>(request.size()));
1187 }
1188
1189 if (!is_connected())
1190 {
1191 if (span)
1192 {
1193 span->set_error("Not connected to server");
1194 }
1195 return error<grpc_message>(
1196 error_codes::network_system::connection_failed,
1197 "Not connected to server",
1198 "grpc::client");
1199 }
1200
1201 // Validate method format
1202 if (method.empty() || method[0] != '/')
1203 {
1204 if (span)
1205 {
1206 span->set_error("Invalid method format");
1207 }
1208 return error<grpc_message>(
1209 error_codes::common_errors::invalid_argument,
1210 "Invalid method format",
1211 "grpc::client",
1212 "Method must start with '/'");
1213 }
1214
1215 // Check deadline
1216 if (options.deadline.has_value())
1217 {
1218 if (std::chrono::system_clock::now() > options.deadline.value())
1219 {
1220 if (span)
1221 {
1222 span->set_error("Deadline exceeded before call started")
1223 .set_attribute("rpc.grpc.status_code", static_cast<int64_t>(status_code::deadline_exceeded));
1224 }
1225 return error<grpc_message>(
1226 static_cast<int>(status_code::deadline_exceeded),
1227 "Deadline exceeded before call started",
1228 "grpc::client");
1229 }
1230 }
1231
1232 // Build gRPC headers
1233 std::vector<http2::http_header> headers;
1234 headers.emplace_back(header_names::content_type, grpc_content_type);
1235 headers.emplace_back(header_names::te, "trailers");
1236 headers.emplace_back(header_names::grpc_accept_encoding,
1237 std::string(compression::identity) + "," +
1238 compression::gzip + "," + compression::deflate);
1239
1240 // Add timeout header if deadline is set
1241 if (options.deadline.has_value())
1242 {
1243 auto now = std::chrono::system_clock::now();
1244 auto remaining = std::chrono::duration_cast<std::chrono::milliseconds>(
1245 options.deadline.value() - now);
1246 if (remaining.count() > 0)
1247 {
1248 headers.emplace_back(header_names::grpc_timeout,
1249 format_timeout(static_cast<uint64_t>(remaining.count())));
1250 }
1251 }
1252
1253 // Add custom metadata
1254 for (const auto& [key, value] : options.metadata)
1255 {
1256 headers.emplace_back(key, value);
1257 }
1258
1259 // Serialize request with gRPC framing
1260 grpc_message request_msg{std::vector<uint8_t>{request.begin(), request.end()}};
1261 auto serialized_request = request_msg.serialize();
1262
1263 // Send HTTP/2 POST request
1264 auto response_result = http2_client_->post(method, serialized_request, headers);
1265 if (response_result.is_err())
1266 {
1267 const auto& err = response_result.error();
1268 if (span)
1269 {
1270 span->set_error(err.message);
1271 }
1272 return error<grpc_message>(err.code, err.message, "grpc::client",
1273 get_error_details(err));
1274 }
1275
1276 const auto& response = response_result.value();
1277
1278 // Check HTTP status - gRPC always uses 200 OK for successful transport
1279 if (response.status_code != 200)
1280 {
1281 if (span)
1282 {
1283 span->set_error("HTTP error: " + std::to_string(response.status_code))
1284 .set_attribute("http.status_code", static_cast<int64_t>(response.status_code));
1285 }
1286 return error<grpc_message>(
1287 static_cast<int>(status_code::unavailable),
1288 "HTTP error: " + std::to_string(response.status_code),
1289 "grpc::client");
1290 }
1291
1292 // Extract gRPC status from trailers/headers
1293 status_code grpc_status = status_code::ok;
1294 std::string grpc_message_str;
1295
1296 for (const auto& header : response.headers)
1297 {
1298 if (header.name == trailer_names::grpc_status)
1299 {
1300 int status_int = 0;
1301 auto [ptr, ec] = std::from_chars(
1302 header.value.data(),
1303 header.value.data() + header.value.size(),
1304 status_int);
1305 if (ec == std::errc())
1306 {
1307 grpc_status = static_cast<status_code>(status_int);
1308 }
1309 }
1310 else if (header.name == trailer_names::grpc_message)
1311 {
1312 grpc_message_str = header.value;
1313 }
1314 }
1315
1316 // Check gRPC status
1317 if (grpc_status != status_code::ok)
1318 {
1319 if (span)
1320 {
1321 span->set_error(grpc_message_str.empty() ?
1322 std::string(status_code_to_string(grpc_status)) : grpc_message_str)
1323 .set_attribute("rpc.grpc.status_code", static_cast<int64_t>(grpc_status));
1324 }
1325 return error<grpc_message>(
1326 static_cast<int>(grpc_status),
1327 grpc_message_str.empty() ?
1328 std::string(status_code_to_string(grpc_status)) : grpc_message_str,
1329 "grpc::client");
1330 }
1331
1332 // Parse response message
1333 if (response.body.empty())
1334 {
1335 // Empty response is valid for some RPCs
1336 if (span)
1337 {
1338 span->set_attribute("rpc.response.size", static_cast<int64_t>(0))
1339 .set_attribute("rpc.grpc.status_code", static_cast<int64_t>(0));
1340 }
1341 return ok(grpc_message{});
1342 }
1343
1344 auto parse_result = grpc_message::parse(response.body);
1345 if (parse_result.is_err())
1346 {
1347 const auto& err = parse_result.error();
1348 if (span)
1349 {
1350 span->set_error(err.message);
1351 }
1352 return error<grpc_message>(err.code, err.message, "grpc::client",
1353 get_error_details(err));
1354 }
1355
1356 if (span)
1357 {
1358 span->set_attribute("rpc.response.size", static_cast<int64_t>(parse_result.value().data.size()))
1359 .set_attribute("rpc.grpc.status_code", static_cast<int64_t>(0));
1360 }
1361 return ok(std::move(parse_result.value()));
1362 }

1607 headers.emplace_back(key, value);
