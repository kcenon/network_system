44 websocket_socket::websocket_socket(std::shared_ptr<tcp_socket> socket,
45 bool is_client)
46 : tcp_socket_(std::move(socket))
47 , protocol_(is_client)
48 {
49 // Set up WebSocket protocol callbacks
50 protocol_.set_message_callback(
51 [this](const ws_message& msg) { handle_protocol_message(msg); });
52
53 protocol_.set_ping_callback([this](const std::vector<uint8_t>& payload)
54 {
55 // RFC 6455: Automatically respond to Ping with Pong
56 auto pong_frame = protocol_.create_pong(
57 std::vector<uint8_t>(payload));
58 tcp_socket_->async_send(
59 std::move(pong_frame),
60 [](std::error_code, std::size_t) {});
61
62 handle_protocol_ping(payload);
63 });
64
65 protocol_.set_pong_callback([this](const std::vector<uint8_t>& payload)
66 { handle_protocol_pong(payload); });
67
68 protocol_.set_close_callback([this](ws_close_code code, const std::string& reason)
69 { handle_protocol_close(code, reason); });
70
71 // Set up TCP socket callbacks to feed data into protocol layer
72 // Use zero-copy span callback to avoid per-read vector allocation
73 tcp_socket_->set_receive_callback_view(
74 [this](std::span<const uint8_t> data) { on_tcp_receive(data); });
75
76 tcp_socket_->set_error_callback(
77 [this](std::error_code ec) { on_tcp_error(ec); });

85 auto websocket_socket::async_handshake(const std::string& host,
86 const std::string& path, uint16_t port,
87 std::function<void(std::error_code)> handler)
88 -> void
89 {
90 // Generate client handshake request
91 auto handshake_request =
92 websocket_handshake::create_client_handshake(host, path, port);
93
94 // Store the expected key for validation
95 auto key_start = handshake_request.find("Sec-WebSocket-Key: ");
96 std::string client_key;
97 if (key_start != std::string::npos)
98 {
99 key_start += 19; // Length of "Sec-WebSocket-Key: "
100 auto key_end = handshake_request.find("\r\n", key_start);
101 client_key = handshake_request.substr(key_start, key_end - key_start);
102 }
103
104 // Send handshake request
105 std::vector<uint8_t> request_data(handshake_request.begin(),
106 handshake_request.end());
107
108 tcp_socket_->async_send(
109 std::move(request_data),
110 [this, client_key, handler](std::error_code ec, std::size_t)
111 {
112 if (ec)
113 {
114 handler(ec);
115 return;
116 }
117
118 // Set up temporary callback to receive handshake response
119 // Use span callback for zero-copy during handshake
120 tcp_socket_->set_receive_callback_view(
121 [this, client_key, handler](std::span<const uint8_t> data)
122 {
123 // Convert response to string
124 std::string response(data.begin(), data.end());
125
126 // Validate server response
127 auto result = websocket_handshake::validate_server_response(
128 response, client_key);
129
130 if (!result.success)
131 {
132 handler(std::make_error_code(std::errc::protocol_error));
133 return;
134 }
135
136 // Handshake successful, update state
137 state_ = ws_state::open;
138
139 // Restore normal receive callback
140 tcp_socket_->set_receive_callback_view(
141 [this](std::span<const uint8_t> data)
142 { on_tcp_receive(data); });
143
144 handler(std::error_code{});
145 });
146
147 // Start reading response
148 tcp_socket_->start_read();
149 });
150 }

152 auto websocket_socket::async_accept(std::function<void(std::error_code)> handler)
153 -> void
154 {
155 // Set up temporary callback to receive handshake request
156 // Use span callback for zero-copy during handshake
157 tcp_socket_->set_receive_callback_view(
158 [this, handler](std::span<const uint8_t> data)
159 {
160 // Convert request to string
161 std::string request(data.begin(), data.end());
162
163 // Parse client request
164 auto result = websocket_handshake::parse_client_request(request);
165
166 if (!result.success)
167 {
168 handler(std::make_error_code(std::errc::protocol_error));
169 return;
170 }
171
172 // Extract client key (headers are stored lowercase by parse_headers)
173 auto it = result.headers.find("sec-websocket-key");
174 if (it == result.headers.end())
175 {
176 handler(std::make_error_code(std::errc::protocol_error));
177 return;
178 }
179
180 std::string client_key = it->second;
181
182 // Generate server response
183 auto response = websocket_handshake::create_server_response(client_key);
184 std::vector<uint8_t> response_data(response.begin(), response.end());
185
186 // Send response
187 tcp_socket_->async_send(
188 std::move(response_data),
189 [this, handler](std::error_code ec, std::size_t)
190 {
191 if (ec)
192 {
193 handler(ec);
194 return;
195 }
196
197 // Handshake successful, update state
198 state_ = ws_state::open;
199
200 // Restore normal receive callback
201 tcp_socket_->set_receive_callback_view(
202 [this](std::span<const uint8_t> data)
203 { on_tcp_receive(data); });
204
205 handler(std::error_code{});
206 });
207 });
208
209 // Start reading request
210 tcp_socket_->start_read();
211 }

215 auto websocket_socket::async_send_text(
216 std::string&& message,
217 std::function<void(std::error_code, std::size_t)> handler) -> VoidResult
218 {
219 if (state_ != ws_state::open)
220 {
221 return error_void(error_codes::network_system::connection_closed,
222 "WebSocket connection not open");
223 }
224
225 // Encode text message into WebSocket frame
226 auto frame_data = protocol_.create_text_message(std::move(message));
227
228 if (frame_data.empty())
229 {
230 return error_void(error_codes::common_errors::invalid_argument,
231 "Invalid UTF-8 in text message");
232 }
233
234 // Send via underlying tcp_socket
235 tcp_socket_->async_send(std::move(frame_data), std::move(handler));
236
237 return ok();
238 }

277 auto websocket_socket::async_close(ws_close_code code, const std::string& reason,
278 std::function<void(std::error_code)> handler)
279 -> void
280 {
281 // Update state to closing
282 state_ = ws_state::closing;
283
284 // Encode close frame
285 auto frame_data = protocol_.create_close(code, std::string(reason));
286
287 // Send via underlying tcp_socket
288 tcp_socket_->async_send(std::move(frame_data),
289 [this, handler](std::error_code ec, std::size_t)
290 {
291 if (ec)
292 {
293 state_ = ws_state::closed;
294 if (handler)
295 {
296 handler(ec);
297 }
298 return;
299 }
300
301 // Close handshake initiated successfully
302 if (handler)
303 {
304 handler(std::error_code{});
305 }
306 });
307 }
