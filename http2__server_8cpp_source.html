59 auto http2_server::start(unsigned short port) -> VoidResult
60 {
61 if (is_running_) {
62 return error_void(
63 error_codes::common_errors::already_exists,
64 "Server already running",
65 "http2_server");
66 }
67
68 try {
69 io_context_ = std::make_unique<asio::io_context>();
70 work_guard_ = std::make_unique<asio::executor_work_guard<asio::io_context::executor_type>>(
71 io_context_->get_executor());
72
73 acceptor_ = std::make_unique<asio::ip::tcp::acceptor>(
74 *io_context_,
75 asio::ip::tcp::endpoint(asio::ip::tcp::v4(), port));
76
77 use_tls_ = false;
78 is_running_ = true;
79
80 // Start I/O thread
81 io_future_ = std::async(std::launch::async, [this]() { run_io(); });
82
83 // Start accepting connections
84 do_accept();
85 start_cleanup_timer();
86
87 return ok();
88 } catch (const std::exception& e) {
89 return error_void(
90 error_codes::network_system::bind_failed,
91 std::string("Failed to start server: ") + e.what(),
92 "http2_server");

96 auto http2_server::start_tls(unsigned short port, const tls_config& config) -> VoidResult
97 {
98 if (is_running_) {
99 return error_void(
100 error_codes::common_errors::already_exists,
101 "Server already running",
102 "http2_server");
103 }
104
105 try {
106 io_context_ = std::make_unique<asio::io_context>();
107 work_guard_ = std::make_unique<asio::executor_work_guard<asio::io_context::executor_type>>(
108 io_context_->get_executor());
109
110 // Setup TLS context
111 ssl_context_ = std::make_unique<asio::ssl::context>(asio::ssl::context::tls_server);
112 ssl_context_->set_options(
113 asio::ssl::context::default_workarounds |
114 asio::ssl::context::no_sslv2 |
115 asio::ssl::context::no_sslv3 |
116 asio::ssl::context::no_tlsv1 |
117 asio::ssl::context::no_tlsv1_1);
118
119 ssl_context_->use_certificate_file(config.cert_file, asio::ssl::context::pem);
120 ssl_context_->use_private_key_file(config.key_file, asio::ssl::context::pem);
121
122 if (!config.ca_file.empty()) {
123 ssl_context_->load_verify_file(config.ca_file);
124 }
125
126 if (config.verify_client) {
127 ssl_context_->set_verify_mode(asio::ssl::verify_peer | asio::ssl::verify_fail_if_no_peer_cert);
128 }
129
130 // Set ALPN callback for HTTP/2
131 SSL_CTX_set_alpn_select_cb(
132 ssl_context_->native_handle(),
133 [](SSL* /*ssl*/, const unsigned char** out, unsigned char* outlen,
134 const unsigned char* in, unsigned int inlen, void* /*arg*/) -> int {
135 // Look for "h2" in client's ALPN list
136 const unsigned char* client = in;
137 const unsigned char* end = in + inlen;
138 while (client < end) {
139 unsigned char len = *client++;
140 if (len == 2 && client + 2 <= end &&
141 client[0] == 'h' && client[1] == '2') {
142 *out = client;
143 *outlen = 2;
144 return SSL_TLSEXT_ERR_OK;
145 }
146 client += len;
147 }
148 return SSL_TLSEXT_ERR_NOACK;
149 },
150 nullptr);
151
152 acceptor_ = std::make_unique<asio::ip::tcp::acceptor>(
153 *io_context_,
154 asio::ip::tcp::endpoint(asio::ip::tcp::v4(), port));
155
156 use_tls_ = true;
157 is_running_ = true;
158
159 // Start I/O thread
160 io_future_ = std::async(std::launch::async, [this]() { run_io(); });
161
162 // Start accepting connections
163 do_accept_tls();
164 start_cleanup_timer();
165
166 return ok();
167 } catch (const std::exception& e) {
168 return error_void(
169 error_codes::network_system::bind_failed,
170 std::string("Failed to start TLS server: ") + e.what(),
171 "http2_server");
172 }
173 }

175 auto http2_server::stop() -> VoidResult
176 {
177 if (!is_running_) {
178 return ok();
179 }
180
181 is_running_ = false;
182
183 // Close acceptor
184 if (acceptor_ && acceptor_->is_open()) {
185 std::error_code ec;
186 acceptor_->close(ec);
187 }
188
189 // Stop all connections
190 {
191 std::lock_guard<std::mutex> lock(connections_mutex_);
192 for (auto& [id, conn] : connections_) {
193 conn->stop();
194 }
195 connections_.clear();
196 }
197
198 // Stop cleanup timer
199 if (cleanup_timer_) {
200 cleanup_timer_->cancel();
201 }
202
203 stop_io();
204
205 // Signal stop
206 try {
207 stop_promise_.set_value();
208 } catch (...) {
209 // Already set
210 }
211
212 return ok();
213 }

292 auto http2_server::handle_accept(std::error_code ec, asio::ip::tcp::socket socket) -> void
293 {
294 if (!is_running_) {
295 return;
296 }
297
298 if (ec) {
299 if (error_handler_) {
300 error_handler_(std::string("Accept error: ") + ec.message());
301 }
302 } else {
303 uint64_t conn_id = next_connection_id_++;
304 auto conn = std::make_shared<http2_server_connection>(
305 conn_id,
306 std::move(socket),
307 settings_,
308 request_handler_,
309 error_handler_);
310
311 add_connection(conn);
312 conn->start();
313 }
314
315 // Continue accepting
316 do_accept();
317 }

319 auto http2_server::handle_accept_tls(std::error_code ec, asio::ip::tcp::socket socket) -> void
320 {
321 if (!is_running_) {
322 return;
323 }
324
325 if (ec) {
326 if (error_handler_) {
327 error_handler_(std::string("Accept error: ") + ec.message());
328 }
329 do_accept_tls();
330 return;
331 }
332
333 auto tls_socket = std::make_unique<asio::ssl::stream<asio::ip::tcp::socket>>(
334 std::move(socket), *ssl_context_);
335
336 auto* raw_socket = tls_socket.get();
337 raw_socket->async_handshake(
338 asio::ssl::stream_base::server,
339 [this, tls_socket = std::move(tls_socket)](std::error_code ec) mutable {
340 if (ec) {
341 if (error_handler_) {
342 error_handler_(std::string("TLS handshake error: ") + ec.message());
343 }
344 } else {
345 uint64_t conn_id = next_connection_id_++;
346 auto conn = std::make_shared<http2_server_connection>(
347 conn_id,
348 std::move(tls_socket),
349 settings_,
350 request_handler_,
351 error_handler_);
352
353 add_connection(conn);
354 conn->start();
355 }
356
357 // Continue accepting
358 do_accept_tls();
359 });
360 }

386 auto http2_server::start_cleanup_timer() -> void
387 {
388 if (!io_context_) {
389 return;
390 }
391
392 cleanup_timer_ = std::make_unique<asio::steady_timer>(*io_context_);
393 cleanup_timer_->expires_after(std::chrono::seconds(30));
394 cleanup_timer_->async_wait([this](std::error_code ec) {
395 if (!ec && is_running_) {
396 cleanup_dead_connections();
397 start_cleanup_timer();
398 }

511 auto http2_server_connection::read_connection_preface() -> void
512 {
513 constexpr size_t PREFACE_SIZE = 24; // "PRI * HTTP/2.0\r\n\r\nSM\r\n\r\n"
514 auto buffer = std::make_shared<std::vector<uint8_t>>(PREFACE_SIZE);
515
516 auto read_handler = [this, buffer](std::error_code ec, std::size_t bytes_read) {
517 if (ec || bytes_read != 24) {
518 if (error_handler_) {
519 error_handler_("Failed to read connection preface");
520 }
521 stop();
522 return;
523 }
524
525 // Verify preface
526 const std::string expected = "PRI * HTTP/2.0\r\n\r\nSM\r\n\r\n";
527 if (std::string(buffer->begin(), buffer->end()) != expected) {
528 if (error_handler_) {
529 error_handler_("Invalid connection preface");
530 }
531 stop();
532 return;
533 }
534
535 preface_received_ = true;
536
537 // Send our SETTINGS frame
538 auto result = send_settings();
539 if (result.is_err()) {
540 if (error_handler_) {
541 error_handler_("Failed to send settings");
542 }
543 stop();
544 return;
545 }
546
547 // Start reading frames
548 start_reading();
549 };
550
551 if (use_tls_) {
552 asio::async_read(*tls_socket_, asio::buffer(*buffer), read_handler);
553 } else {
554 asio::async_read(*plain_socket_, asio::buffer(*buffer), read_handler);
555 }
556 }

558 auto http2_server_connection::send_settings() -> VoidResult
559 {
560 std::vector<setting_parameter> params = {
561 {static_cast<uint16_t>(setting_identifier::max_concurrent_streams),
562 local_settings_.max_concurrent_streams},
563 {static_cast<uint16_t>(setting_identifier::initial_window_size),
564 local_settings_.initial_window_size},
565 {static_cast<uint16_t>(setting_identifier::max_frame_size),
566 local_settings_.max_frame_size},
567 {static_cast<uint16_t>(setting_identifier::header_table_size),
568 local_settings_.header_table_size},
569 };
570
571 settings_frame frame(params, false);
572 return send_frame(frame);
573 }

575 auto http2_server_connection::handle_settings_frame(const settings_frame& frame) -> VoidResult
576 {
577 if (frame.is_ack()) {
578 // Settings acknowledgment received
579 return ok();
580 }
581
582 // Apply peer settings
583 for (const auto& param : frame.settings()) {
584 switch (static_cast<setting_identifier>(param.identifier)) {
585 case setting_identifier::header_table_size:
586 remote_settings_.header_table_size = param.value;
587 encoder_.set_max_table_size(param.value);
588 break;
589 case setting_identifier::enable_push:
590 remote_settings_.enable_push = (param.value != 0);
591 break;
592 case setting_identifier::max_concurrent_streams:
593 remote_settings_.max_concurrent_streams = param.value;
594 break;
595 case setting_identifier::initial_window_size:
596 remote_settings_.initial_window_size = param.value;
597 break;
598 case setting_identifier::max_frame_size:
599 remote_settings_.max_frame_size = param.value;
600 break;
601 case setting_identifier::max_header_list_size:
602 remote_settings_.max_header_list_size = param.value;
603 break;
604 }
605 }
606
607 // Send ACK
608 return send_settings_ack();
609 }

622 auto http2_server_connection::read_frame_header() -> void
623 {
624 if (!is_alive_) {
625 return;
626 }
627
628 auto read_handler = [this](std::error_code ec, std::size_t bytes_read) {
629 if (ec || bytes_read != 9) {
630 if (ec != asio::error::eof && ec != asio::error::operation_aborted) {
631 if (error_handler_) {
632 error_handler_(std::string("Read error: ") + ec.message());
633 }
634 }
635 stop();
636 return;
637 }
638
639 auto header_result = frame_header::parse(frame_header_buffer_);
640 if (header_result.is_err()) {
641 if (error_handler_) {
642 error_handler_("Failed to parse frame header");
643 }
644 stop();
645 return;
646 }
647
648 auto header = header_result.value();
649 if (header.length > 0) {
650 read_frame_payload(header.length);
651 } else {
652 // Process frame with empty payload
653 auto frame_result = frame::parse(frame_header_buffer_);
654 if (frame_result.is_ok()) {
655 process_frame(std::move(frame_result.value()));
656 }
657 read_frame_header();
658 }
659 };
660
661 if (use_tls_) {
662 asio::async_read(*tls_socket_, asio::buffer(frame_header_buffer_), read_handler);
663 } else {
664 asio::async_read(*plain_socket_, asio::buffer(frame_header_buffer_), read_handler);
665 }
666 }
667
668 auto http2_server_connection::read_frame_payload(uint32_t length) -> void
669 {
670 if (!is_alive_) {
671 return;
672 }
673
674 read_buffer_.resize(9 + length);
675 std::copy(frame_header_buffer_.begin(), frame_header_buffer_.end(), read_buffer_.begin());
676
677 auto payload_buffer = asio::buffer(read_buffer_.data() + 9, length);
678
679 auto read_handler = [this](std::error_code ec, std::size_t /*bytes_read*/) {
680 if (ec) {
681 if (ec != asio::error::eof && ec != asio::error::operation_aborted) {
682 if (error_handler_) {
683 error_handler_(std::string("Read payload error: ") + ec.message());
684 }
685 }
686 stop();
687 return;
688 }
689
690 auto frame_result = frame::parse(read_buffer_);
691 if (frame_result.is_ok()) {
692 process_frame(std::move(frame_result.value()));
693 } else {
694 if (error_handler_) {
695 error_handler_("Failed to parse frame");
696 }
697 }
698
699 read_frame_header();
700 };
701
702 if (use_tls_) {
703 asio::async_read(*tls_socket_, payload_buffer, read_handler);
704 } else {
705 asio::async_read(*plain_socket_, payload_buffer, read_handler);
706 }
707 }

730 auto http2_server_connection::process_frame(std::unique_ptr<frame> f) -> VoidResult
731 {
732 auto& header = f->header();
733
734 switch (header.type) {
735 case frame_type::settings: {
736 auto* settings_f = dynamic_cast<settings_frame*>(f.get());
737 if (settings_f) {
738 return handle_settings_frame(*settings_f);
739 }
740 break;
741 }
742 case frame_type::headers: {
743 auto* headers_f = dynamic_cast<headers_frame*>(f.get());
744 if (headers_f) {
745 return handle_headers_frame(*headers_f);
746 }
747 break;
748 }
749 case frame_type::data: {
750 auto* data_f = dynamic_cast<data_frame*>(f.get());
751 if (data_f) {
752 return handle_data_frame(*data_f);
753 }
754 break;
755 }
756 case frame_type::rst_stream: {
757 auto* rst_f = dynamic_cast<rst_stream_frame*>(f.get());
758 if (rst_f) {
759 return handle_rst_stream_frame(*rst_f);
760 }
761 break;
762 }
763 case frame_type::ping: {
764 auto* ping_f = dynamic_cast<ping_frame*>(f.get());
765 if (ping_f) {
766 return handle_ping_frame(*ping_f);
767 }
768 break;
769 }
770 case frame_type::goaway: {
771 auto* goaway_f = dynamic_cast<goaway_frame*>(f.get());
772 if (goaway_f) {
773 return handle_goaway_frame(*goaway_f);
774 }
775 break;
776 }
777 case frame_type::window_update: {
778 auto* wu_f = dynamic_cast<window_update_frame*>(f.get());
779 if (wu_f) {
780 return handle_window_update_frame(*wu_f);
781 }
782 break;
783 }
784 default:
785 // Ignore unknown frame types
786 break;
787 }
788
789 return ok();
790 }

792 auto http2_server_connection::get_or_create_stream(uint32_t stream_id) -> http2_stream*
793 {
794 std::lock_guard<std::mutex> lock(streams_mutex_);
795
796 auto it = streams_.find(stream_id);
797 if (it != streams_.end()) {
798 return &it->second;
799 }
800
801 // Create new stream
802 http2_stream stream;
803 stream.stream_id = stream_id;
804 stream.state = stream_state::open;
805 stream.window_size = static_cast<int32_t>(local_settings_.initial_window_size);
806
807 auto [iter, inserted] = streams_.emplace(stream_id, std::move(stream));

821 auto http2_server_connection::handle_headers_frame(const headers_frame& f) -> VoidResult
822 {
823 uint32_t stream_id = f.header().stream_id;
824 auto* stream = get_or_create_stream(stream_id);
825
826 if (!stream) {
827 return error_void(
828 error_codes::common_errors::internal_error,
829 "Failed to create stream",
830 "http2_server_connection");
831 }
832
833 // Decode headers
834 auto header_block = f.header_block();
835 std::vector<uint8_t> block_data(header_block.begin(), header_block.end());
836 auto headers_result = decoder_.decode(block_data);
837
838 if (headers_result.is_err()) {
839 // Send GOAWAY with COMPRESSION_ERROR
840 goaway_frame goaway(last_stream_id_,
841 static_cast<uint32_t>(error_code::compression_error));
842 send_frame(goaway);
843 stop();
844 return error_void(
845 headers_result.error().code,
846 headers_result.error().message,
847 "http2_server_connection");
848 }
849
850 stream->request_headers = headers_result.value();
851
852 if (f.is_end_stream()) {
853 stream->state = stream_state::half_closed_remote;
854 dispatch_request(stream_id);
855 } else if (f.is_end_headers()) {
856 stream->headers_complete = true;
857 }
858
859 return ok();
860 }

862 auto http2_server_connection::handle_data_frame(const data_frame& f) -> VoidResult
863 {
864 uint32_t stream_id = f.header().stream_id;
865
866 std::lock_guard<std::mutex> lock(streams_mutex_);
867 auto it = streams_.find(stream_id);
868 if (it == streams_.end()) {
869 // Stream doesn't exist
870 rst_stream_frame rst(stream_id, static_cast<uint32_t>(error_code::stream_closed));
871 return send_frame(rst);
872 }
873
874 auto& stream = it->second;
875 auto data = f.data();
876 stream.request_body.insert(stream.request_body.end(), data.begin(), data.end());
877
878 // Send WINDOW_UPDATE if needed
879 size_t data_size = data.size();
880 if (data_size > 0) {
881 // Update connection window
882 window_update_frame conn_wu(0, static_cast<uint32_t>(data_size));
883 send_frame(conn_wu);
884
885 // Update stream window
886 window_update_frame stream_wu(stream_id, static_cast<uint32_t>(data_size));
887 send_frame(stream_wu);
888 }
889
890 if (f.is_end_stream()) {
891 stream.state = stream_state::half_closed_remote;
892 stream.body_complete = true;
893
894 // Need to unlock before dispatching
895 lock.~lock_guard();
896 dispatch_request(stream_id);
897 }
898
899 return ok();
900 }

927 {
928 uint32_t stream_id = f.header().stream_id;
929 int32_t increment = static_cast<int32_t>(f.window_size_increment());
930

932 // Connection-level flow control
933 connection_window_size_ += increment;
934 } else {

937 auto it = streams_.find(stream_id);
938 if (it != streams_.end()) {
939 it->second.window_size += increment;

946 auto http2_server_connection::dispatch_request(uint32_t stream_id) -> void
947 {
948 if (!request_handler_) {
949 return;
950 }
951
952 http2_stream* stream = nullptr;
953 {
954 std::lock_guard<std::mutex> lock(streams_mutex_);
955 auto it = streams_.find(stream_id);
956 if (it == streams_.end()) {
957 return;
958 }
959 stream = &it->second;
960 }
961
962 // Create request from headers
963 auto request = http2_request::from_headers(stream->request_headers);
964 request.body = stream->request_body;
965
966 // Create server stream for response
967 auto encoder_ptr = std::make_shared<hpack_encoder>(encoder_);
968 auto weak_this = weak_from_this();
969
970 http2_server_stream server_stream(
971 stream_id,
972 std::move(request),
973 encoder_ptr,
974 [weak_this](const frame& f) -> VoidResult {
975 auto self = weak_this.lock();
976 if (!self) {
977 return error_void(
978 error_codes::common_errors::internal_error,
979 "Connection closed",
980 "http2_server_stream");
981 }
982 return self->send_frame(f);
983 },
984 remote_settings_.max_frame_size);
985
986 // Call request handler
987 try {
988 request_handler_(server_stream, server_stream.request());
989 } catch (const std::exception& e) {
990 if (error_handler_) {
991 error_handler_(std::string("Request handler exception: ") + e.what());
992 }
993 }
994
995 // Close stream after handler completes
996 close_stream(stream_id);
997 }
