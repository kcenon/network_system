45 auto http2_response::get_header(const std::string& name) const -> std::optional<std::string>
46 {
47 std::string lower_name = name;
48 std::transform(lower_name.begin(), lower_name.end(), lower_name.begin(),
49 [](unsigned char c) { return std::tolower(c); });
50
51 for (const auto& header : headers)
52 {
53 std::string lower_header_name = header.name;
54 std::transform(lower_header_name.begin(), lower_header_name.end(),
55 lower_header_name.begin(),
56 [](unsigned char c) { return std::tolower(c); });
57
58 if (lower_header_name == lower_name)
59 {
60 return header.value;
61 }
62 }
63 return std::nullopt;
64 }

94 auto http2_client::connect(const std::string& host, unsigned short port) -> VoidResult
95 {
96 // Create tracing span for connect operation
97 auto span = tracing::is_tracing_enabled()
98 ? std::make_optional(tracing::trace_context::create_span("http2.client.connect"))
99 : std::nullopt;
100 if (span)
101 {
102 span->set_attribute("net.peer.name", host)
103 .set_attribute("net.peer.port", static_cast<int64_t>(port))
104 .set_attribute("net.transport", "tcp")
105 .set_attribute("http.flavor", "2.0")
106 .set_attribute("client.id", client_id_);
107 }
108
109 if (is_connected_)
110 {
111 if (span)
112 {
113 span->set_error("Already connected");
114 }
115 return error_void(error_codes::common_errors::already_exists,
116 "Already connected", "http2_client::connect");
117 }
118
119 if (host.empty())
120 {
121 if (span)
122 {
123 span->set_error("Host cannot be empty");
124 }
125 return error_void(error_codes::common_errors::invalid_argument,
126 "Host cannot be empty", "http2_client::connect");
127 }
128
129 host_ = host;
130 port_ = port;
131
132 try
133 {
134 // Create I/O context
135 io_context_ = std::make_unique<asio::io_context>();
136
137 // Create SSL context with TLS 1.3
138 ssl_context_ = std::make_unique<asio::ssl::context>(asio::ssl::context::tlsv13_client);
139
140 // Set ALPN for HTTP/2
141 SSL_CTX* native_ctx = ssl_context_->native_handle();
142 static const unsigned char alpn_protos[] = { 2, 'h', '2' };
143 SSL_CTX_set_alpn_protos(native_ctx, alpn_protos, sizeof(alpn_protos));
144
145 // Verify peer certificate
146 ssl_context_->set_default_verify_paths();
147 ssl_context_->set_verify_mode(asio::ssl::verify_peer);
148
149 // Create socket
150 asio::ip::tcp::resolver resolver(*io_context_);
151 auto endpoints = resolver.resolve(host, std::to_string(port));
152
153 socket_ = std::make_unique<asio::ssl::stream<asio::ip::tcp::socket>>(
154 *io_context_, *ssl_context_);
155
156 // Set SNI hostname
157 SSL_set_tlsext_host_name(socket_->native_handle(), host.c_str());
158
159 // Connect TCP
160 asio::connect(socket_->lowest_layer(), endpoints);
161
162 // Perform SSL handshake
163 socket_->handshake(asio::ssl::stream_base::client);
164
165 // Verify ALPN negotiation
166 const unsigned char* alpn_result = nullptr;
167 unsigned int alpn_len = 0;
168 SSL_get0_alpn_selected(socket_->native_handle(), &alpn_result, &alpn_len);
169
170 if (alpn_len != 2 || alpn_result == nullptr ||
171 std::string(reinterpret_cast<const char*>(alpn_result), alpn_len) != "h2")
172 {
173 socket_->lowest_layer().close();
174 if (span)
175 {
176 span->set_error("Server does not support HTTP/2 via ALPN");
177 }
178 return error_void(error_codes::network_system::connection_failed,
179 "Server does not support HTTP/2 via ALPN",
180 "http2_client::connect");
181 }
182
183 is_connected_ = true;
184 is_running_ = true;
185
186 // Send connection preface
187 auto preface_result = send_connection_preface();
188 if (preface_result.is_err())
189 {
190 is_connected_ = false;
191 is_running_ = false;
192 if (span)
193 {
194 span->set_error(preface_result.error().message);
195 }
196 return preface_result;
197 }
198
199 // Send initial SETTINGS
200 auto settings_result = send_settings();
201 if (settings_result.is_err())
202 {
203 is_connected_ = false;
204 is_running_ = false;
205 if (span)
206 {
207 span->set_error(settings_result.error().message);
208 }
209 return settings_result;
210 }
211
212 // Start I/O thread
213 work_guard_ = std::make_unique<asio::executor_work_guard<asio::io_context::executor_type>>(
214 asio::make_work_guard(*io_context_));
215
216 io_future_ = std::async(std::launch::async, [this]() { run_io(); });
217
218 if (span)
219 {
220 span->set_status(tracing::span_status::ok);
221 }
222 return ok();
223 }
224 catch (const std::exception& e)
225 {
226 is_connected_ = false;
227 is_running_ = false;
228 if (span)
229 {
230 span->set_error(std::string("Connection failed: ") + e.what());
231 }
232 return error_void(error_codes::network_system::connection_failed,
233 std::string("Connection failed: ") + e.what(),
234 "http2_client::connect");
235 }
236 }

238 auto http2_client::disconnect() -> VoidResult
239 {
240 if (!is_connected_)
241 {
242 return ok();
243 }
244
245 try
246 {
247 // Send GOAWAY frame
248 if (socket_ && socket_->lowest_layer().is_open())
249 {
250 goaway_frame goaway(next_stream_id_ - 2,
251 static_cast<uint32_t>(error_code::no_error));
252 send_frame(goaway);
253 }
254 }
255 catch (...)
256 {
257 // Ignore errors during GOAWAY
258 }
259
260 stop_io();
261
262 is_connected_ = false;
263
264 return ok();
265 }

334 auto http2_client::start_stream(const std::string& path,
335 const std::vector<http_header>& headers,
336 std::function<void(std::vector<uint8_t>)> on_data,
337 std::function<void(std::vector<http_header>)> on_headers,
338 std::function<void(int)> on_complete)
339 -> Result<uint32_t>
340 {
341 if (!is_connected())
342 {
343 return error<uint32_t>(error_codes::network_system::connection_closed,
344 "Not connected",
345 "http2_client::start_stream");
346 }
347
348 // Create stream with callbacks
349 http2_stream& stream = create_stream();
350 stream.state = stream_state::open;
351 stream.is_streaming = true;
352 stream.on_data = std::move(on_data);
353 stream.on_headers = std::move(on_headers);
354 stream.on_complete = std::move(on_complete);
355
356 // Build headers for POST (streaming requests use POST)
357 auto request_headers = build_headers("POST", path, headers);
358 stream.request_headers = request_headers;
359
360 // Encode headers with HPACK
361 auto encoded_headers = encoder_.encode(request_headers);
362
363 // Send HEADERS frame (not end_stream since we might send more data)
364 headers_frame hf(stream.stream_id, std::move(encoded_headers), false, true);
365
366 auto send_result = send_frame(hf);
367 if (send_result.is_err())
368 {
369 close_stream(stream.stream_id);
370 const auto& err = send_result.error();
371 return error<uint32_t>(err.code, err.message,
372 "http2_client::start_stream",
373 get_error_details(err));
374 }
375
376 uint32_t result_id = stream.stream_id;
377 return ok(std::move(result_id));
378 }

380 auto http2_client::write_stream(uint32_t stream_id,
381 const std::vector<uint8_t>& data,
382 bool end_stream) -> VoidResult
383 {
384 if (!is_connected())
385 {
386 return error_void(error_codes::network_system::connection_closed,
387 "Not connected",
388 "http2_client::write_stream");
389 }
390
391 auto* stream = get_stream(stream_id);
392 if (!stream)
393 {
394 return error_void(error_codes::common_errors::not_found,
395 "Stream not found",
396 "http2_client::write_stream");
397 }
398
399 if (stream->state == stream_state::closed ||
400 stream->state == stream_state::half_closed_local)
401 {
402 return error_void(error_codes::common_errors::internal_error,
403 "Stream not writable",
404 "http2_client::write_stream");
405 }
406
407 // Send DATA frame
408 data_frame df(stream_id, std::vector<uint8_t>(data), end_stream);
409 auto send_result = send_frame(df);
410 if (send_result.is_err())
411 {
412 return send_result;
413 }
414
415 if (end_stream)
416 {
417 stream->state = stream_state::half_closed_local;
418 }
419
420 return ok();
421 }

423 auto http2_client::close_stream_writer(uint32_t stream_id) -> VoidResult
424 {
425 if (!is_connected())
426 {
427 return error_void(error_codes::network_system::connection_closed,
428 "Not connected",
429 "http2_client::close_stream_writer");
430 }
431
432 auto* stream = get_stream(stream_id);
433 if (!stream)
434 {
435 return error_void(error_codes::common_errors::not_found,
436 "Stream not found",
437 "http2_client::close_stream_writer");
438 }
439
440 if (stream->state == stream_state::closed ||
441 stream->state == stream_state::half_closed_local)
442 {
443 return ok(); // Already closed
444 }
445
446 // Send empty DATA frame with END_STREAM
447 data_frame df(stream_id, {}, true);
448 auto send_result = send_frame(df);
449 if (send_result.is_err())
450 {
451 return send_result;
452 }
453
454 stream->state = stream_state::half_closed_local;
455 return ok();
456 }

458 auto http2_client::cancel_stream(uint32_t stream_id) -> VoidResult
459 {
460 if (!is_connected())
461 {
462 return error_void(error_codes::network_system::connection_closed,
463 "Not connected",
464 "http2_client::cancel_stream");
465 }
466

485 return send_result;
486 }
487
488 stream->state = stream_state::closed;
489 return ok();
490 }

524 local_settings_.max_header_list_size}
525 };
526
527 settings_frame settings(params, false);
528 return send_frame(settings);
529 }

531 auto http2_client::handle_settings_frame(const settings_frame& frame) -> VoidResult
532 {
533 if (frame.is_ack())
534 {
535 // Our settings were acknowledged
536 return ok();
537 }
538
539 // Apply remote settings
540 for (const auto& param : frame.settings())
541 {
542 switch (static_cast<setting_identifier>(param.identifier))
543 {
544 case setting_identifier::header_table_size:
545 remote_settings_.header_table_size = param.value;
546 encoder_.set_max_table_size(param.value);
547 break;
548 case setting_identifier::enable_push:
549 remote_settings_.enable_push = (param.value != 0);
550 break;
551 case setting_identifier::max_concurrent_streams:
552 remote_settings_.max_concurrent_streams = param.value;
553 break;
554 case setting_identifier::initial_window_size:
555 remote_settings_.initial_window_size = param.value;
556 break;
557 case setting_identifier::max_frame_size:
558 remote_settings_.max_frame_size = param.value;
559 break;
560 case setting_identifier::max_header_list_size:
561 remote_settings_.max_header_list_size = param.value;
562 break;
563 }
564 }
565
566 return send_settings_ack();
567 }

575 auto http2_client::send_frame(const frame& f) -> VoidResult
576 {
577 if (!socket_ || !socket_->lowest_layer().is_open())
578 {
579 return error_void(error_codes::network_system::connection_closed,
580 "Connection closed", "http2_client::send_frame");

588 }
589 catch (const std::exception& e)

593 "http2_client::send_frame");

597 auto http2_client::read_frame() -> Result<std::unique_ptr<frame>>
598 {
599 if (!socket_ || !socket_->lowest_layer().is_open())
600 {
601 return error<std::unique_ptr<frame>>(error_codes::network_system::connection_closed,
602 "Connection closed",
603 "http2_client::read_frame");
604 }
605
606 try
607 {
608 // Read frame header (9 bytes)
609 std::vector<uint8_t> header_buf(FRAME_HEADER_SIZE);
610 asio::read(*socket_, asio::buffer(header_buf));
611
612 auto header_result = frame_header::parse(header_buf);
613 if (header_result.is_err())
614 {
615 const auto& err = header_result.error();
616 return error<std::unique_ptr<frame>>(err.code, err.message,
617 "http2_client::read_frame",
618 get_error_details(err));
619 }
620
621 const auto& header = header_result.value();
622
623 // Read frame payload
624 std::vector<uint8_t> payload(header.length);
625 if (header.length > 0)
626 {
627 asio::read(*socket_, asio::buffer(payload));
628 }
629
630 // Combine header and payload for parsing
631 std::vector<uint8_t> frame_data;
632 frame_data.reserve(FRAME_HEADER_SIZE + payload.size());
633 frame_data.insert(frame_data.end(), header_buf.begin(), header_buf.end());
634 frame_data.insert(frame_data.end(), payload.begin(), payload.end());
635
636 return frame::parse(frame_data);
637 }
638 catch (const std::exception& e)
639 {
640 return error<std::unique_ptr<frame>>(error_codes::network_system::receive_failed,
641 std::string("Failed to read frame: ") + e.what(),
642 "http2_client::read_frame");
643 }
644 }

646 auto http2_client::process_frame(std::unique_ptr<frame> f) -> VoidResult
647 {
648 if (!f)
649 {
650 return error_void(error_codes::common_errors::invalid_argument,
651 "Null frame", "http2_client::process_frame");
652 }
653
654 switch (f->header().type)
655 {
656 case frame_type::settings:
657 if (auto* sf = dynamic_cast<settings_frame*>(f.get()))
658 {
659 return handle_settings_frame(*sf);
660 }
661 break;
662
663 case frame_type::headers:
664 if (auto* hf = dynamic_cast<headers_frame*>(f.get()))
665 {
666 return handle_headers_frame(*hf);
667 }
668 break;
669
670 case frame_type::data:
671 if (auto* df = dynamic_cast<data_frame*>(f.get()))
672 {
673 return handle_data_frame(*df);
674 }
675 break;
676
677 case frame_type::rst_stream:
678 if (auto* rf = dynamic_cast<rst_stream_frame*>(f.get()))
679 {
680 return handle_rst_stream_frame(*rf);
681 }
682 break;
683
684 case frame_type::goaway:
685 if (auto* gf = dynamic_cast<goaway_frame*>(f.get()))
686 {
687 return handle_goaway_frame(*gf);
688 }
689 break;
690
691 case frame_type::window_update:
692 if (auto* wf = dynamic_cast<window_update_frame*>(f.get()))
693 {
694 return handle_window_update_frame(*wf);
695 }
696 break;
697
698 case frame_type::ping:
699 if (auto* pf = dynamic_cast<ping_frame*>(f.get()))
700 {
701 return handle_ping_frame(*pf);
702 }
703 break;
704
705 default:
706 // Ignore unknown frame types
707 break;
708 }
709
710 return ok();
711 }

751 auto http2_client::send_request(const std::string& method,
752 const std::string& path,
753 const std::vector<http_header>& headers,
754 const std::vector<uint8_t>& body)
755 -> Result<http2_response>
756 {
757 // Create tracing span for HTTP request
758 auto span = tracing::is_tracing_enabled()
759 ? std::make_optional(tracing::trace_context::create_span("http2.request"))
760 : std::nullopt;
761 if (span)
762 {
763 span->set_attribute("http.method", method)
764 .set_attribute("http.url", "https://" + host_ + ":" + std::to_string(port_) + path)
765 .set_attribute("http.flavor", "2.0")
766 .set_attribute("http.request.body_size", static_cast<int64_t>(body.size()))
767 .set_attribute("net.peer.name", host_)
768 .set_attribute("net.peer.port", static_cast<int64_t>(port_));
769 }
770
771 if (!is_connected())
772 {
773 if (span)
774 {
775 span->set_error("Not connected");
776 }
777 return error<http2_response>(error_codes::network_system::connection_closed,
778 "Not connected",
779 "http2_client::send_request");
780 }
781
782 // Create stream
783 http2_stream& stream = create_stream();
784 stream.state = stream_state::open;
785
786 if (span)
787 {
788 span->set_attribute("http2.stream_id", static_cast<int64_t>(stream.stream_id));
789 }
790
791 // Build headers
792 auto request_headers = build_headers(method, path, headers);
793 stream.request_headers = request_headers;
794 stream.request_body = body;
795
796 // Encode headers with HPACK
797 auto encoded_headers = encoder_.encode(request_headers);
798
799 // Send HEADERS frame
800 bool end_stream = body.empty();
801 headers_frame hf(stream.stream_id, std::move(encoded_headers), end_stream, true);
802
803 auto send_result = send_frame(hf);
804 if (send_result.is_err())
805 {
806 close_stream(stream.stream_id);
807 const auto& err = send_result.error();
808 if (span)
809 {
810 span->set_error(err.message);
811 }
812 return error<http2_response>(err.code, err.message,
813 "http2_client::send_request",
814 get_error_details(err));
815 }
816
817 // Send DATA frame if body exists
818 if (!body.empty())
819 {
820 data_frame df(stream.stream_id, std::vector<uint8_t>(body), true);
821 send_result = send_frame(df);
822 if (send_result.is_err())
823 {
824 close_stream(stream.stream_id);
825 const auto& err = send_result.error();
826 if (span)
827 {
828 span->set_error(err.message);
829 }
830 return error<http2_response>(err.code, err.message,
831 "http2_client::send_request",
832 get_error_details(err));
833 }
834 stream.state = stream_state::half_closed_local;
835 }
836 else
837 {
838 stream.state = stream_state::half_closed_local;
839 }
840
841 // Wait for response with timeout
842 auto future = stream.promise.get_future();
843 auto status = future.wait_for(timeout_);
844
845 if (status == std::future_status::timeout)
846 {
847 close_stream(stream.stream_id);
848 if (span)
849 {
850 span->set_error("Request timeout");
851 }
852 return error<http2_response>(error_codes::common_errors::timeout,
853 "Request timeout",
854 "http2_client::send_request");
855 }
856
857 auto response = future.get();
858 if (span)
859 {
860 span->set_attribute("http.status_code", static_cast<int64_t>(response.status_code))
861 .set_attribute("http.response.body_size", static_cast<int64_t>(response.body.size()));
862 if (response.status_code >= 400)
863 {
864 span->set_status(tracing::span_status::error, "HTTP " + std::to_string(response.status_code));
865 }
866 else
867 {
868 span->set_status(tracing::span_status::ok);
869 }
870 }
871 return ok(std::move(response));
872 }

874 auto http2_client::build_headers(const std::string& method,
875 const std::string& path,
876 const std::vector<http_header>& additional)
877 -> std::vector<http_header>
878 {
879 std::vector<http_header> headers;
880
881 // Pseudo-headers (must come first)
882 headers.emplace_back(":method", method);
883 headers.emplace_back(":scheme", "https");
884 headers.emplace_back(":authority", host_);
885 headers.emplace_back(":path", path.empty() ? "/" : path);
886
887 // Standard headers
888 headers.emplace_back("user-agent", "network_system/http2_client");
889
890 // Additional headers
891 for (const auto& h : additional)
892 {
893 // Skip pseudo-headers in additional
894 if (!h.name.empty() && h.name[0] != ':')
895 {
896 headers.push_back(h);
897 }
898 }
899
900 return headers;
901 }

903 auto http2_client::handle_headers_frame(const headers_frame& f) -> VoidResult
904 {
905 uint32_t stream_id = f.header().stream_id;
906 auto* stream = get_stream(stream_id);
907
908 if (!stream)
909 {
910 return error_void(error_codes::common_errors::not_found,
911 "Unknown stream ID",
912 "http2_client::handle_headers_frame");
913 }
914
915 // Decode headers
916 auto decode_result = decoder_.decode(f.header_block());
917 if (decode_result.is_err())
918 {
919 const auto& err = decode_result.error();
920 return error_void(err.code, err.message,
921 "http2_client::handle_headers_frame",
922 get_error_details(err));
923 }
924
925 auto& decoded_headers = decode_result.value();
926
927 // Append to stream headers
928 stream->response_headers.insert(stream->response_headers.end(),
929 decoded_headers.begin(),
930 decoded_headers.end());
931
932 if (f.is_end_headers())
933 {
934 stream->headers_complete = true;
935
936 // Call streaming callback if set
937 if (stream->is_streaming && stream->on_headers)
938 {
939 stream->on_headers(stream->response_headers);
940 }
941 }
942
943 if (f.is_end_stream())
944 {
945 stream->body_complete = true;
946 stream->state = stream_state::closed;
947
948 // Extract status code
949 int status_code = 0;
950 for (const auto& h : stream->response_headers)
951 {
952 if (h.name == ":status")
953 {
954 try
955 {
956 status_code = std::stoi(h.value);
957 }
958 catch (...)
959 {
960 status_code = 0;
961 }
962 break;
963 }
964 }
965
966 // Handle streaming vs non-streaming
967 if (stream->is_streaming)
968 {
969 if (stream->on_complete)
970 {
971 stream->on_complete(status_code);
972 }
973 }
974 else
975 {
976 // Build response for non-streaming
977 http2_response response;
978 response.headers = stream->response_headers;
979 response.body = stream->response_body;
980 response.status_code = status_code;
981 stream->promise.set_value(std::move(response));
982 }
983 }
984
985 return ok();
986 }

988 auto http2_client::handle_data_frame(const data_frame& f) -> VoidResult
989 {
990 uint32_t stream_id = f.header().stream_id;
991 auto* stream = get_stream(stream_id);
992
993 if (!stream)
994 {
995 return error_void(error_codes::common_errors::not_found,
996 "Unknown stream ID",
997 "http2_client::handle_data_frame");
998 }
999
1000 // Get data
1001 auto data = f.data();
1002
1003 // For streaming, call callback immediately; otherwise buffer
1004 if (stream->is_streaming && stream->on_data)
1005 {
1006 stream->on_data(std::vector<uint8_t>(data.begin(), data.end()));
1007 }
1008 else
1009 {
1010 stream->response_body.insert(stream->response_body.end(),
1011 data.begin(), data.end());
1012 }
1013
1014 // Update flow control window
1015 stream->window_size -= static_cast<int32_t>(data.size());
1016 connection_window_size_ -= static_cast<int32_t>(data.size());
1017
1018 // Send WINDOW_UPDATE if needed
1019 if (stream->window_size < static_cast<int32_t>(DEFAULT_WINDOW_SIZE / 2))
1020 {
1021 int32_t increment = DEFAULT_WINDOW_SIZE - stream->window_size;
1022 window_update_frame wuf(stream_id, increment);
1023 send_frame(wuf);
1024 stream->window_size += increment;
1025 }
1026
1027 if (connection_window_size_ < static_cast<int32_t>(DEFAULT_WINDOW_SIZE / 2))
1028 {
1029 int32_t increment = DEFAULT_WINDOW_SIZE - connection_window_size_;
1030 window_update_frame wuf(0, increment); // Stream 0 = connection level
1031 send_frame(wuf);
1032 connection_window_size_ += increment;
1033 }
1034
1035 if (f.is_end_stream())
1036 {
1037 stream->body_complete = true;
1038 stream->state = stream_state::closed;
1039
1040 // Extract status code
1041 int status_code = 0;
1042 for (const auto& h : stream->response_headers)
1043 {
1044 if (h.name == ":status")
1045 {
1046 try
1047 {
1048 status_code = std::stoi(h.value);
1049 }
1050 catch (...)
1051 {
1052 status_code = 0;
1053 }
1054 break;
1055 }
1056 }
1057
1058 // Handle streaming vs non-streaming
1059 if (stream->is_streaming)
1060 {
1061 if (stream->on_complete)
1062 {
1063 stream->on_complete(status_code);
1064 }
1065 }
1066 else
1067 {
1068 http2_response response;
1069 response.headers = stream->response_headers;
1070 response.body = stream->response_body;
1071 response.status_code = status_code;
1072 stream->promise.set_value(std::move(response));
1073 }
1074 }
1075
1076 return ok();
1077 }
