107auto sliding_histogram::mean() const -> double
108{
109 uint64_t c = count();

111 {

113 }

117auto sliding_histogram::percentile(double p) const -> double
118{
119 auto snap = aggregate();
120 if (snap.count == 0)
121 {
122 return 0.0;
123 }
124
125 // Clamp percentile
126 p = std::clamp(p, 0.0, 1.0);
127
128 // Use pre-calculated percentiles if available
129 auto it = snap.percentiles.find(p);
130 if (it != snap.percentiles.end())
131 {
132 return it->second;
133 }
134
135 // Calculate from buckets
136 double target = p * static_cast<double>(snap.count);
137
138 for (size_t i = 0; i < snap.buckets.size(); ++i)
139 {
140 if (static_cast<double>(snap.buckets[i].second) >= target)
141 {
142 double lower_bound = (i == 0) ? 0.0 : snap.buckets[i - 1].first;
143 double upper_bound = snap.buckets[i].first;
144
145 if (std::isinf(upper_bound))
146 {
147 return lower_bound;
148 }
149
150 uint64_t lower_count = (i == 0) ? 0 : snap.buckets[i - 1].second;
151 uint64_t upper_count = snap.buckets[i].second;
152
153 if (upper_count == lower_count)
154 {
155 return lower_bound;
156 }
157
158 double fraction = (target - static_cast<double>(lower_count))
159 / (static_cast<double>(upper_count) - static_cast<double>(lower_count));
160
161 return lower_bound + (fraction * (upper_bound - lower_bound));
162 }
163 }
164
165 return 0.0;
166}

182void sliding_histogram::expire_old_buckets()
183{
184 auto now = std::chrono::steady_clock::now();
185 auto cutoff = now - config_.window_duration;
186
187 while (!buckets_.empty() && buckets_.front()->start_time < cutoff)
188 {
189 buckets_.pop_front();
190 }
191}

193auto sliding_histogram::get_current_bucket() -> time_bucket&
194{
195 auto now = std::chrono::steady_clock::now();
196
197 if (buckets_.empty())
198 {
199 buckets_.push_back(std::make_unique<time_bucket>(config_.hist_config));
200 return *buckets_.back();
201 }
202
203 auto& last = buckets_.back();
204 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(now - last->start_time);
205
206 if (elapsed >= bucket_duration_)
207 {
208 buckets_.push_back(std::make_unique<time_bucket>(config_.hist_config));
209 }
210
211 return *buckets_.back();
212}

214auto sliding_histogram::aggregate() const -> histogram_snapshot
215{
216 std::lock_guard<std::mutex> lock(mutex_);
217 const_cast<sliding_histogram*>(this)->expire_old_buckets();
218
219 histogram_snapshot result;
220 result.count = 0;
221 result.sum = 0.0;
222 result.min_value = std::numeric_limits<double>::infinity();
223 result.max_value = -std::numeric_limits<double>::infinity();
224
225 if (buckets_.empty())
226 {
227 return result;
228 }
229
230 // Aggregate statistics
231 std::vector<std::pair<double, uint64_t>> merged_buckets;
232 bool first_snapshot = true;
233
234 for (const auto& bucket : buckets_)
235 {
236 auto snap = bucket->hist.snapshot();
237 result.count += snap.count;
238 result.sum += snap.sum;
239
240 if (snap.min_value < result.min_value)
241 {
242 result.min_value = snap.min_value;
243 }
244 if (snap.max_value > result.max_value)
245 {
246 result.max_value = snap.max_value;
247 }
248
249 // Merge bucket counts
250 if (first_snapshot)
251 {
252 merged_buckets = snap.buckets;
253 first_snapshot = false;
254 }
255 else
256 {
257 // Need to convert from cumulative to individual counts, add, then back to cumulative
258 std::vector<uint64_t> individual_counts(snap.buckets.size(), 0);
259 for (size_t i = 0; i < snap.buckets.size(); ++i)
260 {
261 uint64_t prev = (i == 0) ? 0 : snap.buckets[i - 1].second;
262 individual_counts[i] = snap.buckets[i].second - prev;
263 }
264
265 std::vector<uint64_t> merged_individual(merged_buckets.size(), 0);
266 for (size_t i = 0; i < merged_buckets.size(); ++i)
267 {
268 uint64_t prev = (i == 0) ? 0 : merged_buckets[i - 1].second;
269 merged_individual[i] = merged_buckets[i].second - prev;
270 }
271
272 // Add individual counts
273 for (size_t i = 0; i < merged_buckets.size() && i < individual_counts.size(); ++i)
274 {
275 merged_individual[i] += individual_counts[i];
276 }
277
278 // Convert back to cumulative
279 uint64_t cumulative = 0;
280 for (size_t i = 0; i < merged_buckets.size(); ++i)
281 {
282 cumulative += merged_individual[i];
283 merged_buckets[i].second = cumulative;
284 }
285 }
286 }
287
288 result.buckets = std::move(merged_buckets);
289
290 // Calculate percentiles from merged buckets
291 if (result.count > 0 && !result.buckets.empty())
292 {
293 auto calc_percentile = [&](double p) -> double
294 {
295 double target = p * static_cast<double>(result.count);
296
297 for (size_t i = 0; i < result.buckets.size(); ++i)
298 {
299 if (static_cast<double>(result.buckets[i].second) >= target)
300 {
301 double lower_bound = (i == 0) ? 0.0 : result.buckets[i - 1].first;
302 double upper_bound = result.buckets[i].first;
303
304 if (std::isinf(upper_bound))
305 {
306 return lower_bound;
307 }
308
309 uint64_t lower_count = (i == 0) ? 0 : result.buckets[i - 1].second;
310 uint64_t upper_count = result.buckets[i].second;
311
312 if (upper_count == lower_count)
313 {
314 return lower_bound;
315 }
316
317 double fraction =
318 (target - static_cast<double>(lower_count))
319 / (static_cast<double>(upper_count) - static_cast<double>(lower_count));
320
321 return lower_bound + (fraction * (upper_bound - lower_bound));
322 }
323 }
324 return 0.0;
325 };
326
327 result.percentiles[0.5] = calc_percentile(0.5);
328 result.percentiles[0.9] = calc_percentile(0.9);
329 result.percentiles[0.95] = calc_percentile(0.95);
330 result.percentiles[0.99] = calc_percentile(0.99);
331 result.percentiles[0.999] = calc_percentile(0.999);
332 }
333
334 return result;
335}
