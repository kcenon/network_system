241 [[nodiscard]] auto is_valid() const -> bool {
242 if (!enabled) {
243 return true; // Valid if disabled
244 }
245
246 // If verification is enabled, CA file/path is required
247 if (verify_mode != certificate_verification::none) {
248 if (!ca_file.has_value() && !ca_path.has_value()) {
249 return false;
250 }
251 }
252
253 // Note: Certificate and private key validation depends on whether
254 // this is a server or client configuration, which is context-dependent.
255 // Server-specific validation should be done by the server class.
256
257 return true;
258 }
