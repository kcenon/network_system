205auto ws_server_adapter::setup_internal_callbacks() -> void
206{
207 if (!server_)
208 {
209 return;
210 }
211
212 // Bridge WebSocket connection callback
213 server_->set_connection_callback(
214 [this](std::shared_ptr<interfaces::i_websocket_session> ws_session)
215 {
216 // Cast to ws_connection for our wrapper
217 auto ws_conn = std::dynamic_pointer_cast<core::ws_connection>(ws_session);
218 if (!ws_conn)
219 {
220 return;
221 }
222
223 auto session = get_or_create_session(ws_conn);
224
225 connection_callback_t callback_copy;
226 {
227 std::lock_guard<std::mutex> lock(callbacks_mutex_);
228 callback_copy = connection_callback_;
229 }
230
231 if (callback_copy)
232 {
233 callback_copy(session);
234 }
235 });
236
237 // Bridge WebSocket disconnection callback (ignore close code and reason)
238 server_->set_disconnection_callback(
239 [this](std::string_view session_id, uint16_t /* code */, std::string_view /* reason */)
240 {
241 // Remove from tracked sessions
242 {
243 std::lock_guard<std::mutex> lock(sessions_mutex_);
244 sessions_.erase(std::string(session_id));
245 }
246
247 disconnection_callback_t callback_copy;
248 {
249 std::lock_guard<std::mutex> lock(callbacks_mutex_);
250 callback_copy = disconnection_callback_;
251 }
252
253 if (callback_copy)
254 {
255 callback_copy(session_id);
256 }
257 });
258
259 // Bridge WebSocket binary message callback
260 server_->set_binary_callback(
261 [this](std::string_view session_id, const std::vector<uint8_t>& data)
262 {
263 receive_callback_t callback_copy;
264 {
265 std::lock_guard<std::mutex> lock(callbacks_mutex_);
266 callback_copy = receive_callback_;
267 }
268
269 if (callback_copy)
270 {
271 callback_copy(session_id, data);
272 }
273 });
274
275 // Bridge WebSocket error callback
276 server_->set_error_callback(
277 [this](std::string_view session_id, std::error_code ec)
278 {
279 error_callback_t callback_copy;
280 {
281 std::lock_guard<std::mutex> lock(callbacks_mutex_);
282 callback_copy = error_callback_;
283 }
284
285 if (callback_copy)
286 {
287 callback_copy(session_id, ec);
288 }
289 });
290}

292auto ws_server_adapter::get_or_create_session(std::shared_ptr<core::ws_connection> connection)
293 -> std::shared_ptr<interfaces::i_session>
294{
295 std::string session_id(connection->id());
296
297 std::lock_guard<std::mutex> lock(sessions_mutex_);
298
299 auto it = sessions_.find(session_id);
300 if (it != sessions_.end())
301 {
302 return it->second;
303 }
304
305 // Create new session wrapper
306 auto session = std::make_shared<ws_session_wrapper>(connection);
307 sessions_[session_id] = session;
308
309 return session;
310}
