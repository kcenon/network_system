53auto ws_connection_adapter::send(std::span<const std::byte> data) -> VoidResult
54{
55 if (!client_ || !client_->is_connected()) {
56 return error_void(
57 static_cast<int>(std::errc::not_connected),
58 "WebSocket is not connected"
59 );
60 }
61
62 std::vector<uint8_t> buffer(data.size());

102auto ws_connection_adapter::connect(const endpoint_info& endpoint) -> VoidResult
103{
104 if (!client_) {
105 return error_void(
106 static_cast<int>(std::errc::invalid_argument),
107 "Client is not initialized"
108 );
109 }
110
111 if (client_->is_connected() || client_->is_running()) {
112 return error_void(
113 static_cast<int>(std::errc::already_connected),
114 "Already connected or connecting"
115 );
116 }
117
118 {
119 std::lock_guard lock(endpoint_mutex_);
120 remote_endpoint_ = endpoint;
121 }
122
123 is_connecting_ = true;
124
125 // Use endpoint's port, default to 80 if not specified
126 uint16_t port = endpoint.port;
127 if (port == 0) {
128 port = 80;
129 }
130
131 auto result = client_->start_client(endpoint.host, port, ws_path_);
132
133 if (!result.is_ok()) {
134 is_connecting_ = false;
135 }
136
137 return result;
138}

140auto ws_connection_adapter::connect(std::string_view url) -> VoidResult
141{
142 if (!client_) {
143 return error_void(
144 static_cast<int>(std::errc::invalid_argument),
145 "Client is not initialized"
146 );
147 }
148
149 if (client_->is_connected() || client_->is_running()) {
150 return error_void(
151 static_cast<int>(std::errc::already_connected),
152 "Already connected or connecting"
153 );
154 }
155
156 std::string host;
157 uint16_t port = 0;
158 std::string path;
159 bool secure = false;
160
161 if (!parse_websocket_url(url, host, port, path, secure)) {
162 return error_void(
163 static_cast<int>(std::errc::invalid_argument),
164 "Invalid WebSocket URL format. Expected: ws://host:port/path or wss://host:port/path"
165 );
166 }
167
168 {
169 std::lock_guard lock(endpoint_mutex_);
170 remote_endpoint_ = endpoint_info{host, port};
171 }
172
173 is_connecting_ = true;
174
175 auto result = client_->start_client(host, port, path);
176
177 if (!result.is_ok()) {
178 is_connecting_ = false;
179 }
180
181 return result;
182}

229auto ws_connection_adapter::setup_internal_callbacks() -> void
230{
231 if (!client_) {
232 return;
233 }
234
235 // Bridge binary message callback to on_data
236 client_->set_binary_callback([this](const std::vector<uint8_t>& data) {
237 std::lock_guard lock(callbacks_mutex_);
238 if (callbacks_.on_data) {
239 std::span<const std::byte> byte_span(
240 reinterpret_cast<const std::byte*>(data.data()),
241 data.size()
242 );
243 callbacks_.on_data(byte_span);
244 }
245 });
246
247 // Also bridge text messages as binary data for unified interface
248 client_->set_text_callback([this](const std::string& text) {
249 std::lock_guard lock(callbacks_mutex_);
250 if (callbacks_.on_data) {
251 std::span<const std::byte> byte_span(
252 reinterpret_cast<const std::byte*>(text.data()),
253 text.size()
254 );
255 callbacks_.on_data(byte_span);
256 }
257 });
258
259 // Bridge connected callback
260 client_->set_connected_callback([this]() {
261 is_connecting_ = false;
262
263 std::lock_guard lock(callbacks_mutex_);
264 if (callbacks_.on_connected) {
265 callbacks_.on_connected();
266 }
267 });
268
269 // Bridge disconnected callback
270 client_->set_disconnected_callback([this](uint16_t /*code*/, std::string_view /*reason*/) {
271 is_connecting_ = false;
272
273 std::lock_guard lock(callbacks_mutex_);
274 if (callbacks_.on_disconnected) {
275 callbacks_.on_disconnected();
276 }
277 });
278
279 // Bridge error callback
280 client_->set_error_callback([this](std::error_code ec) {
281 is_connecting_ = false;
282
283 std::lock_guard lock(callbacks_mutex_);
284 if (callbacks_.on_error) {
285 callbacks_.on_error(ec);
286 }
287 });
288}

290auto ws_connection_adapter::parse_websocket_url(std::string_view url,
291 std::string& host,
292 uint16_t& port,
293 std::string& path,
294 bool& secure) -> bool
295{
296 std::string url_str(url);
297
298 // Check for WebSocket schemes
299 secure = false;
300 std::string remaining;
301
302 if (url_str.substr(0, 6) == "wss://") {
303 secure = true;
304 remaining = url_str.substr(6);
305 } else if (url_str.substr(0, 5) == "ws://") {
306 secure = false;
307 remaining = url_str.substr(5);
308 } else {
309 // No scheme, assume ws://
310 remaining = url_str;
311 }
312
313 // Find path start
314 auto path_pos = remaining.find('/');
315 std::string host_port;
316 if (path_pos != std::string::npos) {
317 host_port = remaining.substr(0, path_pos);
318 path = remaining.substr(path_pos);
319 } else {
320 host_port = remaining;
321 path = "/";
322 }
323
324 // Parse host and port
325 auto colon_pos = host_port.rfind(':');
326
327 // Check if this is IPv6 (contains '[')
328 bool is_ipv6 = host_port.find('[') != std::string::npos;
329
330 if (is_ipv6) {
331 auto bracket_end = host_port.find(']');
332 if (bracket_end == std::string::npos) {
333 return false; // Invalid IPv6 format
334 }
335
336 host = host_port.substr(0, bracket_end + 1);
337
338 if (bracket_end + 1 < host_port.size() && host_port[bracket_end + 1] == ':') {
339 std::string port_str = host_port.substr(bracket_end + 2);
340 try {
341 port = static_cast<uint16_t>(std::stoi(port_str));
342 } catch (...) {
343 return false;
344 }
345 } else {
346 port = secure ? 443 : 80;
347 }
348 } else if (colon_pos != std::string::npos) {
349 host = host_port.substr(0, colon_pos);
350 std::string port_str = host_port.substr(colon_pos + 1);
351 try {
352 port = static_cast<uint16_t>(std::stoi(port_str));
353 } catch (...) {
354 return false;
355 }
356 } else {
357 host = host_port;
358 port = secure ? 443 : 80;
359 }
360
361 return !host.empty();
362}
