131 struct simple_error {
132 int code;
133 std::string message;
134 std::string source;
135 std::string details;
136
137 simple_error(int c, std::string msg, std::string src = "", std::string det = "")
138 : code(c), message(std::move(msg)), source(std::move(src)), details(std::move(det)) {}

141 template<typename T>
142 class Result {
143 public:
144 Result(const T& val) : data_(val) {}
145 Result(T&& val) : data_(std::forward<T>(val)) {}
146 Result(const simple_error& err) : data_(err) {}
147
148 // Static factory methods for API compatibility with common::Result<T>
149 template<typename U = T>
150 static Result<T> ok(U&& value) {
151 return Result<T>(std::forward<U>(value));
152 }
153
154 static Result<T> err(const simple_error& error) {
155 return Result<T>(error);
156 }
157
158 static Result<T> err(int code, const std::string& message, const std::string& source = "") {
159 return Result<T>(simple_error(code, message, source));
160 }
161
162 bool is_ok() const { return std::holds_alternative<T>(data_); }
163 bool is_err() const { return !is_ok(); }
164
165 const T& value() const { return std::get<T>(data_); }
166 T& value() { return std::get<T>(data_); }
167
168 const simple_error& error() const { return std::get<simple_error>(data_); }
169
170 operator bool() const { return is_ok(); }
171
172 private:
173 std::variant<T, simple_error> data_;
174 };
