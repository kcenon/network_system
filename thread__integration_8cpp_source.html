61// basic_thread_pool implementation using thread_system::thread_pool
62class basic_thread_pool::impl {
63public:
64 impl(size_t num_threads) : completed_tasks_(0) {
65 if (num_threads == 0) {
66 num_threads = std::thread::hardware_concurrency();
67 if (num_threads == 0) num_threads = 2; // Fallback
68 }
69
70 // Intentional Leak pattern: Use no-op deleter to prevent destruction
71 // during static destruction phase. This avoids heap corruption when
72 // thread_pool's destructor accesses statically destroyed objects.
73 // Memory impact: ~few KB (reclaimed by OS on process termination)
74 auto* pool = new kcenon::thread::thread_pool("network_basic_pool");
75 pool_ = std::shared_ptr<kcenon::thread::thread_pool>(
76 pool,
77 [](kcenon::thread::thread_pool*) { /* no-op deleter - intentional leak */ }
78 );
79
80 // Add workers to the pool
81 for (size_t i = 0; i < num_threads; ++i) {
82 pool_->enqueue(std::make_unique<kcenon::thread::thread_worker>());
83 }
84
85 pool_->start();
86 }
87
88 ~impl() {
89 stop(true);
90 }
91
92 std::future<void> submit(std::function<void()> task) {
93 auto promise = std::make_shared<std::promise<void>>();
94 auto future = promise->get_future();
95
96 if (!pool_ || !pool_->is_running()) {
97 promise->set_exception(
98 std::make_exception_ptr(
99 std::runtime_error("Thread pool is not running")
100 )
101 );
102 return future;
103 }
104
105 // Note: Capture completed_tasks_ by pointer to avoid capturing 'this',
106 // which can cause heap corruption during static destruction.
107 auto* completed_ptr = &completed_tasks_;
108
109 // Use submit which returns a future and throws on failure
110 try {
111 pool_->submit([task = std::move(task), promise, completed_ptr]() mutable {
112 try {
113 if (task) task();
114 promise->set_value();
115 completed_ptr->fetch_add(1, std::memory_order_relaxed);
116 } catch (...) {
117 promise->set_exception(std::current_exception());
118 }
119 });
120 } catch (const std::exception& e) {
121 promise->set_exception(
122 std::make_exception_ptr(
123 std::runtime_error(
124 std::string("Failed to submit task to thread pool: ") + e.what()
125 )
126 )
127 );
128 }
129
130 return future;
131 }
132
133 std::future<void> submit_delayed(
134 std::function<void()> task,
135 std::chrono::milliseconds delay
136 ) {
137 // Use thread_system's submit_delayed if available (via IExecutor interface)
138#if defined(THREAD_HAS_COMMON_EXECUTOR)
139 return pool_->submit_delayed(std::move(task), delay);
140#else
141 // Fallback: submit a task that sleeps then executes
142 // Note: This is not ideal but maintains API compatibility
143 auto promise = std::make_shared<std::promise<void>>();
144 auto future = promise->get_future();
145
146 if (!pool_ || !pool_->is_running()) {
147 promise->set_exception(
148 std::make_exception_ptr(
149 std::runtime_error("Thread pool is not running")
150 )
151 );
152 return future;
153 }
154
155 // Note: Capture completed_tasks_ by pointer to avoid capturing 'this',
156 // which can cause heap corruption during static destruction.
157 auto* completed_ptr = &completed_tasks_;
158
159 // Use submit which returns a future and throws on failure
160 try {
161 pool_->submit([task = std::move(task), delay, promise, completed_ptr]() mutable {
162 try {
163 std::this_thread::sleep_for(delay);
164 if (task) task();
165 promise->set_value();
166 completed_ptr->fetch_add(1, std::memory_order_relaxed);
167 } catch (...) {
168 promise->set_exception(std::current_exception());
169 }
170 });
171 } catch (const std::exception& e) {
172 promise->set_exception(
173 std::make_exception_ptr(
174 std::runtime_error(
175 std::string("Failed to submit delayed task to thread pool: ") + e.what()
176 )
177 )
178 );
179 }
180
181 return future;
182#endif
183 }
184
185 size_t worker_count() const {
186 return pool_ ? pool_->get_active_worker_count() : 0;
187 }
188
189 bool is_running() const {
190 return pool_ && pool_->is_running();
191 }
192
193 size_t pending_tasks() const {
194 return pool_ ? pool_->get_pending_task_count() : 0;
195 }
196
197 void stop(bool wait_for_tasks) {
198 if (pool_) {
199 pool_->stop(!wait_for_tasks);
200 }
201 }
202
203 size_t get_completed_tasks() const {
204 return completed_tasks_.load();
205 }

207private:
208 std::shared_ptr<kcenon::thread::thread_pool> pool_;
209 std::atomic<size_t> completed_tasks_;
210};

214// Fallback implementation when thread_system is not available
215// This provides a minimal thread pool using std::thread
216class basic_thread_pool::impl {
217 struct DelayedTask {
218 std::chrono::steady_clock::time_point execute_at;
219 std::function<void()> task;
220
221 bool operator>(const DelayedTask& other) const {
222 return execute_at > other.execute_at;
223 }
224 };
225
226public:
227 impl(size_t num_threads)
228 : running_(true), completed_tasks_(0) {
229
230 if (num_threads == 0) {
231 num_threads = std::thread::hardware_concurrency();
232 if (num_threads == 0) num_threads = 2; // Fallback
233 }
234
235 workers_.reserve(num_threads);
236 for (size_t i = 0; i < num_threads; ++i) {
237 workers_.emplace_back([this] { worker_loop(); });
238 }
239
240 scheduler_thread_ = std::thread([this] { scheduler_loop(); });
241 }
242
243 ~impl() {
244 stop(true);
245 }
246
247 std::future<void> submit(std::function<void()> task) {
248 auto promise = std::make_shared<std::promise<void>>();
249 auto future = promise->get_future();
250
251 {
252 std::unique_lock<std::mutex> lock(queue_mutex_);
253 if (!running_) {
254 promise->set_exception(
255 std::make_exception_ptr(
256 std::runtime_error("Thread pool is not running")
257 )
258 );
259 return future;
260 }
261
262 tasks_.emplace([task, promise]() {
263 try {
264 task();
265 promise->set_value();
266 } catch (...) {
267 promise->set_exception(std::current_exception());
268 }
269 });
270 }
271
272 condition_.notify_one();
273 return future;
274 }
275
276 std::future<void> submit_delayed(
277 std::function<void()> task,
278 std::chrono::milliseconds delay
279 ) {
280 auto promise = std::make_shared<std::promise<void>>();
281 auto future = promise->get_future();
282
283 {
284 std::unique_lock<std::mutex> lock(scheduler_mutex_);
285 if (!running_) {
286 promise->set_exception(
287 std::make_exception_ptr(
288 std::runtime_error("Thread pool is not running")
289 )
290 );
291 return future;
292 }
293
294 auto execute_time = std::chrono::steady_clock::now() + delay;
295 delayed_tasks_.push({execute_time, [this, task, promise]() {
296 // Submit to main pool when due
297 submit([task, promise]() {
298 try {
299 task();
300 promise->set_value();
301 } catch (...) {
302 promise->set_exception(std::current_exception());
303 }
304 });
305 }});
306 }
307
308 scheduler_condition_.notify_one();
309 return future;
310 }

313 return workers_.size();
314 }

316 bool is_running() const {
317 return running_.load();
318 }

320 size_t pending_tasks() const {
321 std::unique_lock<std::mutex> lock(queue_mutex_);
322 return tasks_.size();

325 void stop(bool wait_for_tasks) {
326 {
327 std::unique_lock<std::mutex> lock(queue_mutex_);
328 if (!wait_for_tasks) {
329 while (!tasks_.empty()) {
330 tasks_.pop();
331 }
332 }
333 running_ = false;
334 }
335
336 {
337 std::unique_lock<std::mutex> lock(scheduler_mutex_);
338 }
339
340 condition_.notify_all();
341 scheduler_condition_.notify_all();
342
343 for (auto& worker : workers_) {
344 if (worker.joinable()) {
345 worker.join();
346 }
347 }
348 workers_.clear();
349
350 if (scheduler_thread_.joinable()) {
351 scheduler_thread_.join();
352 }
353 }
354
355 size_t get_completed_tasks() const {
356 return completed_tasks_.load();
357 }
358
359private:
360 void worker_loop() {
361 while (true) {
362 std::function<void()> task;
363
364 {
365 std::unique_lock<std::mutex> lock(queue_mutex_);
366 condition_.wait(lock, [this] {
367 return !running_ || !tasks_.empty();
368 });
369
370 if (!running_ && tasks_.empty()) {
371 return;
372 }
373
374 if (!tasks_.empty()) {
375 task = std::move(tasks_.front());
376 tasks_.pop();
377 }
378 }
379
380 if (task) {
381 task();
382 completed_tasks_++;
383 }
384 }
385 }
386
387 void scheduler_loop() {
388 while (running_) {
389 std::unique_lock<std::mutex> lock(scheduler_mutex_);
390
391 if (delayed_tasks_.empty()) {
392 scheduler_condition_.wait(lock, [this] {
393 return !running_ || !delayed_tasks_.empty();
394 });
395 } else {
396 auto now = std::chrono::steady_clock::now();
397 auto& next_task = delayed_tasks_.top();
398
399 if (now >= next_task.execute_at) {
400 auto task = next_task.task;
401 delayed_tasks_.pop();
402 lock.unlock();
403 task(); // Execute (which submits to main pool)
404 continue;
405 } else {
406 scheduler_condition_.wait_until(lock, next_task.execute_at, [this, &next_task] {
407 return !running_ || delayed_tasks_.empty() || delayed_tasks_.top().execute_at < next_task.execute_at;
408 });
409 }
410 }
411 }
412 }
413
414 std::vector<std::thread> workers_;
415 std::thread scheduler_thread_;
416
417 std::queue<std::function<void()>> tasks_;
418 mutable std::mutex queue_mutex_;
419 std::condition_variable condition_;
420
421 std::priority_queue<DelayedTask, std::vector<DelayedTask>, std::greater<DelayedTask>> delayed_tasks_;
422 std::mutex scheduler_mutex_;
423 std::condition_variable scheduler_condition_;
424
425 std::atomic<bool> running_;
426 std::atomic<size_t> completed_tasks_;
427};

504 thread_integration_manager::metrics get_metrics() const {
505 std::unique_lock<std::mutex> lock(mutex_);
506
507 metrics m;
508 if (thread_pool_) {
509 m.worker_threads = thread_pool_->worker_count();
510 m.pending_tasks = thread_pool_->pending_tasks();
511 m.is_running = thread_pool_->is_running();
512
513 if (auto* basic = dynamic_cast<basic_thread_pool*>(thread_pool_.get())) {
514 m.completed_tasks = basic->completed_tasks();
515 }
516 }
517
518 return m;
519 }
