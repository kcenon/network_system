62auto ecn_tracker::process_ecn_counts(const ecn_counts& counts,
63 uint64_t packets_acked,
64 std::chrono::steady_clock::time_point sent_time)
65 -> ecn_result
66{
67 // If ECN has already failed, don't process further
68 if (state_.failed)
69 {
70 return ecn_result::ecn_failure;
71 }
72
73 // RFC 9000 Section 13.4.2.1: Verify ECN counts are valid
74 // ECN counts must be monotonically increasing
75 if (counts.ect0 < state_.counts.ect0 ||
76 counts.ect1 < state_.counts.ect1 ||
77 counts.ecn_ce < state_.counts.ecn_ce)
78 {
79 // ECN counts decreased - this is invalid, disable ECN
80 state_.failed = true;
81 state_.testing = false;
82 state_.capable = false;
83 return ecn_result::ecn_failure;
84 }
85
86 // Validate ECN if still in testing phase
87 if (state_.testing)
88 {
89 if (!validate_ecn(counts, packets_acked))
90 {
91 state_.failed = true;
92 state_.testing = false;
93 state_.capable = false;
94 return ecn_result::ecn_failure;
95 }
96
97 // Check if we have enough data to complete validation
98 uint64_t total_marks = counts.ect0 + counts.ect1 + counts.ecn_ce;
99 if (total_marks >= validation_state::kValidationThreshold)
100 {
101 state_.testing = false;
102 state_.capable = true;
103 }
104 }
105
106 // Check for ECN-CE increase (congestion signal)
107 // RFC 9002 Section 7.1: ECN-CE marks indicate congestion
108 ecn_result result = ecn_result::none;
109 if (counts.ecn_ce > state_.counts.ecn_ce)
110 {
111 // ECN-CE count increased - congestion experienced
112 result = ecn_result::congestion_signal;
113 state_.last_congestion_sent_time = sent_time;
114 }
115
116 // Update stored counts
117 state_.counts = counts;
118
119 return result;
120}

122auto ecn_tracker::validate_ecn(const ecn_counts& counts,
123 uint64_t packets_acked) -> bool
124{
125 // RFC 9000 Section 13.4.2.1:
126 // The total increase in ECT(0), ECT(1), and ECN-CE counts MUST be
127 // at least the number of newly acknowledged packets that were
128 // originally sent with an ECT codepoint.
129
130 uint64_t total_increase =
131 (counts.ect0 - state_.counts.ect0) +
132 (counts.ect1 - state_.counts.ect1) +
133 (counts.ecn_ce - state_.counts.ecn_ce);
134
135 // We sent packets with ECT(0) marking, so we expect the ECN counts
136 // to reflect at least those packets
137 if (total_increase < packets_acked && state_.packets_sent_with_ect > 0)
138 {
139 // ECN counts don't account for all packets - ECN is being stripped
140 // by the network path, or the peer doesn't support ECN properly
141 return false;
142 }
143
144 return true;
145}
