44 {
45 public:
46 explicit impl(size_t pool_size, size_t default_capacity)
47 : pool_size_(pool_size), default_capacity_(default_capacity), total_allocated_(0)
48 {
49 NETWORK_LOG_DEBUG("[buffer_pool] Created with pool_size=" +
50 std::to_string(pool_size) +
51 ", default_capacity=" + std::to_string(default_capacity));
52 }

54 auto acquire(size_t min_capacity, buffer_pool* parent) -> std::shared_ptr<std::vector<uint8_t>>
55 {
56 std::unique_lock<std::mutex> lock(mutex_);
57
58 // Try to find a suitable buffer in the pool
59 for (auto it = available_buffers_.begin(); it != available_buffers_.end(); ++it)
60 {
61 if ((*it)->capacity() >= min_capacity)
62 {
63 auto buffer = std::move(*it);
64 available_buffers_.erase(it);
65
66 // Clear the buffer but keep capacity
67 buffer->clear();
68
69 NETWORK_LOG_TRACE("[buffer_pool] Acquired buffer from pool, capacity=" +
70 std::to_string(buffer->capacity()));
71
72 // Return with custom deleter that returns buffer to pool
73 return std::shared_ptr<std::vector<uint8_t>>(
74 buffer.release(), [parent](std::vector<uint8_t>* buf) {
75 if (parent)
76 {
77 parent->release(buf);
78 }
79 else
80 {
81 delete buf;
82 }
83 });
84 }
85 }
86
87 // No suitable buffer found, create a new one
88 size_t capacity = std::max(min_capacity, default_capacity_);
89 auto buffer = std::make_unique<std::vector<uint8_t>>();
90 buffer->reserve(capacity);
91
92 total_allocated_.fetch_add(1);
93
94 NETWORK_LOG_TRACE("[buffer_pool] Created new buffer, capacity=" +
95 std::to_string(capacity) +
96 ", total_allocated=" + std::to_string(total_allocated_.load()));
97
98 // Return with custom deleter
99 return std::shared_ptr<std::vector<uint8_t>>(
100 buffer.release(), [parent](std::vector<uint8_t>* buf) {
101 if (parent)
102 {
103 parent->release(buf);
104 }
105 else
106 {
107 delete buf;
108 }
109 });
110 }
111
112 auto release(std::vector<uint8_t>* buffer) -> void
113 {
114 if (!buffer)
115 return;
116
117 std::unique_lock<std::mutex> lock(mutex_);
118
119 // If pool is full, just delete the buffer
120 if (available_buffers_.size() >= pool_size_)
121 {
122 NETWORK_LOG_TRACE("[buffer_pool] Pool full, deleting buffer");
123 delete buffer;
124 total_allocated_.fetch_sub(1);
125 return;
126 }
127
128 // Return buffer to pool
129 available_buffers_.emplace_back(buffer);
130
131 NETWORK_LOG_TRACE("[buffer_pool] Returned buffer to pool, available=" +
132 std::to_string(available_buffers_.size()));
133 }
134
135 auto get_stats() const -> std::pair<size_t, size_t>
136 {
137 std::lock_guard<std::mutex> lock(mutex_);
138 return {available_buffers_.size(), total_allocated_.load()};
139 }
140
141 auto clear() -> void
142 {
143 std::unique_lock<std::mutex> lock(mutex_);
144
145 size_t cleared = available_buffers_.size();
146 available_buffers_.clear();
147
148 total_allocated_.fetch_sub(cleared);
149
150 NETWORK_LOG_INFO("[buffer_pool] Cleared " + std::to_string(cleared) + " buffers");
151 }
152
153 private:
154 size_t pool_size_;
155 size_t default_capacity_;
156 std::atomic<size_t> total_allocated_;
157
158 mutable std::mutex mutex_;
159 std::deque<std::unique_ptr<std::vector<uint8_t>>> available_buffers_;
160 };
