59auto tcp_facade::generate_client_id() -> std::string

75auto tcp_facade::validate_client_config(const client_config& config) -> void
76{
77 if (config.host.empty())
78 {
79 throw std::invalid_argument("tcp_facade: host cannot be empty");
80 }
81
82 if (config.port == 0 || config.port > 65535)
83 {
84 throw std::invalid_argument("tcp_facade: port must be between 1 and 65535");
85 }
86
87 if (config.timeout.count() <= 0)
88 {
89 throw std::invalid_argument("tcp_facade: timeout must be positive");
90 }
91}

114auto tcp_facade::create_client(const client_config& config) const
115 -> std::shared_ptr<interfaces::i_protocol_client>
116{
117 // Validate configuration
118 validate_client_config(config);
119
120 // Generate client ID if not provided
121 const std::string client_id = config.client_id.empty() ? generate_client_id() : config.client_id;
122
123 // Create appropriate client type based on SSL flag
124 std::shared_ptr<interfaces::i_protocol_client> client;
125
126 // Create plain messaging client
127 // Note: SSL support will be added after secure_messaging_client implements IProtocolClient
128 if (config.use_ssl)
129 {
130 throw std::runtime_error(
131 "tcp_facade: SSL/TLS support not yet implemented - "
132 "requires secure_messaging_client to implement IProtocolClient interface");
133 }
134
135 client = std::make_shared<core::messaging_client>(client_id);
136
137 // Start the client and connect
138 auto result = client->start(config.host, config.port);
139 if (result.is_err())
140 {
141 throw std::runtime_error("tcp_facade: failed to start client: " + result.error().message);
142 }
143
144 return client;
145}

147auto tcp_facade::create_server(const server_config& config) const
148 -> std::shared_ptr<interfaces::i_protocol_server>
149{
150 // Validate configuration
151 validate_server_config(config);
152
153 // Generate server ID if not provided
154 const std::string server_id = config.server_id.empty() ? generate_server_id() : config.server_id;
155
156 // Create appropriate server type based on SSL flag
157 std::shared_ptr<interfaces::i_protocol_server> server;
158
159 if (config.use_ssl)
160 {
161 // Note: SSL support will be added after secure_messaging_server implements IProtocolServer
162 throw std::runtime_error(
163 "tcp_facade: SSL/TLS support not yet implemented - "
164 "requires secure_messaging_server to implement IProtocolServer interface");
165 }
166
167 // Create adapter wrapping messaging_server
168 server = std::make_shared<internal::adapters::tcp_server_adapter>(server_id);
169
170 return server;
171}

173auto tcp_facade::create_connection_pool(const pool_config& config) const
174 -> std::shared_ptr<core::connection_pool>
175{
176 if (config.host.empty())
177 {
178 throw std::invalid_argument("tcp_facade: host cannot be empty");
179 }
180
181 if (config.port == 0 || config.port > 65535)
182 {
183 throw std::invalid_argument("tcp_facade: port must be between 1 and 65535");
184 }
185
186 if (config.pool_size == 0)
187 {
188 throw std::invalid_argument("tcp_facade: pool_size must be greater than 0");
189 }
190
191 return std::make_shared<core::connection_pool>(
192 config.host,
193 config.port,
194 config.pool_size,
195 config.acquire_timeout);
196}
