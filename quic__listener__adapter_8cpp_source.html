59auto generate_random_cid() -> protocols::quic::connection_id
60{
61 std::random_device rd;
62 std::mt19937 gen(rd());
63 // Use unsigned short instead of uint8_t for MSVC compatibility
64 // C++ standard only allows short, int, long, long long and their unsigned versions
65 std::uniform_int_distribution<unsigned short> dist(0, 255);
66
67 std::vector<uint8_t> cid_bytes(8);
68 for (auto& byte : cid_bytes) {
69 byte = static_cast<uint8_t>(dist(gen));
70 }
71
72 return protocols::quic::connection_id(cid_bytes);
73}

84auto set_nonblocking(int fd) -> bool
85{
86#ifdef _WIN32
87 u_long mode = 1;
88 return ioctlsocket(fd, FIONBIO, &mode) == 0;
89#else
90 int flags = fcntl(fd, F_GETFL, 0);
91 if (flags == -1) {
92 return false;
93 }
94 return fcntl(fd, F_SETFL, flags | O_NONBLOCK) != -1;
95#endif
96}

98auto sockaddr_to_string(const sockaddr_in& addr) -> std::string
99{
100 char ip_str[INET_ADDRSTRLEN];
101 inet_ntop(AF_INET, &addr.sin_addr, ip_str, sizeof(ip_str));
102 return std::string(ip_str) + ":" + std::to_string(ntohs(addr.sin_port));
103}

120auto quic_listener_adapter::start(const endpoint_info& bind_address) -> VoidResult
121{
122 if (running_.load()) {
123 return error_void(
124 static_cast<int>(std::errc::already_connected),
125 "Listener is already running"
126 );
127 }
128
129 // Validate that cert and key files are provided
130 if (config_.cert_file.empty() || config_.key_file.empty()) {
131 return error_void(
132 static_cast<int>(std::errc::invalid_argument),
133 "QUIC server requires cert_file and key_file in configuration"
134 );
135 }
136
137 auto setup_result = setup_socket(bind_address);
138 if (!setup_result.is_ok()) {
139 return setup_result;
140 }
141
142 {
143 std::lock_guard lock(endpoint_mutex_);
144 local_endpoint_ = bind_address;
145 }
146
147 running_ = true;
148 stop_requested_ = false;
149
150 // Start accept thread
151 accept_thread_ = std::thread(&quic_listener_adapter::accept_thread_func, this);
152
153 return ok();
154}

161auto quic_listener_adapter::stop() noexcept -> void
162{
163 stop_requested_ = true;
164
165 // Close all connections
166 {
167 std::lock_guard lock(connections_mutex_);
168 for (auto& [conn_id, info] : connections_) {
169 if (info.conn && info.conn->is_established()) {
170 (void)info.conn->close(0, "Server shutting down");
171 }
172 }
173 connections_.clear();
174 }
175
176 if (accept_thread_.joinable()) {
177 accept_thread_.join();
178 }
179
180 if (socket_fd_ >= 0) {
181 close_socket(socket_fd_);
182 socket_fd_ = -1;
183 }
184
185 running_ = false;
186
187 {
188 std::lock_guard lock(stop_mutex_);
189 }
190 stop_cv_.notify_all();
191}

222auto quic_listener_adapter::send_to(std::string_view connection_id,
223 std::span<const std::byte> data) -> VoidResult
224{
225 std::lock_guard lock(connections_mutex_);
226
227 auto it = connections_.find(std::string(connection_id));
228 if (it == connections_.end()) {
229 return error_void(
230 static_cast<int>(std::errc::no_such_file_or_directory),
231 "Connection not found"
232 );
233 }
234
235 auto& conn_info = it->second;
236 if (!conn_info.conn || !conn_info.conn->is_established()) {
237 return error_void(
238 static_cast<int>(std::errc::not_connected),
239 "Connection is not established"
240 );
241 }
242
243 // Write data to the first available bidirectional stream
244 auto& stream_mgr = conn_info.conn->streams();
245
246 // Create or get stream
247 auto stream_result = stream_mgr.create_bidirectional_stream();
248
249 if (!stream_result.is_ok()) {
250 return error_void(
251 static_cast<int>(std::errc::no_buffer_space),
252 "Failed to create stream"
253 );
254 }
255
256 auto* stream = stream_mgr.get_stream(stream_result.value());
257 if (!stream) {
258 return error_void(
259 static_cast<int>(std::errc::io_error),
260 "Failed to get stream"
261 );
262 }
263
264 std::vector<uint8_t> buffer(data.size());
265 std::memcpy(buffer.data(), data.data(), data.size());
266
267 auto write_result = stream->write(buffer);
268 if (!write_result.is_ok()) {
269 return error_void(
270 static_cast<int>(std::errc::io_error),
271 "Failed to write to stream"
272 );
273 }
274
275 conn_info.last_activity = std::chrono::steady_clock::now();
276
277 return ok();
278}

280auto quic_listener_adapter::broadcast(std::span<const std::byte> data) -> VoidResult
281{
282 std::lock_guard lock(connections_mutex_);
283
284 bool any_success = false;
285 for (auto& [conn_id, conn_info] : connections_) {
286 if (conn_info.conn && conn_info.conn->is_established()) {
287 auto& stream_mgr = conn_info.conn->streams();
288 auto stream_result = stream_mgr.create_bidirectional_stream();
289
290 if (stream_result.is_ok()) {
291 auto* stream = stream_mgr.get_stream(stream_result.value());
292 if (stream) {
293 std::vector<uint8_t> buffer(data.size());
294 std::memcpy(buffer.data(), data.data(), data.size());
295
296 auto write_result = stream->write(buffer);
297 if (write_result.is_ok()) {
298 any_success = true;
299 }
300 }
301 }
302 }
303 }
304
305 if (!any_success && !connections_.empty()) {
306 return error_void(
307 static_cast<int>(std::errc::io_error),
308 "Failed to send to any connection"
309 );
310 }
311
312 return ok();
313}

316{
317 std::lock_guard lock(connections_mutex_);
318
319 auto it = connections_.find(std::string(connection_id));
320 if (it != connections_.end()) {
321 if (it->second.conn) {
322 (void)it->second.conn->close(0, "Connection closed by server");
323 }
324
325 // Notify callback
326 {
327 std::lock_guard cb_lock(callbacks_mutex_);
328 if (callbacks_.on_disconnect) {

330 }
331 }

343auto quic_listener_adapter::setup_socket(const endpoint_info& bind_address) -> VoidResult
344{
345#ifdef _WIN32
346 WSADATA wsaData;
347 if (WSAStartup(MAKEWORD(2, 2), &wsaData) != 0) {
348 return error_void(
349 static_cast<int>(std::errc::io_error),
350 "Failed to initialize Winsock"
351 );
352 }
353#endif
354
355 socket_fd_ = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
356 if (socket_fd_ < 0) {
357 return error_void(
358 static_cast<int>(std::errc::io_error),
359 "Failed to create UDP socket"
360 );
361 }
362
363 // Allow address reuse
364 int reuse = 1;
365 setsockopt(socket_fd_, SOL_SOCKET, SO_REUSEADDR,
366 reinterpret_cast<const char*>(&reuse), sizeof(reuse));
367
368 if (!set_nonblocking(socket_fd_)) {
369 close_socket(socket_fd_);
370 socket_fd_ = -1;
371 return error_void(
372 static_cast<int>(std::errc::io_error),
373 "Failed to set socket to non-blocking mode"
374 );
375 }
376
377 // Bind to address
378 sockaddr_in bind_addr{};
379 bind_addr.sin_family = AF_INET;
380 bind_addr.sin_port = htons(bind_address.port);
381
382 if (bind_address.host == "0.0.0.0" || bind_address.host.empty()) {
383 bind_addr.sin_addr.s_addr = INADDR_ANY;
384 } else {
385 inet_pton(AF_INET, bind_address.host.c_str(), &bind_addr.sin_addr);
386 }
387
388 if (bind(socket_fd_, reinterpret_cast<sockaddr*>(&bind_addr), sizeof(bind_addr)) < 0) {
389 close_socket(socket_fd_);
390 socket_fd_ = -1;
391 return error_void(
392 static_cast<int>(std::errc::address_in_use),
393 "Failed to bind socket to address"
394 );
395 }
396
397 return ok();
398}

400auto quic_listener_adapter::accept_thread_func() -> void
401{
402 constexpr int POLL_TIMEOUT_MS = 10;
403 constexpr size_t MAX_PACKET_SIZE = 65535;
404
405 std::vector<uint8_t> recv_buffer(MAX_PACKET_SIZE);
406
407 while (!stop_requested_.load()) {
408 // Poll for incoming data
409#ifdef _WIN32
410 WSAPOLLFD pfd{};
411 pfd.fd = socket_fd_;
412 pfd.events = POLLIN;
413
414 int poll_result = WSAPoll(&pfd, 1, POLL_TIMEOUT_MS);
415#else
416 pollfd pfd{};
417 pfd.fd = socket_fd_;
418 pfd.events = POLLIN;
419
420 int poll_result = poll(&pfd, 1, POLL_TIMEOUT_MS);
421#endif
422
423 if (poll_result > 0 && (pfd.revents & POLLIN)) {
424 // Receive incoming packet
425 sockaddr_in from_addr{};
426 socklen_t from_len = sizeof(from_addr);
427
428 auto received = recvfrom(socket_fd_, reinterpret_cast<char*>(recv_buffer.data()),
429 static_cast<int>(recv_buffer.size()), 0,
430 reinterpret_cast<sockaddr*>(&from_addr), &from_len);
431
432 if (received > 0) {
433 std::span<const uint8_t> packet_data(recv_buffer.data(),
434 static_cast<size_t>(received));
435
436 // Identify or create connection based on source address
437 std::string client_addr = sockaddr_to_string(from_addr);
438
439 std::lock_guard lock(connections_mutex_);
440
441 auto it = connections_.find(client_addr);
442 if (it == connections_.end()) {
443 // New connection - create QUIC connection
444 auto initial_dcid = generate_random_cid();
445 auto new_conn = std::make_unique<protocols::quic::connection>(
446 true, initial_dcid);
447
448 // Initialize server handshake
449 auto init_result = new_conn->init_server_handshake(
450 config_.cert_file, config_.key_file);
451
452 if (!init_result.is_ok()) {
453 continue;
454 }
455
456 // Set transport parameters
457 protocols::quic::transport_parameters local_params;
458 local_params.initial_max_data = config_.initial_max_data;
459 local_params.initial_max_stream_data_bidi_local =
460 config_.initial_max_stream_data_bidi;
461 local_params.initial_max_stream_data_bidi_remote =
462 config_.initial_max_stream_data_bidi;
463 local_params.initial_max_stream_data_uni =
464 config_.initial_max_stream_data_uni;
465 local_params.initial_max_streams_bidi = config_.max_bidi_streams;
466 local_params.initial_max_streams_uni = config_.max_uni_streams;
467
468 if (config_.idle_timeout.count() > 0) {
469 local_params.max_idle_timeout =
470 static_cast<uint64_t>(config_.idle_timeout.count());
471 }
472
473 new_conn->set_local_params(local_params);
474
475 if (config_.enable_pmtud) {
476 new_conn->enable_pmtud();
477 }
478
479 connection_info info;
480 info.conn = std::move(new_conn);
481 info.last_activity = std::chrono::steady_clock::now();
482
483 connections_[client_addr] = std::move(info);
484
485 // Notify accept callback
486 std::lock_guard cb_lock(callbacks_mutex_);
487 if (callbacks_.on_accept) {
488 callbacks_.on_accept(client_addr);
489 }
490 }
491
492 // Process packet on existing connection
493 auto conn_it = connections_.find(client_addr);
494 if (conn_it != connections_.end() && conn_it->second.conn) {
495 auto result = conn_it->second.conn->receive_packet(packet_data);
496 (void)result;
497 conn_it->second.last_activity = std::chrono::steady_clock::now();
498 }
499 }
500 }
501
502 // Process all connections
503 process_connections();
504 }
505
506 running_ = false;
507
508 {
509 std::lock_guard lock(stop_mutex_);
510 }
511 stop_cv_.notify_all();
512}

514auto quic_listener_adapter::process_connections() -> void
515{
516 std::lock_guard lock(connections_mutex_);
517
518 std::vector<std::string> to_remove;
519
520 for (auto& [conn_id, conn_info] : connections_) {
521 if (!conn_info.conn) {
522 to_remove.push_back(conn_id);
523 continue;
524 }
525
526 // Check for timeouts
527 auto timeout = conn_info.conn->next_timeout();
528 if (timeout && std::chrono::steady_clock::now() >= *timeout) {
529 conn_info.conn->on_timeout();
530 }
531
532 // Generate and send packets
533 auto packets = conn_info.conn->generate_packets();
534
535 // Parse connection ID to get address for sending
536 auto colon_pos = conn_id.rfind(':');
537 if (colon_pos != std::string::npos && !packets.empty()) {
538 std::string host = conn_id.substr(0, colon_pos);
539 uint16_t port = static_cast<uint16_t>(
540 std::stoi(conn_id.substr(colon_pos + 1)));
541
542 sockaddr_in dest_addr{};
543 dest_addr.sin_family = AF_INET;
544 dest_addr.sin_port = htons(port);
545 inet_pton(AF_INET, host.c_str(), &dest_addr.sin_addr);
546
547 for (const auto& packet : packets) {
548 sendto(socket_fd_, reinterpret_cast<const char*>(packet.data()),
549 static_cast<int>(packet.size()), 0,
550 reinterpret_cast<const sockaddr*>(&dest_addr),
551 sizeof(dest_addr));
552 }
553 }
554
555 // Check for received data on streams
556 if (conn_info.conn->is_established()) {
557 auto& stream_mgr = conn_info.conn->streams();
558
559 // Check bidirectional streams (client-initiated: 0, 4, 8, ...)
560 for (uint64_t stream_id = 0; stream_id < 1000; stream_id += 4) {
561 auto* stream = stream_mgr.get_stream(stream_id);
562 if (stream && stream->has_data()) {
563 std::vector<uint8_t> buffer(4096);
564 auto read_result = stream->read(buffer);
565 if (read_result.is_ok() && read_result.value() > 0) {
566 buffer.resize(read_result.value());
567
568 std::lock_guard cb_lock(callbacks_mutex_);
569 if (callbacks_.on_data) {
570 std::span<const std::byte> byte_span(
571 reinterpret_cast<const std::byte*>(buffer.data()),
572 buffer.size()
573 );
574 callbacks_.on_data(conn_id, byte_span);
575 }
576 }
577 }
578 }
579 }
580
581 // Check if connection is closed
582 if (conn_info.conn->is_closed()) {
583 to_remove.push_back(conn_id);
584
585 std::lock_guard cb_lock(callbacks_mutex_);
586 if (callbacks_.on_disconnect) {
587 callbacks_.on_disconnect(conn_id);
588 }
589 }
590 }
591
592 // Remove closed connections
593 for (const auto& conn_id : to_remove) {
594 connections_.erase(conn_id);
595 }
596}

598auto quic_listener_adapter::generate_connection_id(
599 const protocols::quic::connection_id& cid) -> std::string
600{
601 std::ostringstream oss;
602 oss << std::hex;
603 for (size_t i = 0; i < cid.length(); ++i) {
604 oss << std::setfill('0') << std::setw(2)
605 << static_cast<int>(cid.data()[i]);
606 }
607 return oss.str();
