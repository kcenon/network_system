52auto udp_connection_adapter::send(std::span<const std::byte> data) -> VoidResult
53{
54 if (!client_ || !client_->is_running()) {
55 return error_void(
56 static_cast<int>(std::errc::not_connected),
57 "UDP client is not running"
58 );
59 }
60
61 std::vector<uint8_t> buffer(data.size());
62 std::memcpy(buffer.data(), data.data(), data.size());

102auto udp_connection_adapter::connect(const endpoint_info& endpoint) -> VoidResult
103{
104 if (!client_) {
105 return error_void(
106 static_cast<int>(std::errc::invalid_argument),
107 "Client is not initialized"
108 );
109 }
110
111 if (client_->is_running()) {
112 return error_void(
113 static_cast<int>(std::errc::already_connected),
114 "Already running"
115 );
116 }
117
118 {
119 std::lock_guard lock(endpoint_mutex_);
120 remote_endpoint_ = endpoint;
121 }
122
123 is_connecting_ = true;
124 auto result = client_->start_client(endpoint.host, endpoint.port);
125
126 if (result.is_ok()) {
127 is_connecting_ = false;
128 // For UDP, we're immediately "connected" once started
129 std::lock_guard lock(callbacks_mutex_);
130 if (callbacks_.on_connected) {
131 callbacks_.on_connected();
132 }
133 } else {
134 is_connecting_ = false;
135 }
136
137 return result;
138}

140auto udp_connection_adapter::connect(std::string_view url) -> VoidResult
141{
142 // Parse URL for UDP: expect format "udp://host:port" or just "host:port"
143 std::string url_str(url);
144
145 // Remove udp:// prefix if present
146 const std::string udp_prefix = "udp://";
147 if (url_str.substr(0, udp_prefix.size()) == udp_prefix) {
148 url_str = url_str.substr(udp_prefix.size());
149 }
150
151 // Find the last colon for port separation
152 auto colon_pos = url_str.rfind(':');
153 if (colon_pos == std::string::npos) {
154 return error_void(
155 static_cast<int>(std::errc::invalid_argument),
156 "URL must contain port number (format: host:port)"
157 );
158 }
159
160 std::string host = url_str.substr(0, colon_pos);
161 std::string port_str = url_str.substr(colon_pos + 1);
162
163 uint16_t port = 0;
164 try {
165 port = static_cast<uint16_t>(std::stoi(port_str));
166 } catch (...) {
167 return error_void(
168 static_cast<int>(std::errc::invalid_argument),
169 "Invalid port number in URL"
170 );
171 }
172
173 return connect(endpoint_info{host, port});
174}

222auto udp_connection_adapter::setup_internal_callbacks() -> void
223{
224 if (!client_) {
225 return;
226 }
227
228 // Bridge receive callback
229 client_->set_receive_callback(
230 [this](const std::vector<uint8_t>& data,
231 const interfaces::i_udp_client::endpoint_info& sender) {
232 // Update local endpoint information if we receive from the expected target
233 {
234 std::lock_guard ep_lock(endpoint_mutex_);
235 // Note: local_endpoint_ would require socket info which isn't easily available
236 }
237
238 std::lock_guard lock(callbacks_mutex_);
239 if (callbacks_.on_data) {
240 std::span<const std::byte> byte_span(
241 reinterpret_cast<const std::byte*>(data.data()),
242 data.size()
243 );
244 callbacks_.on_data(byte_span);
245 }
246 });
247
248 // Bridge error callback
249 client_->set_error_callback([this](std::error_code ec) {
250 is_connecting_ = false;
251
252 std::lock_guard lock(callbacks_mutex_);
253 if (callbacks_.on_error) {
254 callbacks_.on_error(ec);
255 }
256 });
257}
