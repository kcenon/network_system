55VoidResult ThreadPoolBridge::initialize(const BridgeConfig& config) {
56 if (initialized_.load()) {
57 return error_void(
58 error_codes::common_errors::already_exists,
59 "ThreadPoolBridge already initialized",
60 "ThreadPoolBridge::initialize");
61 }
62
63 if (!pool_) {
64 return error_void(
65 error_codes::common_errors::invalid_argument,
66 "Thread pool is null",
67 "ThreadPoolBridge::initialize");
68 }
69
70 if (!pool_->is_running()) {
71 return error_void(
72 error_codes::common_errors::not_initialized,
73 "Thread pool is not running",
74 "ThreadPoolBridge::initialize");
75 }
76
77 // Check if bridge is enabled (default: true)
78 auto enabled_it = config.properties.find("enabled");
79 if (enabled_it != config.properties.end() && enabled_it->second == "false") {
80 return error_void(
81 error_codes::common_errors::invalid_argument,
82 "Bridge is disabled in configuration",
83 "ThreadPoolBridge::initialize");
84 }
85
86 // Initialize metrics
87 std::lock_guard<std::mutex> lock(metrics_mutex_);
88 cached_metrics_.is_healthy = true;
89 cached_metrics_.last_activity = std::chrono::steady_clock::now();
90 cached_metrics_.custom_metrics["worker_threads"] = static_cast<double>(pool_->worker_count());
91 cached_metrics_.custom_metrics["pending_tasks"] = static_cast<double>(pool_->pending_tasks());
92 cached_metrics_.custom_metrics["backend_type"] = static_cast<double>(backend_type_);
93
94 initialized_.store(true);
95 return ok();
96}

98VoidResult ThreadPoolBridge::shutdown() {
99 if (!initialized_.load()) {
100 return ok(); // Idempotent: already shut down
101 }
102
103 // Update metrics to reflect shutdown state
104 {
105 std::lock_guard<std::mutex> lock(metrics_mutex_);
106 cached_metrics_.is_healthy = false;
107 cached_metrics_.last_activity = std::chrono::steady_clock::now();
108 }
109
110 initialized_.store(false);
111 return ok();
112}

114bool ThreadPoolBridge::is_initialized() const {

118BridgeMetrics ThreadPoolBridge::get_metrics() const {
119 std::lock_guard<std::mutex> lock(metrics_mutex_);
120
121 if (!initialized_.load() || !pool_) {
122 BridgeMetrics metrics;
123 metrics.is_healthy = false;
124 metrics.last_activity = cached_metrics_.last_activity;
125 return metrics;
126 }
127
128 // Update metrics with current thread pool state
129 BridgeMetrics metrics = cached_metrics_;
130 metrics.is_healthy = pool_->is_running();
131 metrics.last_activity = std::chrono::steady_clock::now();
132 metrics.custom_metrics["worker_threads"] = static_cast<double>(pool_->worker_count());
133 metrics.custom_metrics["pending_tasks"] = static_cast<double>(pool_->pending_tasks());
134 metrics.custom_metrics["backend_type"] = static_cast<double>(backend_type_);
135
136 // Update cached metrics for next call
137 cached_metrics_ = metrics;
138
139 return metrics;
140}

150std::shared_ptr<ThreadPoolBridge> ThreadPoolBridge::from_thread_system(
151 const std::string& pool_name) {
152 (void)pool_name; // Pool name is informational only in current implementation
153
154 auto pool = thread_integration_manager::instance().get_thread_pool();
155 if (!pool) {
156 throw std::runtime_error("Failed to get thread pool from thread_integration_manager");
157 }
158
159 return std::make_shared<ThreadPoolBridge>(pool, BackendType::ThreadSystem);
160}
