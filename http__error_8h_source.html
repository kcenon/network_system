105 struct http_error
106 {
107 http_error_code code = http_error_code::internal_server_error;
108 std::string message;
109 std::string detail;
110 std::string request_id;
111 std::chrono::system_clock::time_point timestamp = std::chrono::system_clock::now();
112
117 auto status_code() const -> int { return static_cast<int>(code); }
118
123 auto is_client_error() const -> bool
124 {
125 int status = status_code();
126 return status >= 400 && status < 500;
127 }
128
133 auto is_server_error() const -> bool
134 {
135 int status = status_code();
136 return status >= 500 && status < 600;
137 }
138 };

180 auto to_http_error() const -> http_error
181 {
182 http_error err;
183 err.code = http_error_code::bad_request;
184 err.message = "Bad Request";
185 err.detail = message;
186 if (!context.empty())
187 {
188 err.detail += " near: " + context;
189 }
190 return err;
191 }
