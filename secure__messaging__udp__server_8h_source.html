263 struct endpoint_hash

265 std::size_t operator()(const asio::ip::udp::endpoint& ep) const
266 {
267 auto addr_hash = std::hash<std::string>{}(ep.address().to_string());
268 auto port_hash = std::hash<unsigned short>{}(ep.port());
269 return addr_hash ^ (port_hash << 1);
270 }
