52auto loss_detector::space_index(encryption_level level) const noexcept -> size_t
53{
54 // Map encryption levels to packet number spaces:
55 // Initial -> 0, Handshake -> 1, Zero-RTT/Application -> 2
56 switch (level)
57 {
58 case encryption_level::initial:
59 return 0;
60 case encryption_level::handshake:
61 return 1;
62 case encryption_level::zero_rtt:
63 case encryption_level::application:
64 return 2;
65 default:
66 return 2;
67 }
68}

70auto loss_detector::on_packet_sent(const sent_packet& packet) -> void
71{
72 auto idx = space_index(packet.level);
73 auto& space = spaces_[idx];
74
75 space.sent_packets[packet.packet_number] = packet;
76
77 if (packet.in_flight)
78 {
79 space.bytes_in_flight += packet.sent_bytes;
80 }
81
82 if (packet.ack_eliciting)
83 {
84 space.time_of_last_ack_eliciting = packet.sent_time;
85 }
86
87 set_loss_detection_timer();
88}

90auto loss_detector::on_ack_received(const ack_frame& ack,
91 encryption_level level,
92 std::chrono::steady_clock::time_point recv_time)
93 -> loss_detection_result
94{
95 loss_detection_result result;
96 auto idx = space_index(level);
97 auto& space = spaces_[idx];
98
99 auto largest_newly_acked = ack.largest_acknowledged;
100
101 if (!space.largest_acked_set || largest_newly_acked > space.largest_acked)
102 {
103 space.largest_acked = largest_newly_acked;
104 space.largest_acked_set = true;
105 }
106
107 // Process the ACK ranges to find newly acknowledged packets
108 // First range starts at largest_acknowledged and goes back ack_range_count
109 uint64_t current_pn = ack.largest_acknowledged;
110 size_t first_range_count = ack.ranges.empty() ? 0 : ack.ranges[0].length;
111
112 // Process first ack block (largest_acknowledged down to largest - first_range)
113 for (uint64_t i = 0; i <= first_range_count && current_pn > 0; ++i)
114 {
115 auto it = space.sent_packets.find(current_pn - i);
116 if (it != space.sent_packets.end())
117 {
118 auto& pkt = it->second;
119 if (pkt.in_flight)
120 {
121 space.bytes_in_flight -= pkt.sent_bytes;
122 }
123 result.acked_packets.push_back(std::move(pkt));
124 space.sent_packets.erase(it);
125 }
126 }
127
128 // Also check largest_acknowledged itself
129 {
130 auto it = space.sent_packets.find(ack.largest_acknowledged);
131 if (it != space.sent_packets.end())
132 {
133 auto& pkt = it->second;
134 if (pkt.in_flight)
135 {
136 space.bytes_in_flight -= pkt.sent_bytes;
137 }
138 result.acked_packets.push_back(std::move(pkt));
139 space.sent_packets.erase(it);
140 }
141 }
142
143 // Process additional ACK ranges
144 current_pn = ack.largest_acknowledged;
145 if (!ack.ranges.empty())
146 {
147 current_pn -= ack.ranges[0].length + 1; // After first range
148 }
149
150 for (size_t r = 1; r < ack.ranges.size(); ++r)
151 {
152 // Skip the gap
153 auto gap = ack.ranges[r].gap;
154 if (current_pn >= gap + 2)
155 {
156 current_pn -= gap + 2;
157 }
158 else
159 {
160 break;
161 }
162
163 // Process this range
164 auto range_len = ack.ranges[r].length;
165 for (uint64_t i = 0; i <= range_len && current_pn > 0; ++i)
166 {
167 auto it = space.sent_packets.find(current_pn - i);
168 if (it != space.sent_packets.end())
169 {
170 auto& pkt = it->second;
171 if (pkt.in_flight)
172 {
173 space.bytes_in_flight -= pkt.sent_bytes;
174 }
175 result.acked_packets.push_back(std::move(pkt));
176 space.sent_packets.erase(it);
177 }
178 }
179 if (current_pn >= range_len + 1)
180 {
181 current_pn -= range_len + 1;
182 }
183 else
184 {
185 break;
186 }
187 }
188
189 // Update RTT if we got a sample from the largest acknowledged packet
190 if (!result.acked_packets.empty())
191 {
192 // Find the largest acked packet to compute RTT
193 auto largest_it = std::find_if(
194 result.acked_packets.begin(),
195 result.acked_packets.end(),
196 [largest_newly_acked](const sent_packet& p) {
197 return p.packet_number == largest_newly_acked;
198 });
199
200 if (largest_it != result.acked_packets.end())
201 {
202 auto latest_rtt = std::chrono::duration_cast<std::chrono::microseconds>(
203 recv_time - largest_it->sent_time);
204 auto ack_delay_us = std::chrono::microseconds{
205 static_cast<int64_t>(ack.ack_delay)}; // Encoded value
206 rtt_.update(latest_rtt, ack_delay_us, handshake_confirmed_);
207 }
208
209 // Reset PTO count since we got an ACK
210 pto_count_ = 0;
211 }
212
213 // Detect lost packets
214 auto lost = detect_lost_packets(level, recv_time);
215 if (!lost.empty())
216 {
217 result.event = loss_detection_event::packet_lost;
218 result.lost_packets = std::move(lost);
219 }
220
221 // Process ECN counts if present (RFC 9000 Section 13.4, RFC 9002 Section 7.1)
222 if (ack.ecn && !result.acked_packets.empty())
223 {
224 // Find the earliest sent time among acked packets for congestion recovery tracking
225 auto earliest_sent_time = result.acked_packets[0].sent_time;
226 for (const auto& pkt : result.acked_packets)
227 {
228 if (pkt.sent_time < earliest_sent_time)
229 {
230 earliest_sent_time = pkt.sent_time;
231 }
232 }
233
234 auto ecn_signal = ecn_tracker_.process_ecn_counts(
235 *ack.ecn,
236 result.acked_packets.size(),
237 earliest_sent_time);
238
239 result.ecn_signal = ecn_signal;
240 if (ecn_signal == ecn_result::congestion_signal)
241 {
242 result.ecn_congestion_sent_time = ecn_tracker_.last_congestion_sent_time();
243 }
244 }
245
246 set_loss_detection_timer();
247
248 return result;
249}

251auto loss_detector::detect_lost_packets(encryption_level level,
252 std::chrono::steady_clock::time_point now)
253 -> std::vector<sent_packet>
254{
255 auto idx = space_index(level);
256 auto& space = spaces_[idx];
257 std::vector<sent_packet> lost;
258
259 if (!space.largest_acked_set)
260 {
261 return lost;
262 }
263
264 // Calculate loss delay (RFC 9002 Section 6.1.2)
265 auto smoothed = rtt_.smoothed_rtt();
266 auto min_rtt = rtt_.min_rtt();
267 if (min_rtt == std::chrono::microseconds::max())
268 {
269 min_rtt = smoothed;
270 }
271 auto max_rtt = std::max(smoothed, min_rtt);
272 auto loss_delay_us = static_cast<int64_t>(
273 kTimeThreshold * static_cast<double>(max_rtt.count()));
274 auto granularity_us = std::chrono::duration_cast<std::chrono::microseconds>(
275 kGranularity).count();
276 auto loss_delay = std::chrono::microseconds{
277 std::max(loss_delay_us, granularity_us)};
278
279 auto lost_send_time = now - loss_delay;
280
281 // Reset loss time for this space
282 space.loss_time = std::chrono::steady_clock::time_point{};
283
284 for (auto it = space.sent_packets.begin(); it != space.sent_packets.end();)
285 {
286 auto& [pn, packet] = *it;
287
288 if (pn > space.largest_acked)
289 {
290 ++it;
291 continue;
292 }
293
294 // Check if packet is lost by packet threshold or time threshold
295 bool time_lost = packet.sent_time <= lost_send_time;
296 bool reorder_lost = (space.largest_acked >= pn + kPacketThreshold);
297
298 if (time_lost || reorder_lost)
299 {
300 if (packet.in_flight)
301 {
302 space.bytes_in_flight -= packet.sent_bytes;
303 }
304 lost.push_back(std::move(packet));
305 it = space.sent_packets.erase(it);
306 }
307 else
308 {
309 // This packet might be lost later by time threshold
310 auto potential_loss_time = packet.sent_time + loss_delay;
311 if (space.loss_time == std::chrono::steady_clock::time_point{} ||
312 potential_loss_time < space.loss_time)
313 {
314 space.loss_time = potential_loss_time;
315 }
316 ++it;
317 }
318 }
319
320 return lost;
321}

333auto loss_detector::on_timeout() -> loss_detection_result
334{
335 loss_detection_result result;
336 auto now = std::chrono::steady_clock::now();
337
338 auto [loss_time, loss_level] = get_loss_time_and_space();
339 if (loss_time != std::chrono::steady_clock::time_point{} && loss_time <= now)
340 {
341 // Time threshold loss detection
342 auto lost = detect_lost_packets(loss_level, now);
343 if (!lost.empty())
344 {
345 result.event = loss_detection_event::packet_lost;
346 result.lost_packets = std::move(lost);
347 }
348 }
349 else
350 {
351 // PTO timeout
352 result.event = loss_detection_event::pto_expired;
353 ++pto_count_;
354 }
355
356 set_loss_detection_timer();
357
358 return result;
359}

361auto loss_detector::get_loss_time_and_space() const
362 -> std::pair<std::chrono::steady_clock::time_point, encryption_level>
363{
364 auto earliest_loss_time = std::chrono::steady_clock::time_point::max();
365 auto earliest_level = encryption_level::initial;
366
367 for (size_t i = 0; i < spaces_.size(); ++i)
368 {
369 auto& space = spaces_[i];
370 if (space.loss_time != std::chrono::steady_clock::time_point{} &&
371 space.loss_time < earliest_loss_time)
372 {
373 earliest_loss_time = space.loss_time;
374 switch (i)
375 {
376 case 0:
377 earliest_level = encryption_level::initial;
378 break;
379 case 1:
380 earliest_level = encryption_level::handshake;
381 break;
382 case 2:
383 earliest_level = encryption_level::application;
384 break;
385 }
386 }
387 }
388
389 if (earliest_loss_time == std::chrono::steady_clock::time_point::max())
390 {
391 earliest_loss_time = std::chrono::steady_clock::time_point{};
392 }
393
394 return {earliest_loss_time, earliest_level};
395}

397auto loss_detector::get_pto_time_and_space() const
398 -> std::pair<std::chrono::steady_clock::time_point, encryption_level>
399{
400 auto pto_duration = rtt_.pto() * (1 << pto_count_); // Exponential backoff
401 auto earliest_pto_time = std::chrono::steady_clock::time_point::max();
402 auto earliest_level = encryption_level::application;
403
404 for (size_t i = 0; i < spaces_.size(); ++i)
405 {
406 auto& space = spaces_[i];
407
408 // Skip spaces with no ack-eliciting packets
409 bool has_ack_eliciting = false;
410 for (const auto& [pn, pkt] : space.sent_packets)
411 {
412 if (pkt.ack_eliciting)
413 {
414 has_ack_eliciting = true;
415 break;
416 }
417 }
418
419 if (!has_ack_eliciting)
420 {
421 continue;
422 }
423
424 // Don't include application space if handshake not confirmed
425 if (i == 2 && !handshake_confirmed_)
426 {
427 continue;
428 }
429
430 auto pto_time = space.time_of_last_ack_eliciting + pto_duration;
431 if (pto_time < earliest_pto_time)
432 {
433 earliest_pto_time = pto_time;
434 switch (i)
435 {
436 case 0:
437 earliest_level = encryption_level::initial;
438 break;
439 case 1:
440 earliest_level = encryption_level::handshake;
441 break;
442 case 2:
443 earliest_level = encryption_level::application;
444 break;
445 }
446 }
447 }
448
449 if (earliest_pto_time == std::chrono::steady_clock::time_point::max())
450 {
451 earliest_pto_time = std::chrono::steady_clock::time_point{};
452 }
453
454 return {earliest_pto_time, earliest_level};
455}

457auto loss_detector::set_loss_detection_timer() -> void
458{
459 auto [loss_time, loss_level] = get_loss_time_and_space();
460
461 if (loss_time != std::chrono::steady_clock::time_point{})
462 {
463 // Time threshold loss detection timer
464 loss_detection_timer_ = loss_time;
465 timer_armed_ = true;
466 return;
467 }
468
469 // Check if there are any ack-eliciting packets in flight
470 bool any_ack_eliciting = false;
471 for (const auto& space : spaces_)
472 {
473 for (const auto& [pn, pkt] : space.sent_packets)
474 {
475 if (pkt.ack_eliciting)
476 {
477 any_ack_eliciting = true;
478 break;
479 }
480 }
481 if (any_ack_eliciting)
482 {
483 break;
484 }
485 }
486
487 if (!any_ack_eliciting)
488 {
489 timer_armed_ = false;
490 return;
491 }
492
493 auto [pto_time, pto_level] = get_pto_time_and_space();
494 if (pto_time != std::chrono::steady_clock::time_point{})
495 {
496 loss_detection_timer_ = pto_time;
497 timer_armed_ = true;
498 }
499 else
500 {
501 timer_armed_ = false;
502 }
503}
