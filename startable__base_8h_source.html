157 [[nodiscard]] auto do_start(Args&&... args) -> VoidResult
158 {
159 if (!lifecycle_.try_start())
160 {
161 return error_void(
162 error_codes::common_errors::already_exists,
163 std::string(derived().component_name()) + " is already running",
164 "startable_base::do_start",

168 reset_connection_state();

191 [[nodiscard]] auto do_stop() -> VoidResult
192 {
193 if (!lifecycle_.is_running())
194 {
195 return ok();
196 }
197
198 // Prevent multiple stop calls
199 bool expected = false;
200 if (!stop_initiated_.compare_exchange_strong(expected, true,
201 std::memory_order_acq_rel))
202 {
203 return ok();
204 }
205
206 auto result = derived().do_stop_impl();
207 lifecycle_.mark_stopped();
208
209 // Call derived class's on_stopped hook for callbacks
210 derived().on_stopped();
211
212 return result;
213 }
