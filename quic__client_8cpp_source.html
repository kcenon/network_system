79auto messaging_quic_client::start_client(std::string_view host,
80 unsigned short port,
81 const quic_client_config& config) -> VoidResult
82{
83 // Create tracing span for client start operation
84 auto span = tracing::is_tracing_enabled()
85 ? std::make_optional(tracing::trace_context::create_span("quic.client.start"))
86 : std::nullopt;
87 if (span)
88 {
89 span->set_attribute("net.peer.name", host)
90 .set_attribute("net.peer.port", static_cast<int64_t>(port))
91 .set_attribute("net.transport", "quic")
92 .set_attribute("client.id", client_id_);
93 }
94
95 if (lifecycle_.is_running())
96 {
97 if (span)
98 {
99 span->set_error("QUIC client is already running");
100 }
101 return error_void(
102 error_codes::common_errors::already_exists,
103 "QUIC client is already running",
104 "messaging_quic_client::start_client");
105 }
106
107 if (host.empty())
108 {
109 if (span)
110 {
111 span->set_error("Host cannot be empty");
112 }
113 return error_void(
114 error_codes::common_errors::invalid_argument,
115 "Host cannot be empty",
116 "messaging_quic_client::start_client");
117 }
118
119 config_ = config;
120 lifecycle_.set_running();
121 is_connected_.store(false);
122
123 auto result = do_start_impl(host, port);
124 if (result.is_err())
125 {
126 lifecycle_.mark_stopped();
127 if (span)
128 {
129 span->set_error(result.error().message);
130 }
131 }
132 else
133 {
134 if (span)
135 {
136 span->set_status(tracing::span_status::ok);
137 }
138 }
139
140 return result;
141}

143auto messaging_quic_client::stop_client() -> VoidResult
144{
145 if (!lifecycle_.prepare_stop())

157auto messaging_quic_client::do_start_impl(std::string_view host,
158 unsigned short port) -> VoidResult
159{
160 try
161 {
162 {
163 std::lock_guard<std::mutex> lock(socket_mutex_);
164 socket_.reset();
165 }
166
167 io_context_ = std::make_unique<asio::io_context>();
168 work_guard_ = std::make_unique<
169 asio::executor_work_guard<asio::io_context::executor_type>>(
170 asio::make_work_guard(*io_context_));
171
172 thread_pool_ = network_context::instance().get_thread_pool();
173 if (!thread_pool_)
174 {
175 thread_pool_ = std::make_shared<integration::basic_thread_pool>(2);
176 }
177
178 io_context_future_ = thread_pool_->submit([this]() {
179 try
180 {
181 NETWORK_LOG_INFO(
182 "[messaging_quic_client] Starting io_context on thread pool");
183 io_context_->run();
184 NETWORK_LOG_INFO("[messaging_quic_client] io_context stopped");
185 }
186 catch (const std::exception& e)
187 {
188 NETWORK_LOG_ERROR("[messaging_quic_client] Exception in io_context: " +
189 std::string(e.what()));
190 }
191 });
192
193 do_connect(host, port);
194
195 NETWORK_LOG_INFO("[messaging_quic_client] started. ID=" + client_id_ +
196 " target=" + std::string(host) + ":" +
197 std::to_string(port));
198
199 return ok();
200 }
201 catch (const std::exception& e)
202 {
203 return error_void(
204 error_codes::common_errors::internal_error,
205 "Failed to start client: " + std::string(e.what()),
206 "messaging_quic_client::do_start_impl",
207 "Client ID: " + client_id_ + ", Host: " + std::string(host));
208 }
209}

211auto messaging_quic_client::do_stop_impl() -> VoidResult
212{
213 try
214 {
215 std::shared_ptr<internal::quic_socket> local_socket;
216 {
217 std::lock_guard<std::mutex> lock(socket_mutex_);
218 local_socket = std::move(socket_);
219 }
220 if (local_socket)
221 {
222 local_socket->stop_receive();
223 auto close_result = local_socket->close();
224 if (close_result.is_err())
225 {
226 NETWORK_LOG_WARN("[messaging_quic_client] Close error: " +
227 close_result.error().message);
228 }
229 }
230
231 if (work_guard_)
232 {
233 work_guard_.reset();
234 }
235 if (io_context_)
236 {
237 io_context_->stop();
238 }
239 if (io_context_future_.valid())
240 {
241 io_context_future_.wait();
242 }
243 thread_pool_.reset();
244 io_context_.reset();
245
246 handshake_complete_.store(false);
247
248 NETWORK_LOG_INFO("[messaging_quic_client] stopped.");
249 return ok();
250 }
251 catch (const std::exception& e)
252 {
253 return error_void(
254 error_codes::common_errors::internal_error,
255 "Failed to stop client: " + std::string(e.what()),
256 "messaging_quic_client::do_stop_impl",
257 "Client ID: " + client_id_);
258 }
259}

276auto messaging_quic_client::send_packet(std::vector<uint8_t>&& data) -> VoidResult
277{
278 // Create tracing span for send operation
279 auto span = tracing::is_tracing_enabled()
280 ? std::make_optional(tracing::trace_context::create_span("quic.client.send"))
281 : std::nullopt;
282 if (span)
283 {
284 span->set_attribute("net.transport", "quic")
285 .set_attribute("message.size", static_cast<int64_t>(data.size()))
286 .set_attribute("quic.stream_id", static_cast<int64_t>(default_stream_id_))
287 .set_attribute("client.id", client_id_);
288 }
289
290 if (!is_running())
291 {
292 if (span)
293 {
294 span->set_error("Client is not running");
295 }
296 return error_void(
297 error_codes::network_system::connection_closed,
298 "Client is not running",
299 "messaging_quic_client::send_packet",
300 "Client ID: " + client_id_);
301 }
302
303 if (data.empty())
304 {
305 if (span)
306 {
307 span->set_error("Data cannot be empty");
308 }
309 return error_void(
310 error_codes::common_errors::invalid_argument,
311 "Data cannot be empty",
312 "messaging_quic_client::send_packet",
313 "Client ID: " + client_id_);
314 }
315
316 auto local_socket = get_socket();
317 if (!is_connected() || !local_socket)
318 {
319 if (span)
320 {
321 span->set_error("Client is not connected");
322 }
323 return error_void(
324 error_codes::network_system::connection_closed,
325 "Client is not connected",
326 "messaging_quic_client::send_packet",
327 "Client ID: " + client_id_);
328 }
329
330 auto result = local_socket->send_stream_data(default_stream_id_, std::move(data));
331 if (span)
332 {
333 if (result.is_err())
334 {
335 span->set_error(result.error().message);
336 }
337 else
338 {
339 span->set_status(tracing::span_status::ok);
340 }
341 }
342 return result;
343}

502auto messaging_quic_client::do_connect(std::string_view host,
503 unsigned short port) -> void
504{
505 auto resolver = std::make_shared<asio::ip::udp::resolver>(*io_context_);
506 auto self = shared_from_this();
507
508 NETWORK_LOG_INFO("[messaging_quic_client] Starting async resolve for " +
509 std::string(host) + ":" + std::to_string(port));
510
511 resolver->async_resolve(
512 std::string(host),
513 std::to_string(port),
514 [this, self, resolver, host_str = std::string(host)](
515 std::error_code ec,
516 asio::ip::udp::resolver::results_type results) {
517 NETWORK_LOG_INFO("[messaging_quic_client] Resolve callback invoked");
518 if (ec)
519 {
520 NETWORK_LOG_ERROR("[messaging_quic_client] Resolve error: " +
521 ec.message());
522 on_error(ec);
523 return;
524 }
525
526 if (results.empty())
527 {
528 NETWORK_LOG_ERROR("[messaging_quic_client] No endpoints resolved");
529 on_error(asio::error::host_not_found);
530 return;
531 }
532
533 NETWORK_LOG_INFO("[messaging_quic_client] Resolve successful, creating socket");
534
535 try
536 {
537 auto endpoint = *results.begin();
538
539 // Create UDP socket
540 asio::ip::udp::socket udp_socket(*io_context_, asio::ip::udp::v4());
541
542 // Create QUIC socket as client
543 auto quic_sock = std::make_shared<internal::quic_socket>(
544 std::move(udp_socket),
545 internal::quic_role::client);
546
547 // Set callbacks
548 quic_sock->set_stream_data_callback(
549 [this, self](uint64_t stream_id,
550 std::span<const uint8_t> data,
551 bool fin) {
552 on_stream_data(stream_id, data, fin);
553 });
554
555 quic_sock->set_connected_callback([this, self]() {
556 on_connect();
557 });
558
559 quic_sock->set_error_callback([this, self](std::error_code err) {
560 on_error(err);
561 });
562
563 quic_sock->set_close_callback(
564 [this, self](uint64_t error_code, const std::string& reason) {
565 on_close(error_code, reason);
566 });
567
568 // Store socket
569 {
570 std::lock_guard<std::mutex> lock(socket_mutex_);
571 socket_ = quic_sock;
572 }
573
574 // Connect to server
575 auto connect_result = quic_sock->connect(endpoint, host_str);
576 if (connect_result.is_err())
577 {
578 NETWORK_LOG_ERROR("[messaging_quic_client] Connect failed: " +
579 connect_result.error().message);
580 on_error(std::make_error_code(std::errc::connection_refused));
581 return;
582 }
583
584 // Start receive loop
585 quic_sock->start_receive();
586
587 NETWORK_LOG_INFO("[messaging_quic_client] Connection initiated");
588 }
589 catch (const std::exception& e)
590 {
591 NETWORK_LOG_ERROR("[messaging_quic_client] Exception during connect: " +
592 std::string(e.what()));
593 on_error(std::make_error_code(std::errc::connection_refused));
594 }
595 });
596}

598auto messaging_quic_client::on_connect() -> void
599{
600 NETWORK_LOG_INFO("[messaging_quic_client] Connected successfully.");
601 is_connected_.store(true, std::memory_order_release);
602 handshake_complete_.store(true);
603
604 // Create default stream for send_packet()
605 auto local_socket = get_socket();
606 if (local_socket)
607 {
608 auto stream_result = local_socket->create_stream(false);
609 if (stream_result.is_ok())
610 {
611 default_stream_id_ = stream_result.value();
612 NETWORK_LOG_DEBUG("[messaging_quic_client] Default stream created: " +
613 std::to_string(default_stream_id_));
614 }
615 }
616
617 // Invoke connected callback
618 invoke_connected_callback();
619}

621auto messaging_quic_client::on_stream_data(uint64_t stream_id,
622 std::span<const uint8_t> data,
623 bool fin) -> void
624{
625 if (!is_connected())
626 {
627 return;
628 }
629
630 NETWORK_LOG_DEBUG("[messaging_quic_client] Received " +
631 std::to_string(data.size()) +
632 " bytes on stream " + std::to_string(stream_id));
633
634 std::vector<uint8_t> data_copy(data.begin(), data.end());
635
636 // Invoke stream receive callback
637 invoke_stream_receive_callback(stream_id, data_copy, fin);
638
639 // Also invoke default receive callback for default stream
640 if (stream_id == default_stream_id_)
641 {
642 invoke_receive_callback(data_copy);
643 }
644}
