102class basic_container::impl {
103public:
104 impl() = default;
105
106 std::vector<uint8_t> serialize(const std::any& data) const {
107 if (custom_serializer_) {
108 return custom_serializer_(data);
109 }
110
111 // Default serialization for common types
112 try {
113 if (data.type() == typeid(std::string)) {
114 auto str = std::any_cast<std::string>(data);
115 return std::vector<uint8_t>(str.begin(), str.end());
116 } else if (data.type() == typeid(int)) {
117 auto val = std::any_cast<int>(data);
118 std::vector<uint8_t> bytes(sizeof(int));
119 std::memcpy(bytes.data(), &val, sizeof(int));
120 return bytes;
121 } else if (data.type() == typeid(double)) {
122 auto val = std::any_cast<double>(data);
123 std::vector<uint8_t> bytes(sizeof(double));
124 std::memcpy(bytes.data(), &val, sizeof(double));
125 return bytes;
126 } else if (data.type() == typeid(bool)) {
127 auto val = std::any_cast<bool>(data);
128 return {static_cast<uint8_t>(val ? 1 : 0)};
129 }
130 } catch (const std::bad_any_cast&) {
131 // Return empty vector on error
132 }
133
134 return {};
135 }
136
137 std::any deserialize(const std::vector<uint8_t>& bytes) const {
138 if (custom_deserializer_) {
139 return custom_deserializer_(bytes);
140 }
141
142 // Default deserialization - assume string for simplicity
143 if (!bytes.empty()) {
144 return std::any(std::string(bytes.begin(), bytes.end()));
145 }
146
147 return std::any{};
148 }
149
150 void set_serializer(
151 std::function<std::vector<uint8_t>(const std::any&)> serializer
152 ) {
153 custom_serializer_ = serializer;
154 }
155
156 void set_deserializer(
157 std::function<std::any(const std::vector<uint8_t>&)> deserializer
158 ) {
159 custom_deserializer_ = deserializer;
160 }
161
162private:
163 std::function<std::vector<uint8_t>(const std::any&)> custom_serializer_;
164 std::function<std::any(const std::vector<uint8_t>&)> custom_deserializer_;
165};

202class container_manager::impl {
203public:
204 impl() = default;
205
206 void register_container(
207 const std::string& name,
208 std::shared_ptr<container_interface> container
209 ) {
210 std::unique_lock<std::mutex> lock(mutex_);
211 containers_[name] = container;
212 }
213
214 std::shared_ptr<container_interface> get_container(
215 const std::string& name
216 ) const {
217 std::unique_lock<std::mutex> lock(mutex_);
218 auto it = containers_.find(name);
219 if (it != containers_.end()) {
220 return it->second;
221 }
222 return nullptr;
223 }
224
225 void set_default_container(
226 std::shared_ptr<container_interface> container
227 ) {
228 std::unique_lock<std::mutex> lock(mutex_);
229 default_container_ = container;
230 }
231
232 std::shared_ptr<container_interface> get_default_container() {
233 std::unique_lock<std::mutex> lock(mutex_);
234 if (!default_container_) {
235 default_container_ = std::make_shared<basic_container>();
236 }
237 return default_container_;
238 }
239
240 std::vector<uint8_t> serialize(const std::any& data) {
241 return get_default_container()->serialize(data);
242 }
243
244 std::any deserialize(const std::vector<uint8_t>& bytes) {
245 return get_default_container()->deserialize(bytes);
246 }
247
248 std::vector<std::string> list_containers() const {
249 std::unique_lock<std::mutex> lock(mutex_);
250 std::vector<std::string> names;
251 names.reserve(containers_.size());
252 for (const auto& [name, _] : containers_) {
253 names.push_back(name);
254 }
255 return names;
256 }
257
258private:
259 mutable std::mutex mutex_;
260 std::unordered_map<std::string, std::shared_ptr<container_interface>> containers_;
261 std::shared_ptr<container_interface> default_container_;
262};
