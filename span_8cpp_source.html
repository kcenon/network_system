48struct span::impl
49{
50 std::string name;
51 trace_context context;
52 span_kind kind;
53 span_status status{span_status::unset};
54 std::string status_description;
55 std::map<std::string, attribute_value> attributes;
56 std::vector<span_event> events;
57 std::chrono::steady_clock::time_point start_time;
58 std::chrono::steady_clock::time_point end_time;
59 bool ended{false};
60 trace_context previous_context;
61 span* owner{nullptr};
62
63 impl(std::string_view span_name, trace_context ctx, span_kind span_kind_value)
64 : name(span_name)
65 , context(std::move(ctx))
66 , kind(span_kind_value)
67 , start_time(std::chrono::steady_clock::now())
68 , end_time(start_time)
69 , previous_context(trace_context::current())
70 {
71 // Set this span's context as the current context
72 trace_context::set_current(context);
73 }
74
75 ~impl()
76 {
77 // Note: do_end() is now called from span::~span() before impl_ destruction
78 // to ensure owner->impl_ is still valid when export_span is called.
79 // This is necessary because libc++'s unique_ptr sets its internal pointer
80 // to null before calling the deleter, which would cause owner->impl_ to
81 // be null when export_span tries to access span data.
82 }
83
84 void do_end()
85 {
86 if (ended)
87 {
88 return;
89 }
90
91 ended = true;
92 end_time = std::chrono::steady_clock::now();
93
94 // Restore previous context
95 if (previous_context.is_valid())
96 {
97 trace_context::set_current(previous_context);
98 }
99 else
100 {
101 trace_context::clear_current();
102 }
103
104 // Export the span
105 if (owner)
106 {
107 export_span(*owner);
108 }
109 }
110};

236{

241

314auto span::duration() const noexcept -> std::chrono::nanoseconds
315{
316 if (!impl_)
317 {
318 return std::chrono::nanoseconds::zero();
319 }
320
321 auto end = impl_->ended ? impl_->end_time : std::chrono::steady_clock::now();
322 return std::chrono::duration_cast<std::chrono::nanoseconds>(end - impl_->start_time);
