83// ============================================================================
84// Base Socket Concept
85// ============================================================================
86
105template <typename T>
106concept Socket = requires(T& socket) {
107 { socket.close() } -> std::same_as<void>;
108 { socket.is_closed() } -> std::convertible_to<bool>;
109};

111// ============================================================================
112// Stream Socket Concept (TCP, Secure TCP)
113// ============================================================================
114
139template <typename T>
140concept StreamSocket = Socket<T> && requires(
141 T& socket, std::vector<uint8_t> data,
142 std::function<void(std::error_code, std::size_t)> send_handler,
143 std::function<void(const std::vector<uint8_t>&)> recv_callback,
144 std::function<void(std::span<const uint8_t>)> recv_view_callback,
145 std::function<void(std::error_code)> error_callback) {
146 // Async send operation
147 { socket.async_send(std::move(data), send_handler) } -> std::same_as<void>;
148
149 // Read loop control
150 { socket.start_read() } -> std::same_as<void>;
151 { socket.stop_read() } -> std::same_as<void>;
152
153 // Callback registration
154 { socket.set_receive_callback(recv_callback) } -> std::same_as<void>;
155 { socket.set_receive_callback_view(recv_view_callback) } -> std::same_as<void>;
156 { socket.set_error_callback(error_callback) } -> std::same_as<void>;
157};

187template <typename T>
188concept DatagramSocket = Socket<T> && requires(T& socket) {
189 // Receive loop control
190 { socket.start_receive() } -> std::same_as<void>;
191 { socket.stop_receive() } -> std::same_as<void>;
192
193 // Error callback registration
194 {
195 socket.set_error_callback(std::function<void(std::error_code)>{})
196 } -> std::same_as<void>;
197};

249concept MessageSocket = requires(

251 std::function<void(std::error_code, std::size_t)> handler) {
252 // Connection state

264// ============================================================================
265// Backpressure-Aware Socket Concept
266// ============================================================================
267
291template <typename T>
292concept BackpressureAwareSocket = StreamSocket<T> && requires(
293 T& socket, std::vector<uint8_t> data,
294 std::function<void(std::error_code, std::size_t)> handler,
295 std::function<void(bool)> bp_callback) {
296 // Backpressure state
297 { socket.pending_bytes() } -> std::convertible_to<std::size_t>;
298 { socket.is_backpressure_active() } -> std::convertible_to<bool>;
299
300 // Non-blocking send
301 { socket.try_send(std::move(data), handler) } -> std::convertible_to<bool>;
302
303 // Backpressure notification
304 { socket.set_backpressure_callback(bp_callback) } -> std::same_as<void>;
305};
