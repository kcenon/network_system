48auto generate_unique_id(std::string_view prefix) -> std::string
49{
50 static std::atomic<uint64_t> counter{0};
51 auto now = std::chrono::steady_clock::now().time_since_epoch().count();
52 auto count = counter.fetch_add(1);
53
54 std::ostringstream oss;
55 oss << prefix << "-" << now << "-" << count;
56 return oss.str();
57}

62auto parse_url(std::string_view url) -> std::pair<std::string, uint16_t>
63{
64 std::string url_str(url);
65
66 // Remove quic:// prefix if present
67 const std::string quic_prefix = "quic://";
68 if (url_str.substr(0, quic_prefix.size()) == quic_prefix) {
69 url_str = url_str.substr(quic_prefix.size());
70 }
71
72 // Find the last colon for port separation
73 auto colon_pos = url_str.rfind(':');
74 if (colon_pos == std::string::npos) {
75 return {"", 0};
76 }
77
78 std::string host = url_str.substr(0, colon_pos);
79 std::string port_str = url_str.substr(colon_pos + 1);
80
81 uint16_t port = 0;
82 try {
83 port = static_cast<uint16_t>(std::stoi(port_str));
84 } catch (...) {
85 return {"", 0};
86 }
87
88 return {host, port};
89}

113auto connect(const unified::endpoint_info& endpoint,
114 const quic_config& config,
115 std::string_view id) -> std::unique_ptr<unified::i_connection>
116{
117 // Ensure server_name is set for TLS
118 quic_config cfg = config;
119 if (cfg.server_name.empty()) {
120 cfg.server_name = endpoint.host;
121 }
122
123 auto conn = create_connection(cfg, id);
124 // Initiate connection (async, will complete in background)
125 (void)conn->connect(endpoint);
126 return conn;
127}

129auto connect(std::string_view url,
130 const quic_config& config,
131 std::string_view id) -> std::unique_ptr<unified::i_connection>
132{
133 auto [host, port] = parse_url(url);
134 if (host.empty() || port == 0) {
135 // Return a connection that will fail on connect
136 return create_connection(config, id);
137 }
138
139 // Ensure server_name is set for TLS
140 quic_config cfg = config;
141 if (cfg.server_name.empty()) {
142 cfg.server_name = host;
143 }
144
145 auto conn = create_connection(cfg, id);
146 (void)conn->connect(unified::endpoint_info{host, port});
147 return conn;
148}
