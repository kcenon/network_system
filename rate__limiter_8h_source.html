107class rate_limiter {
108public:
113 explicit rate_limiter(rate_limiter_config config = {})
114 : config_(std::move(config))
115 , last_cleanup_(std::chrono::steady_clock::now()) {}
116
123 [[nodiscard]] bool allow(std::string_view client_id) {
124 std::unique_lock<std::shared_mutex> lock(mutex_);
125
126 auto now = std::chrono::steady_clock::now();
127
128 // Periodic cleanup
129 if (config_.auto_cleanup) {
130 maybe_cleanup(now);
131 }
132
133 std::string key(client_id);
134 auto& bucket = buckets_[key];
135
136 // Initialize new bucket
137 if (bucket.last_refill.time_since_epoch().count() == 0) {
138 bucket.tokens = static_cast<double>(config_.burst_size);
139 bucket.last_refill = now;
140 }
141
142 // Refill tokens based on elapsed time
143 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
144 now - bucket.last_refill);
145
146 double tokens_to_add = elapsed.count() *
147 (static_cast<double>(config_.max_requests_per_second) / 1000.0);
148
149 bucket.tokens = std::min(
150 static_cast<double>(config_.burst_size),
151 bucket.tokens + tokens_to_add);
152 bucket.last_refill = now;
153
154 // Try to consume a token
155 if (bucket.tokens >= 1.0) {
156 bucket.tokens -= 1.0;
157 return true;
158 }
159
160 return false;
161 }
162
174 [[nodiscard]] bool allow(std::string_view client_id, std::string_view session_id) {
175 if (session_id.empty()) {
176 return allow(client_id);
177 }
178 std::string composite_key;
179 composite_key.reserve(client_id.size() + 1 + session_id.size());
180 composite_key.append(client_id);
181 composite_key.push_back(':');
182 composite_key.append(session_id);
183 return allow(std::string_view(composite_key));
184 }
185
192 [[nodiscard]] bool would_allow(std::string_view client_id) const {
193 std::shared_lock<std::shared_mutex> lock(mutex_);
194
195 auto it = buckets_.find(std::string(client_id));
196 if (it == buckets_.end()) {
197 return true; // New client would have full bucket
198 }
199
200 auto now = std::chrono::steady_clock::now();
201 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
202 now - it->second.last_refill);
203
204 double tokens_to_add = elapsed.count() *
205 (static_cast<double>(config_.max_requests_per_second) / 1000.0);
206
207 double available = std::min(
208 static_cast<double>(config_.burst_size),
209 it->second.tokens + tokens_to_add);
210
211 return available >= 1.0;
212 }
213
220 [[nodiscard]] double remaining_tokens(std::string_view client_id) const {
221 std::shared_lock<std::shared_mutex> lock(mutex_);
222
223 auto it = buckets_.find(std::string(client_id));
224 if (it == buckets_.end()) {
225 return static_cast<double>(config_.burst_size);
226 }
227
228 auto now = std::chrono::steady_clock::now();
229 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
230 now - it->second.last_refill);
231
232 double tokens_to_add = elapsed.count() *
233 (static_cast<double>(config_.max_requests_per_second) / 1000.0);
234
235 return std::min(
236 static_cast<double>(config_.burst_size),
237 it->second.tokens + tokens_to_add);
238 }
239
244 void reset(std::string_view client_id) {
245 std::unique_lock<std::shared_mutex> lock(mutex_);
246 buckets_.erase(std::string(client_id));
247 }
248
252 void reset_all() {
253 std::unique_lock<std::shared_mutex> lock(mutex_);
254 buckets_.clear();
255 }
256
261 [[nodiscard]] size_t client_count() const {
262 std::shared_lock<std::shared_mutex> lock(mutex_);
263 return buckets_.size();
264 }
265
270 void set_config(rate_limiter_config config) {
271 std::unique_lock<std::shared_mutex> lock(mutex_);
272 config_ = std::move(config);
273 }
274
279 [[nodiscard]] rate_limiter_config config() const {
280 std::shared_lock<std::shared_mutex> lock(mutex_);
281 return config_;
282 }
283
284private:
285 struct bucket {
286 double tokens = 0.0;
287 std::chrono::steady_clock::time_point last_refill{};
288 };
289
290 void maybe_cleanup(std::chrono::steady_clock::time_point now) {
291 // Cleanup at most once per minute
292 auto since_cleanup = std::chrono::duration_cast<std::chrono::seconds>(
293 now - last_cleanup_);
294
295 if (since_cleanup < std::chrono::seconds(60)) {
296 return;
297 }
298
299 last_cleanup_ = now;
300
301 // Remove stale entries
302 for (auto it = buckets_.begin(); it != buckets_.end();) {
303 auto since_refill = std::chrono::duration_cast<std::chrono::seconds>(
304 now - it->second.last_refill);
305
306 if (since_refill > config_.stale_timeout) {
307 it = buckets_.erase(it);
308 } else {
309 ++it;
310 }
311 }
312 }
313
314 rate_limiter_config config_;
315 mutable std::shared_mutex mutex_;
316 std::unordered_map<std::string, bucket> buckets_;
317 std::chrono::steady_clock::time_point last_cleanup_;
318};

330class connection_limiter {
331public:
336 explicit connection_limiter(size_t max_connections = 1000)
337 : max_connections_(max_connections)
338 , current_connections_(0) {}
339
344 [[nodiscard]] bool can_accept() const noexcept {
345 return current_connections_.load(std::memory_order_acquire) < max_connections_;
346 }
347
352 [[nodiscard]] bool try_accept() noexcept {
353 size_t current = current_connections_.load(std::memory_order_acquire);
354 while (current < max_connections_) {
355 if (current_connections_.compare_exchange_weak(
356 current, current + 1,
357 std::memory_order_acq_rel,
358 std::memory_order_acquire)) {
359 return true;
360 }
361 }
362 return false;
363 }
364
370 void on_connect() noexcept {
371 current_connections_.fetch_add(1, std::memory_order_release);
372 }
373
377 void on_disconnect() noexcept {
378 size_t prev = current_connections_.fetch_sub(1, std::memory_order_release);
379 // Prevent underflow (shouldn't happen in correct usage)
380 if (prev == 0) {
381 current_connections_.fetch_add(1, std::memory_order_release);
382 }
383 }
384
389 [[nodiscard]] size_t current() const noexcept {
390 return current_connections_.load(std::memory_order_acquire);
391 }
392
397 [[nodiscard]] size_t max() const noexcept {
398 return max_connections_;
399 }
400
405 void set_max(size_t max_connections) noexcept {
406 max_connections_ = max_connections;
407 }
408
413 [[nodiscard]] size_t available() const noexcept {
414 size_t current = current_connections_.load(std::memory_order_acquire);
415 return (current < max_connections_) ? (max_connections_ - current) : 0;
416 }
417
422 [[nodiscard]] bool at_capacity() const noexcept {
423 return current_connections_.load(std::memory_order_acquire) >= max_connections_;
424 }
425
426private:
427 size_t max_connections_;
428 std::atomic<size_t> current_connections_;
429};

506class per_client_connection_limiter {
507public:
513 explicit per_client_connection_limiter(
514 size_t max_per_client = 10,
515 size_t max_total = 1000)
516 : max_per_client_(max_per_client)
517 , total_limiter_(max_total) {}
518
524 [[nodiscard]] bool try_accept(std::string_view client_id) {
525 // First check total limit
526 if (!total_limiter_.can_accept()) {
527 return false;
528 }
529
530 std::unique_lock<std::mutex> lock(mutex_);
531 std::string key(client_id);
532
533 auto& count = client_connections_[key];
534 if (count >= max_per_client_) {
535 return false;
536 }
537
538 if (!total_limiter_.try_accept()) {
539 return false;
540 }
541
542 ++count;
543 return true;
544 }
545
550 void release(std::string_view client_id) {
551 std::unique_lock<std::mutex> lock(mutex_);
552 std::string key(client_id);
553
554 auto it = client_connections_.find(key);
555 if (it != client_connections_.end() && it->second > 0) {
556 --it->second;
557 if (it->second == 0) {
558 client_connections_.erase(it);
559 }
560 total_limiter_.on_disconnect();
561 }
562 }
563
569 [[nodiscard]] size_t client_connections(std::string_view client_id) const {
570 std::unique_lock<std::mutex> lock(mutex_);
571 auto it = client_connections_.find(std::string(client_id));
572 return (it != client_connections_.end()) ? it->second : 0;
573 }
574
579 [[nodiscard]] size_t total_connections() const noexcept {
580 return total_limiter_.current();
581 }
582
583private:
584 size_t max_per_client_;
585 connection_limiter total_limiter_;
586 mutable std::mutex mutex_;
587 std::unordered_map<std::string, size_t> client_connections_;
588};
