59 auto messaging_udp_server::start_server(uint16_t port) -> VoidResult
60 {
61 if (lifecycle_.is_running())
62 {
63 return error_void(
64 error_codes::network_system::server_already_running,
65 "UDP server is already running",
66 "messaging_udp_server::start_server");
67 }
68
69 // Mark as running before starting
70 lifecycle_.set_running();
71
72 auto result = do_start_impl(port);
73 if (result.is_err())
74 {
75 lifecycle_.mark_stopped();
76 }
77
78 return result;
79 }

81 auto messaging_udp_server::stop_server() -> VoidResult
82 {
83 if (!lifecycle_.prepare_stop())
84 {
85 return ok(); // Not running or already stopping
86 }
87
88 auto result = do_stop_impl();
89
90 lifecycle_.mark_stopped();
91
92 return result;
93 }

128 auto messaging_udp_server::send_to(
129 const interfaces::i_udp_server::endpoint_info& endpoint,
130 std::vector<uint8_t>&& data,
131 interfaces::i_udp_server::send_callback_t handler) -> VoidResult
132 {
133 if (!lifecycle_.is_running())
134 {
135 return error_void(
136 error_codes::network_system::server_not_started,
137 "UDP server is not running",
138 "messaging_udp_server::send_to",
139 ""
140 );
141 }
142
143 if (!socket_)
144 {
145 return error_void(
146 error_codes::common_errors::internal_error,
147 "Socket not available",
148 "messaging_udp_server::send_to",
149 ""
150 );
151 }
152
153 try
154 {
155 // Convert endpoint_info to asio::ip::udp::endpoint
156 asio::ip::udp::endpoint asio_endpoint(
157 asio::ip::make_address(endpoint.address),
158 endpoint.port
159 );
160
161 socket_->async_send_to(std::move(data), asio_endpoint, std::move(handler));
162 return ok();
163 }
164 catch (const std::exception& e)
165 {
166 return error_void(
167 error_codes::common_errors::internal_error,
168 std::string("Failed to send datagram: ") + e.what(),
169 "messaging_udp_server::send_to",
170 "Target: " + endpoint.address + ":" + std::to_string(endpoint.port)
171 );
172 }

175 auto messaging_udp_server::set_receive_callback(
176 interfaces::i_udp_server::receive_callback_t callback) -> void
177 {
178 if (!callback)
179 {
180 // Clear the callback
181 callbacks_.set<to_index(callback_index::receive)>(nullptr);
182 return;
183 }
184
185 // Adapt the interface callback to the internal callback type
186 // Convert asio::ip::udp::endpoint to endpoint_info
187 callbacks_.set<to_index(callback_index::receive)>(
188 [callback = std::move(callback)](
189 const std::vector<uint8_t>& data,
190 const asio::ip::udp::endpoint& endpoint)
191 {
192 interfaces::i_udp_server::endpoint_info info;
193 info.address = endpoint.address().to_string();
194 info.port = endpoint.port();
195 callback(data, info);
196 });
197 }

213 auto messaging_udp_server::do_start_impl(uint16_t port) -> VoidResult
214 {
215 try
216 {
217 // Create io_context
218 io_context_ = std::make_unique<asio::io_context>();
219
220 // Create and bind UDP socket
221 asio::ip::udp::socket raw_socket(*io_context_, udp::endpoint(udp::v4(), port));
222
223 // Wrap in our udp_socket
224 socket_ = std::make_shared<internal::udp_socket>(std::move(raw_socket));
225
226 // Set callbacks to socket
227 auto receive_cb = get_receive_callback();
228 if (receive_cb)
229 {
230 socket_->set_receive_callback(std::move(receive_cb));
231 }
232
233 auto error_cb = get_error_callback();
234 if (error_cb)
235 {
236 socket_->set_error_callback(std::move(error_cb));
237 }
238
239 // Start receiving
240 socket_->start_receive();
241
242 // Get thread pool from network context
243 thread_pool_ = network_context::instance().get_thread_pool();
244 if (!thread_pool_) {
245 // Fallback: create a temporary thread pool if network_context is not initialized
246 NETWORK_LOG_WARN("[messaging_udp_server] network_context not initialized, creating temporary thread pool");
247 thread_pool_ = std::make_shared<integration::basic_thread_pool>(std::thread::hardware_concurrency());
248 }
249
250 // Start io_context on thread pool
251 io_context_future_ = thread_pool_->submit(
252 [this]()
253 {
254 try
255 {
256 NETWORK_LOG_DEBUG("[messaging_udp_server] io_context started");
257 io_context_->run();
258 NETWORK_LOG_DEBUG("[messaging_udp_server] io_context stopped");
259 }
260 catch (const std::exception& e)
261 {
262 NETWORK_LOG_ERROR(std::string("Worker thread exception: ") + e.what());
263 }
264 });
265
266 NETWORK_LOG_INFO("UDP server started on port " + std::to_string(port));
267
268 return ok();
269 }
270 catch (const std::system_error& e)
271 {
272 return error_void(
273 error_codes::network_system::bind_failed,
274 std::string("Failed to bind UDP socket: ") + e.what(),
275 "messaging_udp_server::do_start_impl",
276 "Port: " + std::to_string(port)
277 );
278 }
279 catch (const std::exception& e)
280 {
281 return error_void(
282 error_codes::common_errors::internal_error,
283 std::string("Failed to start UDP server: ") + e.what(),
284 "messaging_udp_server::do_start_impl",
285 "Port: " + std::to_string(port)
286 );
287 }
288 }
289
290 auto messaging_udp_server::do_stop_impl() -> VoidResult
291 {
292 try
293 {
294 // Stop receiving
295 if (socket_)
296 {
297 socket_->stop_receive();
298 }
299
300 // Stop io_context
301 if (io_context_)
302 {
303 io_context_->stop();
304 }
305
306 // Join worker thread
307 if (io_context_future_.valid())
308 {
309 try {
310 io_context_future_.wait();
311 } catch (const std::exception& e) {
312 NETWORK_LOG_ERROR("[messaging_udp_server] Exception while waiting for io_context: " +
313 std::string(e.what()));
314 }
315 }
316
317 // Clean up
318 socket_.reset();
319 io_context_.reset();
320
321 NETWORK_LOG_INFO("UDP server stopped");
322
323 return ok();
324 }
325 catch (const std::exception& e)
326 {
327 return error_void(
328 error_codes::common_errors::internal_error,
329 std::string("Failed to stop UDP server: ") + e.what(),
330 "messaging_udp_server::do_stop_impl",
331 ""
332 );
333 }
334 }
