116auto ws_client_adapter::send(std::vector<uint8_t>&& data) -> VoidResult
117{
118 if (!client_)
119 {
120 return error_void(
121 error_codes::common_errors::internal_error,
122 "Client not initialized",
123 "ws_client_adapter::send"
124 );
125 }
126
127 if (!client_->is_connected())
128 {
129 return error_void(
130 error_codes::common_errors::invalid_argument,
131 "Client is not connected - call start() first",
132 "ws_client_adapter::send"
133 );
134 }
135
136 // Send as binary WebSocket frame
137 return client_->send_binary(std::move(data));

179auto ws_client_adapter::setup_internal_callbacks() -> void
180{
181 if (!client_)
182 {
183 return;
184 }
185
186 // Bridge WebSocket binary callback to i_protocol_client receive callback
187 client_->set_binary_callback(
188 [this](const std::vector<uint8_t>& data)
189 {
190 std::shared_ptr<interfaces::connection_observer> observer_copy;
191 receive_callback_t callback_copy;
192 {
193 std::lock_guard<std::mutex> lock(callbacks_mutex_);
194 observer_copy = observer_;
195 callback_copy = receive_callback_;
196 }
197
198 if (observer_copy)
199 {
200 observer_copy->on_receive(data);
201 }
202 if (callback_copy)
203 {
204 callback_copy(data);
205 }
206 });
207
208 // Bridge WebSocket connected callback
209 client_->set_connected_callback(
210 [this]()
211 {
212 std::shared_ptr<interfaces::connection_observer> observer_copy;
213 connected_callback_t callback_copy;
214 {
215 std::lock_guard<std::mutex> lock(callbacks_mutex_);
216 observer_copy = observer_;
217 callback_copy = connected_callback_;
218 }
219
220 if (observer_copy)
221 {
222 observer_copy->on_connected();
223 }
224 if (callback_copy)
225 {
226 callback_copy();
227 }
228 });
229
230 // Bridge WebSocket disconnected callback (ignore close code and reason)
231 client_->set_disconnected_callback(
232 [this](uint16_t /* code */, std::string_view /* reason */)
233 {
234 std::shared_ptr<interfaces::connection_observer> observer_copy;
235 disconnected_callback_t callback_copy;
236 {
237 std::lock_guard<std::mutex> lock(callbacks_mutex_);
238 observer_copy = observer_;
239 callback_copy = disconnected_callback_;
240 }
241
242 if (observer_copy)
243 {
244 observer_copy->on_disconnected();
245 }
246 if (callback_copy)
247 {
248 callback_copy();
249 }
250 });
251
252 // Bridge WebSocket error callback
253 client_->set_error_callback(
254 [this](std::error_code ec)
255 {
256 std::shared_ptr<interfaces::connection_observer> observer_copy;
257 error_callback_t callback_copy;
258 {
259 std::lock_guard<std::mutex> lock(callbacks_mutex_);
260 observer_copy = observer_;
261 callback_copy = error_callback_;
262 }
263
264 if (observer_copy)
265 {
266 observer_copy->on_error(ec);
267 }
268 if (callback_copy)
269 {
270 callback_copy(ec);
271 }
272 });
273}
