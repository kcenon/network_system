41{

106auto pmtud_controller::should_probe(
107 std::chrono::steady_clock::time_point now) const noexcept -> bool
108{
109 // Don't probe if disabled or probe already in flight
110 if (state_ == pmtud_state::disabled || probe_in_flight_)
111 {
112 return false;
113 }
114
115 // In base state, always ready to start probing
116 if (state_ == pmtud_state::base)
117 {
118 return true;
119 }
120
121 // In searching state, check if enough time has passed
122 if (state_ == pmtud_state::searching)
123 {
124 auto elapsed = now - last_probe_time_;
125 return elapsed >= config_.probe_interval;
126 }
127
128 // In search_complete state, check if re-validation is needed
129 if (state_ == pmtud_state::search_complete)
130 {
131 auto elapsed = now - search_complete_time_;
132 return elapsed >= config_.confirmation_interval;
133 }
134
135 // In error state, check if enough time for recovery
136 if (state_ == pmtud_state::error)
137 {
138 auto elapsed = now - last_probe_time_;
139 return elapsed >= config_.probe_timeout;
140 }
141
142 return false;
143}

145auto pmtud_controller::probe_size() const noexcept -> std::optional<size_t>
146{
147 if (state_ == pmtud_state::disabled)
148 {
149 return std::nullopt;
150 }
151
152 if (state_ == pmtud_state::base || state_ == pmtud_state::searching)
153 {
154 return probing_mtu_ > 0 ? std::optional<size_t>(probing_mtu_) : std::nullopt;
155 }
156
157 // For re-validation in search_complete state
158 if (state_ == pmtud_state::search_complete)
159 {
160 return current_mtu_;
161 }
162
163 return std::nullopt;
164}

175void pmtud_controller::on_probe_acked(size_t size)
176{
177 probe_in_flight_ = false;
178 consecutive_failures_ = 0;
179
180 // Handle based on current state
181 if (state_ == pmtud_state::base || state_ == pmtud_state::searching)
182 {
183 // Successful probe - update validated MTU
184 if (size > current_mtu_)
185 {
186 current_mtu_ = size;
187 search_low_ = size;
188 }
189
190 // Check if we've found the maximum
191 if (search_high_ - search_low_ <= config_.probe_step)
192 {
193 complete_search();
194 }
195 else
196 {
197 // Continue searching
198 state_ = pmtud_state::searching;
199 probe_count_ = 0;
200 probing_mtu_ = calculate_next_probe_size();
201 }
202 }
203 else if (state_ == pmtud_state::search_complete)
204 {
205 // Re-validation successful
206 search_complete_time_ = std::chrono::steady_clock::now();
207 }
208 else if (state_ == pmtud_state::error)
209 {
210 // Recovery successful
211 state_ = pmtud_state::searching;
212 probe_count_ = 0;
213 probing_mtu_ = calculate_next_probe_size();
214 }
215}

217void pmtud_controller::on_probe_lost(size_t size)
218{
219 probe_in_flight_ = false;
220 ++consecutive_failures_;
221
222 // Check for black hole
223 if (consecutive_failures_ >= kBlackHoleThreshold)
224 {
225 handle_black_hole();
226 return;
227 }
228
229 if (state_ == pmtud_state::base || state_ == pmtud_state::searching)
230 {
231 if (probe_count_ >= config_.max_probes)
232 {
233 // Too many failures at this size - reduce search range
234 search_high_ = size;
235 probe_count_ = 0;
236
237 // Check if we've converged
238 if (search_high_ - search_low_ <= config_.probe_step)
239 {
240 complete_search();
241 }
242 else
243 {
244 probing_mtu_ = calculate_next_probe_size();
245 }
246 }
247 // Otherwise, will retry the same size on next should_probe()
248 }
249 else if (state_ == pmtud_state::search_complete)
250 {
251 // Re-validation failed - MTU may have decreased
252 // Fall back to error state and restart search
253 state_ = pmtud_state::error;
254 search_high_ = current_mtu_;
255 search_low_ = config_.min_mtu;
256 current_mtu_ = config_.min_mtu;
257 probe_count_ = 0;
258 }
259}

261void pmtud_controller::on_packet_too_big(size_t reported_mtu)
262{
263 // RFC 8899: ICMP PTB should trigger immediate MTU reduction
264 if (reported_mtu >= config_.min_mtu && reported_mtu < current_mtu_)
265 {
266 current_mtu_ = reported_mtu;
267 search_high_ = reported_mtu;
268
269 if (state_ == pmtud_state::search_complete)
270 {
271 // Need to restart search with new upper bound
272 state_ = pmtud_state::searching;
273 probe_count_ = 0;
274 probing_mtu_ = calculate_next_probe_size();
275 }
276 }
277 else if (reported_mtu < config_.min_mtu)
278 {
279 // PTB reports MTU below QUIC minimum - possible black hole
280 handle_black_hole();
281 }
282}

284auto pmtud_controller::next_timeout() const noexcept
285 -> std::optional<std::chrono::steady_clock::time_point>
286{
287 if (state_ == pmtud_state::disabled)
288 {
289 return std::nullopt;
290 }
291
292 if (probe_in_flight_)
293 {
294 // Timeout for probe in flight
295 return last_probe_time_ + config_.probe_timeout;
296 }
297
298 if (state_ == pmtud_state::searching || state_ == pmtud_state::base)
299 {
300 return last_probe_time_ + config_.probe_interval;
301 }
302
303 if (state_ == pmtud_state::search_complete)
304 {
305 return search_complete_time_ + config_.confirmation_interval;
306 }
307
308 if (state_ == pmtud_state::error)
309 {
310 return last_probe_time_ + config_.probe_timeout;
311 }
312
313 return std::nullopt;
314}

316void pmtud_controller::on_timeout(std::chrono::steady_clock::time_point now)
317{
318 if (state_ == pmtud_state::disabled)
319 {
320 return;
321 }
322
323 if (probe_in_flight_)
324 {
325 // Probe timeout - treat as loss
326 auto elapsed = now - last_probe_time_;
327 if (elapsed >= config_.probe_timeout)
328 {
329 on_probe_lost(probing_mtu_);
330 }
331 }

335{
336 state_ = pmtud_state::searching;

343auto pmtud_controller::calculate_next_probe_size() const noexcept -> size_t
344{
345 // Binary search - probe the midpoint
346 size_t mid = search_low_ + (search_high_ - search_low_) / 2;
347
348 // Round up to ensure we make progress
349 if (mid == search_low_ && search_high_ > search_low_)
350 {
351 mid = search_low_ + config_.probe_step;
352 }
353
354 return std::min(mid, search_high_);
355}

366{

368 state_ = pmtud_state::error;
369 current_mtu_ = config_.min_mtu;
370 search_low_ = config_.min_mtu;
371 search_high_ = config_.max_probe_mtu;
